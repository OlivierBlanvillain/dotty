101   private def parseStats(sourceCode: String)(implicit ctx: Context): List[untpd.Tree] = {
102     val source = new SourceFile("<console>", sourceCode.toCharArray)
103     val parser = new Parser(source)
104     val stats = parser.blockStatSeq()
105     parser.accept(Tokens.EOF)
106     stats
107   }
108 
109   /** Extract a `ParseResult` from the string `sourceCode` */
110   def apply(sourceCode: String)(implicit ctx: Context): ParseResult =
111     sourceCode match {
112       case "" => Newline
113       case CommandExtract(cmd, arg) => cmd match {
114         case Quit.command => Quit
115         case Help.command => Help
116         case Reset.command => Reset
117         case Imports.command => Imports
118         case Load.command => Load(arg)
119         case TypeOf.command => TypeOf(arg)
120         case _ => UnknownCommand(cmd)
121       }
122       case _ => {
123         val stats = parseStats(sourceCode)
124 
125         if (ctx.reporter.hasErrors) {
126           SyntaxErrors(sourceCode,
127                        ctx.flushBufferedMessages(),
128                        stats)
129         }
130         else
131           Parsed(sourceCode, stats)
132       }
133     }
134 
135   /** Check if the input is incomplete
136    *
137    *  This can be used in order to check if a newline can be inserted without
138    *  having to evaluate the expression
139    */
140   def isIncomplete(sourceCode: String)(implicit ctx: Context): Boolean =
141     sourceCode match {
142       case CommandExtract(_) | "" => false
143       case _ => {
144         val reporter = storeReporter
145         var needsMore = false
146         reporter.withIncompleteHandler(_ => _ => needsMore = true) {
147           parseStats(sourceCode)(ctx.fresh.setReporter(reporter))
148           !reporter.hasErrors && needsMore
149         }
150       }
151     }
152 }
