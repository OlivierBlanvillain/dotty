8 /**
9  * Prints stuff to an ad-hoc logging file when running the repl or terminal in
10  * development mode
11  *
12  * Very handy for the common case where you're debugging terminal interactions
13  * and cannot use `println` because it will stomp all over your already messed
14  * up terminal state and block debugging. With [[Debug]], you can have a
15  * separate terminal open tailing the log file and log as verbosely as you
16  * want without affecting the primary terminal you're using to interact with
17  * Ammonite.
18  */

21     new FileOutputStream(new JFile("terminal/target/log"))
22 
23   def apply(s: Any) =
24     if (System.getProperty("ammonite-sbt-build") == "true")
25       debugOutput.write((System.currentTimeMillis() + "\t\t" + s + "\n").getBytes)
26 }

124 case class LazyList[T](headThunk: () => T, tailThunk: () => LazyList[T]) {
125   var rendered = false
126   lazy val head = {
127     rendered = true
128     headThunk()
129   }
130 
131   lazy val tail = tailThunk()
132 
133   def dropPrefix(prefix: Seq[T]) = {
134     @tailrec def rec(n: Int, l: LazyList[T]): Option[LazyList[T]] = {
135       if (n >= prefix.length) Some(l)
136       else if (prefix(n) == l.head) rec(n + 1, l.tail)
137       else None
138     }
139     rec(0, this)
140   }
141   override def toString = {
142 
143     @tailrec def rec(l: LazyList[T], res: List[T]): List[T] = {
144       if (l.rendered) rec(l.tailThunk(), l.head :: res)
145       else res
146     }
147     s"LazyList(${(rec(this, Nil).reverse ++ Seq("...")).mkString(",")})"
148   }
149 
150   def ~:(other: => T) = LazyList(() => other, () => this)
151 }
152 
153 object LazyList {
154   object ~: {
155     def unapply[T](x: LazyList[T]) = Some((x.head, x.tail))
156   }
157 
158   def continually[T](t: => T): LazyList[T] = LazyList(() => t, () =>continually(t))
159 
160   implicit class CS(ctx: StringContext) {
161     val base = ctx.parts.mkString
162     object p {
163       def unapply(s: LazyList[Int]): Option[LazyList[Int]] = {
164         s.dropPrefix(base.map(_.toInt))
165       }
166     }
167   }

173 object FrontEndUtils {
174   import Ansi.Str._
175 
176   private[this] val newLine = System.lineSeparator()
177 
178   def transpose[A](xs: List[List[A]]): List[List[A]] = {
179     @tailrec def transpose(xs: List[List[A]], result: List[List[A]]): List[List[A]] = {
180       xs.filter(_.nonEmpty) match {
181         case Nil => result
182         case ys  => transpose(ys.map(_.tail), ys.map(_.head) :: result)
183       }
184     }
185 
186     transpose(xs, Nil).reverse
187   }
188 
189   def width = TTY.consoleDim("cols")
190   def height = TTY.consoleDim("lines")
191   def tabulate(snippetsRaw: Seq[Ansi.Str], width: Int): Iterator[String] = {
192     val gap = 2
193     val snippets = if (snippetsRaw.isEmpty) Seq(Ansi.Str.parse("")) else snippetsRaw
194     val maxLength = snippets.maxBy(_.length).length + gap
195     val columns = math.max(1, width / maxLength)
196 
197     val grouped =
198       snippets.toList
199               .grouped(math.ceil(snippets.length * 1.0 / columns).toInt)
200               .toList
201 
202     transpose(grouped).iterator.flatMap { group =>
203       val last = group.last
204       group.init.map(
205         x => x ++ " " * (width / columns - x.length)
206       ) :+ last :+ Ansi.Str.parse(newLine)
207     }
208     .map(_.render)
209   }
210 
211   @tailrec def findPrefix(strings: Seq[String], i: Int = 0): String = {
212     if (strings.count(_.length > i) == 0) strings(0).take(i)
213     else if(strings.collect{ case x if x.length > i => x(i)}.distinct.length > 1)
214       strings(0).take(i)
215     else findPrefix(strings, i + 1)
216   }
217 
218   def printCompletions(completions: Seq[String],
219                        details: Seq[String]): List[String] = {
220 
221     val prelude: List[String] =
222       if (details.length != 0 || completions.length != 0) List(newLine)
223       else Nil
224 
225     val detailsText =
226       if (details.length == 0) Nil
227       else FrontEndUtils.tabulate(details.map(Ansi.Str.parse(_)), FrontEndUtils.width)
228 
229     val completionText =
230       if (completions.length == 0) Nil
231       else FrontEndUtils.tabulate(completions.map(Ansi.Str.parse(_)), FrontEndUtils.width)
232 
233     (prelude ++ detailsText ++ completionText)
234   }
