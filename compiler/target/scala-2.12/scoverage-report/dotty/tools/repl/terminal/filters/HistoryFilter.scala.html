11   * Provides history navigation up and down, saving the current line, a well
12   * as history-search functionality (`Ctrl R` in bash) letting you quickly find
13   * & filter previous commands by entering a sub-string.
14   */

21   /**
22     * `-1` means we haven't started looking at history, `n >= 0` means we're

27   /**
28     * The term we're searching for, if any.
29     *
30     * - `None` means we're not searching for anything, e.g. we're just
31     *   browsing history
32     *
33     * - `Some(term)` where `term` is not empty is what it normally looks
34     *   like when we're searching for something
35     *
36     * - `Some(term)` where `term` is empty only really happens when you
37     *   start searching and delete things, or if you `Ctrl-R` on an empty
38     *   prompt

42   /**

57   def searchHistory(
58     start: Int,
59     increment: Int,
60     buffer: Vector[Char],
61     skipped: Vector[Char]
62   ) = {
63 
64     def nextHistoryIndexFor(v: Vector[Char]) = {
65       HistoryFilter.findNewHistoryIndex(start, v, history(), increment, skipped)
66     }
67 
68     val (newHistoryIndex, newBuffer, newMsg, newCursor) = searchTerm match {
69       // We're not searching for anything, just browsing history.
70       // Pass in Vector.empty so we scroll through all items
71       case None =>
72         val (i, b, c) = nextHistoryIndexFor(Vector.empty)
73         (i, b, "", 99999)
74 
75       // We're searching for some item with a particular search term
76       case Some(b) if b.nonEmpty =>
77         val (i, b1, c) = nextHistoryIndexFor(b)
78 
79         val msg =
80           if (i.nonEmpty) ""
81           else commentStartColor + HistoryFilter.cannotFindSearchMessage + commentEndColor
82 
83         (i, b1, msg, c)
84 
85       // We're searching for nothing in particular; in this case,
86       // show a help message instead of an unhelpful, empty buffer
87       case Some(b) if b.isEmpty =>
88         val msg = commentStartColor + HistoryFilter.emptySearchMessage + commentEndColor
89         // The cursor in this case always goes to zero
90         (Some(start), Vector(), msg, 0)
91 
92     }
93 
94     historyIndex = newHistoryIndex.getOrElse(-1)
95 
96     (newBuffer, newCursor, newMsg)
97   }

99   def activeHistory = searchTerm.nonEmpty || historyIndex != -1

102   def up(b: Vector[Char], c: Int) =
103     searchHistory(historyIndex + 1, 1, b, b)

105   def down(b: Vector[Char], c: Int) =

111   def ctrlR(b: Vector[Char], c: Int) =
112     if (activeSearch) up(b, c)

118   def printableChar(char: Char)(b: Vector[Char], c: Int) = {
119     searchTerm = searchTerm.map(_ :+ char)
120     searchHistory(historyIndex.max(0), 1, b :+ char, Vector())

123   def backspace(b: Vector[Char], c: Int) = {
124     searchTerm = searchTerm.map(_.dropRight(1))

128   /**
129     * Predicate to check if either we're searching for a term or if we're in
130     * history-browsing mode and some predicate is true.
131     *
132     * Very often we want to capture keystrokes in search-mode more aggressively
133     * than in history-mode, e.g. search-mode drops you out more aggressively
134     * than history-mode does, and its up/down keys cycle through history more
135     * aggressively on every keystroke while history-mode only cycles when you
136     * reach the top/bottom line of the multi-line input.
137     */
138   def searchOrHistoryAnd(cond: Boolean) =
139     activeSearch || (activeHistory && cond)

171   def filter0: Filter = Filter.partial("filter0") {
172     // Ways to kick off the history/search if you're not already in it
173 
174     // `Ctrl-R`
175     case TS(18 ~: rest, b, c, _) => wrap(rest, ctrlR(b, c))
176 
177     // `Up` from the first line in the input
178     case TermInfo(TS(p"\u001b[A$rest", b, c, _), w) if firstRow(c, b, w) && !activeHistory =>
179       wrap(rest, startHistory(b, c))
180 
181     // `Ctrl P`
182     case TermInfo(TS(p"\u0010$rest", b, c, _), w) if firstRow(c, b, w) && !activeHistory =>
183       wrap(rest, startHistory(b, c))
184 
185     // `Page-Up` from first character starts history
186     case TermInfo(TS(p"\u001b[5~$rest", b, c, _), w) if c == 0 && !activeHistory =>
187       wrap(rest, startHistory(b, c))
188 
189     // Things you can do when you're already in the history search
190 
191     // Navigating up and down the history. Each up or down searches for
192     // the next thing that matches your current searchTerm
193     // Up
194     case TermInfo(TS(p"\u001b[A$rest", b, c, _), w) if searchOrHistoryAnd(firstRow(c, b, w)) =>
195       wrap(rest, up(b, c))
196 
197     // Ctrl P
198     case TermInfo(TS(p"\u0010$rest", b, c, _), w) if searchOrHistoryAnd(firstRow(c, b, w)) =>
199       wrap(rest, up(b, c))
200 
201     // `Page-Up` from first character cycles history up
202     case TermInfo(TS(p"\u001b[5~$rest", b, c, _), w) if searchOrHistoryAnd(c == 0) =>
203       wrap(rest, up(b, c))
204 
205     // Down
206     case TermInfo(TS(p"\u001b[B$rest", b, c, _), w) if searchOrHistoryAnd(lastRow(c, b, w))  =>
207       wrap(rest, down(b, c))
208 
209     // `Ctrl N`
210 
211     case TermInfo(TS(p"\u000e$rest", b, c, _), w) if searchOrHistoryAnd(lastRow(c, b, w))  =>
212       wrap(rest, down(b, c))
213     // `Page-Down` from last character cycles history down
214     case TermInfo(TS(p"\u001b[6~$rest", b, c, _), w) if searchOrHistoryAnd(c == b.length - 1) =>
215       wrap(rest, down(b, c))
216 
217 
218     // Intercept Backspace and delete a character in search-mode, preserving it, but
219     // letting it fall through and dropping you out of history-mode if you try to make
220     // edits
221     case TS(127 ~: rest, buffer, cursor, _) if activeSearch =>
222       wrap(rest, backspace(buffer, cursor))
223 
224     // Any other control characters drop you out of search mode, but only the
225     // set of `dropHistoryChars` drops you out of history mode
226     case TS(char ~: inputs, buffer, cursor, _)
227       if char.toChar.isControl && searchOrHistoryAnd(dropHistoryChars(char)) =>
228       val newBuffer =
229       // If we're back to -1, it means we've wrapped around and are
230       // displaying the original search term with a wrap-around message
231       // in the terminal. Drop the message and just preserve the search term
232         if (historyIndex == -1) searchTerm.get
233         // If we're searching for an empty string, special-case this and return
234         // an empty buffer rather than the first history item (which would be
235         // the default) because that wouldn't make much sense
236         else if (searchTerm.exists(_.isEmpty)) Vector()
237         // Otherwise, pick whatever history entry we're at and use that
238         else history()(historyIndex).toVector
239       endHistory()
240 
241       TS(char ~: inputs, newBuffer, cursor)
242 
243     // Intercept every other printable character when search is on and
244     // enter it into the current search
245     case TS(char ~: rest, buffer, cursor, _) if activeSearch =>
246       wrap(rest, printableChar(char.toChar)(buffer, cursor))
247 
248     // If you're not in search but are in history, entering any printable
249     // characters kicks you out of it and preserves the current buffer. This
250     // makes it harder for you to accidentally lose work due to history-moves
251     case TS(char ~: rest, buffer, cursor, _) if activeHistory && !char.toChar.isControl =>
252       historyIndex = -1
253       TS(char ~: rest, buffer, cursor)
254   }

259   def mangleBuffer(
260     historyFilter: HistoryFilter,
261     buffer: Ansi.Str,
262     cursor: Int,
263     startColor: Ansi.Attr
264   ) = {
265     if (!historyFilter.activeSearch) buffer
266     else {

268         if (historyFilter.searchTerm.get.isEmpty) (cursor, cursor+1)
269         else {

271 
272           val end = start + (historyFilter.searchTerm.get.length max 1)
273           (start, end)
274         }
275 
276       val newStr = buffer.overlay(startColor, searchStart, searchEnd)
277       newStr
278     }
279   }

281   /**
282     * @param startIndex The first index to start looking from
283     * @param searchTerm The term we're searching from; can be empty
284     * @param history The history we're searching through
285     * @param indexIncrement Which direction to search, +1 or -1
286     * @param skipped Any buffers which we should skip in our search results,
287     *                e.g. because the user has seen them before.
288     */
289   def findNewHistoryIndex(
290     startIndex: Int,
291     searchTerm: Vector[Char],
292     history: IndexedSeq[String],
293     indexIncrement: Int,
294     skipped: Vector[Char]
295   ) = {
296     /**
297       * `Some(i)` means we found a reasonable result at history element `i`
298       * `None` means we couldn't find anything, and should show a not-found
299       * error to the user
300       */
301     def rec(i: Int): Option[Int] = history.lift(i) match {
302       // If i < 0, it means the user is pressing `down` too many times, which
303       // means it doesn't show anything but we shouldn't show an error
304       case None if i < 0 => Some(-1)
305       case None => None
306       case Some(s) if s.contains(searchTerm) && !s.contentEquals(skipped) =>
307         Some(i)
308       case _ => rec(i + indexIncrement)
309     }
310 
311     val newHistoryIndex = rec(startIndex)
312     val foundIndex = newHistoryIndex.find(_ != -1)
313     val newBuffer = foundIndex match {
314       case None => searchTerm
315       case Some(i) => history(i).toVector
316     }
317 
318     val newCursor = foundIndex match {
319       case None => newBuffer.length
320       case Some(i) => history(i).indexOfSlice(searchTerm) + searchTerm.length
321     }
322 
323     (newHistoryIndex, newBuffer, newCursor)
324   }
