14 /**
15  * Filters for simple operation of a terminal: cursor-navigation
16  * (including with all the modifier keys), enter/ctrl-c-exit, etc.

18 object BasicFilters {
19   def all = Filter.merge(
20     navFilter,
21     exitFilter,
22     enterFilter,
23     clearFilter,
24     //loggingFilter,
25     typingFilter
26   )

28   def injectNewLine(b: Vector[Char], c: Int, rest: LazyList[Int], indent: Int = 0) = {
29     val (first, last) = b.splitAt(c)
30     TermState(rest, (first :+ '\n') ++ last ++ Vector.fill(indent)(' '), c + 1 + indent)
31   }
32 
33   def navFilter = Filter.merge(
34     simple(Up)((b, c, m) => moveUp(b, c, m.width)),
35     simple(Down)((b, c, m) => moveDown(b, c, m.width)),
36     simple(Right)((b, c, m) => (b, c + 1)),
37     simple(Left)((b, c, m) => (b, c - 1))
38   )
39 
40   def tabColumn(indent: Int, b: Vector[Char], c: Int, rest: LazyList[Int]) = {
41     val (chunks, chunkStarts, chunkIndex) = FilterTools.findChunks(b, c)
42     val chunkCol = c - chunkStarts(chunkIndex)
43     val spacesToInject = indent - (chunkCol % indent)
44     val (lhs, rhs) = b.splitAt(c)
45     TS(rest, lhs ++ Vector.fill(spacesToInject)(' ') ++ rhs, c + spacesToInject)
46   }
47 
48   def tabFilter(indent: Int): Filter = Filter.partial("tabFilter") {
49     case TS(9 ~: rest, b, c, _) => tabColumn(indent, b, c, rest)
50   }
51 
52   def loggingFilter: Filter = Filter.partial("loggingFilter") {
53     case TS(Ctrl('q') ~: rest, b, c, _) =>
54       println("Char Display Mode Enabled! Ctrl-C to exit")
55       var curr = rest
56       while (curr.head != 3) {
57         println("Char " + curr.head)
58         curr = curr.tail
59       }
60       TS(curr, b, c)
61   }
62 
63   def typingFilter: Filter = Filter.partial("typingFilter") {
64     case TS(p"\u001b[3~$rest", b, c, _) =>
65 //      Debug("fn-delete")
66       val (first, last) = b.splitAt(c)
67       TS(rest, first ++ last.drop(1), c)
68 
69     case TS(127 ~: rest, b, c, _) => // Backspace
70       val (first, last) = b.splitAt(c)
71       TS(rest, first.dropRight(1) ++ last, c - 1)
72 
73     case TS(char ~: rest, b, c, _) =>
74 //      Debug("NORMAL CHAR " + char)
75       val (first, last) = b.splitAt(c)
76       TS(rest, (first :+ char.toChar) ++ last, c + 1)
77   }
78 
79   def doEnter(b: Vector[Char], c: Int, rest: LazyList[Int]) = {
80     val (chunks, chunkStarts, chunkIndex) = FilterTools.findChunks(b, c)
81     if (chunkIndex == chunks.length - 1) Result(b.mkString)
82     else injectNewLine(b, c, rest)
83   }
84 
85   def enterFilter: Filter = Filter.partial("enterFilter") {
86     case TS(13 ~: rest, b, c, _) => doEnter(b, c, rest) // Enter
87     case TS(10 ~: rest, b, c, _) => doEnter(b, c, rest) // Enter
88     case TS(10 ~: 13 ~: rest, b, c, _) => doEnter(b, c, rest) // Enter
89     case TS(13 ~: 10 ~: rest, b, c, _) => doEnter(b, c, rest) // Enter
90   }
91 
92   def exitFilter: Filter = Filter.partial("exitFilter") {
93     case TS(Ctrl('c') ~: rest, b, c, _) =>
94       Interrupt
95     case TS(Ctrl('d') ~: rest, b, c, _) =>
96       // only exit if the line is empty, otherwise, behave like
97       // "delete" (i.e. delete one char to the right)
98       if (b.isEmpty) Exit else {
99         val (first, last) = b.splitAt(c)
100         TS(rest, first ++ last.drop(1), c)
101       }
102     case TS(-1 ~: rest, b, c, _) => Exit   // java.io.Reader.read() produces -1 on EOF
103   }
104 
105   def clearFilter: Filter = Filter.partial("clearFilter") {
106     case TS(Ctrl('l') ~: rest, b, c, _) => ClearScreen(TS(rest, b, c))
107   }
108 
109   def moveStart(b: Vector[Char], c: Int, w: Int) = {
110     val (_, chunkStarts, chunkIndex) = findChunks(b, c)
111     val currentColumn = (c - chunkStarts(chunkIndex)) % w
112     b -> (c - currentColumn)
113   }
114 
115   def moveEnd(b: Vector[Char], c: Int, w: Int) = {
116     val (chunks, chunkStarts, chunkIndex) = findChunks(b, c)
117     val currentColumn = (c - chunkStarts(chunkIndex)) % w
118     val c1 = chunks.lift(chunkIndex + 1) match {
119       case Some(next) =>
120         val boundary = chunkStarts(chunkIndex + 1) - 1
121         if ((boundary - c) > (w - currentColumn)) {
122           val delta= w - currentColumn
123           c + delta
124         }
125         else boundary
126       case None =>
127         c + 1 * 9999
128     }
129     b -> c1
130   }
131 
132   def moveUpDown(
133     b: Vector[Char],
134     c: Int,
135     w: Int,
136     boundaryOffset: Int,
137     nextChunkOffset: Int,
138     checkRes: Int,
139     check: (Int, Int) => Boolean,
140     isDown: Boolean
141   ) = {
142     val (chunks, chunkStarts, chunkIndex) = findChunks(b, c)
143     val offset = chunkStarts(chunkIndex + boundaryOffset)
144     if (check(checkRes, offset)) checkRes
145     else chunks.lift(chunkIndex + nextChunkOffset) match {
146       case None => c + nextChunkOffset * 9999
147       case Some(next) =>
148         val boundary = chunkStarts(chunkIndex + boundaryOffset)
149         val currentColumn = (c - chunkStarts(chunkIndex)) % w
150 
151         if (isDown) boundary + math.min(currentColumn, next)
152         else boundary + math.min(currentColumn - next % w, 0) - 1
153     }
154   }
155 
156   def moveUp(b: Vector[Char], c: Int, w: Int) = {
157     b -> moveUpDown(b, c, w, 0, -1, c - w, _ > _, false)
158   }
159 
160   def moveDown(b: Vector[Char], c: Int, w: Int) = {
161     b -> moveUpDown(b, c, w, 1, 1, c + w, _ <= _, true)
162   }
