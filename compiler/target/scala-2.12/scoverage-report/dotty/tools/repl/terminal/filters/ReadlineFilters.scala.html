69   def transposeLetter(b: Vector[Char], c: Int) =
70     // If there's no letter before the cursor to transpose, don't do anything
71     if (c == 0) (b, c)
72     else if (c == b.length) (b.dropRight(2) ++ b.takeRight(2).reverse, c)
73     else (b.patch(c-1, b.slice(c-1, c+1).reverse, 2), c + 1)

75   def transposeWord(b: Vector[Char], c: Int) = {
76     val leftStart0 = GUILikeFilters.consumeWord(b, c - 1, -1, 1)
77     val leftEnd0   = GUILikeFilters.consumeWord(b, leftStart0, 1, 0)
78     val rightEnd   = GUILikeFilters.consumeWord(b, c, 1, 0)
79     val rightStart = GUILikeFilters.consumeWord(b, rightEnd - 1, -1, 1)
80 
81     // If no word to the left to transpose, do nothing
82     if (leftStart0 == 0 && rightStart == 0) (b, c)
83     else {
84       val (leftStart, leftEnd) =
85         // If there is no word to the *right* to transpose,
86         // transpose the two words to the left instead
87         if (leftEnd0 == b.length && rightEnd == b.length) {
88           val leftStart = GUILikeFilters.consumeWord(b, leftStart0 - 1, -1, 1)
89           val leftEnd   = GUILikeFilters.consumeWord(b, leftStart, 1, 0)
90           (leftStart, leftEnd)
91         } else
92           (leftStart0, leftEnd0)
93 
94       val newB =
95         b.slice(0, leftStart)         ++
96         b.slice(rightStart, rightEnd) ++
97         b.slice(leftEnd, rightStart)  ++
98         b.slice(leftStart, leftEnd)   ++
99         b.slice(rightEnd, b.length)
100 
101       (newB, rightEnd)
102     }
103   }

105   /**
106    * All the cut-pasting logic, though for many people they simply
107    * use these shortcuts for deleting and don't use paste much at all.
108    */
109   case class CutPasteFilter() extends DelegateFilter("CutPasteFilter") {
110     var accumulating = false
111     var currentCut = Vector.empty[Char]
112     def prepend(b: Vector[Char]) = {
113       if (accumulating) currentCut = b ++ currentCut
114       else currentCut = b
115       accumulating = true
116     }
117     def append(b: Vector[Char]) = {
118       if (accumulating) currentCut = currentCut ++ b
119       else currentCut = b
120       accumulating = true
121     }
122     def cutCharLeft(b: Vector[Char], c: Int) = {
123       /* Do not edit current cut. Zsh(zle) & Bash(readline) do not edit the yank ring for Ctrl-h */
124       (b patch(from = c - 1, patch = Nil, replaced = 1), c - 1)
125     }
126 
127     def cutLineLeft(b: Vector[Char], c: Int) = {
128       val (allBeforeCursor, allAfterCursor) = b.splitAt(c)
129       val previousNewlineIndex = allBeforeCursor.lastIndexWhere(_ == '\n')
130       if (previousNewlineIndex == -1) {
131         // beginning of input. no leading newline
132         prepend(allBeforeCursor)
133         (allAfterCursor, 0)
134       } else {
135         val (allBeforeLine, lineBeforeCursor) = allBeforeCursor.splitAt(
136           previousNewlineIndex
137         )
138         val charsBeforeCursorOnLine = lineBeforeCursor.length
139         if (charsBeforeCursorOnLine == 1) {
140           // if only a newline before cursor on line, cut it
141           prepend(lineBeforeCursor)
142           (allBeforeLine ++ allAfterCursor, c - 1)
143         } else {
144           // if there's more on the line before cursor, cut to beginning of line
145           prepend(lineBeforeCursor.tail)
146           val buffer = allBeforeLine ++ "\n" ++ allAfterCursor
147           val cursor = c - (charsBeforeCursorOnLine - 1)
148           (buffer, cursor)
149         }
150       }
151     }
152 
153     def cutLineRight(b: Vector[Char], c: Int) = {
154       val (allBeforeCursor, allAfterCursor) = b.splitAt(c)
155       val nextNewlineIndex = allAfterCursor.indexWhere(_ == '\n')
156       if (nextNewlineIndex == -1) {
157         // end of input. no trailing newline
158         append(allAfterCursor)
159         (allBeforeCursor, c)
160       } else {
161         allAfterCursor.splitAt(
162           nextNewlineIndex
163         ) match {
164           case (Vector(), ('\n' +: allAfterNewline)) =>
165             // if there's only a newline after cursor on line, cut it
166             append(Vector('\n'))
167             (allBeforeCursor ++ allAfterNewline, c)
168           case (restOfLine, allAfterLine) =>
169             // if there's more on the line after cursor, cut to end of line
170             append(restOfLine)
171             (allBeforeCursor ++ allAfterLine, c)
172         }
173       }
174     }
175 
176     def cutWordRight(b: Vector[Char], c: Int) = {
177       val start = GUILikeFilters.consumeWord(b, c, 1, 0)
178       append(b.slice(c, start))
179       (b.take(c) ++ b.drop(start), c)
180     }
181 
182     def cutWordLeft(b: Vector[Char], c: Int) = {
183       val start = GUILikeFilters.consumeWord(b, c - 1, -1, 1)
184       prepend(b.slice(start, c))
185       (b.take(start) ++ b.drop(c), start)
186     }
187 
188     def paste(b: Vector[Char], c: Int) = {
189       accumulating = false
190       (b.take(c) ++ currentCut ++ b.drop(c), c + currentCut.length)
191     }
192 
193     def filter = Filter.merge(
194       simple(Ctrl('u'))((b, c, m) => cutLineLeft(b, c)),
195       simple(Ctrl('k'))((b, c, m) => cutLineRight(b, c)),
196       simple(Alt + "d")((b, c, m) => cutWordRight(b, c)),
197       simple(Ctrl('w'))((b, c, m) => cutWordLeft(b, c)),
198       simple(Alt + "\u007f")((b, c, m) => cutWordLeft(b, c)),
199       // weird hacks to make it run code every time without having to be the one
200       // handling the input; ideally we'd change Filter to be something
201       // other than a PartialFunction, but for now this will do.
202 
203       // If some command goes through that's not appending/prepending to the
204       // kill ring, stop appending and allow the next kill to override it
205       Filter.wrap(identifier) { _ => accumulating = false; None },
206       simple(Ctrl('h'))((b, c, m) => cutCharLeft(b, c)),
207       simple(Ctrl('y'))((b, c, m) => paste(b, c))
208     )
209   }
