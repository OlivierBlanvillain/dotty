21     var mark: Option[Int] = None

23     def setMark(c: Int) = {
24       Debug("setMark\t" + mark + "\t->\t" + c)

28     def doIndent(
29       b: Vector[Char],
30       c: Int,
31       rest: LazyList[Int],
32       slicer: Vector[Char] => Int
33     ) = {
34 
35       val markValue = mark.get
36       val (chunks, chunkStarts, chunkIndex) = FilterTools.findChunks(b, c)
37       val min = chunkStarts.lastIndexWhere(_ <= math.min(c, markValue))
38       val max = chunkStarts.indexWhere(_ > math.max(c, markValue))
39       val splitPoints = chunkStarts.slice(min, max)
40       val frags = (0 +: splitPoints :+ 99999).sliding(2).zipWithIndex
41 
42       var firstOffset = 0
43       val broken =
44         for((Seq(l, r), i) <- frags) yield {
45           val slice = b.slice(l, r)
46           if (i == 0) slice
47           else {
48             val cut = slicer(slice)
49 
50             if (i == 1) firstOffset = cut
51 
52             if (cut < 0) slice.drop(-cut)
53             else Vector.fill(cut)(' ') ++ slice
54           }
55         }
56       val flattened = broken.flatten.toVector
57       val deeperOffset = flattened.length - b.length
58 
59       val (newMark, newC) =
60         if (mark.get > c) (mark.get + deeperOffset, c + firstOffset)
61         else (mark.get + firstOffset, c + deeperOffset)
62 
63       mark = Some(newMark)
64       TS(rest, flattened, newC)
65     }

68 
69       simple(ShiftUp){(b, c, m) => setMark(c); BasicFilters.moveUp(b, c, m.width)},
70       simple(ShiftDown){(b, c, m) => setMark(c); BasicFilters.moveDown(b, c, m.width)},

86           doIndent(b, c, rest,
87             slice => indent
88           )
89 
90         // Intercept every other character.
91         case TS(char ~: inputs, buffer, cursor, _) if mark.isDefined =>
92           // If it's a special command, just cancel the current selection.
93           if (char.toChar.isControl &&
94               char != 127 /*backspace*/ &&
95               char != 13 /*enter*/ &&
96               char != 10 /*enter*/) {
97             mark = None
98             TS(char ~: inputs, buffer, cursor)
99           } else {
100             // If it's a  printable character, delete the current
101             // selection and write the printable character.
102             val Seq(min, max) = Seq(mark.get, cursor).sorted
103             mark = None
104             val newBuffer = buffer.take(min) ++ buffer.drop(max)
105             val newInputs =
106               if (char == 127) inputs
107               else char ~: inputs
108             TS(newInputs, newBuffer, min)
109           }
110       }
111     )

157   def consumeWord(b: Vector[Char], c: Int, delta: Int, offset: Int) = {
158     var current = c
159     while(b.isDefinedAt(current) && !b(current).isLetterOrDigit) current += delta
160     while(b.isDefinedAt(current) && b(current).isLetterOrDigit) current += delta
161     current + offset
162   }
