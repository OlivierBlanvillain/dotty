5 /**
6   * The way you configure your terminal behavior; a trivial wrapper around a
7   * function, though you should provide a good `.toString` method to make
8   * debugging easier. The [[TermInfo]] and [[TermAction]] types are its
9   * interface to the terminal.
10   *
11   * [[Filter]]s are composed sequentially: if a filter returns `None` the next
12   * filter is tried, while if a filter returns `Some` that ends the cascade.
13   * While your `op` function interacts with the terminal purely through
14   * immutable case classes, the Filter itself is free to maintain its own state
15   * and mutate it whenever, even when returning `None` to continue the cascade.
16   */
17 trait Filter {
18   def op(ti: TermInfo): Option[TermAction]
19 
20   /**
21     * the `.toString` of this object, except by making it separate we force
22     * the implementer to provide something and stop them from accidentally
23     * leaving it as the meaningless default.
24     */
25   def identifier: String
26 
27   override def toString = identifier
28 }
29 
30 /**
31   * Convenience constructors to create [[Filter]] instances in a bunch of
32   * different ways
33   */
34 object Filter {
35 
36   /**
37     * Shorthand to construct a filter in the common case where you're
38     * switching on the prefix of the input stream and want to run some
39     * transformation on the buffer/cursor
40     */
41   def simple(prefixes: String*)
42             (f: (Vector[Char], Int, TermInfo) => (Vector[Char], Int)): Filter = new Filter {
43     def op(ti: TermInfo) = {
44       val matchingPrefixOpt =
45         prefixes.iterator
46                 .map{ s => ti.ts.inputs.dropPrefix(s.map(_.toInt)) }
47                 .collectFirst{case Some(s) => s}
48       matchingPrefixOpt.map{ rest =>
49         val (buffer1, cursor1) = f(ti.ts.buffer, ti.ts.cursor, ti)
50         TermState(
51           rest,
52           buffer1,
53           cursor1
54         )
55 
56       }
57     }
58     def identifier = prefixes.mkString(":")
59   }
60   def action(id: String)
61             (prefixes: Seq[String], filter: TermInfo => Boolean = _ => true)
62             (action: TermState => TermAction = x => x): Filter = new Filter {
63     def op(ti: TermInfo) = {
64       prefixes.iterator
65               .map{ prefix => ti.ts.inputs.dropPrefix(prefix.map(_.toInt)) }
66               .collectFirst { case Some(rest) if filter(ti) => action(ti.ts.copy(inputs = rest)) }
67     }
68     def identifier: String = id
69   }
70 
71   def partial(id: String)(f: PartialFunction[TermInfo, TermAction]): Filter =
72     new Filter {
73       def op(ti: TermInfo) = f.lift(ti)
74       def identifier = id
75     }
76 
77   def wrap(id: String)(f: TermInfo => Option[TermAction]): Filter =
78     new Filter {
79       def op(ti: TermInfo) = f(ti)
80       def identifier = id
81     }
82 
83   /**
84     * Merges multiple [[Filter]]s into one.
85     */
86   def merge(pfs: Filter*) = new Filter {
87 
88     def op(v1: TermInfo) = pfs.iterator.map(_.op(v1)).find(_.isDefined).flatten
89 
90     def identifier = pfs.iterator.map(_.identifier).mkString(":")
91   }
92   val empty = Filter.merge()
93 }
