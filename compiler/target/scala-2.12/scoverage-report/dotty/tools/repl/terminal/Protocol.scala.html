16 object TermState {
17   // Using unapply instead exposes a dotty/scalac variation. Because
18   // `TermState` is a case class, scalac generate an unapply with this exact
19   // signature, that is used by the `TermInfo` and `TermAction` unapplies.
20   // With dotty, the generated unapply has type `TermState => TermState`
21   // instead, `unapply(ti: TermAction)` thus becomes a infinite tail
22   // recursion. See #2335.
23   def unapplyWorkaround(ti: TermState): Option[(LazyList[Int], Vector[Char], Int, Ansi.Str)] =
24     Some((ti.inputs, ti.buffer, ti.cursor, ti.msg))
25 
26   def unapply(ti: TermInfo): Option[(LazyList[Int], Vector[Char], Int, Ansi.Str)] =
27     TermState.unapplyWorkaround(ti.ts)
28 
29   def unapply(ti: TermAction): Option[(LazyList[Int], Vector[Char], Int, Ansi.Str)] =
30     ti match {
31       case ts: TermState => TermState.unapplyWorkaround(ts)
32       case _ => None
33     }
34 }
