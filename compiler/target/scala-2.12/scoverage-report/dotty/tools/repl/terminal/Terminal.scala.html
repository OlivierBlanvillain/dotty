8 /**
9  * The core logic around a terminal; it defines the base `filters` API
10  * through which anything (including basic cursor-navigation and typing)
11  * interacts with the terminal.
12  *
13  * Maintains basic invariants, such as "cursor should always be within
14  * the buffer", and "ansi terminal should reflect most up to date TermState"
15  */

18   /**
19    * Computes how tall a line of text is when wrapped at `width`.
20    *
21    * Even 0-character lines still take up one row!
22    *
23    * width = 2
24    * 0 -> 1
25    * 1 -> 1
26    * 2 -> 1
27    * 3 -> 2
28    * 4 -> 2
29    * 5 -> 3
30    */
31   def fragHeight(length: Int, width: Int) = math.max(1, (length - 1) / width + 1)

33   def splitBuffer(buffer: Vector[Char]) = {
34     val frags = mutable.Buffer.empty[Int]
35     frags.append(0)
36     for(c <- buffer){
37       if (c == '\n') frags.append(0)
38       else frags(frags.length - 1) = frags.last + 1
39     }
40     frags
41   }
42   def calculateHeight(buffer: Vector[Char],
43                       width: Int,
44                       prompt: String): Seq[Int] = {
45     val rowLengths = splitBuffer(buffer)
46 
47     calculateHeight0(rowLengths, width - prompt.length)
48   }

50   /**
51    * Given a buffer with characters and newlines, calculates how high
52    * the buffer is and where the cursor goes inside of it.
53    */
54   def calculateHeight0(rowLengths: Seq[Int],
55                        width: Int): Seq[Int] = {
56     val fragHeights =
57       rowLengths
58         .inits
59         .toVector
60         .reverse // We want shortest-to-longest, inits gives longest-to-shortest
61         .filter(_.nonEmpty) // Without the first empty prefix
62         .map{ x =>
63           fragHeight(
64             // If the frag barely fits on one line, give it
65             // an extra spot for the cursor on the next line
66             x.last + 1,
67             width
68           )
69         }
70 //    Debug("fragHeights " + fragHeights)
71     fragHeights
72   }

74   def positionCursor(cursor: Int,
75                      rowLengths: Seq[Int],
76                      fragHeights: Seq[Int],
77                      width: Int) = {
78     var leftoverCursor = cursor
79     //    Debug("leftoverCursor " + leftoverCursor)
80     var totalPreHeight = 0
81     var done = false
82     // Don't check if the cursor exceeds the last chunk, because
83     // even if it does there's nowhere else for it to go
84     for(i <- 0 until rowLengths.length -1 if !done) {
85       // length of frag and the '\n' after it
86       val delta = rowLengths(i) + 1
87       //      Debug("delta " + delta)
88       val nextCursor = leftoverCursor - delta
89       if (nextCursor >= 0) {
90         //        Debug("nextCursor " + nextCursor)
91         leftoverCursor = nextCursor
92         totalPreHeight += fragHeights(i)
93       }else done = true
94     }
95 
96     val cursorY = totalPreHeight + leftoverCursor / width
97     val cursorX = leftoverCursor % width
98 
99     (cursorY, cursorX)
100   }

107   /**

125     /**
126       * Erases the previous line and re-draws it with the new buffer and
127       * cursor.
128       *
129       * Relies on `ups` to know how "tall" the previous line was, to go up
130       * and erase that many rows in the console. Performs a lot of horrific
131       * math all over the place, incredibly prone to off-by-ones, in order
132       * to at the end of the day position the cursor in the right spot.
133       */
134     def redrawLine(buffer: Ansi.Str,
135                    cursor: Int,
136                    ups: Int,
137                    rowLengths: Seq[Int],
138                    fullPrompt: Boolean = true,
139                    newlinePrompt: Boolean = false) = {
140 
141 
142       // Enable this in certain cases (e.g. cursor near the value you are
143       // interested into) see what's going on with all the ansi screen-cursor
144       // movement
145       def debugDelay() = if (false){
146         Thread.sleep(200)
147         writer.flush()
148       }
149 
150 
151       val promptLine =
152         if (fullPrompt) prompt.full
153         else prompt.lastLine
154 
155       val promptWidth = if(newlinePrompt) 0 else prompt.lastLine.length
156       val actualWidth = width - promptWidth
157 
158       ansi.up(ups)
159       ansi.left(9999)
160       ansi.clearScreen(0)
161       writer.write(promptLine.toString)
162       if (newlinePrompt) writer.write("\n")
163 
164       // I'm not sure why this is necessary, but it seems that without it, a
165       // cursor that "barely" overshoots the end of a line, at the end of the
166       // buffer, does not properly wrap and ends up dangling off the
167       // right-edge of the terminal window!
168       //
169       // This causes problems later since the cursor is at the wrong X/Y,
170       // confusing the rest of the math and ending up over-shooting on the
171       // `ansi.up` calls, over-writing earlier lines. This prints a single
172       // space such that instead of dangling it forces the cursor onto the
173       // next line for-realz. If it isn't dangling the extra space is a no-op
174       val lineStuffer = ' '
175       // Under `newlinePrompt`, we print the thing almost-verbatim, since we
176       // want to avoid breaking code by adding random indentation. If not, we
177       // are guaranteed that the lines are short, so we can indent the newlines
178       // without fear of wrapping
179       val newlineReplacement =
180         if (newlinePrompt) {
181 
182           Array(lineStuffer, '\n')
183         } else {
184           val indent = " " * prompt.lastLine.length
185           Array('\n', indent:_*)
186         }
187 
188       writer.write(
189         buffer.render.flatMap{
190           case '\n' => newlineReplacement
191           case x => Array(x)
192         }.toArray
193       )
194       writer.write(lineStuffer)
195 
196       val fragHeights = calculateHeight0(rowLengths, actualWidth)
197       val (cursorY, cursorX) = positionCursor(
198         cursor,
199         rowLengths,
200         fragHeights,
201         actualWidth
202       )
203       ansi.up(fragHeights.sum - 1)
204       ansi.left(9999)
205       ansi.down(cursorY)
206       ansi.right(cursorX)
207       if (!newlinePrompt) ansi.right(prompt.lastLine.length)
208 
209       writer.flush()
210     }

212     @tailrec
213     def readChar(lastState: TermState, ups: Int, fullPrompt: Boolean = true): Option[TermAction] = {
214       val moreInputComing = reader.ready()
215 
216       lazy val (transformedBuffer0, cursorOffset) = displayTransform(
217         lastState.buffer,
218         lastState.cursor
219       )
220 
221       lazy val transformedBuffer = transformedBuffer0 ++ lastState.msg
222       lazy val lastOffsetCursor = lastState.cursor + cursorOffset
223       lazy val rowLengths = splitBuffer(
224         lastState.buffer ++ lastState.msg.plainText
225       )
226       val narrowWidth = width - prompt.lastLine.length
227       val newlinePrompt = rowLengths.exists(_ >= narrowWidth)
228       val promptWidth = if(newlinePrompt) 0 else prompt.lastLine.length
229       val actualWidth = width - promptWidth
230       val newlineUp = if (newlinePrompt) 1 else 0
231       if (!moreInputComing) redrawLine(
232         transformedBuffer,
233         lastOffsetCursor,
234         ups,
235         rowLengths,
236         fullPrompt,
237         newlinePrompt
238       )
239 
240       lazy val (oldCursorY, _) = positionCursor(
241         lastOffsetCursor,
242         rowLengths,
243         calculateHeight0(rowLengths, actualWidth),
244         actualWidth
245       )

247       def updateState(s: LazyList[Int],
248                       b: Vector[Char],
249                       c: Int,
250                       msg: Ansi.Str): (Int, TermState) = {
251 
252         val newCursor = math.max(math.min(c, b.length), 0)
253         val nextUps =
254           if (moreInputComing) ups
255           else oldCursorY + newlineUp
256 
257         val newState = TermState(s, b, newCursor, msg)
258 
259         (nextUps, newState)
260       }
261       // `.get` because we assume that *some* filter is going to match each
262       // character, even if only to dump the character to the screen. If nobody
263       // matches the character then we can feel free to blow up
264       def redrawAndNewline() = {
265         redrawLine(
266           transformedBuffer, lastState.buffer.length,
267           oldCursorY + newlineUp, rowLengths, false, newlinePrompt
268         )
269         writer.write(10)
270         writer.write(13)
271         writer.flush()
272       }
273       filters.op(TermInfo(lastState, actualWidth)).get match {
274         case Printing(TermState(s, b, c, msg), stdout) =>
275           writer.write(stdout)
276           val (nextUps, newState) = updateState(s, b, c, msg)
277           readChar(newState, nextUps)
278 
279         case TermState(s, b, c, msg) =>
280           val (nextUps, newState) = updateState(s, b, c, msg)
281           readChar(newState, nextUps, false)
282 
283         case res: Result =>
284           redrawAndNewline()
285           Some(res)
286 
287         case Interrupt =>
288           redrawAndNewline()
289           Some(Interrupt)
290 
291         case ClearScreen(ts) =>
292           ansi.clearScreen(2)
293           ansi.up(9999)
294           ansi.left(9999)
295           readChar(ts, ups)
296         case Exit =>
297           None
298       }
299     }
300 
301     lazy val ansi = new AnsiNav(writer)
302     lazy val (width, _, initialConfig) = TTY.init()
303     try {
304       readChar(TermState(LazyList.continually(reader.read()), Vector.empty, 0, ""), 0)
305     }
306     finally {
307 
308       // Don't close these! Closing these closes stdin/stdout,
309       // which seems to kill the entire program
310 
311       // reader.close()
312       // writer.close()
313       TTY.stty(initialConfig)
314     }
315   }
