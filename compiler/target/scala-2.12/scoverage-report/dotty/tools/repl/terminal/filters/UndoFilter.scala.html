11 /**
12   * A filter that implements "undo" functionality in the ammonite REPL. It
13   * shares the same `Ctrl -` hotkey that the bash undo, but shares behavior
14   * with the undo behavior in desktop text editors:
15   *
16   * - Multiple `delete`s in a row get collapsed
17   * - In addition to edits you can undo cursor movements: undo will bring your
18   *   cursor back to location of previous edits before it undoes them
19   * - Provides "redo" functionality under `Alt -`/`Esc -`: un-undo the things
20   *   you didn't actually want to undo!
21   *
22   * @param maxUndo: the maximum number of undo-frames that are stored.
23   */
24 case class UndoFilter(maxUndo: Int = 25) extends DelegateFilter("UndoFilter") {
25   /**
26     * The current stack of states that undo/redo would cycle through.
27     *
28     * Not really the appropriate data structure, since when it reaches
29     * `maxUndo` in length we remove one element from the start whenever we
30     * append one element to the end, which costs `O(n)`. On the other hand,
31     * It also costs `O(n)` to maintain the buffer of previous states, and
32     * so `n` is probably going to be pretty small anyway (tens?) so `O(n)`
33     * is perfectly fine.
34     */
35   val undoBuffer = mutable.Buffer[(Vector[Char], Int)](Vector[Char]() -> 0)
36 
37   /**
38     * The current position in the undoStack that the terminal is currently in.
39     */
40   var undoIndex = 0
41   /**
42     * An enum representing what the user is "currently" doing. Used to
43     * collapse sequential actions into one undo step: e.g. 10 plai 
44     * chars typed becomes 1 undo step, or 10 chars deleted becomes one undo
45     * step, but 4 chars typed followed by 3 chars deleted followed by 3 chars
46     * typed gets grouped into 3 different undo steps
47     */
48   var state = UndoState.Default
49   def currentUndo = undoBuffer(undoBuffer.length - undoIndex - 1)
50 
51   def undo(b: Vector[Char], c: Int) = {
52     val msg =
53       if (undoIndex >= undoBuffer.length - 1) UndoFilter.cannotUndoMsg
54       else {
55         undoIndex += 1
56         state = UndoState.Default
57         UndoFilter.undoMsg
58       }
59     val (b1, c1) = currentUndo
60     (b1, c1, msg)
61   }
62 
63   def redo(b: Vector[Char], c: Int) = {
64     val msg =
65       if (undoIndex <= 0) UndoFilter.cannotRedoMsg
66       else {
67         undoIndex -= 1
68         state = UndoState.Default
69         UndoFilter.redoMsg
70       }
71 
72     currentUndo
73     val (b1, c1) = currentUndo
74     (b1, c1, msg)
75   }
76 
77   def wrap(bc: (Vector[Char], Int, Ansi.Str), rest: LazyList[Int]) = {
78     val (b, c, msg) = bc
79     TS(rest, b, c, msg)
80   }
81 
82   def pushUndos(b: Vector[Char], c: Int) = {
83     val (lastB, lastC) = currentUndo
84     // Since we don't have access to the `typingFilter` in this code, we
85     // instead attempt to reverse-engineer "what happened" to the buffer by
86     // comparing the old one with the new.
87     //
88     // It turns out that it's not that hard to identify the few cases we care
89     // about, since they're all result in either 0 or 1 chars being different
90     // between old and new buffers.
91     val newState =
92     // Nothing changed means nothing changed
93       if (lastC == c && lastB == b) state
94       // if cursor advanced 1, and buffer grew by 1 at the cursor, we're typing
95       else if (lastC + 1 == c && lastB == b.patch(c-1, Nil, 1)) UndoState.Typing
96       // cursor moved left 1, and buffer lost 1 char at that point, we're deleting
97       else if (lastC - 1 == c && lastB.patch(c, Nil, 1) == b) UndoState.Deleting
98       // cursor didn't move, and buffer lost 1 char at that point, we're also deleting
99       else if (lastC == c && lastB.patch(c - 1, Nil, 1) == b) UndoState.Deleting
100       // cursor moved around but buffer didn't change, we're navigating
101       else if (lastC != c && lastB == b) UndoState.Navigating
102       // otherwise, sit in the "Default" state where every change is recorded.
103       else UndoState.Default
104 
105     if (state != newState || newState == UndoState.Default && (lastB, lastC) != (b, c)) {
106       // If something changes: either we enter a new `UndoState`, or we're in
107       // the `Default` undo state and the terminal buffer/cursor change, then
108       // truncate the `undoStack` and add a new tuple to the stack that we can
109       // build upon. This means that we lose all ability to re-do actions after
110       // someone starts making edits, which is consistent with most other
111       // editors
112       state = newState
113       undoBuffer.remove(undoBuffer.length - undoIndex, undoIndex)
114       undoIndex = 0
115 
116       if (undoBuffer.length == maxUndo) undoBuffer.remove(0)
117 
118       undoBuffer.append(b -> c)
119     } else if (undoIndex == 0 && (b, c) != undoBuffer(undoBuffer.length - 1)) {
120       undoBuffer(undoBuffer.length - 1) = (b, c)
121     }
122 
123     state = newState
124   }
125 
126   def filter = Filter.merge(
127     Filter.wrap("undoFilterWrapped") {
128       case TS(q ~: rest, b, c, _) =>
129         pushUndos(b, c)
130         None
131     },
132     Filter.partial("undoFilter") {
133       case TS(31 ~: rest, b, c, _) => wrap(undo(b, c), rest)
134       case TS(27 ~: 114 ~: rest, b, c, _) => wrap(undo(b, c), rest)
135       case TS(27 ~: 45 ~: rest, b, c, _) => wrap(redo(b, c), rest)
136     }
137   )
138 }
