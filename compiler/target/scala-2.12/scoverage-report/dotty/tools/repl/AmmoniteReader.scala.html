19 /** Adaptation of the Ammonite shell emulator to fit the Dotty REPL
20  *
21  *  Credit for the code in the `terminal` goes to Li Haoyi
22  *  who wrote most of it as part of Ammonite.
23  *
24  *  @param history is a list of the previous input with the latest entry as head
25  *  @param complete takes a function from cursor point and input string to
26  *                  `Completions`
27  */
28 private[repl] class AmmoniteReader(out: OutputStream,
29                                    history: History,
30                                    complete: (Int, String) => Completions)
31                                   (implicit ctx: Context) {
32 
33   private[this] val reader = new InputStreamReader(System.in)
34   private[this] val writer = new OutputStreamWriter(out)
35 
36   private[this] val cutPasteFilter  = ReadlineFilters.CutPasteFilter()
37   private[this] val selectionFilter = GUILikeFilters.SelectionFilter(indent = 2)
38   private[this] val multilineFilter = Filter.partial("multilineFilter") {
39     case TermState(lb ~: rest, b, c, d) if (lb == 10 || lb == 13) =>
40       val source = b.mkString
41 
42       if (ParseResult.isIncomplete(source))
43         BasicFilters.injectNewLine(b, c, rest, indent = 2)
44       else
45         Result(source) // short-circuit the filters
46   }
47 
48   /** Blockingly read line from `System.in`
49    *
50    *  This entry point into Ammonite handles everything to do with terminal
51    *  emulation. This includes:
52    *
53    *  - Multiline support
54    *  - Copy-pasting
55    *  - History
56    *  - Syntax highlighting
57    *
58    *  To facilitate this, filters are used. The terminal emulation, however,
59    *  does not render output - simply input.
60    */
61   def prompt: ParseResult = {
62     val historyFilter = new HistoryFilter(
63       () => history.toVector,
64       Console.BLUE,
65       AnsiNav.resetForegroundColor
66     )
67 
68     val autocompleteFilter: Filter = Filter.action("autocompleteFilter")(SpecialKeys.Tab :: Nil) {
69       case TermState(rest, b, c, _) =>
70         val Completions(newCursor, completions, details) = complete(c, b.mkString)
71         lazy val prefixDetails = FrontEndUtils.findPrefix(details)
72         val details2 = details.map { det =>
73           val (left, right) = det.splitAt(prefixDetails.length)
74           (Highlighting.Green(left) + Highlighting.Cyan(right)).toString
75         }
76 
77         lazy val prefixComp = FrontEndUtils.findPrefix(completions)
78         val completions2: Seq[String] = for(comp <- completions) yield {
79           val (left, right) = comp.splitAt(prefixComp.length)
80           (Highlighting.Green(left).toString ++ right)
81         }
82 
83         val stdout = FrontEndUtils.printCompletions(completions2, details2).mkString
84 
85         if (details.nonEmpty || completions.isEmpty)
86           Printing(TermState(rest, b, c), stdout)
87         else {
88           val newBuffer = b.take(newCursor) ++ prefixComp ++ b.drop(c)
89           Printing(TermState(rest, newBuffer, newCursor + prefixComp.length), stdout)
90         }
91     }
92 
93     val allFilters = Filter.merge(
94       UndoFilter(),
95       historyFilter,
96       selectionFilter,
97       GUILikeFilters.altFilter,
98       GUILikeFilters.fnFilter,
99       ReadlineFilters.navFilter,
100       cutPasteFilter,
101       autocompleteFilter,
102       multilineFilter,
103       BasicFilters.all
104     )
105 
106     def displayTransform(buffer: Vector[Char], cursor: Int): (Ansi.Str, Int) = {
107       val coloredBuffer =
108         SyntaxHighlighting(buffer)
109 
110       val ansiBuffer = Ansi.Str.parse(coloredBuffer.toVector)
111       val (newBuffer, cursorOffset) = SelectionFilter.mangleBuffer(
112         selectionFilter, ansiBuffer, cursor, Ansi.Reversed.On
113       )
114       val newNewBuffer = HistoryFilter.mangleBuffer(
115         historyFilter, newBuffer, cursor,
116         Ansi.Color.Green
117       )
118 
119       (newNewBuffer, cursorOffset)
120     }
121 
122     val prompt = Console.BLUE + "scala> " + Console.RESET
123 
124     Terminal
125       .readLine(prompt, reader, writer, allFilters, displayTransform)
126       .map {
127         case Result(source) => ParseResult(source)
128         case Interrupt => SigKill
129       }
130       .getOrElse(Quit)
131   }
132 }
