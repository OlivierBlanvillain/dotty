17 /** This rendering object uses `ClassLoader`s to accomplish crossing the 4th
18  *  wall (i.e. fetching back values from the compiled class files put into a
19  *  specific class loader capable of loading from memory) and rendering them.
20  *
21  *  @pre this object should be paired with a compiler session, i.e. when
22  *       `ReplDriver#resetToInitial` is called, the accompanying instance of
23  *       `Rendering` is no longer valid.
24  */
25 private[repl] class Rendering(compiler: ReplCompiler,
26                               parentClassLoader: Option[ClassLoader] = None) {
27

30   /** Class loader used to load compiled code */
31   private[this] def classLoader()(implicit ctx: Context) =
32     if (myClassLoader != null) myClassLoader
33     else {
34       val parent = parentClassLoader.getOrElse {
35         // the compiler's classpath, as URL's
36         val compilerClasspath = ctx.platform.classPath(ctx).asURLs
37         new java.net.URLClassLoader(compilerClasspath.toArray, classOf[ReplDriver].getClassLoader)
38       }
39 
40       myClassLoader = new AbstractFileClassLoader(compiler.directory, parent)
41       // Set the current Java "context" class loader to this rendering class loader
42       Thread.currentThread.setContextClassLoader(myClassLoader)
43       myClassLoader
44     }
45 
46   /** Load the value of the symbol using reflection
47    *
48    *  Calling this method evaluates the expression using reflection
49    */
50   private[this] def valueOf(sym: Symbol)(implicit ctx: Context): Option[String] = {
51     val defn = ctx.definitions
52     val objectName = sym.owner.fullName.encode.toString.dropRight(1) // gotta drop the '$'
53     val resObj: Class[_] = Class.forName(objectName, true, classLoader())
54 
55     val res =
56       resObj
57         .getDeclaredMethods.find(_.getName == sym.name.toString + "Show").get
58         .invoke(null).toString
59 
60     if (!sym.is(Flags.Method) && sym.info == defn.UnitType)
61       None
62     else if (res.startsWith(str.REPL_SESSION_LINE))
63       Some(res.drop(str.REPL_SESSION_LINE.length).dropWhile(c => c.isDigit || c == '$'))
64     else
65       Some(res)
66   }

68   /** Render method definition result */
69   def renderMethod(d: Denotation)(implicit ctx: Context): String =
70     d.symbol.showUser
71 
72   /** Render value definition result */
73   def renderVal(d: Denotation)(implicit ctx: Context): Option[String] = {
74     val dcl = d.symbol.showUser
75 
76     try {
77       val resultValue =
78         if (d.symbol.is(Flags.Lazy)) Some("<lazy>")
79         else valueOf(d.symbol)
80 
81       resultValue.map(value => s"$dcl = $value")
82     }
83     catch { case ex: InvocationTargetException => Some(renderError(ex)) }
84   }
85 
86   /** Render the stack trace of the underlying exception */
87   private def renderError(ex: InvocationTargetException): String = {
88     val cause = ex.getCause match {
89       case ex: ExceptionInInitializerError => ex.getCause
90       case ex => ex
91     }
92     val sw = new StringWriter()
93     val pw = new PrintWriter(sw)
94     cause.printStackTrace(pw)
95     sw.toString
96   }
