37 /** The state of the REPL contains necessary bindings instead of having to have
38  *  mutation
39  *
40  *  The compiler in the REPL needs to do some wrapping in order to compile
41  *  valid code. This wrapping occurs when a single `MemberDef` that cannot be
42  *  top-level needs to be compiled. In order to do this, we need some unique
43  *  identifier for each of these wrappers. That identifier is `objectIndex`.
44  *
45  *  Free expressions such as `1 + 1` needs to have an assignment in order to be
46  *  of use. These expressions are therefore given a identifier on the format
47  *  `resX` where `X` starts at 0 and each new expression that needs an
48  *  identifier is given the increment of the old identifier. This identifier is
49  *  `valIndex`.
50  *
51  *  @param objectIndex the index of the next wrapper
52  *  @param valIndex the index of next value binding for free expressions
53  *  @param history a list of user inputs as strings
54  *  @param imports a list of tuples of imports on tree form and shown form
55  *  @param run the latest run initiated at the start of interpretation. This
56  *             run and its context should be used in order to perform any
57  *             manipulation on `Tree`s and `Symbol`s.
58  */
59 case class State(objectIndex: Int,
60                  valIndex: Int,
61                  history: History,
62                  imports: List[untpd.Import],
63                  run: Run) {

165   /** Extract possible completions at the index of `cursor` in `expr` */
166   protected[this] final def completions(cursor: Int, expr: String, state0: State): Completions = {
167     // TODO move some of this logic to `Interactive`
168     implicit val state = state0.newRun(compiler, rootCtx)
169     compiler
170       .typeCheck(expr, errorsAllowed = true)
171       .map { tree =>
172         implicit val ctx: Context = state.run.runContext
173         val file = new dotc.util.SourceFile("compl", expr)
174         val srcPos = dotc.util.SourcePosition(file, Position(cursor))
175         val (startOffset, completions) = Interactive.completions(SourceTree(tree, file) :: Nil, srcPos)(ctx)
176         val query =
177           if (startOffset < cursor) expr.substring(startOffset, cursor) else ""
178 
179         def filterCompletions(name: String) =
180           (query == "." || name.startsWith(query)) && name != query
181 
182 
183         Completions(
184           Math.min(startOffset, cursor) + { if (query == ".") 1 else 0 },
185           completions.map(_.name.show).distinct.filter(filterCompletions),
186           Nil
187         )
188       }
189       .fold(_ => Completions(cursor, Nil, Nil), x => x)
190   }

244   /** Display definitions from `tree` */
245   private def displayDefinitions(tree: tpd.Tree, newestWrapper: Name)(implicit state: State): State = {
246     implicit val ctx = state.run.runContext
247 
248     def resAndUnit(denot: Denotation) = {
249       import scala.util.{ Try, Success }
250       val sym = denot.symbol
251       val name = sym.name.show
252       val hasValidNumber = Try(name.drop(3).toInt) match {
253         case Success(num) => num < state.valIndex
254         case _ => false
255       }
256       name.startsWith(str.REPL_RES_PREFIX) && hasValidNumber && sym.info == defn.UnitType
257     }
258 
259     def displayMembers(symbol: Symbol) = if (tree.symbol.info.exists) {
260       val info = symbol.info
261       val defs =
262         info.bounds.hi.finalResultType
263           .membersBasedOnFlags(Method, Accessor | ParamAccessor | Synthetic | Private)
264           .filterNot { denot =>
265             denot.symbol.owner == defn.AnyClass ||
266             denot.symbol.owner == defn.ObjectClass ||
267             denot.symbol.isConstructor
268           }
269 
270       val vals =
271         info.fields
272           .filterNot(_.symbol.is(ParamAccessor | Private | Synthetic | Module))
273           .filter(_.symbol.name.is(SimpleNameKind))
274 
275       val typeAliases =
276         info.bounds.hi.typeMembers.filter(_.symbol.info.isInstanceOf[TypeAlias])
277 
278       (
279         typeAliases.map("// defined alias " + _.symbol.showUser) ++
280         defs.map(rendering.renderMethod) ++
281         vals.map(rendering.renderVal).flatten
282       ).foreach(str => out.println(SyntaxHighlighting(str)))
283 
284       state.copy(valIndex = state.valIndex - vals.filter(resAndUnit).length)
285     }
286     else state

288     def isSyntheticCompanion(sym: Symbol) =
289       sym.is(Module) && sym.is(Synthetic)

291     def displayTypeDefs(sym: Symbol) = sym.info.memberClasses
292       .collect {
293         case x if !isSyntheticCompanion(x.symbol) && !x.symbol.name.isReplWrapperName =>

295       }
296       .foreach { sym =>
297         out.println(SyntaxHighlighting("// defined " + sym.showUser))
298       }
299 
300 
301     ctx.atPhase(ctx.typerPhase.next) { implicit ctx =>
302 
303       // Display members of wrapped module:
304       tree.symbol.info.memberClasses
305         .find(_.symbol.name == newestWrapper.moduleClassName)
306         .map { wrapperModule =>
307           displayTypeDefs(wrapperModule.symbol)
308           displayMembers(wrapperModule.symbol)
309         }
310         .getOrElse {
311           // user defined a trait/class/object, so no module needed
312           state
313         }
314     }
315   }

317   /** Interpret `cmd` to action and propagate potentially new `state` */
318   private def interpretCommand(cmd: Command)(implicit state: State): State = cmd match {
319     case UnknownCommand(cmd) => {
320       out.println(s"""Unknown command: "$cmd", run ":help" for a list of commands""")
321       state.withHistory(s"$cmd")
322     }
323 
324     case Help => {
325       out.println(Help.text)
326       state.withHistory(Help.command)
327     }
328 
329     case Reset => {
330       resetToInitial()
331       initState
332     }
333 
334     case Imports => {
335       state.imports.foreach(i => out.println(SyntaxHighlighting(i.show(state.run.runContext))))
336       state.withHistory(Imports.command)
337     }
338 
339     case Load(path) =>
340       val loadCmd = s"${Load.command} $path"
341       val file = new java.io.File(path)
342       if (file.exists) {
343         val contents = scala.io.Source.fromFile(path).mkString
344         ParseResult(contents)(state.run.runContext) match {
345           case parsed: Parsed =>
346             compile(parsed).withHistory(loadCmd)
347           case SyntaxErrors(_, errors, _) =>
348             displayErrors(errors).withHistory(loadCmd)
349           case _ =>
350             state.withHistory(loadCmd)
351         }
352       }
353       else {
354         out.println(s"""Couldn't find file "${file.getCanonicalPath}"""")
355         state.withHistory(loadCmd)
356       }
357 
358     case TypeOf(expr) => {
359       compiler.typeOf(expr).fold(
360         displayErrors,
361         res => out.println(SyntaxHighlighting(res))
362       )
363       state.withHistory(s"${TypeOf.command} $expr")
364     }
365 
366     case Quit =>
367       // end of the world!
368       state
369   }
