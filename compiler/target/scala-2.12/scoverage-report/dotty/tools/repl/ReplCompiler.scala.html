20 /** This subclass of `Compiler` replaces the appropriate phases in order to

72   private def definitions(trees: List[untpd.Tree], state: State): Definitions = {
73     import untpd._
74 
75     implicit val ctx: Context = state.run.runContext

77     def createShow(name: TermName, pos: Position) = {
78       val showName = name ++ "Show"
79       val select = Select(Ident(name), "show".toTermName)
80       val valAsAnyRef = TypeApply(Select(Ident(name), nme.asInstanceOf_),
81                                   List(Ident(tpnme.AnyRef)))
82       val cond = InfixOp(valAsAnyRef,
83                          Ident(nme.EQ),
84                          Literal(Constant(null)))
85       val showWithNullCheck = If(cond, Literal(Constant("null")), select)
86       DefDef(showName, Nil, Nil, TypeTree(), showWithNullCheck).withFlags(Synthetic).withPos(pos)
87     }
88 
89     def createPatDefShows(patDef: PatDef) = {
90       def createDeepShows(tree: untpd.Tree) = {
91         class PatFolder extends UntypedDeepFolder[List[DefDef]] (
92           (acc, tree) => tree match {
93             case Ident(name) if name.isVariableName && name != nme.WILDCARD =>
94               createShow(name.toTermName, tree.pos) :: acc
95             case Bind(name, _) if name.isVariableName && name != nme.WILDCARD =>
96               createShow(name.toTermName, tree.pos) :: acc
97             case _ =>
98               acc
99           }
100         )
101         (new PatFolder).apply(Nil, tree).reverse
102       }
103 
104       // cannot fold over the whole tree because we need to generate show methods
105       // for top level identifier starting with an uppercase (e.g. val X, Y = 2)
106       patDef.pats.flatMap {
107         case id @ Ident(name) if name != nme.WILDCARD =>
108           List(createShow(name.toTermName, id.pos))
109         case bd @ Bind(name, body) if name != nme.WILDCARD =>
110           createShow(name.toTermName, bd.pos) :: createDeepShows(body)
111         case other =>
112           createDeepShows(other)
113       }
114     }
115 
116     var valIdx = state.valIndex
117 
118     val defs = trees.flatMap {
119       case vd: ValDef =>
120         List(vd, createShow(vd.name, vd.pos))
121       case pd: PatDef =>
122         pd :: createPatDefShows(pd)
123       case expr @ Assign(id: Ident, rhs) =>
124         // special case simple reassignment (e.g. x = 3)
125         // in order to print the new value in the REPL
126         val assignName = (id.name ++ str.REPL_ASSIGN_SUFFIX).toTermName
127         val assign = ValDef(assignName, TypeTree(), id).withPos(expr.pos)
128         val show = createShow(assignName, expr.pos)
129         List(expr, assign, show)
130       case expr if expr.isTerm =>
131         val resName = (str.REPL_RES_PREFIX + valIdx).toTermName
132         valIdx += 1
133         val show = createShow(resName, expr.pos)
134         val vd = ValDef(resName, TypeTree(), expr).withPos(expr.pos)
135         List(vd, show)
136       case other =>
137         List(other)
138     }
139 
140     Definitions(
141       state.imports ++ defs,
142       state.copy(
143         objectIndex = state.objectIndex + (if (defs.isEmpty) 0 else 1),
144         valIndex = valIdx
145       )
146     )
147   }

149   /** Wrap trees in an object and add imports from the previous compilations
150    *
151    *  The resulting structure is something like:
152    *
153    *  ```
154    *  package <none> {
155    *    object rs$line$nextId {
156    *      import rs$line${i <- 0 until nextId}._
157    *      import dotty.Show._
158    *
159    *      <trees>
160    *    }
161    *  }
162    *  ```
163    */
164   private def wrapped(defs: Definitions): untpd.PackageDef = {
165     import untpd._
166 
167     assert(defs.stats.nonEmpty)
168 
169     implicit val ctx: Context = defs.state.run.runContext
170 
171     val tmpl = Template(emptyConstructor, Nil, EmptyValDef, defs.stats)
172     val module = ModuleDef(objectName(defs.state), tmpl)
173       .withMods(new Modifiers(Module | Final))
174       .withPos(Position(0, defs.stats.last.pos.end))
175 
176     PackageDef(Ident(nme.EMPTY_PACKAGE), List(module))
177   }

179   private def createUnit(defs: Definitions, sourceCode: String): CompilationUnit = {
180     val unit = new CompilationUnit(new SourceFile(objectName(defs.state).toString, sourceCode))
181     unit.untpdTree = wrapped(defs)

185   private def runCompilationUnit(unit: CompilationUnit, state: State): Result[(CompilationUnit, State)] = {
186     val run = state.run
187     val reporter = state.run.runContext.reporter
188     run.compileUnits(unit :: Nil)
189 
190     if (!reporter.hasErrors) (unit, state).result
191     else run.runContext.flushBufferedMessages().errors
192   }

216     def wrapped(expr: String, sourceFile: SourceFile, state: State)(implicit ctx: Context): Result[untpd.PackageDef] = {
217       def wrap(trees: Seq[untpd.Tree]): untpd.PackageDef = {
218         import untpd._
219 
220         val valdef =
221           ValDef("expr".toTermName, TypeTree(), Block(trees.toList, untpd.unitLiteral))
222 
223         val tmpl = Template(emptyConstructor,
224                             List(Ident(tpnme.Any)),

228         PackageDef(Ident(nme.EMPTY_PACKAGE),
229           TypeDef("EvaluateExpr".toTypeName, tmpl)
230             .withMods(new Modifiers(Final))
231             .withPos(Position(0, expr.length)) :: Nil

233       }

236         case Parsed(sourceCode, trees) =>
237           wrap(trees).result
238         case SyntaxErrors(_, reported, trees) =>
239           if (errorsAllowed) wrap(trees).result
240           else reported.errors
241         case _ => List(
242           new messages.Error(
243             s"Couldn't parse '$expr' to valid scala",
244             sourceFile.atPos(Position(0, expr.length))
245           )
246         ).errors
247       }

250     def unwrapped(tree: tpd.Tree, sourceFile: SourceFile)(implicit ctx: Context): Result[tpd.ValDef] = {
251       def error: Result[tpd.ValDef] =
252         List(new messages.Error(s"Invalid scala expression",
253           sourceFile.atPos(Position(0, sourceFile.content.length)))).errors
254 
255       import tpd._
256       tree match {
257         case PackageDef(_, List(TypeDef(_, tmpl: Template))) =>
258           tmpl.body
259               .collectFirst { case dd: ValDef if dd.name.show == "expr" => dd.result }
260               .getOrElse(error)
261         case _ =>
262           error
263       }
264     }
265 
266 
267     val run = state.run
268     val reporter = state.run.runContext.reporter
269     val src = new SourceFile(s"EvaluateExpr", expr)
270     val runCtx =
271       run.runContext.fresh
272          .setSetting(run.runContext.settings.YstopAfter, List("frontend"))
273 
274     wrapped(expr, src, state)(runCtx).flatMap { pkg =>
275       val unit = new CompilationUnit(src)
276       unit.untpdTree = pkg
277       run.compileUnits(unit :: Nil, runCtx.fresh.setReporter(storeReporter))
278 
279       if (errorsAllowed || !reporter.hasErrors)
280         unwrapped(unit.tpdTree, src)(runCtx)
281       else {
282         reporter.removeBufferedMessages(runCtx).errors
283       }
284     }
