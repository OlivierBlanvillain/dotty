123

249      * Emits one or more conversion instructions based on the types given as arguments.
250      *
251      * @param from The type of the value to be converted into another type.
252      * @param to   The type the value will be converted into.

254      * can-multi-thread

256     final def emitT2T(from: BType, to: BType): Unit = {
257 
258       assert(
259         from.isNonVoidPrimitiveType && to.isNonVoidPrimitiveType,
260         s"Cannot emit primitive conversion from $from to $to"
261       )
262 
263           def pickOne(opcs: Array[Int]): Unit = { // TODO index on to.sort
264             val chosen = (to: @unchecked) match {
265               case BYTE   => opcs(0)
266               case SHORT  => opcs(1)
267               case CHAR   => opcs(2)
268               case INT    => opcs(3)
269               case LONG   => opcs(4)
270               case FLOAT  => opcs(5)
271               case DOUBLE => opcs(6)
272             }
273             if (chosen != -1) { emit(chosen) }
274           }
275 
276       if (from == to) { return }
277       // the only conversion involving BOOL that is allowed is (BOOL -> BOOL)
278       assert(from != BOOL && to != BOOL, s"inconvertible types : $from -> $to")
279 
280       // We're done with BOOL already
281       from match {
282 
283         // using `asm.Type.SHORT` instead of `BType.SHORT` because otherwise "warning: could not emit switch for @switch annotated match"
284 
285         case BYTE  => pickOne(JCodeMethodN.fromByteT2T)
286         case SHORT => pickOne(JCodeMethodN.fromShortT2T)
287         case CHAR  => pickOne(JCodeMethodN.fromCharT2T)
288         case INT   => pickOne(JCodeMethodN.fromIntT2T)
289 
290         case FLOAT  =>
291           import asm.Opcodes.{ F2L, F2D, F2I }
292           to match {
293             case LONG    => emit(F2L)
294             case DOUBLE  => emit(F2D)
295             case _       => emit(F2I); emitT2T(INT, to)
296           }
297 
298         case LONG   =>
299           import asm.Opcodes.{ L2F, L2D, L2I }
300           to match {
301             case FLOAT   => emit(L2F)
302             case DOUBLE  => emit(L2D)
303             case _       => emit(L2I); emitT2T(INT, to)
304           }
305 
306         case DOUBLE =>
307           import asm.Opcodes.{ D2L, D2F, D2I }
308           to match {
309             case FLOAT   => emit(D2F)
310             case LONG    => emit(D2L)
311             case _       => emit(D2I); emitT2T(INT, to)
312           }
313       }
314     } // end of emitT2T()

444     /* Emits one of tableswitch or lookoupswitch.
445      *
446      * can-multi-thread
447      */
448     final def emitSWITCH(keys: Array[Int], branches: Array[asm.Label], defaultBranch: asm.Label, minDensity: Double): Unit = {
449       assert(keys.length == branches.length)
450 
451       // For empty keys, it makes sense emitting LOOKUPSWITCH with defaultBranch only.
452       // Similar to what javac emits for a switch statement consisting only of a default case.
453       if (keys.length == 0) {
454         jmethod.visitLookupSwitchInsn(defaultBranch, keys, branches)
455         return
456       }
457 
458       // sort `keys` by increasing key, keeping `branches` in sync. TODO FIXME use quicksort
459       var i = 1
460       while (i < keys.length) {
461         var j = 1
462         while (j <= keys.length - i) {
463           if (keys(j) < keys(j - 1)) {
464             val tmp     = keys(j)
465             keys(j)     = keys(j - 1)
466             keys(j - 1) = tmp
467             val tmpL        = branches(j)
468             branches(j)     = branches(j - 1)
469             branches(j - 1) = tmpL
470           }
471           j += 1
472         }
473         i += 1
474       }
475 
476       // check for duplicate keys to avoid "VerifyError: unsorted lookupswitch" (SI-6011)
477       i = 1
478       while (i < keys.length) {
479         if (keys(i-1) == keys(i)) {
480           abort("duplicate keys in SWITCH, can't pick arbitrarily one of them to evict, see SI-6011.")
481         }
482         i += 1
483       }
484 
485       val keyMin = keys(0)
486       val keyMax = keys(keys.length - 1)
487 
488       val isDenseEnough: Boolean = {
489         /* Calculate in long to guard against overflow. TODO what overflow? */
490         val keyRangeD: Double = (keyMax.asInstanceOf[Long] - keyMin + 1).asInstanceOf[Double]
491         val klenD:     Double = keys.length
492         val kdensity:  Double = (klenD / keyRangeD)
493 
494         kdensity >= minDensity
495       }
496 
497       if (isDenseEnough) {
498         // use a table in which holes are filled with defaultBranch.
499         val keyRange    = (keyMax - keyMin + 1)
500         val newBranches = new Array[asm.Label](keyRange)
501         var oldPos = 0
502         var i = 0
503         while (i < keyRange) {
504           val key = keyMin + i;
505           if (keys(oldPos) == key) {
506             newBranches(i) = branches(oldPos)
507             oldPos += 1
508           } else {
509             newBranches(i) = defaultBranch
510           }
511           i += 1
512         }
513         assert(oldPos == keys.length, "emitSWITCH")
514         jmethod.visitTableSwitchInsn(keyMin, keyMax, defaultBranch, newBranches: _*)
515       } else {
516         jmethod.visitLookupSwitchInsn(defaultBranch, keys, branches)
517       }
518     }
