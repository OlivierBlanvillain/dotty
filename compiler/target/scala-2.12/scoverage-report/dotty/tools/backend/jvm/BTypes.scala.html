107     /*
108      * Subtype check `this <:< other` on BTypes that takes into account the JVM built-in numeric
109      * promotions (e.g. BYTE to INT). Its operation can be visualized more easily in terms of the
110      * Java bytecode type hierarchy.
111      */
112     final def conformsTo(other: BType): Boolean = {
113       assert(isRef || isPrimitive, s"conformsTo cannot handle $this")
114       assert(other.isRef || other.isPrimitive, s"conformsTo cannot handle $other")
115 
116       this match {
117         case ArrayBType(component) =>
118           if (other == ObjectReference || other == jlCloneableReference || other == jioSerializableReference) true
119           else other match {
120             case ArrayBType(otherComponoent) => component.conformsTo(otherComponoent)
121             case _ => false
122           }
123 
124         case classType: ClassBType =>
125           if (isBoxed) {
126             if (other.isBoxed) this == other
127             else if (other == ObjectReference) true
128             else other match {
129               case otherClassType: ClassBType => classType.isSubtypeOf(otherClassType) // e.g., java/lang/Double conforms to java/lang/Number
130               case _ => false
131             }
132           } else if (isNullType) {
133             if (other.isNothingType) false
134             else if (other.isPrimitive) false
135             else true // Null conforms to all classes (except Nothing) and arrays.
136           } else if (isNothingType) {
137             true
138           } else other match {
139             case otherClassType: ClassBType => classType.isSubtypeOf(otherClassType)
140             // case ArrayBType(_) => this.isNullType   // documentation only, because `if (isNullType)` above covers this case
141             case _ =>
142               // isNothingType ||                      // documentation only, because `if (isNothingType)` above covers this case
143               false
144           }
145 
146         case UNIT =>
147           other == UNIT
148         case BOOL | BYTE | SHORT | CHAR =>
149           this == other || other == INT || other == LONG // TODO Actually, BOOL does NOT conform to LONG. Even with adapt().
150         case _ =>
151           assert(isPrimitive && other.isPrimitive, s"Expected primitive types $this - $other")
152           this == other
153       }
154     }

156     /**
157      * Compute the upper bound of two types.
158      * Takes promotions of numeric primitives into account.
159      */
160     final def maxType(other: BType): BType = this match {
161       case pt: PrimitiveBType => pt.maxValueType(other)
162 
163       case _: ArrayBType | _: ClassBType =>
164         if (isNothingType)       return other
165         if (other.isNothingType) return this
166         if (this == other)       return this
167 
168         assert(other.isRef, s"Cannot compute maxType: $this, $other")
169         // Approximate `lub`. The common type of two references is always ObjectReference.
170         ObjectReference
171     }

201     /**
202      * Some JVM opcodes have typed variants. This method returns the correct opcode according to
203      * the type.
204      *
205      * @param opcode A JVM instruction opcode. This opcode must be one of ILOAD, ISTORE, IALOAD,
206      *               IASTORE, IADD, ISUB, IMUL, IDIV, IREM, INEG, ISHL, ISHR, IUSHR, IAND, IOR
207      *               IXOR and IRETURN.
208      * @return The opcode adapted to this java type. For example, if this type is `float` and
209      *         `opcode` is `IRETURN`, this method returns `FRETURN`.
210      */
211     final def typedOpcode(opcode: Int): Int = {
212       if (opcode == asm.Opcodes.IALOAD || opcode == asm.Opcodes.IASTORE)
213         opcode + loadStoreOpcodeOffset
214       else
215         opcode + typedOpcodeOffset
216     }

250   sealed trait PrimitiveBType extends BType {
251 
252     /**
253      * The upper bound of two primitive types. The `other` type has to be either a primitive
254      * type or Nothing.
255      *
256      * The maxValueType of (Char, Byte) and of (Char, Short) is Int, to encompass the negative
257      * values of Byte and Short. See ticket #2087.
258      */
259     final def maxValueType(other: BType): BType = {
260 
261       def uncomparable: Nothing = throw new AssertionError(s"Cannot compute maxValueType: $this, $other")
262 
263       if (!other.isPrimitive && !other.isNothingType) uncomparable
264 
265       if (other.isNothingType) return this
266       if (this == other)       return this
267 
268       this match {
269         case BYTE =>
270           if (other == CHAR)            INT
271           else if (other.isNumericType) other
272           else                          uncomparable
273 
274         case SHORT =>
275           other match {
276             case BYTE                          => SHORT
277             case CHAR                          => INT
278             case INT  | LONG  | FLOAT | DOUBLE => other
279             case _                             => uncomparable
280           }
281 
282         case CHAR =>
283           other match {
284             case BYTE | SHORT                 => INT
285             case INT  | LONG | FLOAT | DOUBLE => other
286             case _                            => uncomparable
287           }
288 
289         case INT =>
290           other match {
291             case BYTE | SHORT | CHAR   => INT
292             case LONG | FLOAT | DOUBLE => other
293             case _                     => uncomparable
294           }
295 
296         case LONG =>
297           if (other.isIntegralType)  LONG
298           else if (other.isRealType) DOUBLE
299           else                       uncomparable
300 
301         case FLOAT =>
302           if (other == DOUBLE)          DOUBLE
303           else if (other.isNumericType) FLOAT
304           else                          uncomparable
305 
306         case DOUBLE =>
307           if (other.isNumericType) DOUBLE
308           else                     uncomparable
309 
310         case UNIT | BOOL => uncomparable
311       }
312     }
313   }

343   /**
344    * InnerClass and EnclosingMethod attributes (EnclosingMethod is displayed as OUTERCLASS in asm).
345    *
346    * In this summary, "class" means "class or interface".
347    *
348    * JLS: http://docs.oracle.com/javase/specs/jls/se8/html/index.html
349    * JVMS: http://docs.oracle.com/javase/specs/jvms/se8/html/index.html
350    *
351    * Terminology
352    * -----------
353    *
354    *  - Nested class (JLS 8): class whose declaration occurs within the body of another class
355    *
356    *  - Top-level class (JLS 8): non-nested class
357    *
358    *  - Inner class (JLS 8.1.3): nested class that is not (explicitly or implicitly) static
359    *
360    *  - Member class (JLS 8.5): class directly enclosed in the body of a class (and not, for
361    *    example, defined in a method). Member classes cannot be anonymous. May be static.
362    *
363    *  - Local class (JLS 14.3): nested, non-anonymous class that is not a member of a class
364    *    - cannot be static (therefore they are "inner" classes)
365    *    - can be defined in a method, a constructor or in an initializer block
366    *
367    *  - Initializer block (JLS 8.6 / 8.7): block of statements in a java class
368    *    - static initializer: executed before constructor body
369    *    - instance initializer: exectued when class is initialized (instance creation, static
370    *      field access, ...)
371    *
372    *  - A static nested class can be defined as
373    *    - a static member class (explicitly static), or
374    *    - a member class of an interface (implicitly static)
375    *    - local classes are never static, even if they are defined in a static method.
376    *
377    *   Note: it is NOT the case that all inner classes (non-static) have an outer pointer. Example:
378    *     class C { static void foo { class D {} } }
379    *   The class D is an inner class (non-static), but javac does not add an outer pointer to it.
380    *
381    * InnerClass
382    * ----------
383    *
384    * The JVMS 4.7.6 requires an entry for every class mentioned in a CONSTANT_Class_info in the
385    * constant pool (CP) that is not a member of a package (JLS 7.1).
386    *
387    * The JLS 13.1, points 9. / 10. requires: a class must reference (in the CP)
388    *  - its immediately enclosing class
389    *  - all of its member classes
390    *  - all local and anonymous classes that are referenced (or declared) elsewhere (method,
391    *    constructor, initializer block, field initializer)
392    *
393    * In a comment, the 4.7.6 spec says: this implies an entry in the InnerClass attribute for
394    *  - All enclosing classes (except the outermost, which is top-level)
395    *    - My comment: not sure how this is implied, below (*) a Java counter-example.
396    *      In any case, the Java compiler seems to add all enclosing classes, even if they are not
397    *      otherwise mentioned in the CP. So we should do the same.
398    *  - All nested classes (including anonymous and local, but not transitively)
399    *
400    * Fields in the InnerClass entries:
401    *  - inner class: the (nested) class C we are talking about
402    *  - outer class: the class of which C is a member. Has to be null for non-members, i.e. for
403    *                 local and anonymous classes. NOTE: this co-incides with the presence of an
404    *                 EnclosingMethod attribute (see below)
405    *  - inner name:  A string with the simple name of the inner class. Null for anonymous classes.
406    *  - flags:       access property flags, details in JVMS, table in 4.7.6. Static flag: see
407    *                 discussion below.
408    *
409    *
410    * Note 1: when a nested class is present in the InnerClass attribute, all of its enclosing
411    * classes have to be present as well (by the rules above). Example:
412    *
413    *   class Outer { class I1 { class I2 { } } }
414    *   class User { Outer.I1.I2 foo() { } }
415    *
416    * The return type "Outer.I1.I2" puts "Outer$I1$I2" in the CP, therefore the class is added to the
417    * InnerClass attribute. For this entry, the "outer class" field will be "Outer$I1". This in turn
418    * adds "Outer$I1" to the CP, which requires adding that class to the InnerClass attribute.
419    * (For local / anonymous classes this would not be the case, since the "outer class" attribute
420    *  would be empty. However, no class (other than the enclosing class) can refer to them, as they
421    *  have no name.)
422    *
423    * In the current implementation of the Scala compiler, when adding a class to the InnerClass
424    * attribute, all of its enclosing classes will be added as well. Javac seems to do the same,
425    * see (*).
426    *
427    *
428    * Note 2: If a class name is mentioned only in a CONSTANT_Utf8_info, but not in a
429    * CONSTANT_Class_info, the JVMS does not require an entry in the InnerClass attribute. However,
430    * the Java compiler seems to add such classes anyway. For example, when using an annotation, the
431    * annotation class is stored as a CONSTANT_Utf8_info in the CP:
432    *
433    *   @O.Ann void foo() { }
434    *
435    * adds "const #13 = Asciz LO$Ann;;" in the constant pool. The "RuntimeInvisibleAnnotations"
436    * attribute refers to that constant pool entry. Even though there is no other reference to
437    * `O.Ann`, the java compiler adds an entry for that class to the InnerClass attribute (which
438    * entails adding a CONSTANT_Class_info for the class).
439    *
440    *
441    *
442    * EnclosingMethod
443    * ---------------
444    *
445    * JVMS 4.7.7: the attribute must be present "if and only if it represents a local class
446    * or an anonymous class" (i.e. not for member classes).
447    *
448    * The attribute is mis-named, it should be called "EnclosingClass". It has to be defined for all
449    * local and anonymous classes, no matter if there is an enclosing method or not. Accordingly, the
450    * "class" field (see below) must be always defined, while the "method" field may be null.
451    *
452    * NOTE: When a EnclosingMethod attribute is requried (local and anonymous classes), the "outer"
453    * field in the InnerClass table must be null.
454    *
455    * Fields:
456    *  - class:  the enclosing class
457    *  - method: the enclosing method (or constructor). Null if the class is not enclosed by a
458    *            method, i.e. for
459    *             - local or anonymous classes defined in (static or non-static) initializer blocks
460    *             - anonymous classes defined in initializer blocks or field initializers
461    *
462    *            Note: the field is required for anonymous classes defined within local variable
463    *            initializers (within a method), Java example below (**).
464    *
465    *            For local and anonymous classes in initializer blocks or field initializers, and
466    *            class-level anonymous classes, the scala compiler sets the "method" field to null.
467    *
468    *
469    * (*)
470    *   public class Test {
471    *     void foo() {
472    *       class Foo1 {
473    *         // constructor statement block
474    *         {
475    *           class Foo2 {
476    *             class Foo3 { }
477    *           }
478    *         }
479    *       }
480    *     }
481    *   }
482    *
483    * The class file Test$1Foo1$1Foo2$Foo3 has no reference to the class Test$1Foo1, however it
484    * still contains an InnerClass attribute for Test$1Foo1.
485    * Maybe this is just because the Java compiler follows the JVMS comment ("InnerClasses
486    * information for each enclosing class").
487    *
488    *
489    * (**)
490    *   void foo() {
491    *     // anonymous class defined in local variable initializer expression.
492    *     Runnable x = true ? (new Runnable() {
493    *       public void run() { return; }
494    *     }) : null;
495    *   }
496    *
497    * The EnclosingMethod attribute of the anonymous class mentions "foo" in the "method" field.
498    *
499    *
500    * Java Compatibility
501    * ------------------
502    *
503    * In the InnerClass entry for classes in top-level modules, the "outer class" is emitted as the
504    * mirror class (or the existing companion class), i.e. C1 is nested in T (not T$).
505    * For classes nested in a nested object, the "outer class" is the module class: C2 is nested in T$N$
506    * object T {
507    *   class C1
508    *   object N { class C2 }
509    * }
510    *
511    * Reason: java compat. It's a "best effort" "solution". If you want to use "C1" from Java, you
512    * can write "T.C1", and the Java compiler will translate that to the classfile T$C1.
513    *
514    * If we would emit the "outer class" of C1 as "T$", then in Java you'd need to write "T$.C1"
515    * because the java compiler looks at the InnerClass attribute to find if an inner class exists.
516    * However, the Java compiler would then translate the '.' to '$' and you'd get the class name
517    * "T$$C1". This class file obviously does not exist.
518    *
519    * Directly using the encoded class name "T$C1" in Java does not work: since the classfile
520    * describes a nested class, the Java compiler hides it from the classpath and will report
521    * "cannot find symbol T$C1". This means that the class T.N.C2 cannot be referenced from a
522    * Java source file in any way.
523    *
524    *
525    * STATIC flag
526    * -----------
527    *
528    * Java: static member classes have the static flag in the InnerClass attribute, for example B in
529    *   class A { static class B { } }
530    *
531    * The spec is not very clear about when the static flag should be emitted. It says: "Marked or
532    * implicitly static in source."
533    *
534    * The presence of the static flag does NOT coincide with the absence of an "outer" field in the
535    * class. The java compiler never puts the static flag for local classes, even if they don't have
536    * an outer pointer:
537    *
538    *   class A {
539    *     void f()        { class B {} }
540    *     static void g() { calss C {} }
541    *   }
542    *
543    * B has an outer pointer, C doesn't. Both B and C are NOT marked static in the InnerClass table.
544    *
545    * It seems sane to follow the same principle in the Scala compiler. So:
546    *
547    *   package p
548    *   object O1 {
549    *     class C1 // static inner class
550    *     object O2 { // static inner module
551    *       def f = {
552    *         class C2 { // non-static inner class, even though there's no outer pointer
553    *           class C3 // non-static, has an outer pointer
554    *         }
555    *       }
556    *     }
557    *   }
558    *
559    * Mirror Classes
560    * --------------
561    *
562    * TODO: innerclass attributes on mirror class

565   /**
566    * A ClassBType represents a class or interface type. The necessary information to build a
567    * ClassBType is extracted from compiler symbols and types, see BTypesFromSymbols.
568    *
569    * The `offset` and `length` fields are used to represent the internal name of the class. They
570    * are indices into some character array. The internal name can be obtained through the method
571    * `internalNameString`, which is abstract in this component. Name creation is assumed to be
572    * hash-consed, so if two ClassBTypes have the same internal name, they NEED to have the same
573    * `offset` and `length`.
574    *
575    * The actual implementation in subclass BTypesFromSymbols uses the global `chrs` array from the
576    * name table. This representation is efficient because the JVM class name is obtained through
577    * `classSymbol.javaBinaryName`. This already adds the necessary string to the `chrs` array,
578    * so it makes sense to reuse the same name table in the backend.
579    *
580    * ClassBType is not a case class because we want a custom equals method, and because the
581    * extractor extracts the internalName, which is what you typically need.
582    */
583   final class ClassBType(val internalName: String) extends RefBType {
