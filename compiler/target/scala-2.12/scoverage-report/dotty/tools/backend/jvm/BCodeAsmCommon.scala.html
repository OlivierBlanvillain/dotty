15   /**
16    * True if `classSym` is an anonymous class or a local class. I.e., false if `classSym` is a
17    * member class. This method is used to decide if we should emit an EnclosingMethod attribute.
18    * It is also used to decide whether the "owner" field in the InnerClass attribute should be
19    * null.
20    */
21   def isAnonymousOrLocalClass(classSym: Symbol): Boolean = {
22     assert(classSym.isClass, s"not a class: $classSym")
23     // Here used to be an `assert(!classSym.isDelambdafyFunction)`: delambdafy lambda classes are
24     // always top-level. However, SI-8900 shows an example where the weak name-based implementation
25     // of isDelambdafyFunction failed (for a function declared in a package named "lambda").
26     classSym.isAnonymousClass || (classSym.originalOwner != NoSymbol && !classSym.originalOwner.isClass)
27   }

29   /**
30    * Returns the enclosing method for non-member classes. In the following example
31    *
32    * class A {
33    *   def f = {
34    *     class B {
35    *       class C
36    *     }
37    *   }
38    * }
39    *
40    * the method returns Some(f) for B, but None for C, because C is a member class. For non-member
41    * classes that are not enclosed by a method, it returns None:
42    *
43    * class A {
44    *   { class B }
45    * }
46    *
47    * In this case, for B, we return None.
48    *
49    * The EnclosingMethod attribute needs to be added to non-member classes (see doc in BTypes).
50    * This is a source-level property, so we need to use the originalOwner chain to reconstruct it.
51    */
52   private def enclosingMethodForEnclosingMethodAttribute(classSym: Symbol): Option[Symbol] = {
53     assert(classSym.isClass, classSym)
54     def enclosingMethod(sym: Symbol): Option[Symbol] = {
55       if (sym.isClass || sym == NoSymbol) None
56       else if (sym.isMethod) Some(sym)
57       else enclosingMethod(sym.originalOwner)
58     }
59     enclosingMethod(classSym.originalOwner)
60   }
61 
62   /**
63    * The enclosing class for emitting the EnclosingMethod attribute. Since this is a source-level
64    * property, this method looks at the originalOwner chain. See doc in BTypes.
65    */
66   private def enclosingClassForEnclosingMethodAttribute(classSym: Symbol): Symbol = {
67     assert(classSym.isClass, classSym)
68     def enclosingClass(sym: Symbol): Symbol = {
69       if (sym.isClass) sym
70       else enclosingClass(sym.originalOwner)
71     }
72     enclosingClass(classSym.originalOwner)
73   }

75   /*final*/ case class EnclosingMethodEntry(owner: String, name: String, methodDescriptor: String)
76 
77   /**
78    * Data for emitting an EnclosingMethod attribute. None if `classSym` is a member class (not
79    * an anonymous or local class). See doc in BTypes.
80    *
81    * The class is parametrized by two functions to obtain a bytecode class descriptor for a class
82    * symbol, and to obtain a method signature descriptor fro a method symbol. These function depend
83    * on the implementation of GenASM / GenBCode, so they need to be passed in.
84    */
85   def enclosingMethodAttribute(classSym: Symbol, classDesc: Symbol => String, methodDesc: Symbol => String): Option[EnclosingMethodEntry] = {
86     if (isAnonymousOrLocalClass(classSym)) {
87       val methodOpt = enclosingMethodForEnclosingMethodAttribute(classSym)
88       debuglog(s"enclosing method for $classSym is $methodOpt (in ${methodOpt.map(_.enclClass)})")
89       Some(EnclosingMethodEntry(
90         classDesc(enclosingClassForEnclosingMethodAttribute(classSym)),
91         methodOpt.map(_.javaSimpleName.toString).orNull,
92         methodOpt.map(methodDesc).orNull))
93     } else {
94       None
95     }
96   }
97 }

100   def ubytesToCharArray(bytes: Array[Byte]): Array[Char] = {
101     val ca = new Array[Char](bytes.length)
102     var idx = 0
103     while(idx < bytes.length) {
104       val b: Byte = bytes(idx)
105       assert((b & ~0x7f) == 0)
106       ca(idx) = b.asInstanceOf[Char]
107       idx += 1
108     }
109 
110     ca
111   }

113   final def arrEncode(bSeven: Array[Byte]): Array[String] = {
114     var strs: List[String]  = Nil
115     // chop into slices of at most 65535 bytes, counting 0x00 as taking two bytes (as per JVMS 4.4.7 The CONSTANT_Utf8_info Structure)
116     var prevOffset = 0
117     var offset     = 0
118     var encLength  = 0
119     while(offset < bSeven.length) {
120       val deltaEncLength = (if(bSeven(offset) == 0) 2 else 1)
121       val newEncLength = encLength.toLong + deltaEncLength
122       if(newEncLength >= 65535) {
123         val ba     = bSeven.slice(prevOffset, offset)
124         strs     ::= new java.lang.String(ubytesToCharArray(ba))
125         encLength  = 0
126         prevOffset = offset
127       } else {
128         encLength += deltaEncLength
129         offset    += 1
130       }
131     }
132     if(prevOffset < offset) {
133       assert(offset == bSeven.length)
134       val ba = bSeven.slice(prevOffset, offset)
135       strs ::= new java.lang.String(ubytesToCharArray(ba))
136     }
137     assert(strs.size > 1, "encode instead as one String via strEncode()") // TODO too strict?
138     strs.reverse.toArray
139   }
