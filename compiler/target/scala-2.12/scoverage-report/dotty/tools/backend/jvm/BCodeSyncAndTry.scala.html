30     def genSynchronized(tree: Apply, expectedType: BType): BType = tree match {
31       case Apply(fun, args) =>
32       val monitor = locals.makeLocal(ObjectReference, "monitor", Object_Type, tree.pos)
33       val monCleanup = new asm.Label
34 
35       // if the synchronized block returns a result, store it in a local variable.
36       // Just leaving it on the stack is not valid in MSIL (stack is cleaned when leaving try-blocks).
37       val hasResult = (expectedType != UNIT)
38       val monitorResult: Symbol = if (hasResult) locals.makeLocal(tpeTK(args.head), "monitorResult", Object_Type, tree.pos) else null;
39 
40       /* ------ (1) pushing and entering the monitor, also keeping a reference to it in a local var. ------ */
41       genLoadQualifier(fun)
42       bc dup ObjectReference
43       locals.store(monitor)
44       emit(asm.Opcodes.MONITORENTER)
45 
46       /* ------ (2) Synchronized block.
47        *            Reached by fall-through from (1).
48        *            Protected by:
49        *            (2.a) the EH-version of the monitor-exit, and
50        *            (2.b) whatever protects the whole synchronized expression.
51        * ------
52        */
53       val startProtected = currProgramPoint()
54       registerCleanup(monCleanup)
55       genLoad(args.head, expectedType /* toTypeKind(tree.tpe.resultType) */)
56       unregisterCleanup(monCleanup)
57       if (hasResult) { locals.store(monitorResult) }
58       nopIfNeeded(startProtected)
59       val endProtected = currProgramPoint()
60 
61       /* ------ (3) monitor-exit after normal, non-early-return, termination of (2).
62        *            Reached by fall-through from (2).
63        *            Protected by whatever protects the whole synchronized expression.
64        * ------
65        */
66       locals.load(monitor)
67       emit(asm.Opcodes.MONITOREXIT)
68       if (hasResult) { locals.load(monitorResult) }
69       val postHandler = new asm.Label
70       bc goTo postHandler
71 
72       /* ------ (4) exception-handler version of monitor-exit code.
73        *            Reached upon abrupt termination of (2).
74        *            Protected by whatever protects the whole synchronized expression.
75        *            null => "any" exception in bytecode, like we emit for finally.
76        *            Important not to use j/l/Throwable which dooms the method to a life of interpretation! (SD-233)
77        * ------
78        */
79       protect(startProtected, endProtected, currProgramPoint(), null)
80       locals.load(monitor)
81       emit(asm.Opcodes.MONITOREXIT)
82       emit(asm.Opcodes.ATHROW)
83 
84       /* ------ (5) cleanup version of monitor-exit code.
85        *            Reached upon early-return from (2).
86        *            Protected by whatever protects the whole synchronized expression.
87        * ------
88        */
89       if (shouldEmitCleanup) {
90         markProgramPoint(monCleanup)
91         locals.load(monitor)
92         emit(asm.Opcodes.MONITOREXIT)
93         pendingCleanups()
94       }
95 
96       /* ------ (6) normal exit of the synchronized expression.
97        *            Reached after normal, non-early-return, termination of (3).
98        *            Protected by whatever protects the whole synchronized expression.
99        * ------
100        */
101       mnode visitLabel postHandler
102 
103       lineNumber(tree)
104 
105       expectedType
106     }

118     /*
119      *  Emitting try-catch is easy, emitting try-catch-finally not quite so.
120      *  A finally-block (which always has type Unit, thus leaving the operand stack unchanged)
121      *  affects control-transfer from protected regions, as follows:
122      *
123      *    (a) `return` statement:
124      *
125      *        First, the value to return (if any) is evaluated.
126      *        Afterwards, all enclosing finally-blocks are run, from innermost to outermost.
127      *        Only then is the return value (if any) returned.
128      *
129      *        Some terminology:
130      *          (a.1) Executing a return statement that is protected
131      *                by one or more finally-blocks is called "early return"
132      *          (a.2) the chain of code sections (a code section for each enclosing finally-block)
133      *                to run upon early returns is called "cleanup chain"
134      *
135      *        As an additional spin, consider a return statement in a finally-block.
136      *        In this case, the value to return depends on how control arrived at that statement:
137      *        in case it arrived via a previous return, the previous return enjoys priority:
138      *        the value to return is given by that statement.
139      *
140      *    (b) A finally-block protects both the try-clause and the catch-clauses.
141      *
142      *           Sidenote:
143      *             A try-clause may contain an empty block. On CLR, a finally-block has special semantics
144      *             regarding Abort interruptions; but on the JVM it's safe to elide an exception-handler
145      *             that protects an "empty" range ("empty" as in "containing NOPs only",
146      *             see `asm.optimiz.DanglingExcHandlers` and SI-6720).
147      *
148      *        This means a finally-block indicates instructions that can be reached:
149      *          (b.1) Upon normal (non-early-returning) completion of the try-clause or a catch-clause
150      *                In this case, the next-program-point is that following the try-catch-finally expression.
151      *          (b.2) Upon early-return initiated in the try-clause or a catch-clause
152      *                In this case, the next-program-point is the enclosing cleanup section (if any), otherwise return.
153      *          (b.3) Upon abrupt termination (due to unhandled exception) of the try-clause or a catch-clause
154      *                In this case, the unhandled exception must be re-thrown after running the finally-block.
155      *
156      *    (c) finally-blocks are implicit to `synchronized` (a finally-block is added to just release the lock)
157      *        that's why `genSynchronized()` too emits cleanup-sections.
158      *
159      *  A number of code patterns can be emitted to realize the intended semantics.
160      *
161      *  A popular alternative (GenICode, javac) consists in duplicating the cleanup-chain at each early-return position.
162      *  The principle at work being that once control is transferred to a cleanup-section,
163      *  control will always stay within the cleanup-chain.
164      *  That is, barring an exception being thrown in a cleanup-section, in which case the enclosing try-block
165      *  (reached via abrupt termination) takes over.
166      *
167      *  The observations above hint at another code layout, less verbose, for the cleanup-chain.
168      *
169      *  The code layout that GenBCode emits takes into account that once a cleanup section has been reached,
170      *  jumping to the next cleanup-section (and so on, until the outermost one) realizes the correct semantics.
171      *
172      *  There is still code duplication in that two cleanup-chains are needed (but this is unavoidable, anyway):
173      *  one for normal control flow and another chain consisting of exception handlers.
174      *  The in-line comments below refer to them as
175      *    - "early-return-cleanups" and
176      *    - "exception-handler-version-of-finally-block" respectively.
177      *
178      */
179     def genLoadTry(tree: Try): BType = tree match {
180       case Try(block, catches, finalizer) =>
181       val kind = tpeTK(tree)
182 
183       val caseHandlers: List[EHClause] =
184         for (CaseDef(pat, _, caseBody) <- catches) yield {
185           pat match {
186             case Typed(Ident(`nme_WILDCARD`), tpt)  => NamelessEH(tpeTK(tpt).asClassBType, caseBody)
187             case Ident(`nme_WILDCARD`)              => NamelessEH(ThrowableReference,  caseBody)
188             case Bind(_, _)                       => BoundEH   (pat.symbol, caseBody)
189           }
190         }
191 
192       // ------ (0) locals used later ------
193 
194       /*
195        * `postHandlers` is a program point denoting:
196        *     (a) the finally-clause conceptually reached via fall-through from try-catch-finally
197        *         (in case a finally-block is present); or
198        *     (b) the program point right after the try-catch
199        *         (in case there's no finally-block).
200        * The name choice emphasizes that the code section lies "after all exception handlers",
201        * where "all exception handlers" includes those derived from catch-clauses as well as from finally-blocks.
202        */
203       val postHandlers = new asm.Label
204 
205       val hasFinally   = (finalizer != EmptyTree)
206 
207       /*
208        * used in the finally-clause reached via fall-through from try-catch, if any.
209        */
210       val guardResult  = hasFinally && (kind != UNIT) && mayCleanStack(finalizer)
211 
212       /*
213        * please notice `tmp` has type tree.tpe, while `earlyReturnVar` has the method return type.
214        * Because those two types can be different, dedicated vars are needed.
215        */
216       val tmp          = if (guardResult) locals.makeLocal(tpeTK(tree), "tmp", tree.tpe, tree.pos) else null;
217 
218       /*
219        * upon early return from the try-body or one of its EHs (but not the EH-version of the finally-clause)
220        * AND hasFinally, a cleanup is needed.
221        */
222       val finCleanup   = if (hasFinally) new asm.Label else null
223 
224       /* ------ (1) try-block, protected by:
225        *                       (1.a) the EHs due to case-clauses,   emitted in (2),
226        *                       (1.b) the EH  due to finally-clause, emitted in (3.A)
227        *                       (1.c) whatever protects the whole try-catch-finally expression.
228        * ------
229        */
230 
231       val startTryBody = currProgramPoint()
232       registerCleanup(finCleanup)
233       genLoad(block, kind)
234       unregisterCleanup(finCleanup)
235       nopIfNeeded(startTryBody)
236       val endTryBody = currProgramPoint()
237       bc goTo postHandlers
238 
239       /* ------ (2) One EH for each case-clause (this does not include the EH-version of the finally-clause)
240        *            An EH in (2) is reached upon abrupt termination of (1).
241        *            An EH in (2) is protected by:
242        *                         (2.a) the EH-version of the finally-clause, if any.
243        *                         (2.b) whatever protects the whole try-catch-finally expression.
244        * ------
245        */
246 
247       for (ch <- caseHandlers) {
248 
249         // (2.a) emit case clause proper
250         val startHandler = currProgramPoint()
251         var endHandler: asm.Label = null
252         var excType: ClassBType = null
253         registerCleanup(finCleanup)
254         ch match {
255           case NamelessEH(typeToDrop, caseBody) =>
256             bc drop typeToDrop
257             genLoad(caseBody, kind) // adapts caseBody to `kind`, thus it can be stored, if `guardResult`, in `tmp`.
258             nopIfNeeded(startHandler)
259             endHandler = currProgramPoint()
260             excType = typeToDrop
261 
262           case BoundEH   (patSymbol,  caseBody) =>
263             // test/files/run/contrib674.scala , a local-var already exists for patSymbol.
264             // rather than creating on first-access, we do it right away to emit debug-info for the created local var.
265             val Local(patTK, _, patIdx, _) = locals.getOrMakeLocal(patSymbol)
266             bc.store(patIdx, patTK)
267             genLoad(caseBody, kind)
268             nopIfNeeded(startHandler)
269             endHandler = currProgramPoint()
270             emitLocalVarScope(patSymbol, startHandler, endHandler)
271             excType = patTK.asClassBType
272         }
273         unregisterCleanup(finCleanup)
274         // (2.b)  mark the try-body as protected by this case clause.
275         protect(startTryBody, endTryBody, startHandler, excType)
276         // (2.c) emit jump to the program point where the finally-clause-for-normal-exit starts, or in effect `after` if no finally-clause was given.
277         bc goTo postHandlers
278 
279       }
280 
281       /* ------ (3.A) The exception-handler-version of the finally-clause.
282        *              Reached upon abrupt termination of (1) or one of the EHs in (2).
283        *              Protected only by whatever protects the whole try-catch-finally expression.
284        * ------
285        */
286 
287       // a note on terminology: this is not "postHandlers", despite appearences.
288       // "postHandlers" as in the source-code view. And from that perspective, both (3.A) and (3.B) are invisible implementation artifacts.
289       if (hasFinally) {
290         nopIfNeeded(startTryBody)
291         val finalHandler = currProgramPoint() // version of the finally-clause reached via unhandled exception.
292         protect(startTryBody, finalHandler, finalHandler, null)
293         val Local(eTK, _, eIdx, _) = locals(locals.makeLocal(ThrowableReference, "exc", Throwable_Type, finalizer.pos))
294         bc.store(eIdx, eTK)
295         emitFinalizer(finalizer, null, isDuplicate = true)
296         bc.load(eIdx, eTK)
297         emit(asm.Opcodes.ATHROW)
298       }
299 
300       /* ------ (3.B) Cleanup-version of the finally-clause.
301        *              Reached upon early RETURN from (1) or upon early RETURN from one of the EHs in (2)
302        *              (and only from there, ie reached only upon early RETURN from
303        *               program regions bracketed by registerCleanup/unregisterCleanup).
304        *              Protected only by whatever protects the whole try-catch-finally expression.
305        *
306        *              Given that control arrives to a cleanup section only upon early RETURN,
307        *              the value to return (if any) is always available. Therefore, a further RETURN
308        *              found in a cleanup section is always ignored (a warning is displayed, @see `genReturn()`).
309        *              In order for `genReturn()` to know whether the return statement is enclosed in a cleanup section,
310        *              the variable `insideCleanupBlock` is used.
311        * ------
312        */
313 
314       // this is not "postHandlers" either.
315       // `shouldEmitCleanup` can be set, and at the same time this try expression may lack a finally-clause.
316       // In other words, all combinations of (hasFinally, shouldEmitCleanup) are valid.
317       if (hasFinally && shouldEmitCleanup) {
318         val savedInsideCleanup = insideCleanupBlock
319         insideCleanupBlock = true
320         markProgramPoint(finCleanup)
321         // regarding return value, the protocol is: in place of a `return-stmt`, a sequence of `adapt, store, jump` are inserted.
322         emitFinalizer(finalizer, null, isDuplicate = true)
323         pendingCleanups()
324         insideCleanupBlock = savedInsideCleanup
325       }
326 
327       /* ------ (4) finally-clause-for-normal-nonEarlyReturn-exit
328        *            Reached upon normal, non-early-return termination of (1) or of an EH in (2).
329        *            Protected only by whatever protects the whole try-catch-finally expression.
330        * TODO explain what happens upon RETURN contained in (4)
331        * ------
332        */
333 
334       markProgramPoint(postHandlers)
335       if (hasFinally) {
336         emitFinalizer(finalizer, tmp, isDuplicate = false) // the only invocation of emitFinalizer with `isDuplicate == false`
337       }
338 
339       kind
340     } // end of genLoadTry()

342     /* if no more pending cleanups, all that remains to do is return. Otherwise jump to the next (outer) pending cleanup. */
343     private def pendingCleanups(): Unit = {
344       cleanups match {
345         case Nil =>
346           if (earlyReturnVar != null) {
347             locals.load(earlyReturnVar)
348             bc.emitRETURN(locals(earlyReturnVar).tk)
349           } else {
350             bc emitRETURN UNIT
351           }
352           shouldEmitCleanup = false
353 
354         case nextCleanup :: _ =>
355           bc goTo nextCleanup
356       }
357     }

359     def protect(start: asm.Label, end: asm.Label, handler: asm.Label, excType: ClassBType): Unit = {
360       val excInternalName: String =
361         if (excType == null) null
362         else excType.internalName
363       assert(start != end, "protecting a range of zero instructions leads to illegal class format. Solution: add a NOP to that range.")
364       mnode.visitTryCatchBlock(start, end, handler, excInternalName)
365     }

367     /* `tmp` (if non-null) is the symbol of the local-var used to preserve the result of the try-body, see `guardResult` */
368     def emitFinalizer(finalizer: Tree, tmp: Symbol, isDuplicate: Boolean): Unit = {
369       var saved: immutable.Map[ /* LabelDef */ Symbol, asm.Label ] = null
370       if (isDuplicate) {
371         saved = jumpDest
372         for(ldef <- labelDefsAtOrUnder(finalizer)) {
373           jumpDest -= ldef.symbol
374         }
375       }
376       // when duplicating, the above guarantees new asm.Labels are used for LabelDefs contained in the finalizer (their vars are reused, that's ok)
377       if (tmp != null) { locals.store(tmp) }
378       genLoad(finalizer, UNIT)
379       if (tmp != null) { locals.load(tmp)  }
380       if (isDuplicate) {
381         jumpDest = saved
382       }
383     }

388         case _ => false
