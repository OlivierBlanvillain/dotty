243   private def emitArgument(av:   AnnotationVisitor,
244                            name: String,
245                            arg:  Tree, bcodeStore: BCodeHelpers)(innerClasesStore: bcodeStore.BCInnerClassGen): Unit = {
246     (normalizeArgument(arg): @unchecked) match {
247       case Literal(const @ Constant(_)) =>
248         const.tag match {
249           case BooleanTag | ByteTag | ShortTag | CharTag | IntTag | LongTag | FloatTag | DoubleTag => av.visit(name, const.value)
250           case StringTag =>
251             assert(const.value != null, const) // TODO this invariant isn't documented in `case class Constant`
252             av.visit(name, const.stringValue) // `stringValue` special-cases null, but that execution path isn't exercised for a const with StringTag
253           case ClazzTag => av.visit(name, const.typeValue.toTypeKind(bcodeStore)(innerClasesStore).toASMType)
254           case EnumTag =>
255             val edesc = innerClasesStore.typeDescriptor(const.tpe.asInstanceOf[bcodeStore.int.Type]) // the class descriptor of the enumeration class.
256             val evalue = const.symbolValue.name.mangledString // value the actual enumeration value.
257             av.visitEnum(name, edesc, evalue)
258         }
259       case t: TypeApply if (t.fun.symbol == Predef_classOf) =>
260         av.visit(name, t.args.head.tpe.classSymbol.denot.info.toTypeKind(bcodeStore)(innerClasesStore).toASMType)
261       case t: tpd.Select =>
262         if (t.symbol.denot.owner.is(Flags.Enum)) {
263           val edesc = innerClasesStore.typeDescriptor(t.tpe.asInstanceOf[bcodeStore.int.Type]) // the class descriptor of the enumeration class.
264           val evalue = t.symbol.name.mangledString // value the actual enumeration value.
265           av.visitEnum(name, edesc, evalue)
266         } else {
267             // println(i"not an enum: ${t.symbol} / ${t.symbol.denot.owner} / ${t.symbol.denot.owner.isTerm} / ${t.symbol.denot.owner.flags}")
268             assert(toDenot(t.symbol).name.is(DefaultGetterName),
269               s"${toDenot(t.symbol).name.debugString}") // this should be default getter. do not emmit.
270         }
271       case t: SeqLiteral =>
272         val arrAnnotV: AnnotationVisitor = av.visitArray(name)
273         for (arg <- t.elems) { emitArgument(arrAnnotV, null, arg, bcodeStore)(innerClasesStore) }
274         arrAnnotV.visitEnd()
275 
276       case Apply(fun, args) if fun.symbol == defn.ArrayClass.primaryConstructor ||
277         toDenot(fun.symbol).owner == defn.ArrayClass.linkedClass && fun.symbol.name == nme_apply =>
278         val arrAnnotV: AnnotationVisitor = av.visitArray(name)
279 
280         var actualArgs = if (fun.tpe.isImplicitMethod) {
281           // generic array method, need to get implicit argument out of the way
282           fun.asInstanceOf[Apply].args
283         } else args
284 
285         val flatArgs = actualArgs.flatMap { arg =>
286           normalizeArgument(arg) match {
287             case t: tpd.SeqLiteral => t.elems
288             case e => List(e)
289           }
290         }
291         for(arg <- flatArgs) {
292           emitArgument(arrAnnotV, null, arg, bcodeStore)(innerClasesStore)
293         }
294         arrAnnotV.visitEnd()
295 /*
296       case sb @ ScalaSigBytes(bytes) =>
297         // see http://www.scala-lang.org/sid/10 (Storage of pickled Scala signatures in class files)
298         // also JVMS Sec. 4.7.16.1 The element_value structure and JVMS Sec. 4.4.7 The CONSTANT_Utf8_info Structure.
299         if (sb.fitsInOneString) {
300           av.visit(name, BCodeAsmCommon.strEncode(sb))
301         } else {
302           val arrAnnotV: asm.AnnotationVisitor = av.visitArray(name)
303           for(arg <- BCodeAsmCommon.arrEncode(sb)) { arrAnnotV.visit(name, arg) }
304           arrAnnotV.visitEnd()
305         }          // for the lazy val in ScalaSigBytes to be GC'ed, the invoker of emitAnnotations() should hold the ScalaSigBytes in a method-local var that doesn't escape.
306 */
307       case t @ Apply(constr, args) if t.tpe.derivesFrom(JavaAnnotationClass) =>
308         val typ = t.tpe.classSymbol.denot.info
309         val assocs = assocsFromApply(t)
310         val desc = innerClasesStore.typeDescriptor(typ.asInstanceOf[bcodeStore.int.Type]) // the class descriptor of the nested annotation class
311         val nestedVisitor = av.visitAnnotation(name, desc)
312         emitAssocs(nestedVisitor, assocs, bcodeStore)(innerClasesStore)
313     }
314   }

551     // scala/bug#3452 Static forwarder generation uses the same erased signature as the method if forwards to.
552     // By rights, it should use the signature as-seen-from the module class, and add suitable
553     // primitive and value-class boxing/unboxing.
554     // But for now, just like we did in mixin, we just avoid writing a wrong generic signature

559     if (erasedMemberType =:= sym.denot.info)

564   private def getGenericSignature(sym: Symbol, owner: Symbol, memberTpe: Type)(implicit ctx: Context): Option[String] =
565     if (needsGenericSignature(sym)) {
566       val erasedTypeSym = sym.denot.info.typeSymbol
567       if (erasedTypeSym.isPrimitiveValueClass) {
568         None
569       } else {
570         val jsOpt = GenericSignatures.javaSig(sym, memberTpe)
571         if (ctx.settings.XverifySignatures.value) {
572           jsOpt.foreach(verifySignature(sym, _))
573         }
574 
575         jsOpt
576       }
577     } else {
578       None
