31   /*
32    * There's a dedicated PlainClassBuilder for each CompilationUnit,
33    * which simplifies the initialization of per-class data structures in `genPlainClass()` which in turn delegates to `initJClass()`
34    *
35    * The entry-point to emitting bytecode instructions is `genDefDef()` where the per-method data structures are initialized,
36    * including `resetMethodBookkeeping()` and `initJMethod()`.
37    * Once that's been done, and assuming the method being visited isn't abstract, `emitNormalMethodBody()` populates
38    * the ASM MethodNode instance with ASM AbstractInsnNodes.
39    *
40    * Given that CleanUp delivers trees that produce values on the stack,
41    * the entry-point to all-things instruction-emit is `genLoad()`.
42    * There, an operation taking N arguments results in recursively emitting instructions to lead each of them,
43    * followed by emitting instructions to process those arguments (to be found at run-time on the operand-stack).
44    *
45    * In a few cases the above recipe deserves more details, as provided in the documentation for:
46    *   - `genLoadTry()`
47    *   - `genSynchronized()
48    *   - `jumpDest` , `cleanups` , `labelDefsAtOrUnder`
49    */

131 
132       val ps = claszSymbol.info.parents
133       val superClass: String = if (ps.isEmpty) ObjectReference.internalName else internalName(ps.head.typeSymbol)
134       val interfaceNames = classBTypeFromSymbol(claszSymbol).info.interfaces map {
135         case classBType =>
136           if (classBType.isNestedClass) { innerClassBufferASM += classBType }
137           classBType.internalName
138       }

140       val flags = javaFlags(claszSymbol)
141 
142       val thisSignature = getGenericSignature(claszSymbol, claszSymbol.owner)
143       cnode.visit(classfileVersion, flags,
144                   thisName, thisSignature,
145                   superClass, interfaceNames.toArray)

147       if (emitSource) {
148         cnode.visitSource(sourceFileFor(cunit), null /* SourceDebugExtension */)
149       }
150

152         case Some(EnclosingMethodEntry(className, methodName, methodDescriptor)) =>
153           cnode.visitOuterClass(className, methodName, methodDescriptor)
154         case _ => ()
155       }
156 
157       val ssa = getAnnotPickle(thisName, claszSymbol)
158       cnode.visitAttribute(if (ssa.isDefined) pickleMarkerLocal else pickleMarkerForeign)
159       emitAnnotations(cnode, claszSymbol.annotations ++ ssa)
160 
161       if (isCZStaticModule || isCZParcelable) {
162 
163         if (isCZStaticModule) { addModuleInstanceField() }
164 
165       } else {
166 
167         val skipStaticForwarders = (claszSymbol.isInterface || claszSymbol.isModule || noForwarders)

176               addForwarders(cnode, thisName, lmoc.moduleClass)
177             }
178           }
179         }
180 
181       }
182 
183       // the invoker is responsible for adding a class-static constructor.

275     /*
276      *  A jump is represented as an Apply node whose symbol denotes a LabelDef, the target of the jump.
277      *  The `jumpDest` map is used to:
278      *    (a) find the asm.Label for the target, given an Apply node's symbol;
279      *    (b) anchor an asm.Label in the instruction stream, given a LabelDef node.
280      *  In other words, (a) is necessary when visiting a jump-source, and (b) when visiting a jump-target.
281      *  A related map is `labelDef`: it has the same keys as `jumpDest` but its values are LabelDef nodes not asm.Labels.
282      *
283      */

288         val pp = new asm.Label

294     /*
295      *  A program point may be lexically nested (at some depth)
296      *    (a) in the try-clause of a try-with-finally expression
297      *    (b) in a synchronized block.
298      *  Each of the constructs above establishes a "cleanup block" to execute upon
299      *  both normal-exit, early-return, and abrupt-termination of the instructions it encloses.
300      *
301      *  The `cleanups` LIFO queue represents the nesting of active (for the current program point)
302      *  pending cleanups. For each such cleanup an asm.Label indicates the start of its cleanup-block.
303      *  At any given time during traversal of the method body,
304      *  the head of `cleanups` denotes the cleanup-block for the closest enclosing try-with-finally or synchronized-expression.
305      *
306      *  `cleanups` is used:
307      *
308      *    (1) upon visiting a Return statement.
309      *        In case of pending cleanups, we can't just emit a RETURN instruction, but must instead:
310      *          - store the result (if any) in `earlyReturnVar`, and
311      *          - jump to the next pending cleanup.
312      *        See `genReturn()`
313      *
314      *    (2) upon emitting a try-with-finally or a synchronized-expr,
315      *        In these cases, the targets of the above jumps are emitted,
316      *        provided an early exit was actually encountered somewhere in the protected clauses.
317      *        See `genLoadTry()` and `genSynchronized()`
318      *
319      *  The code thus emitted for jumps and targets covers the early-return case.
320      *  The case of abrupt (ie exceptional) termination is covered by exception handlers
321      *  emitted for that purpose as described in `genLoadTry()` and `genSynchronized()`.
322      */
323     var cleanups: List[asm.Label] = Nil
324     def registerCleanup(finCleanup: asm.Label): Unit = {
325       if (finCleanup != null) { cleanups = finCleanup :: cleanups }
326     }
327     def unregisterCleanup(finCleanup: asm.Label): Unit = {
328       if (finCleanup != null) {
329         assert(cleanups.head eq finCleanup,
330                s"Bad nesting of cleanup operations: $cleanups trying to unregister: $finCleanup")
331         cleanups = cleanups.tail
332       }
333     }

409     /*
410      *  The semantics of try-with-finally and synchronized-expr require their cleanup code
411      *  to be present in three forms in the emitted bytecode:
412      *    (a) as normal-exit code, reached via fall-through from the last program point being protected,
413      *    (b) as code reached upon early-return from an enclosed return statement.
414      *        The only difference between (a) and (b) is their next program-point:
415      *          the former must continue with fall-through while
416      *          the latter must continue to the next early-return cleanup (if any, otherwise return from the method).
417      *        Otherwise they are identical.
418      *    (c) as exception-handler, reached via exceptional control flow,
419      *        which rethrows the caught exception once it's done with the cleanup code.
420      *
421      *  A particular cleanup may in general contain LabelDefs. Care is needed when duplicating such jump-targets,
422      *  so as to preserve agreement wit the (also duplicated) jump-sources.
423      *  This is achieved based on the bookkeeping provided by two maps:
424      *    - `labelDefsAtOrUnder` lists all LabelDefs enclosed by a given Tree node (the key)
425      *    - `labelDef` provides the LabelDef node whose symbol is used as key.
426      *       As a sidenote, a related map is `jumpDest`: it has the same keys as `labelDef` but its values are asm.Labels not LabelDef nodes.
427      *
428      *  Details in `emitFinalizer()`, which is invoked from `genLoadTry()` and `genSynchronized()`.
429      */

544     def genDefDef(dd: DefDef): Unit = dd match {
545       case DefDef(_, _, _, vparamss, _, rhs) =>
546       // the only method whose implementation is not emitted: getClass()
547       if (dd.symbol.isGetClass) { return }
548       assert(mnode == null, "GenBCode detected nested method.")
549 
550       methSymbol  = dd.symbol
551       jMethodName = methSymbol.javaSimpleName.toString
552       returnType  = asmMethodType(dd.symbol).returnType
553       isMethSymStaticCtor = methSymbol.isStaticConstructor
554 
555       resetMethodBookkeeping(dd)
556 
557       // add method-local vars for params
558 
559       assert(vparamss.isEmpty || vparamss.tail.isEmpty, s"Malformed parameter list: $vparamss")
560       val params = if (vparamss.isEmpty) Nil else vparamss.head
561       for (p <- params) { locals.makeLocal(p.symbol) }
562       // debug assert((params.map(p => locals(p.symbol).tk)) == asmMethodType(methSymbol).getArgumentTypes.toList, "debug")
563 
564       if (params.size > MaximumJvmParameters) {
565         // SI-7324
566         error(methSymbol.pos, s"Platform restriction: a parameter list's length cannot exceed $MaximumJvmParameters.")
567         return
568       }
569 
570       val isNative         = methSymbol.hasAnnotation(NativeAttr)
571       val isAbstractMethod = (methSymbol.isDeferred || (methSymbol.owner.isInterface && !methSymbol.isJavaDefaultMethod))
572       val flags = GenBCodeOps.mkFlags(
573         javaFlags(methSymbol),
574         if (isAbstractMethod)        asm.Opcodes.ACC_ABSTRACT   else 0,
575         if (methSymbol.isStrictFP)   asm.Opcodes.ACC_STRICT     else 0,
576         if (isNative)                asm.Opcodes.ACC_NATIVE     else 0  // native methods of objects are generated in mirror classes
577       )
578 
579       // TODO needed? for(ann <- m.symbol.annotations) { ann.symbol.initialize }
580       initJMethod(flags, params.map(p => p.symbol.annotations))
581 
582       /* Add method-local vars for LabelDef-params.
583        *
584        * This makes sure that:
585        *   (1) upon visiting any "forward-jumping" Apply (ie visited before its target LabelDef), and after
586        *   (2) grabbing the corresponding param symbols,
587        * those param-symbols can be used to access method-local vars.
588        *
589        * When duplicating a finally-contained LabelDef, another program-point is needed for the copy (each such copy has its own asm.Label),
590        * but the same vars (given by the LabelDef's params) can be reused,
591        * because no LabelDef ends up nested within itself after such duplication.
592        */
593       for(ld <- labelDefsAtOrUnder(rhs); LabelDef(_, ldpl ,_) = ld; ldp <- ldpl; if !locals.contains(ldp)) {
594         // the tail-calls xform results in symbols shared btw method-params and labelDef-params, thus the guard above.
