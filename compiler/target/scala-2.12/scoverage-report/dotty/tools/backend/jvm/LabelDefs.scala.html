11 /**
12  * Verifies that each Label DefDef has only a single address to jump back and
13  * reorders them such that they are not nested and this address is a
14  * fall-through address for the JVM.
15  *
16  * ```scala
17  * <label> def foo(i: Int) = {
18  *   <label> def bar = 0
19  *   <label> def dough(i: Int) = if (i == 0) bar else foo(i-1)
20  *   dough(i)
21  * }
22  *
23  * foo(100)
24  * ```
25  *
26  * will get rewritten to:
27  *
28  * ```scala
29  * <label> def foo(i: Int) = dough(i)
30  * <label> def dough(i: Int) = if (i == 0) bar else foo(i-1)
31  * <label> def bar = 2
32  *   foo(100)
33  * ```
34  *
35  * Proposed way to generate this pattern in backend is:
36  *
37  * ```scala
38  * foo(100)
39  * <jump foo>
40  * <label> def foo(i: Int) = dough(i)
41  * // <jump a>                           // unreachable
42  * <label> def dough(i: Int) = if (i == 0) bar else foo(i-1)
43  * // <jump a>                           // unreachable
44  * <label> def bar = 2
45  * // <jump a>                           // unreachable
46  * <asm point a>
47  * ```
48  *
49  * Unreachable jumps will be eliminated by local dead code analysis.
50  * After JVM is smart enough to remove next-line jumps
51  *
52  * Note that his phase Ychecking this phase required softening scoping rules
53  * as it intentionally allowed to break scoping rules inside methods for labels.
54  * This is modified by setting `labelsReordered` flag in Phases.
55  *
56  * @author Dmitry Petrashko
57  */

63   override def transformDefDef(tree: DefDef)(implicit ctx: Context): Tree = {
64     if (tree.symbol is Label) tree
65     else {
66       val labelDefs = collectLabelDefs(tree.rhs)
67 
68       def putLabelDefsNearCallees = new TreeMap() {
69         override def transform(tree: Tree)(implicit ctx: Context): Tree = {
70           tree match {
71             case t: Apply if labelDefs.contains(t.symbol) =>
72               val labelDef = labelDefs(t.symbol)
73               labelDefs -= t.symbol
74               val labelDef2 = cpy.DefDef(labelDef)(rhs = transform(labelDef.rhs))
75               Block(labelDef2:: Nil, t)
76             case t: DefDef =>
77               assert(t.symbol is Label)
78               EmptyTree
79             case _ => if (!labelDefs.isEmpty) super.transform(tree) else tree
80           }
81         }
82       }
83 
84       if (labelDefs.isEmpty) tree
85       else cpy.DefDef(tree)(rhs = putLabelDefsNearCallees.transform(tree.rhs))
86     }
87   }
88 
89   private def collectLabelDefs(tree: Tree)(implicit ctx: Context): MutableSymbolMap[DefDef] = {
90     // labelSymbol -> Defining tree
91     val labelDefs = newMutableSymbolMap[DefDef]
92     new TreeTraverser {
93       override def traverse(tree: Tree)(implicit ctx: Context): Unit = tree match {
94         case t: DefDef =>
95           assert(t.symbol is Label)
96           labelDefs(t.symbol) = t
97           traverseChildren(t)
98         case _ => traverseChildren(tree)
99       }
100     }.traverse(tree)
101     labelDefs
102   }
