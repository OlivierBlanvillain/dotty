38 class CollectEntryPoints extends MiniPhase {
39   def phaseName: String = "Collect entry points"
40 
41   override def transformDefDef(tree: tpd.DefDef)(implicit ctx: Context): tpd.Tree = {
42     if ((tree.symbol ne NoSymbol) && CollectEntryPoints.isJavaEntryPoint(tree.symbol)) {
43       ctx.genBCodePhase.asInstanceOf[GenBCode].registerEntryPoint(tree.symbol)
44     }
45     tree
46   }
47 }

49 object CollectEntryPoints{
50   def isJavaMainMethod(sym: Symbol)(implicit ctx: Context) = {
51     (sym.name == nme.main) && (sym.info match {
52       case r@MethodTpe(_, List(defn.ArrayOf(t)), _) =>
53         (t.widenDealias =:= defn.StringType) && (
54         r.resultType.widenDealias =:= defn.UnitType)
55       case _ => false
56     })
57   }
58 
59   def isJavaEntryPoint(sym: Symbol)(implicit ctx: Context): Boolean = {
60     import Types.MethodType
61     val d = ctx.definitions
62     val StringType = d.StringType
63     // The given class has a main method.
64     def hasJavaMainMethod(sym: Symbol): Boolean =
65       (toDenot(sym).info member nme.main).alternatives exists(x => isJavaMainMethod(x.symbol))
66 
67     def fail(msg: String, pos: Position = sym.pos) = {
68       ctx.warning(          sym.name +
69         s" has a main method with parameter type Array[String], but ${toDenot(sym).fullName} will not be a runnable program.\n  Reason: $msg",
70         sourcePos(sym.pos)
71         // TODO: make this next claim true, if possible
72         //   by generating valid main methods as static in module classes
73         //   not sure what the jvm allows here
74         // + "  You can still run the program by calling it as " + javaName(sym) + " instead."
75       )
76       false
77     }
78     def failNoForwarder(msg: String) = {
79       fail(s"$msg, which means no static forwarder can be generated.\n")
80     }
81     val possibles = if (sym.flags is Flags.Module) (toDenot(sym).info nonPrivateMember nme.main).alternatives else Nil
82     val hasApproximate = possibles exists { m =>
83       m.info match {
84         case MethodTpe(_, p :: Nil, _) => p.typeSymbol == defn.ArrayClass
85         case _                         => false
86       }
87     }
88     // At this point it's a module with a main-looking method, so either succeed or warn that it isn't.
89     hasApproximate && {
90       // Before erasure so we can identify generic mains.
91       {
92         // implicit val c = ctx.withPhase(ctx.erasurePhase)
93 
94         val companion     = sym.asClass.moduleClass
95 
96         if (hasJavaMainMethod(companion))
97           failNoForwarder("companion contains its own main method")
98         else if (toDenot(companion).info.member(nme.main) != NoDenotation)
99         // this is only because forwarders aren't smart enough yet
100           failNoForwarder("companion contains its own main method (implementation restriction: no main is allowed, regardless of signature)")
101         else if (companion.flags is Flags.Trait)
102           failNoForwarder("companion is a trait")
103         // Now either succeed, or issue some additional warnings for things which look like
104         // attempts to be java main methods.
105         else (possibles exists(x=> isJavaMainMethod(x.symbol))) || {
106           possibles exists { m =>
107             toDenot(m.symbol).info match {
108               case t: PolyType =>
109                 fail("main methods cannot be generic.")
110               case MethodTpe(paramNames, paramTypes, resultType) =>
111                 if (resultType :: paramTypes exists (_.typeSymbol.isAbstractType))
112                   fail("main methods cannot refer to type parameters or abstract types.", m.symbol.pos)
113                 else
114                   isJavaMainMethod(m.symbol) || fail("main method must have exact signature (Array[String])Unit", m.symbol.pos)
115               case tp =>
116                 fail(s"don't know what this is: $tp", m.symbol.pos)
117             }
118           }
119         }
120       }
121     }
122   }
