10 /** Collect all super calls to trait members.
11  *
12  *  For each super reference to trait member, register a call from the current class to the
13  *  owner of the referenced member.
14  *
15  *  This information is used to know if it is safe to remove a redundant mixin class.
16  *  A redundant mixin class is one that is implemented by another mixin class. As the
17  *  methods in a redundant mixin class could be implemented with a default abstract method,
18  *  the redundant mixin class could be required as a parent by the JVM.
19  */
20 class CollectSuperCalls extends MiniPhase {
21   import tpd._
22 
23   def phaseName: String = "collectSuperCalls"
24 
25   override def transformSelect(tree: Select)(implicit ctx: Context): Tree = {
26     tree.qualifier match {
27       case sup: Super =>
28         if (tree.symbol.owner.is(Trait))
29           registerSuperCall(ctx.owner.enclosingClass.asClass, tree.symbol.owner.asClass)
30       case _ =>
31     }
32     tree
33   }
34 
35   private def registerSuperCall(sym: ClassSymbol, calls: ClassSymbol)(implicit ctx: Context) = {
36     ctx.genBCodePhase match {
37       case genBCodePhase: GenBCode =>
38         genBCodePhase.registerSuperCall(sym, calls)
39       case _ =>
40     }
41   }
