112    * `refedInnerClasses` should contain those inner classes defined as direct member classes of `jclass`

200     /**
201      * The class internal name for a given class symbol. If the symbol describes a nested class, the
202      * ClassBType is added to the innerClassBufferASM.
203      */
204     final def internalName(sym: Symbol): String = {
205       // For each java class, the scala compiler creates a class and a module (thus a module class).
206       // If the `sym` is a java module class, we use the java class instead. This ensures that we
207       // register the class (instead of the module class) in innerClassBufferASM.
208       // The two symbols have the same name, so the resulting internalName is the same.
209       val classSym = if (sym.isJavaDefined && sym.isModuleClass) sym.linkedClassOfClass else sym
210       getClassBTypeAndRegisterInnerClass(classSym).internalName
211     }

223     /**
224      * The ClassBType for a class symbol. If the class is nested, the ClassBType is added to the
225      * innerClassBufferASM.
226      *
227      * The class symbol scala.Nothing is mapped to the class scala.runtime.Nothing$. Similarly,
228      * scala.Null is mapped to scala.runtime.Null$. This is because there exist no class files
229      * for the Nothing / Null. If used for example as a parameter type, we use the runtime classes
230      * in the classfile method signature.
231      *
232      * Note that the referenced class symbol may be an implementation class. For example when
233      * compiling a mixed-in method that forwards to the static method in the implementation class,
234      * the class descriptor of the receiver (the implementation class) is obtained by creating the
235      * ClassBType.

242       else {
243        val r = classBTypeFromSymbol(sym)
244         if (r.isNestedClass) innerClassBufferASM += r

246       }

320     /* Add a forwarder for method m. Used only from addForwarders().
321      *
322      * must-single-thread
323      */
324     private def addForwarder(jclass: asm.ClassVisitor, module: Symbol, m: Symbol): Unit = {
325       val moduleName     = internalName(module)
326       val methodInfo     = module.thisType.memberInfo(m)
327       val paramJavaTypes: List[BType] = methodInfo.paramTypes map toTypeKind
328       // val paramNames     = 0 until paramJavaTypes.length map ("x_" + _)
329 
330       /* Forwarders must not be marked final,
331        *  as the JVM will not allow redefinition of a final static method,
332        *  and we don't know what classes might be subclassing the companion class.  See SI-4827.
333        */
334       // TODO: evaluate the other flags we might be dropping on the floor here.
335       // TODO: ACC_SYNTHETIC ?
336       val flags = GenBCodeOps.PublicStatic | (
337         if (m.isVarargsMethod) asm.Opcodes.ACC_VARARGS else 0
338       )
339 
340       // TODO needed? for(ann <- m.annotations) { ann.symbol.initialize }
341       val jgensig = getStaticForwarderGenericSignature(m, module)
342       val (throws, others) = m.annotations partition (_.symbol == ThrowsClass)
343       val thrownExceptions: List[String] = getExceptions(throws)
344 
345       val jReturnType = toTypeKind(methodInfo.resultType)
346       val mdesc = MethodBType(paramJavaTypes, jReturnType).descriptor
347       val mirrorMethodName = m.javaSimpleName.toString
348       val mirrorMethod: asm.MethodVisitor = jclass.visitMethod(
349         flags,
350         mirrorMethodName,
351         mdesc,
352         jgensig,
353         mkArrayS(thrownExceptions)
354       )
355 
356       emitAnnotations(mirrorMethod, others)
357       emitParamAnnotations(mirrorMethod, m.info.params.map(_.annotations))
358 
359       mirrorMethod.visitCode()
360 
361       mirrorMethod.visitFieldInsn(asm.Opcodes.GETSTATIC, moduleName, MODULE_INSTANCE_FIELD, symDescriptor(module))
362 
363       var index = 0
364       for(jparamType <- paramJavaTypes) {
365         mirrorMethod.visitVarInsn(jparamType.typedOpcode(asm.Opcodes.ILOAD), index)
366         assert(!jparamType.isInstanceOf[MethodBType], jparamType)
367         index += jparamType.size
368       }
369 
370       mirrorMethod.visitMethodInsn(asm.Opcodes.INVOKEVIRTUAL, moduleName, mirrorMethodName, asmMethodType(m).descriptor, false)
371       mirrorMethod.visitInsn(jReturnType.typedOpcode(asm.Opcodes.IRETURN))
372 
373       mirrorMethod.visitMaxs(0, 0) // just to follow protocol, dummy arguments
374       mirrorMethod.visitEnd()
375 
376     }

378     /* Add forwarders for all methods defined in `module` that don't conflict
379      *  with methods in the companion class of `module`. A conflict arises when
380      *  a method with the same name is defined both in a class and its companion object:
381      *  method signature is not taken into account.
382      *
383      * must-single-thread
384      */
385     def addForwarders(jclass: asm.ClassVisitor, jclassName: String, moduleClass: Symbol): Unit = {
386       assert(moduleClass.isModuleClass, moduleClass)
387       debuglog(s"Dumping mirror class for object: $moduleClass")
388 
389       val linkedClass  = moduleClass.companionClass
390       lazy val conflictingNames: Set[Name] = {
391         // Dotty deviation: needed to add ": Symbol" because of https://github.com/lampepfl/dotty/issues/2143
392         (linkedClass.info.members collect { case sym if sym.name.isTermName => sym.name }).toSet
393       }
394       debuglog(s"Potentially conflicting names for forwarders: $conflictingNames")
395 
396       for (m <- moduleClass.info.membersBasedOnFlags(ExcludedForwarderFlags, Flag_METHOD)) {
397         if (m.isType || m.isDeferred || (m.owner eq ObjectClass) || m.isConstructor)
398           debuglog(s"No forwarder for '$m' from $jclassName to '$moduleClass'")
399         else if (conflictingNames(m.name))
400           log(s"No forwarder for $m due to conflict with ${linkedClass.info.member(m.name)}")
401         else if (m.hasAccessBoundary)
402           log(s"No forwarder for non-public member $m")
403         else {
404           log(s"Adding static forwarder for '$m' from $jclassName to '$moduleClass'")
405           addForwarder(jclass, moduleClass, m)
406         }
407       }
408     }

412      *   "The Exceptions attribute indicates which checked exceptions a method may throw.
413      *    There may be at most one Exceptions attribute in each method_info structure."

464     /* Generate a mirror class for a top-level module. A mirror class is a class
465      *  containing only static methods that forward to the corresponding method
466      *  on the MODULE instance of the given Scala object.  It will only be
467      *  generated if there is no companion class: if there is, an attempt will
468      *  instead be made to add the forwarder methods to the companion class.
469      *
470      *  must-single-thread
471      */
472     def genMirrorClass(moduleClass: Symbol, cunit: CompilationUnit): asm.tree.ClassNode = {
473       assert(moduleClass.isModuleClass)
474       assert(moduleClass.companionClass == NoSymbol, moduleClass)
475       innerClassBufferASM.clear()
476       this.cunit = cunit
477       val moduleName = internalName(moduleClass) // + "$"
478       val mirrorName = moduleName.substring(0, moduleName.length() - 1)
479 
480       val flags = (asm.Opcodes.ACC_SUPER | asm.Opcodes.ACC_PUBLIC | asm.Opcodes.ACC_FINAL)
481       val mirrorClass = new asm.tree.ClassNode
482       mirrorClass.visit(
483         classfileVersion,
484         flags,
485         mirrorName,
486         null /* no java-generic-signature */,
487         ObjectReference.internalName,
488         EMPTY_STRING_ARRAY
489       )
490 
491       if (emitSource) {
492         mirrorClass.visitSource("" + sourceFileFor(cunit),
493                                 null /* SourceDebugExtension */)
494       }
495 
496       val ssa = getAnnotPickle(mirrorName, moduleClass.companionSymbol)
497       mirrorClass.visitAttribute(if (ssa.isDefined) pickleMarkerLocal else pickleMarkerForeign)
498       emitAnnotations(mirrorClass, moduleClass.annotations ++ ssa)
499 
500       addForwarders(mirrorClass, mirrorName, moduleClass)
501 
502       innerClassBufferASM ++= classBTypeFromSymbol(moduleClass).info.memberClasses
503       addInnerClassesASM(mirrorClass, innerClassBufferASM.toList)
504 
505       mirrorClass.visitEnd()
506 
507       moduleClass.name // this side-effect is necessary, really.
508 
509       mirrorClass
510     }
