11 /**
12  * This class mainly contains the method classBTypeFromSymbol, which extracts the necessary
13  * information from a symbol and its type to create the correpsonding ClassBType. It requires
14  * access to the compiler (global parameter).
15  *
16  * The mixin CoreBTypes defines core BTypes that are used in the backend. Building these BTypes
17  * uses classBTypeFromSymbol, hence requires access to the compiler (global).
18  *
19  * BTypesFromSymbols extends BTypes because the implementation of BTypes requires access to some
20  * of the core btypes. They are declared in BTypes as abstract members. Note that BTypes does
21  * not have access to the compiler instance.
22  */

49   final def classBTypeFromSymbol(classSym: Symbol): ClassBType = {
50     assert(classSym != NoSymbol, "Cannot create ClassBType from NoSymbol")
51     assert(classSym.isClass, s"Cannot create ClassBType from non-class symbol $classSym")
52     assert(
53       (!primitiveTypeMap.contains(classSym) || isCompilingPrimitive) &&
54       (classSym != NothingClass && classSym != NullClass),
55       s"Cannot create ClassBType for special class symbol ${classSym.fullName}")
56 
57     convertedClasses.getOrElse(classSym, {
58       val internalName = classSym.javaBinaryName
59       // We first create and add the ClassBType to the hash map before computing its info. This
60       // allows initializing cylic dependencies, see the comment on variable ClassBType._info.
61       val classBType = new ClassBType(internalName)
62       convertedClasses(classSym) = classBType
63       setClassInfo(classSym, classBType)
64     })
65   }

67   private def setClassInfo(classSym: Symbol, classBType: ClassBType): ClassBType = {
68     val superClassSym = if (classSym.isImplClass) ObjectClass else classSym.superClass
69     assert(
70       if (classSym == ObjectClass)
71         superClassSym == NoSymbol
72       else if (classSym.isInterface)
73         superClassSym == ObjectClass
74       else
75         // A ClassBType for a primitive class (scala.Boolean et al) is only created when compiling these classes.
76         ((superClassSym != NoSymbol) && !superClassSym.isInterface) || (isCompilingPrimitive && primitiveTypeMap.contains(classSym)),
77       s"Bad superClass for $classSym: $superClassSym"
78     )
79     val superClass = if (superClassSym == NoSymbol) None
80                      else Some(classBTypeFromSymbol(superClassSym))
81 
82     val interfaces = classSym.superInterfaces.map(classBTypeFromSymbol)
83 
84     val flags = javaFlags(classSym)
85 
86     /* The InnerClass table of a class C must contain all nested classes of C, even if they are only
87      * declared but not otherwise referenced in C (from the bytecode or a method / field signature).
88      * We collect them here.
89      *
90      * Nested classes that are also referenced in C will be added to the innerClassBufferASM during
91      * code generation, but those duplicates will be eliminated when emitting the InnerClass
92      * attribute.
93      *
94      * Why doe we need to collect classes into innerClassBufferASM at all? To collect references to
95      * nested classes, but NOT nested in C, that are used within C.
96      */
97     val nestedClassSymbols = {
98       // The lambdalift phase lifts all nested classes to the enclosing class, so if we collect
99       // member classes right after lambdalift, we obtain all nested classes, including local and
100       // anonymous ones.
101       val nestedClasses = classSym.nestedClasses
102 
103       // If this is a top-level class, and it has a companion object, the member classes of the
104       // companion are added as members of the class. For example:
105       //   class C { }
106       //   object C {
107       //     class D
108       //     def f = { class E }
109       //   }
110       // The class D is added as a member of class C. The reason is that the InnerClass attribute
111       // for D will containt class "C" and NOT the module class "C$" as the outer class of D.
112       // This is done by buildNestedInfo, the reason is Java compatibility, see comment in BTypes.
113       // For consistency, the InnerClass entry for D needs to be present in C - to Java it looks
114       // like D is a member of C, not C$.
115       val linkedClass = classSym.linkedClass
116       val companionModuleMembers = classSym.companionModuleMembers
117 
118       nestedClasses ++ companionModuleMembers
119     }
120 
121     /**
122      * For nested java classes, the scala compiler creates both a class and a module (and therefore
123      * a module class) symbol. For example, in `class A { class B {} }`, the nestedClassSymbols
124      * for A contain both the class B and the module class B.
125      * Here we get rid of the module class B, making sure that the class B is present.
126      */
127     val nestedClassSymbolsNoJavaModuleClasses = nestedClassSymbols.filter(s => {
128       if (s.isJavaDefined && s.isModuleClass) {
129         // We could also search in nestedClassSymbols for s.linkedClassOfClass, but sometimes that
130         // returns NoSymbol, so it doesn't work.
131         val nb = nestedClassSymbols.count(mc => mc.name == s.name && mc.owner == s.owner)
132         // this assertion is specific to how ScalaC works. It doesn't apply to dotty, as n dotty there will be B & B$
133         // assert(nb == 2, s"Java member module without member class: $s - $nestedClassSymbols")
134         false
135       } else true
136     })
137 
138     val memberClasses = nestedClassSymbolsNoJavaModuleClasses.map(classBTypeFromSymbol)
139 
140     val nestedInfo = buildNestedInfo(classSym)
141 
142     classBType.info = ClassInfo(superClass, interfaces, flags, memberClasses, nestedInfo)
143     classBType
144   }

147   private def buildNestedInfo(innerClassSym: Symbol): Option[NestedInfo] = {
148     assert(innerClassSym.isClass, s"Cannot build NestedInfo for non-class symbol $innerClassSym")
149 
150     val isNested = !innerClassSym.rawowner.isPackageClass
151     if (!isNested) None
152     else {
153       // See comment in BTypes, when is a class marked static in the InnerClass table.
154       val isStaticNestedClass = innerClassSym.originalOwner.isOriginallyStaticOwner
155 
156       // After lambdalift (which is where we are), the rawowoner field contains the enclosing class.
157       val enclosingClassSym = innerClassSym.enclosingClassSym
158       val enclosingClass: ClassBType = classBTypeFromSymbol(enclosingClassSym)
159 
160       val outerName: Option[String] = {
161         if (isAnonymousOrLocalClass(innerClassSym)) {
162           None
163         } else {
164           val outerName = innerClassSym.rawowner.javaBinaryName
165           // Java compatibility. See the big comment in BTypes that summarizes the InnerClass spec.
166           val outerNameModule =
167             if (innerClassSym.rawowner.isTopLevelModuleClass) dropModule(outerName)
168             else outerName
169           Some(outerNameModule.toString)
170         }
171       }
172 
173       val innerName: Option[String] = {
174         if (innerClassSym.isAnonymousClass || innerClassSym.isAnonymousFunction) None
175         else Some(innerClassSym.rawname + innerClassSym.moduleSuffix) // moduleSuffix for module classes
176       }
177 
178       Some(NestedInfo(enclosingClass, outerName, innerName, isStaticNestedClass))
179     }
180   }

182   /**
183    * Return the Java modifiers for the given symbol.
184    * Java modifiers for classes:
185    *  - public, abstract, final, strictfp (not used)
186    * for interfaces:
187    *  - the same as for classes, without 'final'
188    * for fields:
189    *  - public, private (*)
190    *  - static, final
191    * for methods:
192    *  - the same as for fields, plus:
193    *  - abstract, synchronized (not used), strictfp (not used), native (not used)
194    * for all:
195    *  - deprecated
196    *
197    *  (*) protected cannot be used, since inner classes 'see' protected members,
198    *      and they would fail verification after lifted.
199    */
200   final def javaFlags(sym: Symbol): Int = {
201 
202 
203     val privateFlag = sym.getsJavaPrivateFlag
204 
205     val finalFlag = sym.getsJavaFinalFlag
206 
207     // Primitives are "abstract final" to prohibit instantiation
208     // without having to provide any implementations, but that is an
209     // illegal combination of modifiers at the bytecode level so
210     // suppress final if abstract if present.
211     import asm.Opcodes._
212     GenBCodeOps.mkFlags(
213       if (privateFlag) ACC_PRIVATE else ACC_PUBLIC,
214       if (sym.isDeferred || sym.hasAbstractFlag) ACC_ABSTRACT else 0,
215       if (sym.isInterface) ACC_INTERFACE else 0,
216       if (finalFlag && !sym.hasAbstractFlag) ACC_FINAL else 0,
217       if (sym.isStaticMember) ACC_STATIC else 0,
218       if (sym.isBridge) ACC_BRIDGE | ACC_SYNTHETIC else 0,
219       if (sym.isArtifact) ACC_SYNTHETIC else 0,
220       if (sym.isClass && !sym.isInterface) ACC_SUPER else 0,
221       if (sym.hasEnumFlag) ACC_ENUM else 0,
222       if (sym.isVarargsMethod) ACC_VARARGS else 0,
223       if (sym.isSynchronized) ACC_SYNCHRONIZED else 0,
224       if (sym.isDeprecated) asm.Opcodes.ACC_DEPRECATED else 0
225     )
226   }

228   def javaFieldFlags(sym: Symbol) = {
229     javaFlags(sym) | GenBCodeOps.mkFlags(
230       if (sym hasAnnotation TransientAttr) asm.Opcodes.ACC_TRANSIENT else 0,
231       if (sym hasAnnotation VolatileAttr)  asm.Opcodes.ACC_VOLATILE  else 0,
232       if (sym.isMutable) 0 else asm.Opcodes.ACC_FINAL
233     )
