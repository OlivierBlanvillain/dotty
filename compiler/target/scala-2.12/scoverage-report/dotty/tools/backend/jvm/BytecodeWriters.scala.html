28   def getFile(base: AbstractFile, clsName: String, suffix: String): AbstractFile = {
29     def ensureDirectory(dir: AbstractFile): AbstractFile =
30       if (dir.isDirectory) dir
31       else throw new FileConflictException(s"${base.path}/$clsName$suffix: ${dir.path} is not a directory", dir)
32     var dir = base
33     val pathParts = clsName.split("[./]").toList
34     for (part <- pathParts.init) dir = ensureDirectory(dir) subdirectoryNamed part
35     ensureDirectory(dir) fileNamed pathParts.last + suffix
36   }
37   def getFile(sym: Symbol, clsName: String, suffix: String): AbstractFile =
38     getFile(sym.outputDirectory, clsName, suffix)

75   /*
76    * The ASM textual representation for bytecode overcomes disadvantages of javap ouput in three areas:
77    *    (a) pickle dingbats undecipherable to the naked eye;
78    *    (b) two constant pools, while having identical contents, are displayed differently due to physical layout.
79    *    (c) stack maps (classfile version 50 and up) are displayed in encoded form by javap,
80    *        their expansion by ASM is more readable.
81    *
82    * */
83   trait AsmpBytecodeWriter extends BytecodeWriter {
84     import scala.tools.asm
85 
86     private val baseDir = Directory(int.emitAsmp.get).createDirectory()
87 
88     private def emitAsmp(jclassBytes: Array[Byte], asmpFile: io.File): Unit = {
89       val pw = asmpFile.printWriter()
90       try {
91         val cnode = new asm.tree.ClassNode()
92         val cr    = new asm.ClassReader(jclassBytes)
93         cr.accept(cnode, 0)
94         val trace = new scala.tools.asm.util.TraceClassVisitor(new java.io.PrintWriter(new java.io.StringWriter()))
95         cnode.accept(trace)
96         trace.p.print(pw)
97       }
98       finally pw.close()
99     }
100 
101     abstract override def writeClass(label: String, jclassName: String, jclassBytes: Array[Byte], outfile: AbstractFile): Unit = {
102       super.writeClass(label, jclassName, jclassBytes, outfile)
103 
104       val segments = jclassName.split("[./]")
105       val asmpFile = segments.foldLeft(baseDir: Path)(_ / _) changeExtension "asmp" toFile;
106 
107       asmpFile.parent.createDirectory()
108       emitAsmp(jclassBytes, asmpFile)
109     }
110   }
