39   private def dirName(path: String)  = splitPath(path, front = true)

41   private def splitPath(path0: String, front: Boolean): String = {
42     val isDir = path0.charAt(path0.length - 1) == '/'
43     val path  = if (isDir) path0.substring(0, path0.length - 1) else path0
44     val idx   = path.lastIndexOf('/')
45 
46     if (idx < 0)
47       if (front) "/"
48       else path
49     else
50       if (front) path.substring(0, idx + 1)
51       else path.substring(idx + 1)
52   }

89   private def ensureDir(dirs: mutable.Map[String, DirEntry], path: String, zipEntry: ZipEntry): DirEntry =
90     //OPT inlined from getOrElseUpdate; saves ~50K closures on test run.
91     // was:
92     // dirs.getOrElseUpdate(path, {
93     //   val parent = ensureDir(dirs, dirName(path), null)
94     //   val dir    = new DirEntry(path)
95     //   parent.entries(baseName(path)) = dir

99       case Some(v) => v

121   private[this] class LazyEntry(
122     name: String,
123     time: Long,
124     size: Int
125   ) extends Entry(name) {
126     override def lastModified: Long = time // could be stale
127     override def input: InputStream = {
128       val zipFile  = openZipFile()
129       val entry    = zipFile.getEntry(name)
130       val delegate = zipFile.getInputStream(entry)
131       new FilterInputStream(delegate) {
132         override def close(): Unit = { zipFile.close() }
133       }
134     }
135     override def sizeOption: Option[Int] = Some(size) // could be stale
136   }

138   // keeps a file handle open to ZipFile, which forbids file mutation
139   // on Windows, and leaks memory on all OS (typically by stopping
140   // classloaders from being garbage collected). But is slightly
141   // faster than LazyEntry.
142   private[this] class LeakyEntry(
143     zipFile: ZipFile,
144     zipEntry: ZipEntry
145   ) extends Entry(zipEntry.getName) {
146     override def lastModified: Long = zipEntry.getTime
147     override def input: InputStream = zipFile.getInputStream(zipEntry)
148     override def sizeOption: Option[Int] = Some(zipEntry.getSize.toInt)
149   }
150 
151   @volatile lazy val (root, allDirs) = {
152     val root = new DirEntry("/")
153     val dirs = mutable.HashMap[String, DirEntry]("/" -> root)
154     val zipFile = openZipFile()
155     val entries = zipFile.entries()
156 
157     try {
158       while (entries.hasMoreElements) {
159         val zipEntry = entries.nextElement
160         val dir = getDir(dirs, zipEntry)
161         if (zipEntry.isDirectory) dir
162         else {
163           val f =
164             if (ZipArchive.closeZipFile)
165               new LazyEntry(
166                 zipEntry.getName(),
167                 zipEntry.getTime(),
168                 zipEntry.getSize().toInt
169               )
170             else
171               new LeakyEntry(zipFile, zipEntry)
172 
173           dir.entries(f.name) = f
174         }
175       }
176     } finally {
177       if (ZipArchive.closeZipFile) zipFile.close()
178     }
179     (root, dirs)
180   }

198 final class ManifestResources(val url: URL) extends ZipArchive(null) {
199   def iterator() = {
200     val root     = new DirEntry("/")
201     val dirs     = mutable.HashMap[String, DirEntry]("/" -> root)
202     val manifest = new Manifest(input)
203     val iter     = manifest.getEntries().keySet().iterator().asScala.filter(_.endsWith(".class")).map(new ZipEntry(_))
204 
205     for (zipEntry <- iter) {
206       val dir = getDir(dirs, zipEntry)
207       if (!zipEntry.isDirectory) {
208         val f = new Entry(zipEntry.getName) {
209           override def lastModified = zipEntry.getTime()
210           override def input        = resourceInputStream(path)
211           override def sizeOption   = None
212         }
213         dir.entries(f.name) = f
214       }
215     }
216 
217     try root.iterator
218     finally dirs.clear()
219   }
220 
221   def name  = path
222   def path: String = {
223     val s = url.getPath
224     val n = s.lastIndexOf('!')
225     s.substring(0, n)
226   }
227   def input = url.openStream()
228   def lastModified =
229     try url.openConnection().getLastModified()
230     catch { case _: IOException => 0 }
231 
232   override def canEqual(other: Any) = other.isInstanceOf[ManifestResources]
233   override def hashCode() = url.hashCode
234   override def equals(that: Any) = that match {

236     case _                => false

239   private def resourceInputStream(path: String): InputStream = {
240     new FilterInputStream(null) {
241       override def read(): Int = {
242         if(in == null) in = Thread.currentThread().getContextClassLoader().getResourceAsStream(path)
243         if(in == null) throw new RuntimeException(path + " not found")
244         super.read()
245       }
246 
247       override def close(): Unit = {
248         super.close()
249         in = null
250       }
251     }
252   }
