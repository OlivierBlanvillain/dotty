30 /** An abstraction for files.  For character data, a Codec
31  *  can be supplied at either creation time or when a method
32  *  involving character data is called (with the latter taking
33  *  precedence if supplied.) If neither is available, the value
34  *  of scala.io.Codec.default is used.
35  *
36  *  @author  Paul Phillips
37  *  @since   2.8
38  *
39  *  ''Note:  This is library is considered experimental and should not be used unless you know what you are doing.''
40  */
41 class File(jpath: JPath)(implicit constructorCodec: Codec) extends Path(jpath) with Streamable.Chars {
42   override val creationCodec = constructorCodec
43 
44   override def addExtension(ext: String): File = super.addExtension(ext).toFile
45   override def toAbsolute: File = if (isAbsolute) this else super.toAbsolute.toFile
46   override def toDirectory: Directory = new Directory(jpath)
47   override def toFile: File = this
48   override def normalize: File = super.normalize.toFile
49   override def length = super[Path].length
50   override def walkFilter(cond: Path => Boolean): Iterator[Path] =
51     if (cond(this)) Iterator.single(this) else Iterator.empty
52 
53   /** Obtains an InputStream. */
54   def inputStream() = Files.newInputStream(jpath)
55 
56   /** Obtains a OutputStream. */
57   def outputStream(append: Boolean = false) =
58     if (append) Files.newOutputStream(jpath, CREATE, APPEND)
59     else Files.newOutputStream(jpath, CREATE, TRUNCATE_EXISTING)
60   def bufferedOutput(append: Boolean = false) = new BufferedOutputStream(outputStream(append))
61 
62   /** Obtains an OutputStreamWriter wrapped around a FileOutputStream.
63    *  This should behave like a less broken version of java.io.FileWriter,
64    *  in that unlike the java version you can specify the encoding.
65    */
66   def writer(append: Boolean, codec: Codec): OutputStreamWriter =
67     new OutputStreamWriter(outputStream(append), codec.charSet)
68 
69   /** Wraps a BufferedWriter around the result of writer().
70    */
71   def bufferedWriter(): BufferedWriter = bufferedWriter(append = false)
72   def bufferedWriter(append: Boolean): BufferedWriter = bufferedWriter(append, creationCodec)
73   def bufferedWriter(append: Boolean, codec: Codec): BufferedWriter =
74     new BufferedWriter(writer(append, codec))
75 
76   def printWriter(): PrintWriter = new PrintWriter(bufferedWriter(), true)
77 
78   /** Creates a new file and writes all the Strings to it. */
79   def writeAll(strings: String*): Unit = {
80     val out = bufferedWriter()
81     try strings foreach (out write _)
82     finally out.close()
83   }
84 
85   def appendAll(strings: String*): Unit = {
86     val out = bufferedWriter(append = true)
87     try strings foreach (out write _)
88     finally out.close()
89   }
90 
91   /** Calls println on each string (so it adds a newline in the PrintWriter fashion.) */
92   def printlnAll(strings: String*): Unit = {
93     val out = printWriter()
94     try strings foreach (out println _)
95     finally out.close()
96   }
97 
98   def safeSlurp(): Option[String] =
99     try Some(slurp())
100     catch { case _: IOException => None }
101 
102   /** Reflection since we're into the java 6+ API.
103    */
104   def setExecutable(executable: Boolean, ownerOnly: Boolean = true): Boolean = {
105     type JBoolean = java.lang.Boolean
106     val method =
107       try classOf[JFile].getMethod("setExecutable", classOf[Boolean], classOf[Boolean])
108       catch { case _: NoSuchMethodException => return false }
109 
110     try method.invoke(jpath.toFile, executable: JBoolean, ownerOnly: JBoolean).asInstanceOf[JBoolean].booleanValue
111     catch { case _: Exception => false }
112   }
113 }
