58 /**
59  * <p>
60  *   This class and its children serve to unify handling of files and
61  *   directories. These files and directories may or may not have some
62  *   real counter part within the file system. For example, some file
63  *   handles reference files within a zip archive or virtual ones
64  *   that exist only in memory.
65  * </p>
66  * <p>
67  *   Every abstract file has a path (i.e. a full name) and a name
68  *   (i.e. a short name) and may be backed by some real File. There are
69  *   two different kinds of abstract files: regular files and
70  *   directories. Regular files may be read and have a last modification
71  *   time. Directories may list their content and look for subfiles with
72  *   a specified name or path and of a specified kind.
73  * </p>
74  * <p>
75  *   The interface does <b>not</b> allow to access the content.
76  *   The class `symtab.classfile.AbstractFileReader` accesses
77  *   bytes, knowing that the character set of classfiles is UTF-8. For
78  *   all other cases, the class `SourceFile` is used, which honors
79  *   `global.settings.encoding.value`.
80  * </p>
81  *
82  * ''Note:  This library is considered experimental and should not be used unless you know what you are doing.''
83  */

163   /** Returns contents of file (if applicable) in a byte array.
164    */
165   @throws(classOf[IOException])
166   def toByteArray: Array[Byte] = {
167     val in = input
168     sizeOption match {
169       case Some(size) =>
170         var rest = size
171         val arr = new Array[Byte](rest)
172         while (rest > 0) {
173           val res = in.read(arr, arr.length - rest, rest)
174           if (res == -1)
175             throw new IOException("read error")
176           rest -= res
177         }
178         in.close()
179         arr
180       case None =>
181         val out = new ByteArrayOutputStream()
182         var c = in.read()
183         while(c != -1) {
184           out.write(c)
185           c = in.read()
186         }
187         in.close()
188         out.toByteArray()
189     }
190   }

214   private def lookup(getFile: (AbstractFile, String, Boolean) => AbstractFile,
215                      path0: String,
216                      directory: Boolean): AbstractFile = {
217     val separator = java.io.File.separatorChar
218     // trim trailing '/'s
219     val path: String = if (path0.last == separator) path0 dropRight 1 else path0
220     val length = path.length()
221     assert(length > 0 && !(path.last == separator), path)
222     var file = this
223     var start = 0
224     while (true) {
225       val index = path.indexOf(separator, start)
226       assert(index < 0 || start < index, ((path, directory, start, index)))
227       val name = path.substring(start, if (index < 0) length else index)
228       file = getFile(file, name, if (index < 0) directory else true)
229       if ((file eq null) || index < 0) return file
230       start = index + 1
231     }
232     file
233   }

235   private def fileOrSubdirectoryNamed(name: String, isDir: Boolean): AbstractFile = {
236     val lookup = lookupName(name, isDir)
237     if (lookup != null) lookup

240       val path = jpath.resolve(name)

242       else Files.createFile(path)

244     }
