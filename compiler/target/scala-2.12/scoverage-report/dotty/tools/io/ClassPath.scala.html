61   def findClass(className: String): Option[ClassRepresentation] = {
62     // A default implementation which should be overridden, if we can create the more efficient
63     // solution for a given type of ClassPath
64     val (pkg, simpleClassName) = PackageNameUtils.separatePkgAndClassNames(className)
65 
66     val foundClassFromClassFiles = classes(pkg).find(_.name == simpleClassName)
67     def findClassInSources = sources(pkg).find(_.name == simpleClassName)
68 
69     foundClassFromClassFiles orElse findClassInSources
70   }

100   /** Expand single path entry */
101   private def expandS(pattern: String): List[String] = {
102     val wildSuffix = File.separator + "*"
103 
104     /* Get all subdirectories, jars, zips out of a directory. */
105     def lsDir(dir: Directory, filt: String => Boolean = _ => true) =
106       dir.list.filter(x => filt(x.name) && (x.isDirectory || isJarOrZip(x))).map(_.path).toList
107 
108     if (pattern == "*") lsDir(Directory("."))
109     else if (pattern endsWith wildSuffix) lsDir(Directory(pattern dropRight 2))
110     else if (pattern contains '*') {
111       try {
112         val regexp = ("^" + pattern.replaceAllLiterally("""\*""", """.*""") + "$").r
113         lsDir(Directory(pattern).parent, regexp.findFirstIn(_).isDefined)
114       }
115       catch { case _: PatternSyntaxException => List(pattern) }
116     }
117     else List(pattern)
118   }
119 
120   /** Split classpath using platform-dependent path separator */
121   def split(path: String): List[String] = (path split pathSeparator).toList.filterNot(_ == "").distinct
122 
123   /** Join classpath using platform-dependent path separator */
124   def join(paths: String*): String  = paths filterNot (_ == "") mkString pathSeparator
125 
126   /** Split the classpath, apply a transformation function, and reassemble it. */
127   def map(cp: String, f: String => String): String = join(split(cp) map f: _*)
128 
129   /** Expand path and possibly expanding stars */
130   def expandPath(path: String, expandStar: Boolean = true): List[String] =
131     if (expandStar) split(path) flatMap expandS
132     else split(path)
133 
134   /** Expand dir out to contents, a la extdir */
135   def expandDir(extdir: String): List[String] = {
136     AbstractFile getDirectory extdir match {
137       case null => Nil
138       case dir  => dir.filter(_.isClassContainer).map(x => new java.io.File(dir.file, x.name).getPath).toList
139     }
140   }
141 
142   /** Expand manifest jar classpath entries: these are either urls, or paths
143    *  relative to the location of the jar.
144    */
145   def expandManifestPath(jarPath: String): List[URL] = {
146     val file = File(jarPath)
147     if (!file.isFile) return Nil
148 
149     val baseDir = file.parent
150     new Jar(file).classPathElements map (elem =>
151       specToURL(elem) getOrElse (baseDir / elem).toURL
152     )
153   }
154 
155   def specToURL(spec: String): Option[URL] =
156     try Some(new URL(spec))
157     catch { case _: MalformedURLException => None }
158 
159   def manifests: List[java.net.URL] = {
160     import scala.collection.JavaConverters._
161     val resources = Thread.currentThread().getContextClassLoader().getResources("META-INF/MANIFEST.MF")
162     resources.asScala.filter(_.getProtocol == "jar").toList
163   }
