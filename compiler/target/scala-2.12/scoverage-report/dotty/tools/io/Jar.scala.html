78 class JarWriter(val file: File, val manifest: Manifest) {
79   private lazy val out = new JarOutputStream(file.outputStream(), manifest)
80 
81   /** Adds a jar entry for the given path and returns an output
82    *  stream to which the data should immediately be written.
83    *  This unusual interface exists to work with fjbg.
84    */
85   def newOutputStream(path: String): DataOutputStream = {
86     val entry = new JarEntry(path)
87     out putNextEntry entry
88     new DataOutputStream(out)
89   }
90 
91   def writeAllFrom(dir: Directory): Unit = {
92     try dir.list foreach (x => addEntry(x, ""))
93     finally out.close()
94   }
95   def addStream(entry: JarEntry, in: InputStream): Unit =  {
96     out putNextEntry entry
97     try transfer(in, out)
98     finally out.closeEntry()
99   }
100   def addFile(file: File, prefix: String): Unit =  {
101     val entry = new JarEntry(prefix + file.name)
102     addStream(entry, file.inputStream())
103   }
104   def addEntry(entry: Path, prefix: String): Unit =  {
105     if (entry.isFile) addFile(entry.toFile, prefix)
106     else addDirectory(entry.toDirectory, prefix + entry.name + "/")
107   }
108   def addDirectory(entry: Directory, prefix: String): Unit =  {
109     entry.list foreach (p => addEntry(p, prefix))
110   }
111 
112   private def transfer(in: InputStream, out: OutputStream) = {
113     val buf = new Array[Byte](10240)
114     @tailrec def loop(): Unit = in.read(buf, 0, buf.length) match {
115       case -1 => in.close()
116       case n  => out.write(buf, 0, n) ; loop()
117     }
118     loop()
119   }
120 
121   def close() = out.close()
122 }
