18 /** This class implements an abstract file backed by a File.
19  *
20  * ''Note:  This library is considered experimental and should not be used unless you know what you are doing.''
21  */
22 class PlainFile(val givenPath: Path) extends AbstractFile {
23   assert(path ne null)

56   /** Returns all abstract subfiles of this abstract directory. */
57   def iterator: Iterator[AbstractFile] = {
58     // Optimization: Assume that the file was not deleted and did not have permissions changed
59     // between the call to `list` and the iteration. This saves a call to `exists`.
60     def existsFast(path: Path) = path match {
61       case (_: Directory | _: File) => true
62       case _ => path.exists
63     }
64     givenPath.toDirectory.list.filter(existsFast).map(new PlainFile(_))
65   }
66 
67   /**
68    * Returns the abstract file in this abstract directory with the
69    * specified name. If there is no such file, returns null. The
70    * argument "directory" tells whether to look for a directory or
71    * or a regular file.
72    */
73   def lookupName(name: String, directory: Boolean): AbstractFile = {
74     val child = givenPath / name
75     if ((child.isDirectory && directory) || (child.isFile && !directory)) new PlainFile(child)
76     else null
77   }
78 
79   /** Does this abstract file denote an existing file? */
80   def create(): Unit = if (!exists) givenPath.createFile()
81 
82   /** Delete the underlying file or directory (recursively). */
83   def delete(): Unit =
84     if (givenPath.isFile) givenPath.delete()
85     else if (givenPath.isDirectory) givenPath.toDirectory.deleteRecursively()
86 
87   /** Returns a plain file with the given name. It does not
88    *  check that it exists.
89    */
90   def lookupNameUnchecked(name: String, directory: Boolean): AbstractFile =
91     new PlainFile(givenPath / name)
