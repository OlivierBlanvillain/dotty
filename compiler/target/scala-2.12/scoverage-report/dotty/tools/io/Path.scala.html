19 /** An abstraction for filesystem paths.  The differences between
20  *  Path, File, and Directory are primarily to communicate intent.
21  *  Since the filesystem can change at any time, there is no way to
22  *  reliably associate Files only with files and so on.  Any Path
23  *  can be converted to a File or Directory (and thus gain access to
24  *  the additional entity specific methods) by calling toFile or
25  *  toDirectory, which has no effect on the filesystem.
26  *
27  *  Also available are createFile and createDirectory, which attempt
28  *  to create the path in question.
29  *
30  *  @author  Paul Phillips
31  *  @since   2.8
32  *
33  *  ''Note:  This library is considered experimental and should not be used unless you know what you are doing.''
34  */

37   def isExtensionJarOrZip(name: String): Boolean = {
38     val ext = extension(name)
39     ext == "jar" || ext == "zip"
40   }
41   def extension(name: String): String = {

52   def onlyFiles(xs: Iterator[Path]): Iterator[File] = xs filter (_.isFile) map (_.toFile)

99   /** If this path is a directory, recursively iterate over its contents.
100    *  The supplied condition is a filter which is applied to each element,
101    *  with that branch of the tree being closed off if it is false.
102    *  So for example if the condition is false for some subdirectory, nothing
103    *  under that directory will be in the Iterator. If it's true, all files for
104    *  which the condition holds and are directly in that subdirectory are in the
105    *  Iterator, and all sub-subdirectories are recursively evaluated
106    */
107   def walkFilter(cond: Path => Boolean): Iterator[Path] =
108     if (isFile) toFile walkFilter cond
109     else if (isDirectory) toDirectory walkFilter cond
110     else Iterator.empty
