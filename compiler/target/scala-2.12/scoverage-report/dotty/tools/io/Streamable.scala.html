15 /** Traits for objects which can be represented as Streams.
16  *
17  *  @author Paul Phillips
18  *  @since  2.8
19  *
20  *  ''Note:  This library is considered experimental and should not be used unless you know what you are doing.''
21  */
22 object Streamable {
23   /** Traits which can be viewed as a sequence of bytes.  Source types
24    *  which know their length should override def length: Long for more
25    *  efficient method implementations.
26    *
27    *  ''Note:  This library is considered experimental and should not be used unless you know what you are doing.''
28    *
29    *  Note that this code was not written with resource management in mind.
30    *  Several methods (such as `chars` and `lines`) create InputStreams they
31    *  don't close
32    */
33   trait Bytes {
34     def inputStream(): InputStream
35     def length: Long = -1
36 
37     def bufferedInput() = new BufferedInputStream(inputStream())
38     def bytes(): Iterator[Byte] = bytesAsInts() map (_.toByte)
39     def bytesAsInts(): Iterator[Int] = {
40       val in = bufferedInput()
41       Iterator continually in.read() takeWhile (_ != -1)
42     }
43 
44     /** This method aspires to be the fastest way to read
45      *  a stream of known length into memory.
46      */
47     def toByteArray(): Array[Byte] = {
48       // if we don't know the length, fall back on relative inefficiency
49       if (length == -1L)
50         return (new ArrayBuffer[Byte]() ++= bytes()).toArray
51 
52       val arr = new Array[Byte](length.toInt)
53       val len = arr.length
54       lazy val in = bufferedInput()
55       var offset = 0
56 
57       def loop(): Unit = {
58         if (offset < len) {
59           val read = in.read(arr, offset, len - offset)
60           if (read >= 0) {
61             offset += read
62             loop()
63           }
64         }
65       }
66       try loop()
67       finally in.close()
68 
69       if (offset == arr.length) arr
70       else fail("Could not read entire source (%d of %d bytes)".format(offset, len))
71     }
72   }
73 
74   /** For objects which can be viewed as Chars.
75    *
76    * ''Note:  This library is considered experimental and should not be used unless you know what you are doing.''
77    */
78   trait Chars extends Bytes {
79     /** Calls to methods requiring byte<->char transformations should be offered
80      *  in a form which allows specifying the codec.  When it is not specified,
81      *  the one discovered at creation time will be used, which will always find the
82      *  one in scala.io.Codec if no other is available.  This can be overridden
83      *  to use a different default.
84      */
85     def creationCodec: Codec = implicitly[Codec]
86 
87     /** Caller is responsible for closing the returned BufferedSource. */
88     def chars(codec: Codec): BufferedSource = Source.fromInputStream(inputStream())(codec)
89 
90     /** Beware! Leaks an InputStream which will not be closed until it gets finalized. */
91     def lines(): Iterator[String] = lines(creationCodec)
92 
93     /** Beware! Leaks an InputStream which will not be closed until it gets finalized. */
94     def lines(codec: Codec): Iterator[String] = chars(codec).getLines()
95 
96     /** Obtains an InputStreamReader wrapped around a FileInputStream.
97      */
98     def reader(codec: Codec): InputStreamReader = new InputStreamReader(inputStream(), codec.charSet)
99 
100     /** Wraps a BufferedReader around the result of reader().
101      */
102     def bufferedReader(): BufferedReader = bufferedReader(creationCodec)
103     def bufferedReader(codec: Codec) = new BufferedReader(reader(codec))
104 
105     /** Creates a BufferedReader and applies the closure, automatically closing it on completion.
106      */
107     def applyReader[T](f: BufferedReader => T): T = {
108       val in = bufferedReader()
109       try f(in)
110       finally in.close()
111     }
112 
113     /** Convenience function to import entire file into a String.
114      */
115     def slurp(): String = slurp(creationCodec)
116     def slurp(codec: Codec) = {
117       val src = chars(codec)
118       try src.mkString finally src.close()  // Always Be Closing
119     }
120   }
121 
122   /** Call a function on something Closeable, finally closing it. */
123   def closing[T <: JCloseable, U](stream: T)(f: T => U): U =
124     try f(stream)
125     finally stream.close()
126 
127   def bytes(is: => InputStream): Array[Byte] =
128     (new Bytes {
129       def inputStream() = is
130     }).toByteArray()
131 
132   def slurp(is: => InputStream)(implicit codec: Codec): String =
133     new Chars { def inputStream() = is } slurp codec
134 
135   def slurp(url: URL)(implicit codec: Codec): String =
136     slurp(url.openStream())
