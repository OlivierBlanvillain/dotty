12 /**
13  * A classpath unifying multiple class- and sourcepath entries.
14  * The Classpath can obtain entries for classes and sources independently
15  * so it tries to do operations quite optimally - iterating only these collections
16  * which are needed in the given moment and only as far as it's necessary.
17  *
18  * @param aggregates classpath instances containing entries which this class processes
19  */
20 case class AggregateClassPath(aggregates: Seq[ClassPath]) extends ClassPath {
21   override def findClassFile(className: String): Option[AbstractFile] = {
22     val (pkg, _) = PackageNameUtils.separatePkgAndClassNames(className)
23     aggregatesForPackage(pkg).iterator.map(_.findClassFile(className)).collectFirst {
24       case Some(x) => x
25     }
26   }
27   private[this] val packageIndex: collection.mutable.Map[String, Seq[ClassPath]] = collection.mutable.Map()
28   private def aggregatesForPackage(pkg: String): Seq[ClassPath] = packageIndex.synchronized {
29     packageIndex.getOrElseUpdate(pkg, aggregates.filter(_.hasPackage(pkg)))
30   }
31 
32   override def findClass(className: String): Option[ClassRepresentation] = {
33     val (pkg, _) = PackageNameUtils.separatePkgAndClassNames(className)
34 
35     def findEntry(isSource: Boolean): Option[ClassRepresentation] = {
36       aggregatesForPackage(pkg).iterator.map(_.findClass(className)).collectFirst {
37         case Some(s: SourceFileEntry) if isSource => s
38         case Some(s: ClassFileEntry) if !isSource => s
39       }
40     }
41 
42     val classEntry = findEntry(isSource = false)
43     val sourceEntry = findEntry(isSource = true)
44 
45     (classEntry, sourceEntry) match {
46       case (Some(c: ClassFileEntry), Some(s: SourceFileEntry)) => Some(ClassAndSourceFilesEntry(c.file, s.file))
47       case (c @ Some(_), _) => c
48       case (_, s) => s
49     }
50   }
51 
52   override def asURLs: Seq[URL] = aggregates.flatMap(_.asURLs)
53 
54   override def asClassPathStrings: Seq[String] = aggregates.map(_.asClassPathString).distinct
55 
56   override def asSourcePathString: String = ClassPath.join(aggregates map (_.asSourcePathString): _*)
57 
58   override private[dotty] def packages(inPackage: String): Seq[PackageEntry] = {
59     val aggregatedPackages = aggregates.flatMap(_.packages(inPackage)).distinct
60     aggregatedPackages
61   }
62 
63   override private[dotty] def classes(inPackage: String): Seq[ClassFileEntry] =
64     getDistinctEntries(_.classes(inPackage))
65 
66   override private[dotty] def sources(inPackage: String): Seq[SourceFileEntry] =
67     getDistinctEntries(_.sources(inPackage))
68 
69   override private[dotty] def hasPackage(pkg: String) = aggregates.exists(_.hasPackage(pkg))
70   override private[dotty] def list(inPackage: String): ClassPathEntries = {
71     val (packages, classesAndSources) = aggregates.map { cp =>
72       try {
73         cp.list(inPackage)
74       } catch {
75         case ex: java.io.IOException =>
76           val e = new FatalError(ex.getMessage)
77           e.initCause(ex)
78           throw e
79       }
80     }.unzip
81     val distinctPackages = packages.flatten.distinct
82     val distinctClassesAndSources = mergeClassesAndSources(classesAndSources: _*)
83     ClassPathEntries(distinctPackages, distinctClassesAndSources)
84   }
85 
86   /**
87    * Returns only one entry for each name. If there's both a source and a class entry, it
88    * creates an entry containing both of them. If there would be more than one class or source
89    * entries for the same class it always would use the first entry of each type found on a classpath.
90    */
91   private def mergeClassesAndSources(entries: Seq[ClassRepresentation]*): Seq[ClassRepresentation] = {
92     // based on the implementation from MergedClassPath
93     var count = 0
94     val indices = collection.mutable.HashMap[String, Int]()
95     val mergedEntries = new ArrayBuffer[ClassRepresentation](1024)
96 
97     for {
98       partOfEntries <- entries
99       entry <- partOfEntries
100     } {
101       val name = entry.name
102       if (indices contains name) {
103         val index = indices(name)
104         val existing = mergedEntries(index)
105 
106         if (existing.binary.isEmpty && entry.binary.isDefined)
107           mergedEntries(index) = ClassAndSourceFilesEntry(entry.binary.get, existing.source.get)
108         if (existing.source.isEmpty && entry.source.isDefined)
109           mergedEntries(index) = ClassAndSourceFilesEntry(existing.binary.get, entry.source.get)
110       }
111       else {
112         indices(name) = count
113         mergedEntries += entry
114         count += 1
115       }
116     }
117     mergedEntries.toIndexedSeq
118   }
119 
120   private def getDistinctEntries[EntryType <: ClassRepresentation](getEntries: ClassPath => Seq[EntryType]): Seq[EntryType] = {
121     val seenNames = collection.mutable.HashSet[String]()
122     val entriesBuffer = new ArrayBuffer[EntryType](1024)
123     for {
124       cp <- aggregates
125       entry <- getEntries(cp) if !seenNames.contains(entry.name)
126     } {
127       entriesBuffer += entry
128       seenNames += entry.name
129     }
130     entriesBuffer.toIndexedSeq
131   }
132 }

134 object AggregateClassPath {
135   def createAggregate(parts: ClassPath*): ClassPath = {
136     val elems = new ArrayBuffer[ClassPath]()
137     parts foreach {
138       case AggregateClassPath(ps) => elems ++= ps
139       case p => elems += p
140     }
141     if (elems.size == 1) elems.head
142     else AggregateClassPath(elems.toIndexedSeq)
143   }
