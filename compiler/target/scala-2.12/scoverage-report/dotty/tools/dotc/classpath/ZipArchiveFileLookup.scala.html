13 /**
14  * A trait allowing to look for classpath entries of given type in zip and jar files.
15  * It provides common logic for classes handling class and source files.
16  * It's aware of things like e.g. META-INF directory which is correctly skipped.
17  */
18 trait ZipArchiveFileLookup[FileEntryType <: ClassRepresentation] extends ClassPath {
19   val zipFile: File

22

28   override private[dotty] def packages(inPackage: String): Seq[PackageEntry] = {
29     val prefix = PackageNameUtils.packagePrefix(inPackage)
30     for {
31       dirEntry <- findDirEntry(inPackage).toSeq
32       entry <- dirEntry.iterator if entry.isPackage
33     } yield PackageEntryImpl(prefix + entry.name)
34   }
35 
36   protected def files(inPackage: String): Seq[FileEntryType] =
37     for {
38       dirEntry <- findDirEntry(inPackage).toSeq
39       entry <- dirEntry.iterator if isRequiredFileType(entry)
40     } yield createFileEntry(entry)
41 
42   protected def file(inPackage: String, name: String): Option[FileEntryType] =
43     for {
44       dirEntry <- findDirEntry(inPackage)
45       entry <- Option(dirEntry.lookupName(name, directory = false))
46       if isRequiredFileType(entry)
47     } yield createFileEntry(entry)
48 
49   override private[dotty] def hasPackage(pkg: String) = findDirEntry(pkg).isDefined
50   override private[dotty] def list(inPackage: String): ClassPathEntries = {
51     val foundDirEntry = findDirEntry(inPackage)
52 
53     foundDirEntry map { dirEntry =>
54       val pkgBuf = collection.mutable.ArrayBuffer.empty[PackageEntry]
55       val fileBuf = collection.mutable.ArrayBuffer.empty[FileEntryType]
56       val prefix = PackageNameUtils.packagePrefix(inPackage)
57 
58       for (entry <- dirEntry.iterator) {
59         if (entry.isPackage)
60           pkgBuf += PackageEntryImpl(prefix + entry.name)
61         else if (isRequiredFileType(entry))
62           fileBuf += createFileEntry(entry)
63       }
64       ClassPathEntries(pkgBuf, fileBuf)
65     } getOrElse ClassPathEntries(Seq.empty, Seq.empty)
66   }
67 
68   private def findDirEntry(pkg: String): Option[archive.DirEntry] = {
69     val dirName = s"${FileUtils.dirPath(pkg)}/"
70     archive.allDirs.get(dirName)
71   }
