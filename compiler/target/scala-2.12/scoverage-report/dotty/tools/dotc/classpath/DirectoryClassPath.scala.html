17 /**
18  * A trait allowing to look for classpath entries in directories. It provides common logic for
19  * classes handling class and source files.
20  * It makes use of the fact that in the case of nested directories it's easy to find a file
21  * when we have a name of a package.
22  * It abstracts over the file representation to work with both JFile and AbstractFile.
23  */
24 trait DirectoryLookup[FileEntryType <: ClassRepresentation] extends ClassPath {
25   type F
26 
27   val dir: F
28 
29   protected def emptyFiles: Array[F] // avoids reifying ClassTag[F]
30   protected def getSubDir(dirName: String): Option[F]
31   protected def listChildren(dir: F, filter: Option[F => Boolean] = None): Array[F]
32   protected def getName(f: F): String
33   protected def toAbstractFile(f: F): AbstractFile
34   protected def isPackage(f: F): Boolean
35 
36   protected def createFileEntry(file: AbstractFile): FileEntryType
37   protected def isMatchingFile(f: F): Boolean
38 
39   private def getDirectory(forPackage: String): Option[F] = {
40     if (forPackage == ClassPath.RootPackage) {
41       Some(dir)
42     } else {
43       val packageDirName = FileUtils.dirPath(forPackage)
44       getSubDir(packageDirName)
45     }
46   }
47 
48   override private[dotty] def hasPackage(pkg: String) = getDirectory(pkg).isDefined
49 
50   private[dotty] def packages(inPackage: String): Seq[PackageEntry] = {
51     val dirForPackage = getDirectory(inPackage)
52     val nestedDirs: Array[F] = dirForPackage match {
53       case None => emptyFiles
54       case Some(directory) => listChildren(directory, Some(isPackage))
55     }
56     val prefix = PackageNameUtils.packagePrefix(inPackage)
57     nestedDirs.map(f => PackageEntryImpl(prefix + getName(f)))
58   }
59 
60   protected def files(inPackage: String): Seq[FileEntryType] = {
61     val dirForPackage = getDirectory(inPackage)
62     val files: Array[F] = dirForPackage match {
63       case None => emptyFiles
64       case Some(directory) => listChildren(directory, Some(isMatchingFile))
65     }
66     files.map(f => createFileEntry(toAbstractFile(f)))
67   }
68 
69   private[dotty] def list(inPackage: String): ClassPathEntries = {
70     val dirForPackage = getDirectory(inPackage)
71     val files: Array[F] = dirForPackage match {
72       case None => emptyFiles
73       case Some(directory) => listChildren(directory)
74     }
75     val packagePrefix = PackageNameUtils.packagePrefix(inPackage)
76     val packageBuf = collection.mutable.ArrayBuffer.empty[PackageEntry]
77     val fileBuf = collection.mutable.ArrayBuffer.empty[FileEntryType]
78     for (file <- files) {
79       if (isPackage(file))
80         packageBuf += PackageEntryImpl(packagePrefix + getName(file))
81       else if (isMatchingFile(file))
82         fileBuf += createFileEntry(toAbstractFile(file))
83     }
84     ClassPathEntries(packageBuf, fileBuf)
85   }
86 }

88 trait JFileDirectoryLookup[FileEntryType <: ClassRepresentation] extends DirectoryLookup[FileEntryType] {
89   type F = JFile
90 
91   protected def emptyFiles: Array[JFile] = Array.empty
92   protected def getSubDir(packageDirName: String): Option[JFile] = {
93     val packageDir = new JFile(dir, packageDirName)
94     if (packageDir.exists && packageDir.isDirectory) Some(packageDir)
95     else None
96   }
97   protected def listChildren(dir: JFile, filter: Option[JFile => Boolean]): Array[JFile] = {
98     val listing = filter match {
99       case Some(f) => dir.listFiles(mkFileFilter(f))
100       case None => dir.listFiles()
101     }
102 
103     if (listing != null) {
104       // Sort by file name for stable order of directory .class entries in package scope.
105       // This gives stable results ordering of base type sequences for unrelated classes
106       // with the same base type depth.
107       //
108       // Notably, this will stably infer`Product with Serializable`
109       // as the type of `case class C(); case class D(); List(C(), D()).head`, rather than the opposite order.
110       // On Mac, the HFS performs this sorting transparently, but on Linux the order is unspecified.
111       //
112       // Note this behaviour can be enabled in javac with `javac -XDsortfiles`, but that's only
113       // intended to improve determinism of the compiler for compiler hackers.
114       java.util.Arrays.sort(listing,
115         new java.util.Comparator[JFile] {
116           def compare(o1: JFile, o2: JFile) = o1.getName.compareTo(o2.getName)
117         })
118       listing
119     } else Array()
120   }
121   protected def getName(f: JFile): String = f.getName
122   protected def toAbstractFile(f: JFile): AbstractFile = new PlainFile(new dotty.tools.io.File(f.toPath))
123   protected def isPackage(f: JFile): Boolean = f.isPackage
124 
125   assert(dir != null, "Directory file in DirectoryFileLookup cannot be null")
126 
127   def asURLs: Seq[URL] = Seq(dir.toURI.toURL)
128   def asClassPathStrings: Seq[String] = Seq(dir.getPath)
129 }

144 /**
145   * Implementation `ClassPath` based on the JDK 9 encapsulated runtime modules (JEP-220)
146   *
147   * https://bugs.openjdk.java.net/browse/JDK-8066492 is the most up to date reference
148   * for the structure of the jrt:// filesystem.
149   *
150   * The implementation assumes that no classes exist in the empty package.
151   */
152 final class JrtClassPath(fs: java.nio.file.FileSystem) extends ClassPath with NoSourcePaths {
153   import java.nio.file.Path, java.nio.file._
154   type F = Path
155   private val dir: Path = fs.getPath("/packages")
156 
157   // e.g. "java.lang" -> Seq("/modules/java.base")
158   private val packageToModuleBases: Map[String, Seq[Path]] = {
159     val ps = Files.newDirectoryStream(dir).iterator().asScala
160     def lookup(pack: Path): Seq[Path] = {
161       Files.list(pack).iterator().asScala.map(l => if (Files.isSymbolicLink(l)) Files.readSymbolicLink(l) else l).toList
162     }
163     ps.map(p => (p.toString.stripPrefix("/packages/"), lookup(p))).toMap
164   }
165 
166   /** Empty string represents root package */
167   override private[dotty] def hasPackage(pkg: String) = packageToModuleBases.contains(pkg)
168 
169   override private[dotty] def packages(inPackage: String): Seq[PackageEntry] = {
170     def matches(packageDottedName: String) =
171       if (packageDottedName.contains("."))
172         packageOf(packageDottedName) == inPackage
173       else inPackage == ""
174     packageToModuleBases.keysIterator.filter(matches).map(PackageEntryImpl(_)).toVector
175   }
176   private[dotty] def classes(inPackage: String): Seq[ClassFileEntry] = {
177     if (inPackage == "") Nil
178     else {
179       packageToModuleBases.getOrElse(inPackage, Nil).flatMap(x =>
180         Files.list(x.resolve(inPackage.replace('.', '/'))).iterator().asScala.filter(_.getFileName.toString.endsWith(".class"))).map(x =>
181         ClassFileEntryImpl(new PlainFile(new dotty.tools.io.File(x)))).toVector
182     }
183   }
184 
185   override private[dotty] def list(inPackage: String): ClassPathEntries =
186     if (inPackage == "") ClassPathEntries(packages(inPackage), Nil)
187     else ClassPathEntries(packages(inPackage), classes(inPackage))
188 
189   def asURLs: Seq[URL] = Seq(dir.toUri.toURL)
190   // We don't yet have a scheme to represent the JDK modules in our `-classpath`.
191   // java models them as entries in the new "module path", we'll probably need to follow this.
192   def asClassPathStrings: Seq[String] = Nil
193 
194   def findClassFile(className: String): Option[AbstractFile] = {
195     if (!className.contains(".")) None
196     else {
197       val inPackage = packageOf(className)
198       packageToModuleBases.getOrElse(inPackage, Nil).iterator.flatMap{x =>
199         val file = x.resolve(className.replace('.', '/') + ".class")
200         if (Files.exists(file)) new PlainFile(new dotty.tools.io.File(file)) :: Nil else Nil
201       }.take(1).toList.headOption
202     }
203   }
204   private def packageOf(dottedClassName: String): String =
205     dottedClassName.substring(0, dottedClassName.lastIndexOf("."))
206 }

211   def findClassFile(className: String): Option[AbstractFile] = {
212     val relativePath = FileUtils.dirPath(className)
213     val classFile = new JFile(s"$dir/$relativePath.class")
214     if (classFile.exists) {
215       val wrappedClassFile = new dotty.tools.io.File(classFile.toPath)
216       val abstractClassFile = new PlainFile(wrappedClassFile)
217       Some(abstractClassFile)
218     } else None
219   }

222   protected def isMatchingFile(f: JFile): Boolean = f.isClass
223

238       .map(ext => new JFile(s"$dir/$relativePath.$ext"))

242       val wrappedSourceFile = new dotty.tools.io.File(file.toPath)
243       val abstractSourceFile = new PlainFile(wrappedSourceFile)
244       abstractSourceFile
245     }
246   }
