16 /**
17  * A trait providing an optional cache for classpath entries obtained from zip and jar files.
18  * It allows us to e.g. reduce significantly memory used by PresentationCompilers in Scala IDE
19  * when there are a lot of projects having a lot of common dependencies.
20  */
21 sealed trait ZipAndJarFileLookupFactory {
22   private val cache = new FileBasedCache[ClassPath]
23 
24   def create(zipFile: AbstractFile)(implicit ctx: Context): ClassPath = {
25     if (ctx.settings.YdisableFlatCpCaching.value || zipFile.file == null) createForZipFile(zipFile)
26     else createUsingCache(zipFile)
27   }
28 
29   protected def createForZipFile(zipFile: AbstractFile): ClassPath
30 
31   private def createUsingCache(zipFile: AbstractFile): ClassPath =
32     cache.getOrCreate(zipFile.file.toPath, () => createForZipFile(zipFile))
33 }

35 /**
36  * Manages creation of classpath for class files placed in zip and jar files.
37  * It should be the only way of creating them as it provides caching.
38  */
39 object ZipAndJarClassPathFactory extends ZipAndJarFileLookupFactory {
40   private case class ZipArchiveClassPath(zipFile: File)
41     extends ZipArchiveFileLookup[ClassFileEntryImpl]
42     with NoSourcePaths {
43 
44     override def findClassFile(className: String): Option[AbstractFile] = {
45       val (pkg, simpleClassName) = PackageNameUtils.separatePkgAndClassNames(className)
46       file(pkg, simpleClassName + ".class").map(_.file)
47     }
48 
49     // This method is performance sensitive as it is used by SBT's ExtractDependencies phase.
50     override def findClass(className: String): Option[ClassRepresentation] = {
51       val (pkg, simpleClassName) = PackageNameUtils.separatePkgAndClassNames(className)
52       file(pkg, simpleClassName + ".class")
53     }
54 
55     override private[dotty] def classes(inPackage: String): Seq[ClassFileEntry] = files(inPackage)
56 
57     override protected def createFileEntry(file: FileZipArchive#Entry): ClassFileEntryImpl = ClassFileEntryImpl(file)
58     override protected def isRequiredFileType(file: AbstractFile): Boolean = file.isClass
59   }

61   /**
62    * This type of classpath is closely related to the support for JSR-223.
63    * Its usage can be observed e.g. when running:
64    * jrunscript -classpath scala-compiler.jar;scala-reflect.jar;scala-library.jar -l scala
65    * with a particularly prepared scala-library.jar. It should have all classes listed in the manifest like e.g. this entry:
66    * Name: scala/Function2$mcFJD$sp.class
67    */
68   private case class ManifestResourcesClassPath(file: ManifestResources) extends ClassPath with NoSourcePaths {
69     override def findClassFile(className: String): Option[AbstractFile] = {
70       val (pkg, simpleClassName) = PackageNameUtils.separatePkgAndClassNames(className)
71       classes(pkg).find(_.name == simpleClassName).map(_.file)
72     }
73 
74     override def asClassPathStrings: Seq[String] = Seq(file.path)
75 
76     override def asURLs: Seq[URL] = file.toURLs()
77 
78     import ManifestResourcesClassPath.PackageFileInfo
79     import ManifestResourcesClassPath.PackageInfo
80 
81     /**
82      * A cache mapping package name to abstract file for package directory and subpackages of given package.
83      *
84      * ManifestResources can iterate through the collections of entries from e.g. remote jar file.
85      * We can't just specify the path to the concrete directory etc. so we can't just 'jump' into
86      * given package, when it's needed. On the other hand we can iterate over entries to get
87      * AbstractFiles, iterate over entries of these files etc.
88      *
89      * Instead of traversing a tree of AbstractFiles once and caching all entries or traversing each time,
90      * when we need subpackages of a given package or its classes, we traverse once and cache only packages.
91      * Classes for given package can be then easily loaded when they are needed.
92      */
93     private lazy val cachedPackages: collection.mutable.HashMap[String, PackageFileInfo] = {
94       val packages = collection.mutable.HashMap[String, PackageFileInfo]()
95 
96       def getSubpackages(dir: AbstractFile): List[AbstractFile] =
97         (for (file <- dir if file.isPackage) yield file)(collection.breakOut)
98 
99       @tailrec
100       def traverse(packagePrefix: String,
101                    filesForPrefix: List[AbstractFile],
102                    subpackagesQueue: collection.mutable.Queue[PackageInfo]): Unit = filesForPrefix match {
103         case pkgFile :: remainingFiles =>
104           val subpackages = getSubpackages(pkgFile)
105           val fullPkgName = packagePrefix + pkgFile.name
106           packages.put(fullPkgName, PackageFileInfo(pkgFile, subpackages))
107           val newPackagePrefix = fullPkgName + "."
108           subpackagesQueue.enqueue(PackageInfo(newPackagePrefix, subpackages))
109           traverse(packagePrefix, remainingFiles, subpackagesQueue)
110         case Nil if subpackagesQueue.nonEmpty =>
111           val PackageInfo(packagePrefix, filesForPrefix) = subpackagesQueue.dequeue()
112           traverse(packagePrefix, filesForPrefix, subpackagesQueue)
113         case _ =>
114       }
115 
116       val subpackages = getSubpackages(file)
117       packages.put(ClassPath.RootPackage, PackageFileInfo(file, subpackages))
118       traverse(ClassPath.RootPackage, subpackages, collection.mutable.Queue())
119       packages
120     }
121 
122     override private[dotty] def packages(inPackage: String): Seq[PackageEntry] = cachedPackages.get(inPackage) match {
123       case None => Seq.empty
124       case Some(PackageFileInfo(_, subpackages)) =>
125         val prefix = PackageNameUtils.packagePrefix(inPackage)
126         subpackages.map(packageFile => PackageEntryImpl(prefix + packageFile.name))
127     }
128 
129     override private[dotty] def classes(inPackage: String): Seq[ClassFileEntry] = cachedPackages.get(inPackage) match {
130       case None => Seq.empty
131       case Some(PackageFileInfo(pkg, _)) =>
132         (for (file <- pkg if file.isClass) yield ClassFileEntryImpl(file))(collection.breakOut)
133     }
134 
135     override private[dotty] def hasPackage(pkg: String) = cachedPackages.contains(pkg)
136     override private[dotty] def list(inPackage: String): ClassPathEntries = ClassPathEntries(packages(inPackage), classes(inPackage))
137   }
138 
139   private object ManifestResourcesClassPath {
140     case class PackageFileInfo(packageFile: AbstractFile, subpackages: Seq[AbstractFile])
141     case class PackageInfo(packageName: String, subpackages: List[AbstractFile])
142   }
143 
144   override protected def createForZipFile(zipFile: AbstractFile): ClassPath =
145     if (zipFile.file == null) createWithoutUnderlyingFile(zipFile)
146     else ZipArchiveClassPath(zipFile.file)
147 
148   private def createWithoutUnderlyingFile(zipFile: AbstractFile) = zipFile match {
149     case manifestRes: ManifestResources =>
150       ManifestResourcesClassPath(manifestRes)
151     case _ =>
152       val errorMsg = s"Abstract files which don't have an underlying file and are not ManifestResources are not supported. There was $zipFile"
153       throw new IllegalArgumentException(errorMsg)
154   }
155 }

177 final class FileBasedCache[T] {
178   private case class Stamp(lastModified: FileTime, fileKey: Object)
179   private val cache = collection.mutable.Map.empty[java.nio.file.Path, (Stamp, T)]
180 
181   def getOrCreate(path: java.nio.file.Path, create: () => T): T = cache.synchronized {
182     val attrs = Files.readAttributes(path, classOf[BasicFileAttributes])
183     val lastModified = attrs.lastModifiedTime()
184     // only null on some platforms, but that's okay, we just use the last modified timestamp as our stamp
185     val fileKey = attrs.fileKey()
186     val stamp = Stamp(lastModified, fileKey)
187     cache.get(path) match {
188       case Some((cachedStamp, cached)) if cachedStamp == stamp => cached
189       case _ =>
190         val value = create()
191         cache.put(path, (stamp, value))
192         value
193     }
194   }
195 
196   def clear(): Unit = cache.synchronized {
197     // TODO support closing
198     // cache.valuesIterator.foreach(_.close())
199     cache.clear()
200   }
