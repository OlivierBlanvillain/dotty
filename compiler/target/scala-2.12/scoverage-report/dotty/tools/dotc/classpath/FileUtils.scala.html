40   def stripSourceExtension(fileName: String): String = {
41     if (endsScala(fileName)) stripClassExtension(fileName)
42     else if (endsJava(fileName)) stripJavaExtension(fileName)
43     else throw new FatalError("Unexpected source file ending: " + fileName)
44   }

46   def dirPath(forPackage: String) = forPackage.replace('.', '/')
47 
48   def endsClass(fileName: String): Boolean =
49     fileName.length > 6 && fileName.substring(fileName.length - 6) == ".class"
50 
51   def endsScalaOrJava(fileName: String): Boolean =
52     endsScala(fileName) || endsJava(fileName)
53 
54   def endsJava(fileName: String): Boolean =
55     fileName.length > 5 && fileName.substring(fileName.length - 5) == ".java"
56 
57   def endsScala(fileName: String): Boolean =
58     fileName.length > 6 && fileName.substring(fileName.length - 6) == ".scala"
59 
60   def stripClassExtension(fileName: String): String =
61     fileName.substring(0, fileName.length - 6) // equivalent of fileName.length - ".class".length
62 
63   def stripJavaExtension(fileName: String): String =
64     fileName.substring(0, fileName.length - 5)
65 
66   // probably it should match a pattern like [a-z_]{1}[a-z0-9_]* but it cannot be changed
67   // because then some tests in partest don't pass
68   def mayBeValidPackage(dirName: String): Boolean =
69     (dirName != "META-INF") && (dirName != "") && (dirName.charAt(0) != '.')
70 
71   def mkFileFilter(f: JFile => Boolean) = new FileFilter {
72     def accept(pathname: JFile): Boolean = f(pathname)
73   }
