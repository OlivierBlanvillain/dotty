12 /**
13  * Provides factory methods for classpath. When creating classpath instances for a given path,
14  * it uses proper type of classpath depending on a types of particular files containing sources or classes.
15  */
16 class ClassPathFactory {
17   /**
18     * Create a new classpath based on the abstract file.
19     */
20   def newClassPath(file: AbstractFile)(implicit ctx: Context): ClassPath = ClassPathFactory.newClassPath(file)
21 
22   /**
23     * Creators for sub classpaths which preserve this context.
24     */
25   def sourcesInPath(path: String)(implicit ctx: Context): List[ClassPath] =
26     for {
27       file <- expandPath(path, expandStar = false)
28       dir <- Option(AbstractFile getDirectory file)
29     } yield createSourcePath(dir)
30 
31 
32   def expandPath(path: String, expandStar: Boolean = true): List[String] = dotty.tools.io.ClassPath.expandPath(path, expandStar)
33 
34   def expandDir(extdir: String): List[String] = dotty.tools.io.ClassPath.expandDir(extdir)
35 
36   def contentsOfDirsInPath(path: String)(implicit ctx: Context): List[ClassPath] =
37     for {
38       dir <- expandPath(path, expandStar = false)
39       name <- expandDir(dir)
40       entry <- Option(AbstractFile.getDirectory(name))
41     } yield newClassPath(entry)
42 
43   def classesInExpandedPath(path: String)(implicit ctx: Context): IndexedSeq[ClassPath] =
44     classesInPathImpl(path, expand = true).toIndexedSeq
45 
46   def classesInPath(path: String)(implicit ctx: Context) = classesInPathImpl(path, expand = false)
47 
48   def classesInManifest(useManifestClassPath: Boolean)(implicit ctx: Context) =
49     if (useManifestClassPath) dotty.tools.io.ClassPath.manifests.map(url => newClassPath(AbstractFile getResources url))
50     else Nil
51 
52   // Internal
53   protected def classesInPathImpl(path: String, expand: Boolean)(implicit ctx: Context) =
54     for {
55       file <- expandPath(path, expand)
56       dir <- {
57         def asImage = if (file.endsWith(".jimage")) Some(AbstractFile.getFile(file)) else None
58         Option(AbstractFile.getDirectory(file)).orElse(asImage)
59       }
60     } yield newClassPath(dir)
61 
62   private def createSourcePath(file: AbstractFile)(implicit ctx: Context): ClassPath =
63     if (file.isJarOrZip)
64       ZipAndJarSourcePathFactory.create(file)
65     else if (file.isDirectory)
66       new DirectorySourcePath(file.file)
67     else
68       sys.error(s"Unsupported sourcepath element: $file")
69 }
70 
71 object ClassPathFactory {
72   def newClassPath(file: AbstractFile)(implicit ctx: Context): ClassPath = file match {
73     case vd: VirtualDirectory => VirtualDirectoryClassPath(vd)
74     case _ =>
75       if (file.isJarOrZip)
76         ZipAndJarClassPathFactory.create(file)
77       else if (file.isDirectory)
78         new DirectoryClassPath(file.file)
79       else
80         sys.error(s"Unsupported classpath element: $file")
81   }
82 }
