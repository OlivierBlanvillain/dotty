12 /** A map that applies three functions and a substitution together to a tree and
13  *  makes sure they are coordinated so that the result is well-typed. The functions are
14  *  @param  typeMap  A function from Type to Type that gets applied to the
15  *                   type of every tree node and to all locally defined symbols,
16  *                   followed by the substitution [substFrom := substTo].
17  *  @param treeMap   A transformer that translates all encountered subtrees in
18  *                   prefix traversal orders
19  *  @param oldOwners Previous owners. If a top-level local symbol in the mapped tree
20  *                   has one of these as an owner, the owner is replaced by the corresponding
21  *                   symbol in `newOwners`.
22  *  @param newOwners New owners, replacing previous owners.
23  *  @param substFrom The symbols that need to be substituted.
24  *  @param substTo   The substitution targets.
25  *
26  *  The reason the substitution is broken out from the rest of the type map is
27  *  that all symbols have to be substituted at the same time. If we do not do this,
28  *  we risk data races on named types. Example: Say we have `outer#1.inner#2` and we
29  *  have two substitutions S1 = [outer#1 := outer#3], S2 = [inner#2 := inner#4] where
30  *  hashtags precede symbol ids. If we do S1 first, we get outer#2.inner#3. If we then
31  *  do S2 we get outer#2.inner#4. But that means that the named type outer#2.inner
32  *  gets two different denotations in the same period. Hence, if -Yno-double-bindings is
33  *  set, we would get a data race assertion error.
34  */

51     private def mapPrefix(from: List[Symbol], to: List[Symbol], tp: Type): Type = from match {
52       case Nil => tp
53       case (cls: ClassSymbol) :: from1 => mapPrefix(from1, to.tail, tp.substThis(cls, to.head.thisType))

147   /** The current tree map composed with a substitution [from -> to] */
148   def withSubstitution(from: List[Symbol], to: List[Symbol]): TreeTypeMap =
149     if (from eq to) this
150     else {
151       // assert that substitution stays idempotent, assuming its parts are
152       // TODO: It might be better to cater for the asserted-away conditions, by
153       // setting up a proper substitution abstraction with a compose operator that

176   /** The tree map with the substitution between originals `syms`
177    *  and mapped symbols `mapped`. Also goes into mapped classes
178    *  and substitutes their declarations.
179    */
180   def withMappedSyms(syms: List[Symbol], mapped: List[Symbol]): TreeTypeMap = {
181     val symsChanged = syms ne mapped
182     val substMap = withSubstitution(syms, mapped)
183     val fullMap = (substMap /: mapped.filter(_.isClass)) { (tmap, cls) =>
184       val origDcls = cls.info.decls.toList
185       val mappedDcls = ctx.mapSymbols(origDcls, tmap)
186       val tmap1 = tmap.withMappedSyms(origDcls, mappedDcls)
187       if (symsChanged) (origDcls, mappedDcls).zipped.foreach(cls.asClass.replace)
188       tmap1
189     }
190     if (symsChanged || (fullMap eq substMap)) fullMap
191     else withMappedSyms(syms, mapAlways = true)
192   }
