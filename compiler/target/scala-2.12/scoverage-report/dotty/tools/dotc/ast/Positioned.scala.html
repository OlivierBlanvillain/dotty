56   /** If any children of this node do not have positions,
57    *  fit their positions between the positions of the known subtrees
58    *  and transitively visit their children.
59    *  The method is likely time-critical because it is invoked on any node
60    *  we create, so we want to avoid object allocations in the common case.
61    *  The method is naturally expressed as two mutually (tail-)recursive
62    *  functions, one which computes the next element to consider or terminates if there
63    *  is none and the other which propagates the position information to that element.
64    *  But since mutual tail recursion is not supported in Scala, we express it instead
65    *  as a while loop with a termination by return in the middle.
66    */
67   private def setChildPositions(pos: Position): Unit = {
68     var n = productArity                    // subnodes are analyzed right to left
69     var elems: List[Any] = Nil              // children in lists still to be considered, from right to left
70     var end = pos.end                       // the last defined offset, fill in positions up to this offset
71     var outstanding: List[Positioned] = Nil // nodes that need their positions filled once a start position
72                                             // is known, from left to right.
73     def fillIn(ps: List[Positioned], start: Int, end: Int): Unit = ps match {
74       case p :: ps1 =>
75         // If a tree has no position or a zero-extent position, it should be
76         // synthetic. We can preserve this invariant by always setting a
77         // zero-extent position for these trees here.
78         if (!p.pos.exists || p.pos.isZeroExtent) {
79           p.setPos(Position(start, start))
80           fillIn(ps1, start, end)
81         } else {
82           p.setPos(Position(start, end))
83           fillIn(ps1, end, end)
84         }
85       case nil =>
86     }
87     while (true) {
88       var nextChild: Any = null // the next child to be considered
89       if (elems.nonEmpty) {
90         nextChild = elems.head
91         elems = elems.tail
92       }
93       else if (n > 0) {
94         n = n - 1
95         nextChild = productElement(n)
96       }
97       else {
98         fillIn(outstanding, pos.start, end)
99         return
100       }
101       nextChild match {
102         case p: Positioned =>
103           if (p.pos.exists) {
104             fillIn(outstanding, p.pos.end, end)
105             outstanding = Nil
106             end = p.pos.start
107           }
108           else outstanding = p :: outstanding
109         case xs: List[_] =>
110           elems = elems ::: xs.reverse
111         case _ =>
112       }
113     }
114   }

132   private def unionPos(pos: Position, xs: List[_]): Position = xs match {
133     case (p: Positioned) :: xs1 => unionPos(pos union p.pos, xs1)
134     case (xs0: List[_]) :: xs1 => unionPos(unionPos(pos, xs0), xs1)
135     case _ :: xs1 => unionPos(pos, xs1)
136     case _ => pos
137   }

139   def contains(that: Positioned): Boolean = {
140     def isParent(x: Any): Boolean = x match {
141       case x: Positioned =>
142         x contains that
143       case xs: List[_] =>
144         xs exists isParent
145       case _ =>
146         false
147     }

149       (this.pos contains that.pos) && {
150         var n = productArity
151         var found = false
152         while (!found && n > 0) {
153           n -= 1
154           found = isParent(productElement(n))
155         }
156         found

160   /** Check that all positioned items in this tree satisfy the following conditions:
161    *  - Parent positions contain child positions
162    *  - If item is a non-empty tree, it has a position
163    */
164   def checkPos(nonOverlapping: Boolean)(implicit ctx: Context): Unit = try {
165     import untpd._
166     var lastPositioned: Positioned = null
167     var lastPos = NoPosition
168     def check(p: Any): Unit = p match {
169       case p: Positioned =>
170         assert(pos contains p.pos,
171           s"""position error, parent position does not contain child positon
172              |parent          = $this,
173              |parent position = $pos,
174              |child           = $p,
175              |child position  = ${p.pos}""".stripMargin)
176         p match {
177           case tree: Tree if !tree.isEmpty =>
178             assert(tree.pos.exists,
179               s"position error: position not set for $tree # ${tree.uniqueId}")
180           case _ =>
181         }
182         if (nonOverlapping) {
183           this match {
184             case _: XMLBlock =>
185               // FIXME: Trees generated by the XML parser do not satisfy `checkPos`
186             case _: WildcardFunction
187             if lastPositioned.isInstanceOf[ValDef] && !p.isInstanceOf[ValDef] =>
188               // ignore transition from last wildcard parameter to body
189             case _ =>
190               assert(!lastPos.exists || !p.pos.exists || lastPos.end <= p.pos.start,
191                 s"""position error, child positions overlap or in wrong order
192                    |parent             = $this
193                    |1st child          = $lastPositioned
194                    |1st child position = $lastPos
195                    |2nd child          = $p
196                    |2nd child position = ${p.pos}""".stripMargin)
197           }
198           lastPositioned = p
199           lastPos = p.pos
200         }
201         p.checkPos(nonOverlapping)
202       case xs: List[_] =>
203         xs.foreach(check)
204       case _ =>
205     }
206     this match {
207       case tree: DefDef if tree.name == nme.CONSTRUCTOR && tree.mods.is(JavaDefined) =>
208         // Special treatment for constructors coming from Java:
209         // Leave out tparams, they are copied with wrong positions from parent class
210         check(tree.mods)
211         check(tree.vparamss)
212       case _ =>
213         val end = productArity
214         var n = 0
215         while (n < end) {
216           check(productElement(n))
217           n += 1
218         }
219     }
220   } catch {
221     case ex: AssertionError =>
222       println(i"error while checking $this")
223       throw ex
224   }
