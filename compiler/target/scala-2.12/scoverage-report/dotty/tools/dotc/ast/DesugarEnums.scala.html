41   /** A reference to the enum class `E`, possibly followed by type arguments.
42    *  Each covariant type parameter is approximated by its lower bound.
43    *  Each contravariant type parameter is approximated by its upper bound.
44    *  It is an error if a type parameter is non-variant, or if its approximation
45    *  refers to pther type parameters.
46    */
47   def interpolatedEnumParent(pos: Position)(implicit ctx: Context): Tree = {
48     val tparams = enumClass.typeParams
49     def isGround(tp: Type) = tp.subst(tparams, tparams.map(_ => NoType)) eq tp
50     val targs = tparams map { tparam =>
51       if (tparam.variance > 0 && isGround(tparam.info.bounds.lo))
52         tparam.info.bounds.lo
53       else if (tparam.variance < 0 && isGround(tparam.info.bounds.hi))
54         tparam.info.bounds.hi
55       else {
56         def problem =
57           if (tparam.variance == 0) "is non variant"
58           else "has bounds that depend on a type parameter in the same parameter list"
59         errorType(i"""cannot determine type argument for enum parent $enumClass,
60                      |type parameter $tparam $problem""", pos)
61       }
62     }
63     TypeTree(enumClass.typeRef.appliedTo(targs)).withPos(pos)
64   }

123   /** The return type of an enum case apply method and any widening methods in which
124    *  the apply's right hand side will be wrapped. For parents of the form
125    *
126    *      extends E(args) with T1(args1) with ... TN(argsN)
127    *
128    *  and type parameters `tparams` the generated widen method is
129    *
130    *      def C$to$E[tparams](x$1: E[tparams] with T1 with ... TN) = x$1
131    *
132    *  @param cdef            The case definition
133    *  @param parents         The declared parents of the enum case
134    *  @param tparams         The type parameters of the enum case
135    *  @param appliedEnumRef  The enum class applied to `tparams`.
136    */

161           tpt = TypeTree(),

163         (TypeTree(), widenDef :: Nil)
164       case _ =>

169   /** A pair consisting of
170    *   - the next enum tag
171    *   - scaffolding containing the necessary definitions for singleton enum cases
172    *     unless that scaffolding was already generated by a previous call to `nextEnumKind`.
173    */
174   def nextEnumTag(kind: CaseKind.Value)(implicit ctx: Context): (Int, List[Tree]) = {
175     val (count, seenKind) = ctx.tree.removeAttachment(EnumCaseCount).getOrElse((0, CaseKind.Class))
176     val minKind = if (kind < seenKind) kind else seenKind
177     ctx.tree.pushAttachment(EnumCaseCount, (count + 1, minKind))
178     val scaffolding =
179       if (enumClass.typeParams.nonEmpty || kind >= seenKind) Nil
180       else if (kind == CaseKind.Object) enumScaffolding
181       else if (seenKind == CaseKind.Object) enumValueCreator :: Nil
182       else enumScaffolding :+ enumValueCreator
183     (count, scaffolding)
184   }

195   /** Expand a module definition representing a parameterless enum case */
196   def expandEnumModule(name: TermName, impl: Template, mods: Modifiers, pos: Position)(implicit ctx: Context): Tree =
197     if (impl.parents.isEmpty)
198       if (impl.body.isEmpty)
199         expandSimpleEnumCase(name, mods, pos)
200       else {
201         val parent = interpolatedEnumParent(pos)
202         expandEnumModule(name, cpy.Template(impl)(parents = parent :: Nil), mods, pos)
203       }
204     else {
205       def toStringMeth =
206         DefDef(nme.toString_, Nil, Nil, TypeTree(defn.StringType), Literal(Constant(name.toString)))
207           .withFlags(Override)
208       val (tagMeth, scaffolding) = enumTagMeth(CaseKind.Object)
209       val impl1 = cpy.Template(impl)(body =
210         impl.body ++ List(tagMeth, toStringMeth) ++ registerCall)
211       val vdef = ValDef(name, TypeTree(), New(impl1)).withMods(mods | Final)
212       flatTree(scaffolding ::: vdef :: Nil).withPos(pos)
213     }
