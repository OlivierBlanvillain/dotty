299   def New(tpt: Tree, argss: List[List[Tree]])(implicit ctx: Context): Tree = {
300     val (tycon, targs) = tpt match {
301       case AppliedTypeTree(tycon, targs) =>
302         (tycon, targs)
303       case TypedSplice(AppliedTypeTree(tycon, targs)) =>
304         (TypedSplice(tycon), targs map (TypedSplice(_)))
305       case TypedSplice(tpt1: tpd.Tree) =>
306         val tycon = tpt1.tpe.typeConstructor
307         val argTypes = tpt1.tpe.argTypesLo
308         def wrap(tpe: Type) = TypeTree(tpe) withPos tpt.pos
309         (wrap(tycon), argTypes map wrap)
310       case _ =>
311         (tpt, Nil)
312     }
313     var prefix: Tree = Select(New(tycon), nme.CONSTRUCTOR)
314     if (targs.nonEmpty) prefix = TypeApply(prefix, targs)
315     ensureApplied((prefix /: argss)(Apply(_, _)))
316   }

490   abstract class UntypedTreeMap(cpy: UntypedTreeCopier = untpd.cpy) extends TreeMap(cpy) {
491     override def transform(tree: Tree)(implicit ctx: Context): Tree = tree match {
492       case ModuleDef(name, impl) =>
493         cpy.ModuleDef(tree)(name, transformSub(impl))
494       case ParsedTry(expr, handler, finalizer) =>
495         cpy.ParsedTry(tree)(transform(expr), transform(handler), transform(finalizer))
496       case SymbolLit(str) =>

498       case InterpolatedString(id, segments) =>
499         cpy.InterpolatedString(tree)(id, segments.mapConserve(transform))
500       case Function(args, body) =>
501         cpy.Function(tree)(transform(args), transform(body))
502       case InfixOp(left, op, right) =>
503         cpy.InfixOp(tree)(transform(left), op, transform(right))
504       case PostfixOp(od, op) =>
505         cpy.PostfixOp(tree)(transform(od), op)
506       case PrefixOp(op, od) =>

508       case Parens(t) =>
509         cpy.Parens(tree)(transform(t))
510       case Tuple(trees) =>
511         cpy.Tuple(tree)(transform(trees))
512       case Throw(expr) =>
513         cpy.Throw(tree)(transform(expr))
514       case Quote(expr) =>
515         cpy.Quote(tree)(transform(expr))
516       case WhileDo(cond, body) =>
