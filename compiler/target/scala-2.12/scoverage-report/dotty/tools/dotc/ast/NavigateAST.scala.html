12   /** The untyped tree corresponding to typed tree `tree` in the compilation
13    *  unit specified by `ctx`
14    */
15   def toUntyped(tree: tpd.Tree)(implicit ctx: Context): untpd.Tree =
16     untypedPath(tree, exactMatch = true) match {
17       case (utree: untpd.Tree) :: _ =>
18         utree
19       case _ =>
20         val loosePath = untypedPath(tree, exactMatch = false)
21         throw new
22           Error(i"""no untyped tree for $tree, pos = ${tree.pos}
23                    |best matching path =\n$loosePath%\n====\n%
24                    |path positions = ${loosePath.map(_.pos)}""")
25     }

27   /** The reverse path of untyped trees starting with a tree that closest matches
28    *  `tree` and ending in the untyped tree at the root of the compilation unit
29    *  specified by `ctx`.
30    *  @param exactMatch    If `true`, the path must start with a node that exactly
31    *                       matches `tree`, or `Nil` is returned.
32    *                       If `false` the path might start with a node enclosing
33    *                       the logical position of `tree`.
34    *  Note: A complication concerns member definitions. ValDefs and DefDefs
35    *  have after desugaring a position that spans just the name of the symbol being
36    *  defined and nothing else. So we look instead for an untyped tree approximating the
37    *  envelope of the definition, and declare success if we find another DefTree.
38    */
39   def untypedPath(tree: tpd.Tree, exactMatch: Boolean = false)(implicit ctx: Context): List[Positioned] =
40     tree match {
41       case tree: MemberDef[_] =>
42         untypedPath(tree.pos) match {
43           case path @ (last: DefTree[_]) :: _ => path
44           case path if !exactMatch => path
45           case _ => Nil
46         }
47       case _ =>
48         untypedPath(tree.pos) match {
49           case (path @ last :: _) if last.pos == tree.pos || !exactMatch => path
50           case _ => Nil
51         }
52     }
53 
54   /** The reverse part of the untyped root of the compilation unit of `ctx` to
55    *  position `pos`.
56    */
57   def untypedPath(pos: Position)(implicit ctx: Context): List[Positioned] =
58     pathTo(pos, ctx.compilationUnit.untpdTree)

61   /** The reverse path from node `from` to the node that closest encloses position `pos`,
62    *  or `Nil` if no such path exists. If a non-empty path is returned it starts with
63    *  the node closest enclosing `pos` and ends with `from`.
64    *
65    *  @param skipZeroExtent  If true, skip over zero-extent nodes in the search. These nodes
66    *                         do not correspond to code the user wrote since their start and
67    *                         end point are the same, so this is useful when trying to reconcile
68    *                         nodes with source code.
69    */
70   def pathTo(pos: Position, from: Positioned, skipZeroExtent: Boolean = false)(implicit ctx: Context): List[Positioned] = {
71     def childPath(it: Iterator[Any], path: List[Positioned]): List[Positioned] = {
72       while (it.hasNext) {
73         val path1 = it.next() match {
74           case p: Positioned => singlePath(p, path)
75           case xs: List[_] => childPath(xs.iterator, path)
76           case _ => path
77         }
78         if (path1 ne path) return path1
79       }
80       path
81     }
82     def singlePath(p: Positioned, path: List[Positioned]): List[Positioned] =
83       if (p.pos.exists && !(skipZeroExtent && p.pos.isZeroExtent) && p.pos.contains(pos)) {
84         // FIXME: We shouldn't be manually forcing trees here, we should replace
85         // our usage of `productIterator` by something in `Positioned` that takes
86         // care of low-level details like this for us.
87         p match {
88           case p: WithLazyField[_] =>
89             p.forceIfLazy
90           case _ =>
91         }
92         childPath(p.productIterator, p :: path)
93       } else path
94     singlePath(from, Nil)
95   }
