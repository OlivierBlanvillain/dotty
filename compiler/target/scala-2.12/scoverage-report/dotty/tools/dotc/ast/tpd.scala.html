221   def ClassDef(cls: ClassSymbol, constr: DefDef, body: List[Tree], superArgs: List[Tree] = Nil)(implicit ctx: Context): TypeDef = {
222     val firstParent :: otherParents = cls.info.parents
223     val superRef =
224       if (cls is Trait) TypeTree(firstParent)
225       else {
226         def isApplicable(ctpe: Type): Boolean = ctpe match {
227           case ctpe: PolyType =>
228             isApplicable(ctpe.instantiate(firstParent.argTypes))
229           case ctpe: MethodType =>
230             (superArgs corresponds ctpe.paramInfos)(_.tpe <:< _)
231           case _ =>
232             false
233         }
234         val constr = firstParent.decl(nme.CONSTRUCTOR).suchThat(constr => isApplicable(constr.info))
235         New(firstParent, constr.symbol.asTerm, superArgs)
236       }
237     val parents = superRef :: otherParents.map(TypeTree(_))
238 
239     val selfType =
240       if (cls.classInfo.selfInfo ne NoType) ValDef(ctx.newSelfSym(cls))
241       else EmptyValDef
242     def isOwnTypeParam(stat: Tree) =
243       (stat.symbol is TypeParam) && stat.symbol.owner == cls
244     val bodyTypeParams = body filter isOwnTypeParam map (_.symbol)
245     val newTypeParams =
246       for (tparam <- cls.typeParams if !(bodyTypeParams contains tparam))
247       yield TypeDef(tparam)
248     val findLocalDummy = new FindLocalDummyAccumulator(cls)
249     val localDummy = ((NoSymbol: Symbol) /: body)(findLocalDummy.apply)
250       .orElse(ctx.newLocalDummy(cls))
251     val impl = untpd.Template(constr, parents, selfType, newTypeParams ++ body)
252       .withType(localDummy.termRef)
253     ta.assignType(untpd.TypeDef(cls.name, impl), cls)
254   }

257    *
258    *      new parents { forwarders }
259    *
260    *  where `forwarders` contains forwarders for all functions in `fns`.
261    *  @param parents    a non-empty list of class types
262    *  @param fns        a non-empty of functions for which forwarders should be defined in the class.

266   def AnonClass(parents: List[Type], fns: List[TermSymbol], methNames: List[TermName])(implicit ctx: Context): Block = {
267     val owner = fns.head.owner
268     val parents1 =
269       if (parents.head.classSymbol.is(Trait)) parents.head.parents.head :: parents
270       else parents
271     val cls = ctx.newNormalizedClassSymbol(owner, tpnme.ANON_CLASS, Synthetic, parents1,
272         coord = fns.map(_.pos).reduceLeft(_ union _))
273     val constr = ctx.newConstructor(cls, Synthetic, Nil, Nil).entered
274     def forwarder(fn: TermSymbol, name: TermName) = {
275       val fwdMeth = fn.copy(cls, name, Synthetic | Method).entered.asTerm
276       DefDef(fwdMeth, prefss => ref(fn).appliedToArgss(prefss))
277     }
278     val forwarders = (fns, methNames).zipped.map(forwarder)
279     val cdef = ClassDef(cls, DefDef(constr), forwarders)
280     Block(cdef :: Nil, New(cls.typeRef, Nil))
281   }

404    *     <module> class obj$ extends parents { this: obj.type => decls }

407    *  What's interesting here is that the block is well typed
408    *  (because class obj$ is hoistable), but the type of the `obj` val is
409    *  not expressible. What needs to happen in general when
410    *  inferring the type of a val from its RHS, is: if the type contains

932       }
