332   /** The purity level of this expression.
333    *  @return   SimplyPure  if expression has no side effects and cannot contain local definitions
334    *            Pure        if expression has no side effects
335    *            Idempotent  if running the expression a second time has no side effects
336    *            Impure      otherwise
337    *
338    *  Note that purity and idempotency are different. References to modules and lazy
339    *  vals are impure (side-effecting) both because side-effecting code may be executed and because the first reference
340    *  takes a different code path than all to follow; but they are idempotent
341    *  because running the expression a second time gives the cached result.
342    */
343   private def exprPurity(tree: Tree)(implicit ctx: Context): PurityLevel = unsplice(tree) match {
344     case EmptyTree
345        | This(_)
346        | Super(_, _)
347        | Literal(_)
348        | Closure(_, _, _) =>
349       SimplyPure
350     case Ident(_) =>
351       refPurity(tree)
352     case Select(qual, _) =>
353       refPurity(tree).min(exprPurity(qual))
354     case TypeApply(fn, _) =>
355       exprPurity(fn)
356 /*
357  * Not sure we'll need that. Comment out until we find out
358     case Apply(Select(free @ Ident(_), nme.apply), _) if free.symbol.name endsWith nme.REIFY_FREE_VALUE_SUFFIX =>
359       // see a detailed explanation of this trick in `GenSymbols.reifyFreeTerm`
360       free.symbol.hasStableFlag && isIdempotentExpr(free)
361 */
362     case Apply(fn, args) =>
363       def isKnownPureOp(sym: Symbol) =
364         sym.owner.isPrimitiveValueClass || sym.owner == defn.StringClass
365       // Note: After uncurry, field accesses are represented as Apply(getter, Nil),
366       // so an Apply can also be pure.
367       if (args.isEmpty && fn.symbol.is(Stable)) exprPurity(fn)
368       else if (tree.tpe.isInstanceOf[ConstantType] && isKnownPureOp(tree.symbol))
369         // A constant expression with pure arguments is pure.
370         minOf(exprPurity(fn), args.map(exprPurity)) `min` Pure
371       else Impure
372     case Typed(expr, _) =>
373       exprPurity(expr)
374     case Block(stats, expr) =>
375       minOf(exprPurity(expr), stats.map(statPurity))
376     case Inlined(_, bindings, expr) =>
377       minOf(exprPurity(expr), bindings.map(statPurity))
378     case NamedArg(_, expr) =>
379       exprPurity(expr)
380     case _ =>
381       Impure
382   }
