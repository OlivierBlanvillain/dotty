41   /** Trees take a parameter indicating what the type of their `tpe` field
42    *  is. Two choices: `Type` or `Untyped`.
43    *  Untyped trees have type `Tree[Untyped]`.
44    *
45    *  Tree typing uses a copy-on-write implementation:
46    *
47    *   - You can never observe a `tpe` which is `null` (throws an exception)
48    *   - So when creating a typed tree with `withType` we can re-use
49    *     the existing tree transparently, assigning its `tpe` field,
50    *     provided it was `null` before.
51    *   - It is impossible to embed untyped trees in typed ones.
52    *   - Typed trees can be embedded in untyped ones provided they are rooted
53    *     in a TypedSplice node.
54    *   - Type checking an untyped tree should remove all embedded `TypedSplice`
55    *     nodes.
56    */

126     /** Check that typed trees don't refer to untyped ones, except if
127      *   - the parent tree is an import, or
128      *   - the child tree is an identifier, or
129      *   - errors were reported
130      */
131     private def checkChildrenTyped(it: Iterator[Any])(implicit ctx: Context): Unit =
132       if (!this.isInstanceOf[Import[_]])
133         while (it.hasNext)
134           it.next() match {
135             case x: Ident[_] => // untyped idents are used in a number of places in typed trees
136             case x: Tree[_] =>
137               assert(x.hasType || ctx.reporter.errorsReported,
138                      s"$this has untyped child $x")
139             case xs: List[_] => checkChildrenTyped(xs.iterator)
140             case _ =>
141           }

766   def flatten[T >: Untyped](trees: List[Tree[T]]): List[Tree[T]] = {
767     var buf: ListBuffer[Tree[T]] = null
768     var xs = trees
769     while (xs.nonEmpty) {
770       xs.head match {
771         case Thicket(elems) =>
772           if (buf == null) {
773             buf = new ListBuffer
774             var ys = trees
775             while (ys ne xs) {
776               buf += ys.head
777               ys = ys.tail
778             }
779           }
780           for (elem <- elems) {
781             assert(!elem.isInstanceOf[Thicket[_]])
782             buf += elem
783           }
784         case tree =>
785           if (buf != null) buf += tree
786       }
787       xs = xs.tail
788     }
789     if (buf != null) buf.toList else trees
790   }
