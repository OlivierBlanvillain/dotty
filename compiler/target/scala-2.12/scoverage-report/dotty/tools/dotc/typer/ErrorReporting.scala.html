112     def whyNoMatchStr(found: Type, expected: Type) = {
113       def dropJavaMethod(tp: Type): Type = tp match {
114         case tp: PolyType =>
115           tp.derivedLambdaType(resType = dropJavaMethod(tp.resultType))
116         case tp: MethodType if tp.isJavaMethod =>
117           MethodType(tp.paramNames, tp.paramInfos, dropJavaMethod(tp.resultType))
118         case tp => tp
119       }
120       val found1 = dropJavaMethod(found)
121       val expected1 = dropJavaMethod(expected)
122       if ((found1 eq found) != (expected eq expected1) && (found1 <:< expected1))
123         "\n(Note that Scala's and Java's representation of this type differs)"
124       else if (found.topType != expected.topType)
125         "\n(Note that the types are in different universes, see Phantom types)"
126       else if (ctx.settings.explainTypes.value)

132     def typeMismatchMsg(found: Type, expected: Type, postScript: String = "") = {
133       // replace constrained TypeParamRefs and their typevars by their bounds where possible
134       // the idea is that if the bounds are also not-subtypes of each other to report
135       // the type mismatch on the bounds instead of the original TypeParamRefs, since
136       // these are usually easier to analyze.
137       object reported extends TypeMap {
138         def setVariance(v: Int) = variance = v
139         val constraint = ctx.typerState.constraint
140         def apply(tp: Type): Type = tp match {
141           case tp: TypeParamRef =>
142             constraint.entry(tp) match {
143               case bounds: TypeBounds =>
144                 if (variance < 0) apply(constraint.fullUpperBound(tp))
145                 else if (variance > 0) apply(constraint.fullLowerBound(tp))
146                 else tp
147               case NoType => tp
148               case instType => apply(instType)
149             }
150           case tp: TypeVar => apply(tp.stripTypeVar)
151           case _ => mapOver(tp)
152         }
153       }
154       val found1 = reported(found)
155       reported.setVariance(-1)
156       val expected1 = reported(expected)
157       val (found2, expected2) =
158         if (found1 frozen_<:< expected1) (found, expected) else (found1, expected1)
159       TypeMismatch(found2, expected2, whyNoMatchStr(found, expected), postScript)
160     }
