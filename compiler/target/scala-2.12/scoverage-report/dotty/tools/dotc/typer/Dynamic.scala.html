25 /** Handles programmable member selections of `Dynamic` instances and values
26  *  with structural types. Two functionalities:
27  *
28  * 1. Translates selection that does not typecheck according to the scala.Dynamic rules:
29  *    foo.bar(baz) = quux                   ~~> foo.selectDynamic(bar).update(baz, quux)
30  *    foo.bar = baz                         ~~> foo.updateDynamic("bar")(baz)
31  *    foo.bar(x = bazX, y = bazY, baz, ...) ~~> foo.applyDynamicNamed("bar")(("x", bazX), ("y", bazY), ("", baz), ...)
32  *    foo.bar(baz0, baz1, ...)              ~~> foo.applyDynamic(bar)(baz0, baz1, ...)
33  *    foo.bar                               ~~> foo.selectDynamic(bar)
34  *
35  *  The first matching rule of is applied.
36  *
37  * 2. Translates member selections on structural types to calls of `selectDynamic`
38  *    or `selectDynamicMethod` on a `Selectable` instance. @See handleStructural.
39  *
40  */

45   /** Translate selection that does not typecheck according to the normal rules into a applyDynamic/applyDynamicNamed.
46    *    foo.bar(baz0, baz1, ...)                       ~~> foo.applyDynamic(bar)(baz0, baz1, ...)
47    *    foo.bar[T0, ...](baz0, baz1, ...)              ~~> foo.applyDynamic[T0, ...](bar)(baz0, baz1, ...)
48    *    foo.bar(x = bazX, y = bazY, baz, ...)          ~~> foo.applyDynamicNamed("bar")(("x", bazX), ("y", bazY), ("", baz), ...)
49    *    foo.bar[T0, ...](x = bazX, y = bazY, baz, ...) ~~> foo.applyDynamicNamed[T0, ...]("bar")(("x", bazX), ("y", bazY), ("", baz), ...)
50    */
51   def typedDynamicApply(tree: untpd.Apply, pt: Type)(implicit ctx: Context): Tree = {
52     def typedDynamicApply(qual: untpd.Tree, name: Name, targs: List[untpd.Tree]): Tree = {
53       def isNamedArg(arg: untpd.Tree): Boolean = arg match { case NamedArg(_, _) => true; case _ => false }
54       val args = tree.args
55       val dynName = if (args.exists(isNamedArg)) nme.applyDynamicNamed else nme.applyDynamic
56       if (dynName == nme.applyDynamicNamed && untpd.isWildcardStarArgList(args))
57         errorTree(tree, "applyDynamicNamed does not support passing a vararg parameter")
58       else {
59         def namedArgTuple(name: String, arg: untpd.Tree) = untpd.Tuple(List(Literal(Constant(name)), arg))
60         def namedArgs = args.map {
61           case NamedArg(argName, arg) => namedArgTuple(argName.toString, arg)
62           case arg => namedArgTuple("", arg)
63         }
64         val args1 = if (dynName == nme.applyDynamic) args else namedArgs
65         typedApply(untpd.Apply(coreDynamic(qual, dynName, name, targs), args1), pt)
66       }
67     }
68 
69      tree.fun match {
70       case Select(qual, name) if !isDynamicMethod(name) =>
71         typedDynamicApply(qual, name, Nil)
72       case TypeApply(Select(qual, name), targs) if !isDynamicMethod(name) =>
73         typedDynamicApply(qual, name, targs)
74       case TypeApply(fun, targs) =>
75         typedDynamicApply(fun, nme.apply, targs)
76       case fun =>
77         typedDynamicApply(fun, nme.apply, Nil)
78     }
79   }

81   /** Translate selection that does not typecheck according to the normal rules into a selectDynamic.
82    *    foo.bar          ~~> foo.selectDynamic(bar)
83    *    foo.bar[T0, ...] ~~> foo.selectDynamic[T0, ...](bar)
84    *
85    *  Note: inner part of translation foo.bar(baz) = quux ~~> foo.selectDynamic(bar).update(baz, quux) is achieved
86    *  through an existing transformation of in typedAssign [foo.bar(baz) = quux ~~> foo.bar.update(baz, quux)].
87    */

95     def typedDynamicAssign(qual: untpd.Tree, name: Name, targs: List[untpd.Tree]): Tree =

107   private def coreDynamic(qual: untpd.Tree, dynName: Name, name: Name, targs: List[untpd.Tree])(implicit ctx: Context): untpd.Apply = {
108     val select = untpd.Select(qual, dynName)
109     val selectWithTypes =
110       if (targs.isEmpty) select
111       else untpd.TypeApply(select, targs)
112     untpd.Apply(selectWithTypes, Literal(Constant(name.toString)))
113   }

115   /** Handle reflection-based dispatch for members of structural types.
116    *  Given `x.a`, where `x` is of (widened) type `T` and `x.a` is of type `U`:
117    *
118    *  If `U` is a value type, map `x.a` to the equivalent of:
119    *
120    *     (x: Selectable).selectDynamic(x, "a").asInstanceOf[U]
121    *
122    *  If `U` is a method type (T1,...,Tn)R, map `x.a` to the equivalent of:
123    *
124    *     (x: Selectable).selectDynamicMethod("a", CT1, ..., CTn).asInstanceOf[(T1,...,Tn) => R]
125    *
126    *  where CT1,...,CTn are the class tags representing the erasure of T1,...,Tn.
127    *
128    *  It's an error if U is neither a value nor a method type, or a dependent method
129    *  type, or of too large arity (limit is Definitions.MaxStructuralMethodArity).
130    */
131   def handleStructural(tree: Tree)(implicit ctx: Context): Tree = {
132     val Select(qual, name) = tree
133 
134     def structuralCall(selectorName: TermName, formals: List[Tree]) = {
135       val selectable = adapt(qual, defn.SelectableType)
136       val scall = untpd.Apply(
137         untpd.TypedSplice(selectable.select(selectorName)),
138         (Literal(Constant(name.toString)) :: formals).map(untpd.TypedSplice(_)))
139       typed(scall)
140     }
141 
142     def fail(reason: String) =
143       errorTree(tree, em"Structural access not allowed on method $name because it $reason")
144 
145     tree.tpe.widen match {
146       case tpe: MethodType =>
147         if (tpe.isParamDependent)
148           fail(i"has a method type with inter-parameter dependencies")
149         else if (tpe.paramNames.length > Definitions.MaxStructuralMethodArity)
150           fail(i"""takes too many parameters.
151                   |Structural types only support methods taking up to ${Definitions.MaxStructuralMethodArity} arguments""")
152         else {
153           val ctags = tpe.paramInfos.map(pt =>
154             implicitArgTree(defn.ClassTagType.appliedTo(pt :: Nil), tree.pos.endPos))
155           structuralCall(nme.selectDynamicMethod, ctags).asInstance(tpe.toFunctionType())
156         }
157       case tpe: ValueType =>
158         structuralCall(nme.selectDynamic, Nil).asInstance(tpe)
159       case tpe: PolyType =>
160         fail("is polymorphic")
161       case tpe =>
162         fail(i"has an unsupported type: $tpe")
163     }
164   }
