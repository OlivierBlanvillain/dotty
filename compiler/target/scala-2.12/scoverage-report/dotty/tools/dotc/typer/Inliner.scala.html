416     // make sure prefix is executed if it is impure
417     if (!isIdempotentExpr(prefix)) registerType(meth.owner.thisType)
418 
419     // Register types of all leaves of inlined body so that the `paramProxy` and `thisProxy` maps are defined.
420     rhs.foreachSubTree(registerLeaf)

422     // The class that the this-proxy `selfSym` represents
423     def classOf(selfSym: Symbol) = selfSym.info.widen.classSymbol
424 
425     // The total nesting depth of the class represented by `selfSym`.
426     def outerLevel(selfSym: Symbol): Int = classOf(selfSym).ownersIterator.length
427 
428     // All needed this-proxies, paired-with and sorted-by nesting depth of
429     // the classes they represent (innermost first)
430     val sortedProxies = thisProxy.toList.map {
431       case (cls, proxy) => (outerLevel(cls), proxy.symbol)
432     } sortBy (-_._1)
433 
434     // Compute val-definitions for all this-proxies and append them to `bindingsBuf`
435     var lastSelf: Symbol = NoSymbol
436     var lastLevel: Int = 0
437     for ((level, selfSym) <- sortedProxies) {
438       lazy val rhsClsSym = selfSym.info.widenDealias.classSymbol
439       val rhs =
440         if (lastSelf.exists)
441           ref(lastSelf).outerSelect(lastLevel - level, selfSym.info)
442         else if (rhsClsSym.is(Module) && rhsClsSym.isStatic)
443           ref(rhsClsSym.sourceModule)
444         else
445           prefix
446       bindingsBuf += ValDef(selfSym.asTerm, rhs)
447       inlining.println(i"proxy at $level: $selfSym = ${bindingsBuf.last}")
448       lastSelf = selfSym
449       lastLevel = level
450     }

452     // The type map to apply to the inlined tree. This maps references to this-types
453     // and parameters to type references of their arguments or proxies.
454     val typeMap = new TypeMap {
455       def apply(t: Type) = t match {
456         case t: ThisType => thisProxy.getOrElse(t.cls, t)
457         case t: TypeRef => paramProxy.getOrElse(t, mapOver(t))
458         case t: SingletonType => paramProxy.getOrElse(t, mapOver(t))
459         case t => mapOver(t)
460       }
461       override def mapClassInfo(tp: ClassInfo) = mapFullClassInfo(tp)
462     }

465     // and parameters to references of their arguments or their proxies.

467       tree match {
468       case _: This =>
469         tree.tpe match {
470           case thistpe: ThisType =>
471             thisProxy.get(thistpe.cls) match {
472               case Some(t) => ref(t).withPos(tree.pos)
473               case None => tree
474             }
475           case _ => tree
476         }
477       case _: Ident =>
478         paramProxy.get(tree.tpe) match {
479           case Some(t) if tree.isTerm && t.isSingleton => singleton(t).withPos(tree.pos)
480           case Some(t) if tree.isType => TypeTree(t).withPos(tree.pos)
481           case None => tree
482         }
483       case _ => tree
484     }}
485 
486     val inlineCtx = inlineContext(call)
487     // The complete translation maps references to `this` and parameters to
488     // corresponding arguments or proxies on the type and term level. It also changes
489     // the owner from the inlined method to the current owner.
490     val inliner = new TreeTypeMap(typeMap, treeMap, meth :: Nil, ctx.owner :: Nil)(inlineCtx)
491 
492     val expansion = inliner(rhs.withPos(call.pos))
493     trace(i"inlining $call\n, BINDINGS =\n${bindingsBuf.toList}%\n%\nEXPANSION =\n$expansion", inlining, show = true) {
494 
495       // The final expansion runs a typing pass over the inlined tree. See InlineTyper for details.
496       val expansion1 = InlineTyper.typed(expansion, pt)(inlineCtx)
497 
498       /** Does given definition bind a closure that will be inlined? */

501         case _ => false
502       }
503 
504       /** All bindings in `bindingsBuf` except bindings of inlineable closures */
505       val bindings = bindingsBuf.toList.filterNot(bindsDeadInlineable).map(_.withPos(call.pos))
506 
507       tpd.Inlined(call, bindings, expansion1)
508     }
