74   /** The accumulator which forces type variables using the policy encoded in `force`
75    *  and returns whether the type is fully defined. The direction in which
76    *  a type variable is instantiated is determined as follows:
77    *   1. T is minimized if the constraint over T is only from below (i.e.
78    *      constrained lower bound != given lower bound and
79    *      constrained upper bound == given upper bound).
80    *   2. T is maximized if the constraint over T is only from above (i.e.
81    *      constrained upper bound != given upper bound and
82    *      constrained lower bound == given lower bound).
83    *  If (1) and (2) do not apply:
84    *   3. T is maximized if it appears only contravariantly in the given type.
85    *   4. T is minimized in all other cases.
86    *
87    *  The instantiation is done in two phases:
88    *  1st Phase: Try to instantiate minimizable type variables to
89    *  their lower bound. Record whether successful.
90    *  2nd Phase: If first phase was successful, instantiate all remaining type variables
91    *  to their upper bound.
92    */
93   private class IsFullyDefinedAccumulator(force: ForceDegree.Value)(implicit ctx: Context) extends TypeAccumulator[Boolean] {
94     private def instantiate(tvar: TypeVar, fromBelow: Boolean): Type = {
95       val inst = tvar.instantiate(fromBelow)
96       typr.println(i"forced instantiation of ${tvar.origin} = $inst")
97       inst
98     }
99     private[this] var toMaximize: Boolean = false
100     def apply(x: Boolean, tp: Type): Boolean = tp.dealias match {
101       case _: WildcardType | _: ProtoType =>
102         false
103       case tvar: TypeVar
104       if !tvar.isInstantiated && ctx.typerState.constraint.contains(tvar) =>
105         force.appliesTo(tvar) && {
106           val direction = instDirection(tvar.origin)
107           if (direction != 0) {
108             //if (direction > 0) println(s"inst $tvar dir = up")
109             instantiate(tvar, direction < 0)
110           }
111           else {
112             val minimize =
113               force.minimizeAll ||
114               variance >= 0 && !(
115                 force == ForceDegree.noBottom &&
116                 defn.isBottomType(ctx.typeComparer.approximation(tvar.origin, fromBelow = true)))
117             if (minimize) instantiate(tvar, fromBelow = true)
118             else toMaximize = true
119           }
120           foldOver(x, tvar)
121         }
122       case tp =>
123         foldOver(x, tp)
124     }
125 
126     private class UpperInstantiator(implicit ctx: Context) extends TypeAccumulator[Unit] {
127       def apply(x: Unit, tp: Type): Unit = {
128         tp match {
129           case tvar: TypeVar if !tvar.isInstantiated =>
130             instantiate(tvar, fromBelow = false)
131           case _ =>
132         }
133         foldOver(x, tp)
134       }
135     }

137     def process(tp: Type): Boolean = {
138       val res = apply(true, tp)
139       if (res && toMaximize) new UpperInstantiator().apply((), tp)
140       res
141     }

156   /** The list of uninstantiated type variables bound by some prefix of type `T` which
157    *  occur in at least one formal parameter type of a prefix application.
158    *  Considered prefixes are:
159    *    - The function `f` of an application node `f(e1, .., en)`
160    *    - The function `f` of a type application node `f[T1, ..., Tn]`
161    *    - The prefix `p` of a selection `p.f`.
162    *    - The result expression `e` of a block `{s1; .. sn; e}`.
163    */
164   def tvarsInParams(tree: Tree)(implicit ctx: Context): List[TypeVar] = {
165     @tailrec def boundVars(tree: Tree, acc: List[TypeVar]): List[TypeVar] = tree match {
166       case Apply(fn, _) => boundVars(fn, acc)
167       case TypeApply(fn, targs) =>
168         val tvars = targs.tpes.collect {
169           case tvar: TypeVar if !tvar.isInstantiated => tvar
170         }
171         boundVars(fn, acc ::: tvars)
172       case Select(pre, _) => boundVars(pre, acc)
173       case Block(_, expr) => boundVars(expr, acc)
174       case _ => acc
175     }
176     @tailrec def occurring(tree: Tree, toTest: List[TypeVar], acc: List[TypeVar]): List[TypeVar] =
177       if (toTest.isEmpty) acc
178       else tree match {
179         case Apply(fn, _) =>
180           fn.tpe.widen match {
181             case mtp: MethodType =>
182               val (occ, nocc) = toTest.partition(tvar => mtp.paramInfos.exists(tvar.occursIn))
183               occurring(fn, nocc, occ ::: acc)
184             case _ =>
185               occurring(fn, toTest, acc)
186           }
187         case TypeApply(fn, targs) => occurring(fn, toTest, acc)
188         case Select(pre, _) => occurring(pre, toTest, acc)
189         case Block(_, expr) => occurring(expr, toTest, acc)
190         case _ => acc
191       }
192     occurring(tree, boundVars(tree, Nil), Nil)
193   }

195   /** The instantiation direction for given poly param computed
196    *  from the constraint:
197    *  @return   1 (maximize) if constraint is uniformly from above,
198    *           -1 (minimize) if constraint is uniformly from below,
199    *            0 if unconstrained, or constraint is from below and above.
200    */
201   private def instDirection(param: TypeParamRef)(implicit ctx: Context): Int = {
202     val constrained = ctx.typerState.constraint.fullBounds(param)
203     val original = param.binder.paramInfos(param.paramNum)
204     val cmp = ctx.typeComparer
205     val approxBelow =
206       if (!cmp.isSubTypeWhenFrozen(constrained.lo, original.lo)) 1 else 0
207     val approxAbove =
208       if (!cmp.isSubTypeWhenFrozen(original.hi, constrained.hi)) 1 else 0
209     approxAbove - approxBelow
210   }

226   /** Instantiate undetermined type variables so that type `tp` is maximized.
227    *  @return   The list of type symbols that were created
228    *            to instantiate undetermined type variables that occur non-variantly
229    */
230   def maximizeType(tp: Type, pos: Position, fromScala2x: Boolean)(implicit ctx: Context): List[Symbol] = Stats.track("maximizeType") {
231     val vs = variances(tp, alwaysTrue)
232     val patternBound = new mutable.ListBuffer[Symbol]
233     vs foreachBinding { (tvar, v) =>
234       if (v == 1) tvar.instantiate(fromBelow = false)
235       else if (v == -1) tvar.instantiate(fromBelow = true)
236       else {
237         val bounds = ctx.typerState.constraint.fullBounds(tvar.origin)
238         if (bounds.hi <:< bounds.lo || bounds.hi.classSymbol.is(Final) || fromScala2x)
239           tvar.instantiate(fromBelow = false)
240         else {
241           val wildCard = ctx.newPatternBoundSymbol(WildcardParamName.fresh().toTypeName, bounds, pos)
242           tvar.instantiateWith(wildCard.typeRef)
243           patternBound += wildCard
244         }
245       }
246     }

252   /** All occurrences of type vars in this type that satisfy predicate
253    *  `include` mapped to their variances (-1/0/1) in this type, where
254    *  -1 means: only covariant occurrences
255    *  +1 means: only covariant occurrences
256    *  0 means: mixed or non-variant occurrences
257    *
258    *  Note: We intentionally use a relaxed version of variance here,
259    *  where the variance does not change under a prefix of a named type
260    *  (the strict version makes prefixes invariant). This turns out to be
261    *  better for type inference. In a nutshell, if a type variable occurs
262    *  like this:
263    *
264    *     (U? >: x.type) # T
265    *
266    *  we want to instantiate U to x.type right away. No need to wait further.
267    */
268   private def variances(tp: Type, include: TypeVar => Boolean)(implicit ctx: Context): VarianceMap = Stats.track("variances") {
269     val constraint = ctx.typerState.constraint
270 
271     object accu extends TypeAccumulator[VarianceMap] {
272       def setVariance(v: Int) = variance = v
273       def apply(vmap: VarianceMap, t: Type): VarianceMap = t match {
274         case t: TypeVar
275         if !t.isInstantiated && (ctx.typerState.constraint contains t) && include(t) =>
276           val v = vmap(t)
277           if (v == null) vmap.updated(t, variance)
278           else if (v == variance || v == 0) vmap
279           else vmap.updated(t, 0)
280         case _ =>
281           foldOver(vmap, t)
282       }
283     }
284 
285     /** Include in `vmap` type variables occurring in the constraints of type variables
286      *  already in `vmap`. Specifically:
287      *   - if `tvar` is covariant in `vmap`, include all variables in its lower bound
288      *     (because they influence the minimal solution of `tvar`),
289      *   - if `tvar` is contravariant in `vmap`, include all variables in its upper bound
290      *     at flipped variances (because they influence the maximal solution of `tvar`),
291      *   - if `tvar` is nonvariant in `vmap`, include all variables in its upper and lower
292      *     bounds as non-variant.
293      *  Do this in a fixpoint iteration until `vmap` stabilizes.
294      */
295     def propagate(vmap: VarianceMap): VarianceMap = {
296       var vmap1 = vmap
297       def traverse(tp: Type) = { vmap1 = accu(vmap1, tp) }
298       vmap.foreachBinding { (tvar, v) =>
299         val param = tvar.origin
300         val e = constraint.entry(param)
301         accu.setVariance(v)
302         if (v >= 0) {
303           traverse(e.bounds.lo)
304           constraint.lower(param).foreach(p => traverse(constraint.typeVarOfParam(p)))
305         }
306         if (v <= 0) {
307           traverse(e.bounds.hi)
308           constraint.upper(param).foreach(p => traverse(constraint.typeVarOfParam(p)))
309         }
310       }
311       if (vmap1 eq vmap) vmap else propagate(vmap1)
312     }
313 
314     propagate(accu(SimpleIdentityMap.Empty, tp))
315   }

322   /** Interpolate those undetermined type variables in the widened type of this tree
323    *  which are introduced by type application contained in the tree.
324    *  If such a variable appears covariantly in type `tp` or does not appear at all,
325    *  approximate it by its lower bound. Otherwise, if it appears contravariantly
326    *  in type `tp` approximate it by its upper bound.
327    *  @param ownedBy  if it is different from NoSymbol, all type variables owned by
328    *                  `ownedBy` qualify, independent of position.
329    *                  Without that second condition, it can be that certain variables escape
330    *                  interpolation, for instance when their tree was eta-lifted, so
331    *                  the typechecked tree is no longer the tree in which the variable
332    *                  was declared. A concrete example of this phenomenon can be
333    *                  observed when compiling core.TypeOps#asSeenFrom.
334    */
335   def interpolateUndetVars(tree: Tree, ownedBy: Symbol, pt: Type)(implicit ctx: Context): Unit = {
336     val constraint = ctx.typerState.constraint
337     val qualifies = (tvar: TypeVar) =>
338       (tree contains tvar.bindingTree) || ownedBy.exists && tvar.owner == ownedBy
339     def interpolate() = Stats.track("interpolateUndetVars") {
340       val tp = tree.tpe.widen
341       constr.println(s"interpolate undet vars in ${tp.show}, pos = ${tree.pos}, mode = ${ctx.mode}, undets = ${constraint.uninstVars map (tvar => s"${tvar.show}@${tvar.bindingTree.pos}")}")
342       constr.println(s"qualifying undet vars: ${constraint.uninstVars filter qualifies map (tvar => s"$tvar / ${tvar.show}")}, constraint: ${constraint.show}")
343 
344       val vs = variances(tp, qualifies)
345       val hasUnreportedErrors = ctx.typerState.reporter match {
346         case r: StoreReporter if r.hasErrors => true
347         case _ => false
348       }
349 
350       var isConstrained = tree.isInstanceOf[Apply] || tree.tpe.isInstanceOf[MethodOrPoly]
351 
352       def ensureConstrained() = if (!isConstrained) {
353         isConstrained = true
354         constrainResult(tree.tpe, pt)
355       }
356 
357       // Avoid interpolating variables if typerstate has unreported errors.
358       // Reason: The errors might reflect unsatisfiable constraints. In that
359       // case interpolating without taking account the constraints risks producing
360       // nonsensical types that then in turn produce incomprehensible errors.
361       // An example is in neg/i1240.scala. Without the condition in the next code line
362       // we get for
363       //
364       //      val y: List[List[String]] = List(List(1))
365       //
366       //     i1430.scala:5: error: type mismatch:
367       //     found   : Int(1)
368       //     required: Nothing
369       //     val y: List[List[String]] = List(List(1))
370       //                                           ^
371       // With the condition, we get the much more sensical:
372       //
373       //     i1430.scala:5: error: type mismatch:
374       //     found   : Int(1)
375       //     required: String
376       //     val y: List[List[String]] = List(List(1))
377       if (!hasUnreportedErrors)
378         vs foreachBinding { (tvar, v) =>
379           if (v != 0 && ctx.typerState.constraint.contains(tvar)) {
380             // previous interpolations could have already instantiated `tvar`
381             // through unification, that's why we have to check again whether `tvar`
382             // is contained in the current constraint.
383             typr.println(s"interpolate ${if (v == 1) "co" else "contra"}variant ${tvar.show} in ${tp.show}")
384             ensureConstrained()
385             tvar.instantiate(fromBelow = v == 1)
386           }
387         }
388       for (tvar <- constraint.uninstVars)
389         if (!(vs contains tvar) && qualifies(tvar)) {
390           typr.println(s"instantiating non-occurring ${tvar.show} in ${tp.show} / $tp")
391           ensureConstrained()
392           tvar.instantiate(fromBelow = tvar.hasLowerBound)
393         }
394     }
395     if (constraint.uninstVars exists qualifies) interpolate()
396   }
