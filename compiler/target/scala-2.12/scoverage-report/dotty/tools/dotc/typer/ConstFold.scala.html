19   /** If tree is a constant operation, replace with result. */
20   def apply(tree: Tree)(implicit ctx: Context): Tree = finish(tree) {
21     tree match {
22       case Apply(Select(xt, op), yt :: Nil) =>
23         xt.tpe.widenTermRefExpr match {
24           case ConstantType(x) =>
25             yt.tpe.widenTermRefExpr match {
26               case ConstantType(y) => foldBinop(op, x, y)
27               case _ => null
28             }
29           case _ => null
30         }
31       case Select(xt, op) =>
32         xt.tpe.widenTermRefExpr match {
33           case ConstantType(x) => foldUnop(op, x)
34           case _ => null
35         }
36       case _ => null
37     }
38   }

51   private def finish(tree: Tree)(compX: => Constant)(implicit ctx: Context): Tree =
52     try {
53       val x = compX
54       if (x ne null) tree withType ConstantType(x)
55       else tree
56     } catch {
57       case _: ArithmeticException => tree   // the code will crash at runtime,
58                                             // but that is better than the
59                                             // compiler itself crashing
60     }

62   private def foldUnop(op: Name, x: Constant): Constant = (op, x.tag) match {
63     case (nme.UNARY_!, BooleanTag) => Constant(!x.booleanValue)
64 
65     case (nme.UNARY_~ , IntTag    ) => Constant(~x.intValue)
66     case (nme.UNARY_~ , LongTag   ) => Constant(~x.longValue)
67 
68     case (nme.UNARY_+ , IntTag    ) => Constant(x.intValue)
69     case (nme.UNARY_+ , LongTag   ) => Constant(x.longValue)
70     case (nme.UNARY_+ , FloatTag  ) => Constant(x.floatValue)
71     case (nme.UNARY_+ , DoubleTag ) => Constant(x.doubleValue)

163   private def foldBinop(op: Name, x: Constant, y: Constant): Constant = {
164     val optag =
165       if (x.tag == y.tag) x.tag
166       else if (x.isNumeric && y.isNumeric) math.max(x.tag, y.tag)
167       else NoTag
168 
169     try optag match {
170       case BooleanTag                               => foldBooleanOp(op, x, y)
171       case ByteTag | ShortTag | CharTag | IntTag    => foldSubrangeOp(op, x, y)
172       case LongTag                                  => foldLongOp(op, x, y)
173       case FloatTag                                 => foldFloatOp(op, x, y)
174       case DoubleTag                                => foldDoubleOp(op, x, y)
175       case StringTag if op == nme.ADD               => Constant(x.stringValue + y.stringValue)
176       case _                                        => null
177     }
178     catch {
179       case ex: ArithmeticException => null
180     }
181   }
