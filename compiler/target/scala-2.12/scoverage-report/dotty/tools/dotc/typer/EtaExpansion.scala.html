21 /** A class that handles argument lifting. Argument lifting is needed in the following
22  *  scenarios:
23  *   - eta expansion
24  *   - applications with default arguments
25  *   - applications with out-of-order named arguments
26  *  Lifting generally lifts impure expressions only, except in the case of possible
27  *  default arguments, where we lift also complex pure expressions, since in that case
28  *  arguments can be duplicated as arguments to default argument methods.
29  */
30 abstract class Lifter {
31   import tpd._
32 
33   /** Test indicating `expr` does not need lifting */
34   def noLift(expr: Tree)(implicit ctx: Context): Boolean
35 
36   /** The corresponding lifter for pass-by-name arguments */
37   protected def exprLifter: Lifter = NoLift
38 
39   /** The flags of a lifted definition */
40   protected def liftedFlags: FlagSet = EmptyFlags
41 
42   /** The tree of a lifted definition */
43   protected def liftedDef(sym: TermSymbol, rhs: Tree)(implicit ctx: Context): MemberDef = ValDef(sym, rhs)
44 
45   private def lift(defs: mutable.ListBuffer[Tree], expr: Tree, prefix: TermName = EmptyTermName)(implicit ctx: Context): Tree =
46     if (noLift(expr)) expr
47     else {
48       val name = UniqueName.fresh(prefix)
49       var liftedType = fullyDefinedType(expr.tpe.widen, "lifted expression", expr.pos)
50       if (liftedFlags.is(Method)) liftedType = ExprType(liftedType)
51       val lifted = ctx.newSymbol(ctx.owner, name, liftedFlags, liftedType, coord = positionCoord(expr.pos))
52       defs += liftedDef(lifted, expr).withPos(expr.pos.focus)
53       ref(lifted.termRef).withPos(expr.pos)
54     }
55 
56   /** Lift out common part of lhs tree taking part in an operator assignment such as
57    *
58    *     lhs += expr
59    */
60   def liftAssigned(defs: mutable.ListBuffer[Tree], tree: Tree)(implicit ctx: Context): Tree = tree match {
61     case Apply(MaybePoly(fn @ Select(pre, name), targs), args) =>
62       cpy.Apply(tree)(
63         cpy.Select(fn)(
64           lift(defs, pre), name).appliedToTypeTrees(targs),
65           liftArgs(defs, fn.tpe, args))
66     case Select(pre, name) =>
67       cpy.Select(tree)(lift(defs, pre), name)
68     case _ =>
69       tree
70   }
71 
72   /** Lift a function argument, stripping any NamedArg wrapper */
73   private def liftArg(defs: mutable.ListBuffer[Tree], arg: Tree, prefix: TermName = EmptyTermName)(implicit ctx: Context): Tree =
74     arg match {
75       case arg @ NamedArg(name, arg1) => cpy.NamedArg(arg)(name, lift(defs, arg1, prefix))
76       case arg => lift(defs, arg, prefix)
77     }
78 
79   /** Lift arguments that are not-idempotent into ValDefs in buffer `defs`
80    *  and replace by the idents of so created ValDefs.
81    */
82   def liftArgs(defs: mutable.ListBuffer[Tree], methRef: Type, args: List[Tree])(implicit ctx: Context) =
83     methRef.widen match {
84       case mt: MethodType =>
85         (args, mt.paramNames, mt.paramInfos).zipped.map { (arg, name, tp) =>
86           val lifter = if (tp.isInstanceOf[ExprType]) exprLifter else this
87           lifter.liftArg(defs, arg, if (name.firstPart contains '$') EmptyTermName else name)
88         }
89       case _ =>
90         args.map(liftArg(defs, _))
91     }
92 
93   /** Lift out function prefix and all arguments from application
94    *
95    *     pre.f(arg1, ..., argN)   becomes
96    *
97    *     val x0 = pre
98    *     val x1 = arg1
99    *     ...
100    *     val xN = argN
101    *     x0.f(x1, ..., xN)
102    *
103    *  But leave pure expressions alone.
104    *
105    */
106   def liftApp(defs: mutable.ListBuffer[Tree], tree: Tree)(implicit ctx: Context): Tree = tree match {
107     case Apply(fn, args) =>
108       cpy.Apply(tree)(liftApp(defs, fn), liftArgs(defs, fn.tpe, args))
109     case TypeApply(fn, targs) =>
110       cpy.TypeApply(tree)(liftApp(defs, fn), targs)
111     case Select(pre, name) if isPureRef(tree) =>
112       cpy.Select(tree)(liftPrefix(defs, pre), name)
113     case Block(stats, expr) =>
114       liftApp(defs ++= stats, expr)
115     case New(tpt) =>
116       tree
117     case _ =>
118       lift(defs, tree)
119   }
120 
121   /** Lift prefix `pre` of an application `pre.f(...)` to
122    *
123    *     val x0 = pre
124    *     x0.f(...)
125    *
126    *  unless `pre` is a `New` or `pre` is idempotent.
127    */
128   def liftPrefix(defs: mutable.ListBuffer[Tree], tree: Tree)(implicit ctx: Context): Tree = tree match {
129     case New(_) => tree
130     case _ => if (isIdempotentExpr(tree)) tree else lift(defs, tree)
131   }
132 }

162   /** Eta-expanding a tree means converting a method reference to a function value.
163    *  @param    tree       The tree to expand
164    *  @param    mt         The type of the method reference
165    *  @param    xarity     The arity of the expected function type
166    *  and assume the lifted application of `tree` (@see liftApp) is
167    *
168    *         { val xs = es; expr }
169    *
170    *  If xarity matches the number of parameters in `mt`, the eta-expansion is
171    *
172    *         { val xs = es; (x1, ..., xn) => expr(x1, ..., xn) }
173    *
174    * Note that the function value's parameters are untyped, hence the type will
175    * be supplied by the environment (or if missing be supplied by the target
176    * method as a fallback). On the other hand, if `xarity` is different from
177    * the number of parameters in `mt`, then we cannot propagate parameter types
178    * from the expected type, and we fallback to using the method's original
179    * parameter types instead.
180    *
181    * In either case, the result is an untyped tree, with `es` and `expr` as typed splices.
182    */
183   def etaExpand(tree: Tree, mt: MethodType, xarity: Int)(implicit ctx: Context): untpd.Tree = {
184     import untpd._
185     assert(!ctx.isAfterTyper)
186     val defs = new mutable.ListBuffer[tpd.Tree]
187     val lifted: Tree = TypedSplice(liftApp(defs, tree))
188     val paramTypes: List[Tree] =
189       if (mt.paramInfos.length == xarity) mt.paramInfos map (_ => TypeTree())
190       else mt.paramInfos map TypeTree
191     val params = (mt.paramNames, paramTypes).zipped.map((name, tpe) =>
192       ValDef(name, tpe, EmptyTree).withFlags(Synthetic | Param).withPos(tree.pos.startPos))
193     var ids: List[Tree] = mt.paramNames map (name => Ident(name).withPos(tree.pos.startPos))
194     if (mt.paramInfos.nonEmpty && mt.paramInfos.last.isRepeatedParam)
195       ids = ids.init :+ repeated(ids.last)
196     var body: Tree = Apply(lifted, ids)
197     mt.resultType match {
198       case rt: MethodType if !rt.isImplicitMethod => body = PostfixOp(body, Ident(nme.WILDCARD))
199       case _ =>
200     }
201     val fn = untpd.Function(params, body)
202     if (defs.nonEmpty) untpd.Block(defs.toList map (untpd.TypedSplice(_)), fn) else fn
203   }
