16   def rootImport(refFn: () => TermRef)(implicit ctx: Context) = {
17     val selectors = untpd.Ident(nme.WILDCARD) :: Nil

24 /** Info relating to an import clause
25  *  @param   sym          The import symbol defined by the clause
26  *  @param   selectors    The selector clauses
27  *  @param   symNameOpt   Optionally, the name of the import symbol. None for root imports.
28  *                        Defined for all explicit imports from ident or select nodes.
29  *  @param   isRootImport true if this is one of the implicit imports of scala, java.lang,
30  *                        scala.Predef or dotty.DottyPredef in the start context, false otherwise.
31  */

69   /** Compute info relating to the selector list */
70   private def ensureInitialized(): Unit = if (myExcluded == null) {
71     myExcluded = Set()
72     myMapped = SimpleIdentityMap.Empty
73     myOriginals = Set()
74     def recur(sels: List[untpd.Tree]): Unit = sels match {
75       case sel :: sels1 =>
76         sel match {
77           case Thicket(Ident(name: TermName) :: Ident(nme.WILDCARD) :: Nil) =>
78             myExcluded += name
79           case Thicket(Ident(from: TermName) :: Ident(to: TermName) :: Nil) =>
80             myMapped = myMapped.updated(to, from)
81             myExcluded += from
82             myOriginals += from
83           case Ident(nme.WILDCARD) =>
84             myWildcardImport = true
85           case Ident(name: TermName) =>
86             myMapped = myMapped.updated(name, name)
87             myOriginals += name
88         }
89         recur(sels1)
90       case nil =>
91     }
92     recur(selectors)
93   }

95   /** The implicit references imported by this import clause */
96   def importedImplicits(implicit ctx: Context): List[ImplicitRef] = {
97     val pre = site
98     if (isWildcardImport) {
99       val refs = pre.implicitMembers
100       if (excluded.isEmpty) refs
101       else refs filterNot (ref => excluded contains ref.name.toTermName)
102     } else
103       for {
104         renamed <- reverseMapping.keys
105         denot <- pre.member(reverseMapping(renamed)).altsWith(_ is Implicit)
106       } yield {
107         val original = reverseMapping(renamed)
108         val ref = TermRef(pre, original, denot)
109         if (renamed == original) ref
110         else new RenamedImplicitRef(ref, renamed)
111       }
112   }

114   /** The root import symbol hidden by this symbol, or NoSymbol if no such symbol is hidden.
115    *  Note: this computation needs to work even for un-initialized import infos, and
116    *  is not allowed to force initialization.
117    *
118    *  TODO: Once we have fully bootstrapped, I would prefer if we expressed
119    *  unimport with an `override` modifier, and generalized it to all imports.
120    *  I believe this would be more transparent than the current set of conditions. E.g.
121    *
122    *      override import Predef.{any2stringAdd => _, StringAdd => _, _} // disables String +
123    *      override import java.lang.{}                                   // disables all imports
124    */
125   def unimported(implicit ctx: Context): Symbol = {
126     if (myUnimported == null) {
127       lazy val sym = site.termSymbol
128       def maybeShadowsRoot = symNameOpt match {
129         case Some(symName) => defn.ShadowableImportNames.contains(symName)
130         case None => false
131       }
132       myUnimported =
133         if (maybeShadowsRoot && defn.RootImportTypes.exists(_.symbol == sym)) sym
134         else NoSymbol
135       assert(myUnimported != null)
136     }
137     myUnimported
138   }
139   private[this] var myUnimported: Symbol = _
