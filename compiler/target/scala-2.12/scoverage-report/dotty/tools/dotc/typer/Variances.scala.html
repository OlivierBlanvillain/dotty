12   type Variance = FlagSet
13   val Bivariant = VarianceFlags
14   val Invariant = EmptyFlags

16   /** Flip between covariant and contravariant */
17   def flip(v: Variance): Variance = {
18     if (v == Covariant) Contravariant
19     else if (v == Contravariant) Covariant
20     else v
21   }
22 
23   /** Map everything below Bivariant to Invariant */
24   def cut(v: Variance): Variance =
25     if (v == Bivariant) v else Invariant
26 
27   def compose(v: Variance, boundsVariance: Int) =
28     if (boundsVariance == 1) v
29     else if (boundsVariance == -1) flip(v)
30     else cut(v)

32   /** Compute variance of type parameter `tparam' in types of all symbols `sym'. */
33   def varianceInSyms(syms: List[Symbol])(tparam: Symbol)(implicit ctx: Context): Variance =
34     (Bivariant /: syms) ((v, sym) => v & varianceInSym(sym)(tparam))

36   /** Compute variance of type parameter `tparam' in type of symbol `sym'. */

41   /** Compute variance of type parameter `tparam' in all types `tps'. */
42   def varianceInTypes(tps: List[Type])(tparam: Symbol)(implicit ctx: Context): Variance =

45   /** Compute variance of type parameter `tparam' in all type arguments
46    *  <code>tps</code> which correspond to formal type parameters `tparams1'.
47    */
48   def varianceInArgs(tps: List[Type], tparams1: List[Symbol])(tparam: Symbol)(implicit ctx: Context): Variance = {
49     var v: Variance = Bivariant;
50     for ((tp, tparam1) <- tps zip tparams1) {
51       val v1 = varianceInType(tp)(tparam)
52       v = v & (if (tparam1.is(Covariant)) v1
53            else if (tparam1.is(Contravariant)) flip(v1)

55     }

69   /** Compute variance of type parameter <code>tparam</code> in type <code>tp</code>. */
70   def varianceInType(tp: Type)(tparam: Symbol)(implicit ctx: Context): Variance = tp match {
71     case TermRef(pre, _) =>
72       varianceInType(pre)(tparam)
73     case tp @ TypeRef(pre, _) =>
74       if (tp.symbol == tparam) Covariant else varianceInType(pre)(tparam)
75     case tp @ TypeBounds(lo, hi) =>
76       if (lo eq hi) cut(varianceInType(hi)(tparam))
77       else flip(varianceInType(lo)(tparam)) & varianceInType(hi)(tparam)
78     case tp @ RefinedType(parent, _, rinfo) =>
79       varianceInType(parent)(tparam) & varianceInType(rinfo)(tparam)
80     case tp: RecType =>
81       varianceInType(tp.parent)(tparam)
82     case tp: MethodOrPoly =>
83       flip(varianceInTypes(tp.paramInfos)(tparam)) & varianceInType(tp.resultType)(tparam)
84     case ExprType(restpe) =>
85       varianceInType(restpe)(tparam)
86     case tp @ AppliedType(tycon, args) =>
87       def varianceInArgs(v: Variance, args: List[Type], tparams: List[ParamInfo]): Variance =
88         args match {
89           case arg :: args1 =>
90             varianceInArgs(
91               v & compose(varianceInType(arg)(tparam), tparams.head.paramVariance),
92               args1, tparams.tail)
93           case nil =>
94             v
95         }
96       varianceInArgs(varianceInType(tycon)(tparam), args, tycon.typeParams)
97     case AnnotatedType(tp, annot) =>
98       varianceInType(tp)(tparam) & varianceInAnnot(annot)(tparam)
99     case tp: AndOrType =>
100       varianceInType(tp.tp1)(tparam) & varianceInType(tp.tp2)(tparam)
101     case _ =>
102       Bivariant
103   }
104 
105   def varianceString(v: Variance) =
106     if (v is Covariant) "covariant"
107     else if (v is Contravariant) "contravariant"
108     else "invariant"
109

111     if (v > 0) "+"
112     else if (v < 0) "-"
113     else ""
