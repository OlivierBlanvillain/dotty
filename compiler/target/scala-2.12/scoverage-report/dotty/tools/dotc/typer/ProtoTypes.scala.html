28 
29     /** Is there an implicit conversion from `tp` to `pt`? */
30     def viewExists(tp: Type, pt: Type)(implicit ctx: Context): Boolean
31 
32     /** A type `tp` is compatible with a type `pt` if one of the following holds:
33      *    1. `tp` is a subtype of `pt`
34      *    2. `pt` is by name parameter type, and `tp` is compatible with its underlying type
35      *    3. there is an implicit conversion from `tp` to `pt`.
36      *    4. `tp` is a numeric subtype of `pt` (this case applies even if implicit conversions are disabled)
37      */
38     def isCompatible(tp: Type, pt: Type)(implicit ctx: Context): Boolean =
39       (tp.widenExpr relaxed_<:< pt.widenExpr) || viewExists(tp, pt)

41     /** Test compatibility after normalization in a fresh typerstate. */
42     def normalizedCompatible(tp: Type, pt: Type)(implicit ctx: Context) =
43       ctx.test { implicit ctx =>
44         val normTp = normalize(tp, pt)
45         isCompatible(normTp, pt) || pt.isRef(defn.UnitClass) && normTp.isParameterless
46       }

48     private def disregardProto(pt: Type)(implicit ctx: Context): Boolean = pt.dealias match {
49       case _: OrType => true
50       case pt => pt.isRef(defn.UnitClass)

53     /** Check that the result type of the current method
54      *  fits the given expected result type.
55      */
56     def constrainResult(mt: Type, pt: Type)(implicit ctx: Context): Boolean = {
57       val savedConstraint = ctx.typerState.constraint
58       val res = pt match {
59         case pt: FunProto =>
60           mt match {
61             case mt: MethodType => constrainResult(resultTypeApprox(mt), pt.resultType)
62             case _ => true
63           }
64         case _: ValueTypeOrProto if !disregardProto(pt) =>
65           isCompatible(normalize(mt, pt), pt)
66         case pt: WildcardType if pt.optBounds.exists =>
67           isCompatible(normalize(mt, pt), pt)
68         case _ =>

70       }
71       if (!res) ctx.typerState.resetConstraintTo(savedConstraint)

381    *  If the constraint contains already some of these parameters in its domain,
382    *  make a copy of the type lambda and add the copy's type parameters instead.

436    *   - skips implicit parameters of methods and functions;

438    *   - converts non-dependent method types to the corresponding function types
439    *     unless the expected type is an ApplyingProto or IgnoredProto.
440    *   - dereferences parameterless method types
441    *   - dereferences nullary method types provided the corresponding function type
442    *     is not a subtype of the expected type.
443    * Note: We need to take account of the possibility of inserting a () argument list in normalization. Otherwise, a type with a
444    *     def toString(): String
445    * member would not count as a valid solution for ?{toString: String}. This would then lead to an implicit
446    * insertion, with a nice explosion of inference search because of course every implicit result has some sort
447    * of toString method. The problem is solved by dereferencing nullary method types if the corresponding
448    * function type is not compatible with the prototype.
449    */

478   /** Approximate occurrences of parameter types and uninstantiated typevars
479    *  by wildcard types.
480    */
481   final def wildApprox(tp: Type, theMap: WildApproxMap, seen: Set[TypeParamRef])(implicit ctx: Context): Type = tp match {
482     case tp: NamedType => // default case, inlined for speed
483       if (tp.symbol.isStatic) tp
484       else tp.derivedSelect(wildApprox(tp.prefix, theMap, seen))
485     case tp @ AppliedType(tycon, args) =>
486       wildApprox(tycon, theMap, seen) match {
487         case _: WildcardType => WildcardType // this ensures we get a * type
488         case tycon1 => tp.derivedAppliedType(tycon1, args.mapConserve(wildApprox(_, theMap, seen)))
489       }
490     case tp: RefinedType => // default case, inlined for speed
491       tp.derivedRefinedType(
492           wildApprox(tp.parent, theMap, seen),
493           tp.refinedName,
494           wildApprox(tp.refinedInfo, theMap, seen))
495     case tp: TypeAlias => // default case, inlined for speed
496       tp.derivedTypeAlias(wildApprox(tp.alias, theMap, seen))
497     case tp @ TypeParamRef(poly, pnum) =>
498       def wildApproxBounds(bounds: TypeBounds) =
499         if (seen.contains(tp)) WildcardType
500         else WildcardType(wildApprox(bounds, theMap, seen + tp).bounds)
501       def unconstrainedApprox = wildApproxBounds(poly.paramInfos(pnum))
502       def approxPoly =
503         if (ctx.mode.is(Mode.TypevarsMissContext)) unconstrainedApprox
504         else
505           ctx.typerState.constraint.entry(tp) match {
506             case bounds: TypeBounds => wildApproxBounds(bounds)
507             case NoType             => unconstrainedApprox
508             case inst               => wildApprox(inst, theMap, seen)
509           }
510       approxPoly
511     case TermParamRef(mt, pnum) =>
512       WildcardType(TypeBounds.upper(wildApprox(mt.paramInfos(pnum), theMap, seen)))
513     case tp: TypeVar =>
514       wildApprox(tp.underlying, theMap, seen)
515     case tp: AndType =>
516       def approxAnd = {
517         val tp1a = wildApprox(tp.tp1, theMap, seen)
518         val tp2a = wildApprox(tp.tp2, theMap, seen)
519         def wildBounds(tp: Type) =
520           if (tp.isInstanceOf[WildcardType]) tp.bounds else TypeBounds.upper(tp)
521         if (tp1a.isInstanceOf[WildcardType] || tp2a.isInstanceOf[WildcardType])
522           WildcardType(wildBounds(tp1a) & wildBounds(tp2a))
523         else
524           tp.derivedAndType(tp1a, tp2a)
525       }
526       approxAnd
527     case tp: OrType =>
528       def approxOr = {
529         val tp1a = wildApprox(tp.tp1, theMap, seen)
530         val tp2a = wildApprox(tp.tp2, theMap, seen)
531         if (tp1a.isInstanceOf[WildcardType] || tp2a.isInstanceOf[WildcardType])
532           WildcardType(tp1a.bounds | tp2a.bounds)
533         else
534           tp.derivedOrType(tp1a, tp2a)
535       }
536       approxOr
537     case tp: SelectionProto =>
538       tp.derivedSelectionProto(tp.name, wildApprox(tp.memberProto, theMap, seen), NoViewsAllowed)
539     case tp: ViewProto =>
540       tp.derivedViewProto(
541           wildApprox(tp.argType, theMap, seen),
542           wildApprox(tp.resultType, theMap, seen))
543     case  _: ThisType | _: BoundType | NoPrefix => // default case, inlined for speed
544       tp
545     case _ =>
546       (if (theMap != null && seen.eq(theMap.seen)) theMap else new WildApproxMap(seen))
547         .mapOver(tp)
548   }
