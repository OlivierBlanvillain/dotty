16 /** A version of Typer that keeps all symbols defined and referenced in a
17  *  previously typed tree.
18  *
19  *  All definition nodes keep their symbols. All leaf nodes for idents, selects,
20  *  and TypeTrees keep their types. Indexing is a no-op.
21  *
22  *  Otherwise, everything is as in Typer.
23  */
24 class ReTyper extends Typer {
25   import tpd._

27   /** Checks that the given tree has been typed */
28   protected def promote(tree: untpd.Tree)(implicit ctx: Context): tree.ThisTree[Type] = {
29     assert(tree.hasType, i"$tree ${tree.getClass} ${tree.uniqueId}")
30     tree.withType(tree.typeOpt)
31   }

33   override def typedIdent(tree: untpd.Ident, pt: Type)(implicit ctx: Context): Tree =

36   override def typedSelect(tree: untpd.Select, pt: Type)(implicit ctx: Context): Tree = {

40     untpd.cpy.Select(tree)(qual1, tree.name).withType(tree.typeOpt)
41   }
