176     /** Check info `tp` for cycles. Throw CyclicReference for illegal cycles,
177      *  break direct cycle with a LazyRef for legal, F-bounded cycles.

179     def checkInfo(tp: Type): Type = tp match {
180       case tp @ TypeAlias(alias) =>
181         try tp.derivedTypeAlias(apply(alias))
182         finally {
183           where = "alias"
184           lastChecked = alias
185         }

192         nestedCycleOK = true
193         try tp.derivedTypeBounds(lo1, apply(hi))

200         tp
201     }

203     private def apply(tp: Type, cycleOK: Boolean, nestedCycleOK: Boolean): Type = {
204       val savedCycleOK = this.cycleOK
205       val savedNestedCycleOK = this.nestedCycleOK
206       this.cycleOK = cycleOK
207       this.nestedCycleOK = nestedCycleOK

215     def apply(tp: Type): Type = tp match {
216       case tp: TermRef =>
217         this(tp.info)
218         mapOver(tp)
219       case tp @ AppliedType(tycon, args) =>
220         tp.derivedAppliedType(this(tycon), args.map(this(_, nestedCycleOK, nestedCycleOK)))
221       case tp @ RefinedType(parent, name, rinfo) =>
222         tp.derivedRefinedType(this(parent), name, this(rinfo, nestedCycleOK, nestedCycleOK))
223       case tp: RecType =>
224         tp.rebind(this(tp.parent))
225       case tp @ TypeRef(pre, _) =>
226         try {
227           // A prefix is interesting if it might contain (transitively) a reference
228           // to symbol `sym` itself. We only check references with interesting
229           // prefixes for cycles. This pruning is done in order not to force
230           // global symbols when doing the cyclicity check.
231           def isInteresting(prefix: Type): Boolean = prefix.stripTypeVar match {
232             case NoPrefix => true
233             case prefix: ThisType => sym.owner.isClass && prefix.cls.isContainedIn(sym.owner)
234             case prefix: NamedType =>
235               (!sym.is(Private) && prefix.derivesFrom(sym.owner)) ||
236               (!prefix.symbol.isStaticOwner && isInteresting(prefix.prefix))
237             case SuperType(thistp, _) => isInteresting(thistp)
238             case AndType(tp1, tp2) => isInteresting(tp1) || isInteresting(tp2)
239             case OrType(tp1, tp2) => isInteresting(tp1) && isInteresting(tp2)
240             case _: RefinedOrRecType | _: AppliedType => true
241             case _ => false
242           }
243           if (isInteresting(pre)) {
244             val pre1 = this(pre, false, false)
245             if (locked.contains(tp) || tp.symbol.infoOrCompleter.isInstanceOf[NoCompleter])
246               throw CyclicReference(tp.symbol)
247             locked += tp
248             try checkInfo(tp.info)
249             finally locked -= tp
250             tp.withPrefix(pre1)
251           }
252           else tp
253         } catch {
254           case ex: CyclicReference =>
255             ctx.debuglog(i"cycle detected for $tp, $nestedCycleOK, $cycleOK")
256             if (cycleOK) LazyRef(_ => tp)
257             else if (reportErrors) throw ex
258             else tp
259         }
260       case _ => mapOver(tp)
261     }

388   /** Check the type signature of the symbol `M` defined by `tree` does not refer
389    *  to a private type or value which is invisible at a point where `M` is still
390    *  visible.
391    *
392    *  As an exception, we allow references to type aliases if the underlying
393    *  type of the alias is not a leak, and if `sym` is not a type. The rationale
394    *  for this is that the inferred type of a term symbol might contain leaky
395    *  aliases which should be removed (see leak-inferred.scala for an example),
396    *  but a type symbol definition will not contain leaky aliases unless the
397    *  user wrote them, so we can ask the user to change his definition. The more
398    *  practical reason for not transforming types is that `checkNoPrivateLeaks`
399    *  can force a lot of denotations, and this restriction means that we never
400    *  need to run `TypeAssigner#avoidPrivateLeaks` on type symbols when
401    *  unpickling, which avoids some issues related to forcing order.
402    *
403    *  See i997.scala for negative tests, and i1130.scala for a case where it
404    *  matters that we transform leaky aliases away.
405    *
406    *  @return The `info` of `sym`, with problematic aliases expanded away.
407    */
408   def checkNoPrivateLeaks(sym: Symbol, pos: Position)(implicit ctx: Context): Type = {
409     class NotPrivate extends TypeMap {
410       var errors: List[() => String] = Nil
411 
412       def accessBoundary(sym: Symbol): Symbol =
413         if (sym.is(Private) || !sym.owner.isClass) sym.owner
414         else if (sym.privateWithin.exists) sym.privateWithin
415         else if (sym.is(Package)) sym
416         else accessBoundary(sym.owner)
417 
418       val symBoundary = accessBoundary(sym)
419 
420       /** Is `other` leaked outside its access boundary ?
421        *  @pre  The signature of `sym` refers to `other`
422        */
423       def isLeaked(other: Symbol) =
424         other.is(Private, butNot = TypeParam) && {
425           val otherBoundary = other.owner
426           val otherLinkedBoundary = otherBoundary.linkedClass
427           !(symBoundary.isContainedIn(otherBoundary) ||
428             otherLinkedBoundary.exists && symBoundary.isContainedIn(otherLinkedBoundary))
429         }
430 
431       def apply(tp: Type): Type = tp match {
432         case tp: NamedType =>
433           val prevErrors = errors
434           var tp1 =
435             if (isLeaked(tp.symbol)) {
436               errors =
437                 (() => em"non-private $sym refers to private ${tp.symbol}\nin its type signature ${sym.info}") :: errors
438               tp
439             }
440             else mapOver(tp)
441           if ((errors ne prevErrors) && !sym.isType && tp.info.isAlias) {
442             // try to dealias to avoid a leak error
443             val savedErrors = errors
444             errors = prevErrors
445             val tp2 = apply(tp.superType)
446             if (errors eq prevErrors) tp1 = tp2
447             else errors = savedErrors
448           }
449           tp1
450         case tp: ClassInfo =>
451           tp.derivedClassInfo(
452             prefix = apply(tp.prefix),
453             classParents =
454               tp.parents.map { p =>
455                 apply(p).stripAnnots match {
456                   case ref: TypeRef => ref
457                   case ref: AppliedType => ref
458                   case _ => defn.ObjectType // can happen if class files are missing
459                 }
460               }
461             )
462         case _ =>
463           mapOver(tp)
464       }
465     }
466     val notPrivate = new NotPrivate
467     val info = notPrivate(sym.info)
468     notPrivate.errors.foreach(error => ctx.errorOrMigrationWarning(error(), pos))
469     info
470   }
