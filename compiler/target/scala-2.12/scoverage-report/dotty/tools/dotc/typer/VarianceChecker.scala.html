14 /** Provides `check` method to check that all top-level definitions
15  *  in tree are variance correct. Does not recurse inside methods.
16  *  The method should be invoked once for each Template.
17  */
18 object VarianceChecker {
19   case class VarianceError(tvar: Symbol, required: Variance)
20   def check(tree: tpd.Tree)(implicit ctx: Context) =
21     new VarianceChecker()(ctx).Traverser.traverse(tree)
22 }

24 class VarianceChecker()(implicit ctx: Context) {
25   import VarianceChecker._
26   import tpd._
27 
28   private object Validator extends TypeAccumulator[Option[VarianceError]] {
29     private[this] var base: Symbol = _
30 
31     /** Is no variance checking needed within definition of `base`? */
32     def ignoreVarianceIn(base: Symbol): Boolean = (
33          base.isTerm
34       || base.is(Package)
35       || base.is(Local)
36     )
37 
38     /** The variance of a symbol occurrence of `tvar` seen at the level of the definition of `base`.
39      *  The search proceeds from `base` to the owner of `tvar`.
40      *  Initially the state is covariant, but it might change along the search.
41      */
42     def relativeVariance(tvar: Symbol, base: Symbol, v: Variance = Covariant): Variance = /*trace(i"relative variance of $tvar wrt $base, so far: $v")*/ {
43       if (base == tvar.owner) v
44       else if ((base is Param) && base.owner.isTerm)
45         relativeVariance(tvar, paramOuter(base.owner), flip(v))
46       else if (ignoreVarianceIn(base.owner)) Bivariant
47       else if (base.isAliasType) relativeVariance(tvar, base.owner, Invariant)
48       else relativeVariance(tvar, base.owner, v)
49     }
50 
51     /** The next level to take into account when determining the
52      *  relative variance with a method parameter as base. The method
53      *  is always skipped. If the method is a constructor, we also skip
54      *  its class owner, because constructors are not checked for variance
55      *  relative to the type parameters of their own class. On the other
56      *  hand constructors do count for checking the variance of type parameters
57      *  of enclosing classes. I believe the Scala 2 rules are too lenient in
58      *  that respect.
59      */
60     private def paramOuter(meth: Symbol) =
61       if (meth.isConstructor) meth.owner.owner else meth.owner
62 
63     /** Check variance of abstract type `tvar` when referred from `base`. */
64     private def checkVarianceOfSymbol(tvar: Symbol): Option[VarianceError] = {
65       val relative = relativeVariance(tvar, base)
66       if (relative == Bivariant) None
67       else {
68         val required = compose(relative, this.variance)
69         def tvar_s = s"$tvar (${varianceString(tvar.flags)} ${tvar.showLocated})"
70         def base_s = s"$base in ${base.owner}" + (if (base.owner.isClass) "" else " in " + base.owner.enclosingClass)
71         ctx.log(s"verifying $tvar_s is ${varianceString(required)} at $base_s")
72         ctx.log(s"relative variance: ${varianceString(relative)}")
73         ctx.log(s"current variance: ${this.variance}")
74         ctx.log(s"owner chain: ${base.ownersIterator.toList}")
75         if (tvar is required) None
76         else Some(VarianceError(tvar, required))
77       }
78     }
79 
80     /** For PolyTypes, type parameters are skipped because they are defined
81      *  explicitly (their TypeDefs will be passed here.) For MethodTypes, the
82      *  same is true of the parameters (ValDefs).
83      */
84     def apply(status: Option[VarianceError], tp: Type): Option[VarianceError] = trace(s"variance checking $tp of $base at $variance", variances) {
85       if (status.isDefined) status
86       else tp match {
87         case tp: TypeRef =>
88           val sym = tp.symbol
89           if (sym.variance != 0 && base.isContainedIn(sym.owner)) checkVarianceOfSymbol(sym)
90           else if (sym.isAliasType) this(status, sym.info.bounds.hi)
91           else foldOver(status, tp)
92         case tp: MethodOrPoly =>
93           this(status, tp.resultType) // params will be checked in their TypeDef or ValDef nodes.
94         case AnnotatedType(_, annot) if annot.symbol == defn.UncheckedVarianceAnnot =>
95           status
96         //case tp: ClassInfo =>
97         //  ???  not clear what to do here yet. presumably, it's all checked at local typedefs
98         case _ =>
99           foldOver(status, tp)
100       }
101     }
102 
103     def validateDefinition(base: Symbol): Option[VarianceError] = {
104       val saved = this.base
105       this.base = base
106       try apply(None, base.info)
107       finally this.base = saved
108     }
109   }
110 
111   private object Traverser extends TreeTraverser {
112     def checkVariance(sym: Symbol, pos: Position) = Validator.validateDefinition(sym) match {
113       case Some(VarianceError(tvar, required)) =>
114         def msg = i"${varianceString(tvar.flags)} $tvar occurs in ${varianceString(required)} position in type ${sym.info} of $sym"
115         if (ctx.scala2Mode && sym.owner.isConstructor) {
116           ctx.migrationWarning(s"According to new variance rules, this is no longer accepted; need to annotate with @uncheckedVariance:\n$msg", pos)
117           patch(Position(pos.end), " @scala.annotation.unchecked.uncheckedVariance") // TODO use an import or shorten if possible
118         }
119         else ctx.error(msg, pos)
120       case None =>
121     }
122 
123     override def traverse(tree: Tree)(implicit ctx: Context) = {
124       def sym = tree.symbol
125       // No variance check for private/protected[this] methods/values.
126       def skip =
127         !sym.exists ||
128         sym.is(Local) || // !!! watch out for protected local!
129         sym.is(TypeParam) && sym.owner.isClass // already taken care of in primary constructor of class
130       tree match {
131         case defn: MemberDef if skip =>
132           ctx.debuglog(s"Skipping variance check of ${sym.showDcl}")
133         case tree: TypeDef =>
134           checkVariance(sym, tree.pos)
135         case tree: ValDef =>
136           checkVariance(sym, tree.pos)
137         case DefDef(_, tparams, vparamss, _, _) =>
138           checkVariance(sym, tree.pos)
139           tparams foreach traverse
140           vparamss foreach (_ foreach traverse)
141         case Template(_, _, _, body) =>
142           traverseChildren(tree)
143         case _ =>
144       }
145     }
146   }
