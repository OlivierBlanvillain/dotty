41   def parse(implicit ctx: Context) = monitor("parsing") {
42     val unit = ctx.compilationUnit
43     unit.untpdTree =
44       if (unit.isJava) new JavaParser(unit.source).parse()
45       else new Parser(unit.source).parse()
46     val printer = if (ctx.settings.Xprint.value.contains("parser")) default else typr
47     printer.println("parsed:\n" + unit.untpdTree.show)
48     if (Config.checkPositions)
49       unit.untpdTree.checkPos(nonOverlapping = !unit.isJava && !ctx.reporter.hasErrors)
50   }

64   def typeCheck(implicit ctx: Context) = monitor("typechecking") {

72   private def firstTopLevelDef(trees: List[tpd.Tree])(implicit ctx: Context): Symbol = trees match {
73     case PackageDef(_, defs) :: _    => firstTopLevelDef(defs)
74     case Import(_, _) :: defs        => firstTopLevelDef(defs)
75     case (tree @ TypeDef(_, _)) :: _ => tree.symbol
76     case _ => NoSymbol
77   }

79   protected def discardAfterTyper(unit: CompilationUnit)(implicit ctx: Context) =
80     unit.isJava || firstTopLevelDef(unit.tpdTree :: Nil).isPrimitiveValueClass

82   override def runOn(units: List[CompilationUnit])(implicit ctx: Context): List[CompilationUnit] = {
83     val unitContexts = for (unit <- units) yield {
84       ctx.inform(s"compiling ${unit.source}")
85       ctx.fresh.setCompilationUnit(unit)
86     }
87     unitContexts foreach (parse(_))
88     record("parsedTrees", ast.Trees.ntrees)
89     remaining = unitContexts
90     while (remaining.nonEmpty) {
91       enterSyms(remaining.head)
92       remaining = remaining.tail
93     }
94     unitContexts.foreach(enterAnnotations(_))
95     unitContexts.foreach(typeCheck(_))
96     record("total trees after typer", ast.Trees.ntrees)
97     unitContexts.map(_.compilationUnit).filterNot(discardAfterTyper)
98   }
