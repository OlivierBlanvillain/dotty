626       // enter a dummy companion (`denot.isAbsent` returns true) in scope. This

651           }

658     val ctxWithStats = (ctx /: stats) ((ctx, stat) => indexExpanded(stat)(ctx))

691   /** Add annotations of `stat` to `sym`.
692    *  This method can be called twice on a symbol (e.g. once
693    *  during the `annotate` phase and then again during completion).
694    *  Therefore, care needs to be taken not to add annotations again
695    *  that are already added to the symbol.
696    */
697   def addAnnotations(sym: Symbol, stat: MemberDef)(implicit ctx: Context) = {
698     // (1) The context in which an annotation of a top-level class or module is evaluated
699     // is the closest enclosing context which has the enclosing package as owner.
700     // (2) The context in which an annotation for any other symbol is evaluated is the
701     // closest enclosing context which has the owner of the class enclosing the symbol as owner.
702     // E.g in
703     //
704     //     package p
705     //     import a.b
706     //     class C {
707     //       import d.e
708     //       @ann m() ...
709     //     }
710     //
711     // `@ann` is evaluated in the context just outside `C`, where the `a.b`
712     // import is visible but the `d.e` import is forgotten. This measure is necessary
713     // in order to avoid cycles.
714     lazy val annotCtx = {
715       var target = sym.owner.lexicallyEnclosingClass
716       if (!target.is(PackageClass)) target = target.owner
717       var c = ctx
718       while (c.owner != target) c = c.outer
719       c
720     }
721     for (annotTree <- untpd.modsDeco(stat).mods.annotations) {
722       val cls = typedAheadAnnotation(annotTree)(annotCtx)
723       if (sym.unforcedAnnotation(cls).isEmpty) {
724         val ann = Annotation.deferred(cls, implicit ctx => typedAnnotation(annotTree))
725         sym.addAnnotation(ann)
726         if (cls == defn.InlineAnnot && sym.is(Method, butNot = Accessor))
727           sym.setFlag(Inline)
728       }
729     }
730   }

873        * only if parent type contains uninstantiated type parameters.
874        */
875       def parentType(parent: untpd.Tree)(implicit ctx: Context): Type =
876         if (parent.isType)
877           typedAheadType(parent, AnyTypeConstructorProto).tpe
878         else {
879           val (core, targs) = stripApply(parent) match {

895        * (1) It must be a class type with a stable prefix (@see checkClassTypeWithStablePrefix)

903         else {

910               case _ => ""
911             }

914           }

919             if (pclazz.is(Sealed) && pclazz.associatedFile != cls.associatedFile)

922           }
923         }
924       }
925

927 
928       val selfInfo =
929         if (self.isEmpty) NoType
930         else if (cls.is(Module)) {
931           val moduleType = cls.owner.thisType select sourceModule
932           if (self.name == nme.WILDCARD) moduleType
933           else recordSym(

944       // which have type trees deriving from its parameters can be

946       // accessors, that's why the constructor needs to be completed before
947       // the parent types are elaborated.

963     }

1008   /** The type signature of a ValDef or DefDef
1009    *  @param mdef     The definition
1010    *  @param sym      Its symbol
1011    *  @param paramFn  A wrapping function that produces the type of the
1012    *                  defined symbol, given its final return type
1013    */
1014   def valOrDefDefSig(mdef: ValOrDefDef, sym: Symbol, typeParams: List[Symbol], paramss: List[List[Symbol]], paramFn: Type => Type)(implicit ctx: Context): Type = {
1015 
1016     def inferredType = {
1017       /** A type for this definition that might be inherited from elsewhere:
1018        *  If this is a setter parameter, the corresponding getter type.
1019        *  If this is a class member, the conjunction of all result types
1020        *  of overridden methods.
1021        *  NoType if neither case holds.
1022        */
1023       val inherited =
1024         if (sym.owner.isTerm) NoType
1025         else {
1026           // TODO: Look only at member of supertype instead?
1027           lazy val schema = paramFn(WildcardType)
1028           val site = sym.owner.thisType
1029           ((NoType: Type) /: sym.owner.info.baseClasses.tail) { (tp, cls) =>
1030             def instantiatedResType(info: Type, tparams: List[Symbol], paramss: List[List[Symbol]]): Type = info match {
1031               case info: PolyType =>
1032                 if (info.paramNames.length == typeParams.length)
1033                   instantiatedResType(info.instantiate(tparams.map(_.typeRef)), Nil, paramss)
1034                 else NoType
1035               case info: MethodType =>
1036                 paramss match {
1037                   case params :: paramss1 if info.paramNames.length == params.length =>
1038                     instantiatedResType(info.instantiate(params.map(_.termRef)), tparams, paramss1)
1039                   case _ =>
1040                     NoType
1041                 }
1042               case _ =>
1043                 if (tparams.isEmpty && paramss.isEmpty) info.widenExpr
1044                 else NoType
1045             }
1046             val iRawInfo =
1047               cls.info.nonPrivateDecl(sym.name).matchingDenotation(site, schema).info
1048             val iResType = instantiatedResType(iRawInfo, typeParams, paramss).asSeenFrom(site, cls)
1049             if (iResType.exists)
1050               typr.println(i"using inherited type for ${mdef.name}; raw: $iRawInfo, inherited: $iResType")
1051             tp & iResType
1052           }
1053         }
1054 
1055       /** The proto-type to be used when inferring the result type from
1056        *  the right hand side. This is `WildcardType` except if the definition
1057        *  is a default getter. In that case, the proto-type is the type of
1058        *  the corresponding parameter where bound parameters are replaced by
1059        *  Wildcards.
1060        */
1061       def rhsProto = sym.asTerm.name collect {
1062         case DefaultGetterName(original, idx) =>
1063           val meth: Denotation =
1064             if (original.isConstructorName && (sym.owner is ModuleClass))
1065               sym.owner.companionClass.info.decl(nme.CONSTRUCTOR)
1066             else
1067               ctx.defContext(sym).denotNamed(original)
1068           def paramProto(paramss: List[List[Type]], idx: Int): Type = paramss match {
1069             case params :: paramss1 =>
1070               if (idx < params.length) wildApprox(params(idx), null, Set.empty)
1071               else paramProto(paramss1, idx - params.length)
1072             case nil =>
1073               WildcardType
1074           }
1075           val defaultAlts = meth.altsWith(_.hasDefaultParams)
1076           if (defaultAlts.length == 1)
1077             paramProto(defaultAlts.head.info.widen.paramInfoss, idx)
1078           else
1079             WildcardType
1080       } getOrElse WildcardType
1081 
1082       // println(s"final inherited for $sym: ${inherited.toString}") !!!
1083       // println(s"owner = ${sym.owner}, decls = ${sym.owner.info.decls.show}")
1084       def isInline = sym.is(FinalOrInline, butNot = Method | Mutable)
1085 
1086       // Widen rhs type and eliminate `|' but keep ConstantTypes if
1087       // definition is inline (i.e. final in Scala2) and keep module singleton types
1088       // instead of widening to the underlying module class types.
1089       def widenRhs(tp: Type): Type = tp.widenTermRefExpr match {
1090         case ctp: ConstantType if isInline => ctp
1091         case ref: TypeRef if ref.symbol.is(ModuleClass) => tp
1092         case _ => tp.widen.widenUnion
1093       }
1094 
1095       // Replace aliases to Unit by Unit itself. If we leave the alias in
1096       // it would be erased to BoxedUnit.
1097       def dealiasIfUnit(tp: Type) = if (tp.isRef(defn.UnitClass)) defn.UnitType else tp
1098 
1099       val rhsCtx = ctx.addMode(Mode.InferringReturnType)
1100       def rhsType = typedAheadExpr(mdef.rhs, inherited orElse rhsProto)(rhsCtx).tpe
1101 
1102       // Approximate a type `tp` with a type that does not contain skolem types.
1103       val deskolemize = new ApproximatingTypeMap {
1104         def apply(tp: Type) = /*trace(i"deskolemize($tp) at $variance", show = true)*/ {
1105           tp match {
1106             case tp: SkolemType => range(tp.bottomType, atVariance(1)(apply(tp.info)))
1107             case _ => mapOver(tp)
1108           }
1109         }
1110       }
1111 
1112       def cookedRhsType = deskolemize(dealiasIfUnit(widenRhs(rhsType)))
1113       lazy val lhsType = fullyDefinedType(cookedRhsType, "right-hand side", mdef.pos)
1114       //if (sym.name.toString == "y") println(i"rhs = $rhsType, cooked = $cookedRhsType")
1115       if (inherited.exists)
1116         if (sym.is(Final, butNot = Method) && lhsType.isInstanceOf[ConstantType])
1117           lhsType // keep constant types that fill in for a non-constant (to be revised when inline has landed).
1118         else inherited
1119       else {
1120         if (sym is Implicit)
1121           mdef match {
1122             case _: DefDef => missingType(sym, "result ")
1123             case _: ValDef if sym.owner.isType => missingType(sym, "")
1124             case _ =>
1125           }
1126         lhsType orElse WildcardType
1127       }
1128     }
1129 
1130     val tptProto = mdef.tpt match {
1131       case _: untpd.DerivedTypeTree =>
1132         WildcardType
1133       case TypeTree() =>
1134         checkMembersOK(inferredType, mdef.pos)
1135       case DependentTypeTree(tpFun) =>
1136         tpFun(paramss.head)
1137       case TypedSplice(tpt: TypeTree) if !isFullyDefined(tpt.tpe, ForceDegree.none) =>
1138         val rhsType = typedAheadExpr(mdef.rhs, tpt.tpe).tpe
1139         mdef match {
1140           case mdef: DefDef if mdef.name == nme.ANON_FUN =>
1141             val hygienicType = avoid(rhsType, paramss.flatten)
1142             if (!hygienicType.isValueType || !(hygienicType <:< tpt.tpe))
1143               ctx.error(i"return type ${tpt.tpe} of lambda cannot be made hygienic;\n" +
1144                 i"it is not a supertype of the hygienic type $hygienicType", mdef.pos)
1145             //println(i"lifting $rhsType over $paramss -> $hygienicType = ${tpt.tpe}")
1146             //println(TypeComparer.explained { implicit ctx => hygienicType <:< tpt.tpe })
1147           case _ =>
1148         }
1149         WildcardType
1150       case _ =>
1151         WildcardType
1152     }
1153     paramFn(typedAheadType(mdef.tpt, tptProto).tpe)
