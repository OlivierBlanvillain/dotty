41   /** An abstraction of a class info, consisting of
42    *   - the intersection of its parents,
43    *   - refined by all non-private fields, methods, and type members,
44    *   - abstracted over all type parameters (into a type lambda)
45    *   - where all references to `this` of the class are closed over in a RecType.
46    */
47   def classBound(info: ClassInfo)(implicit ctx: Context): Type = {
48     val cls = info.cls
49     val parentType = info.parents.reduceLeft(ctx.typeComparer.andType(_, _))
50 
51     def addRefinement(parent: Type, decl: Symbol) = {
52       val inherited =
53         parentType.findMember(decl.name, cls.thisType, excluded = Private)
54           .suchThat(decl.matches(_))
55       val inheritedInfo = inherited.info
56       if (inheritedInfo.exists && decl.info <:< inheritedInfo && !(inheritedInfo <:< decl.info)) {
57         val r = RefinedType(parent, decl.name, decl.info)
58         typr.println(i"add ref $parent $decl --> " + r)
59         r
60       }
61       else
62         parent
63     }
64 
65     def close(tp: Type) = RecType.closeOver(rt => tp.substThis(cls, rt.recThis))
66 
67     def isRefinable(sym: Symbol) = !sym.is(Private) && !sym.isConstructor
68     val refinableDecls = info.decls.filter(isRefinable)
69     val raw = (parentType /: refinableDecls)(addRefinement)
70     HKTypeLambda.fromParams(cls.typeParams, raw) match {
71       case tl: HKTypeLambda => tl.derivedLambdaType(resType = close(tl.resType))
72       case tp => close(tp)
73     }
74   }

76   /** An upper approximation of the given type `tp` that does not refer to any symbol in `symsToAvoid`.
77    *  We need to approximate with ranges:
78    *
79    *    term references to symbols in `symsToAvoid`,
80    *    term references that have a widened type of which some part refers
81    *    to a symbol in `symsToAvoid`,
82    *    type references to symbols in `symsToAvoid`,
83    *    this types of classes in `symsToAvoid`.
84    *
85    *  Type variables that would be interpolated to a type that
86    *  needs to be widened are replaced by the widened interpolation instance.
87    */
88   def avoid(tp: Type, symsToAvoid: => List[Symbol])(implicit ctx: Context): Type = {
89     val widenMap = new ApproximatingTypeMap {
90       lazy val forbidden = symsToAvoid.toSet
91       def toAvoid(sym: Symbol) = !sym.isStatic && forbidden.contains(sym)
92       def partsToAvoid = new NamedPartsAccumulator(tp => toAvoid(tp.symbol))
93       def apply(tp: Type): Type = tp match {
94         case tp: TermRef
95         if toAvoid(tp.symbol) || partsToAvoid(mutable.Set.empty, tp.info).nonEmpty =>
96           tp.info.widenExpr.dealias match {
97             case info: SingletonType => apply(info)
98             case info => range(tp.info.bottomType, apply(info))
99           }
100         case tp: TypeRef if toAvoid(tp.symbol) =>
101           tp.info match {
102             case TypeAlias(alias) =>
103               apply(alias)
104             case TypeBounds(lo, hi) =>
105               range(atVariance(-variance)(apply(lo)), apply(hi))
106             case info: ClassInfo =>
107               range(tp.bottomType, apply(classBound(info)))
108             case _ =>
109               range(tp.bottomType, tp.topType) // should happen only in error cases
110           }
111         case tp: ThisType if toAvoid(tp.cls) =>
112           range(tp.bottomType, apply(classBound(tp.cls.classInfo)))
113         case tp: SkolemType if partsToAvoid(mutable.Set.empty, tp.info).nonEmpty =>
114           range(tp.info.bottomType, apply(tp.info))
115         case tp: TypeVar if ctx.typerState.constraint.contains(tp) =>
116           val lo = ctx.typeComparer.instanceType(
117             tp.origin, fromBelow = variance > 0 || variance == 0 && tp.hasLowerBound)
118           val lo1 = apply(lo)
119           if (lo1 ne lo) lo1 else tp
120         case _ =>
121           mapOver(tp)
122       }
123 
124       /** Three deviations from standard derivedSelect:
125        *   1. We first try a widening conversion to the type's info with
126        *      the original prefix. Since the original prefix is known to
127        *      be a subtype of the returned prefix, this can improve results.
128        *   2. Then, if the approximation result is a singleton reference C#x.type, we
129        *      replace by the widened type, which is usually more natural.
130        *   3. Finally, we need to handle the case where the prefix type does not have a member
131        *      named `tp.name` anymmore. In that case, we need to fall back to Bot..Top.
132        */
133       override def derivedSelect(tp: NamedType, pre: Type) =
134         if (pre eq tp.prefix)
135           tp
136         else tryWiden(tp, tp.prefix).orElse {
137           if (tp.isTerm && variance > 0 && !pre.isSingleton)
138           	apply(tp.info.widenExpr)
139           else if (upper(pre).member(tp.name).exists)
140             super.derivedSelect(tp, pre)
141           else
142             range(tp.bottomType, tp.topType)
143         }
144     }
145 
146     widenMap(tp)
147   }

174   /** If `tpe` is a named type, check that its denotation is accessible in the
175    *  current context. Return the type with those alternatives as denotations
176    *  which are accessible.
177    *
178    *  Also performs the following normalizations on the type `tpe`.
179    *  (1) parameter accessors are always dereferenced.
180    *  (2) if the owner of the denotation is a package object, it is assured
181    *      that the package object shows up as the prefix.
182    */
183   def ensureAccessible(tpe: Type, superAccess: Boolean, pos: Position)(implicit ctx: Context): Type = {
184     def test(tpe: Type, firstTry: Boolean): Type = tpe match {
185       case tpe: NamedType =>
186         val pre = tpe.prefix
187         val name = tpe.name
188         val d = tpe.denot.accessibleFrom(pre, superAccess)
189         if (!d.exists) {
190           // it could be that we found an inaccessible private member, but there is
191           // an inherited non-private member with the same name and signature.
192           val d2 = pre.nonPrivateMember(name)
193           if (reallyExists(d2) && firstTry)
194             test(NamedType(pre, name, d2), false)
195           else if (pre.derivesFrom(defn.DynamicClass)) {
196             TryDynamicCallType
197           } else {
198             val alts = tpe.denot.alternatives.map(_.symbol).filter(_.exists)
199             var packageAccess = false
200             val what = alts match {
201               case Nil =>
202                 i"$name cannot be accessed as a member of $pre"
203               case sym :: Nil =>
204                 if (sym.owner.is(Package)) {
205                   packageAccess = true
206                   i"${sym.showLocated} cannot be accessed"
207                 }
208                 else {
209                   val symStr = if (sym.owner == pre.typeSymbol) sym.show else sym.showLocated
210                   i"$symStr cannot be accessed as a member of $pre"
211                 }
212               case _ =>
213                 em"none of the overloaded alternatives named $name can be accessed as members of $pre"
214             }
215             val where =
216               if (!ctx.owner.exists) ""
217               else if (packageAccess) i" from nested ${ctx.owner.enclosingPackageClass}"
218               else i" from ${ctx.owner.enclosingClass}"
219             val whyNot = new StringBuffer
220             alts foreach (_.isAccessibleFrom(pre, superAccess, whyNot))
221             if (tpe.isError) tpe
222             else errorType(ex"$what$where.$whyNot", pos)
223           }
224         }
225         else ctx.makePackageObjPrefixExplicit(tpe withDenot d)
226       case _ =>
227         tpe
228     }
229     test(tpe, true)
230   }

376   def assignType(tree: untpd.TypeApply, fn: Tree, args: List[Tree])(implicit ctx: Context) = {
377     val ownType = fn.tpe.widen match {
378       case pt: TypeLambda =>
379         val paramNames = pt.paramNames
380         if (hasNamedArg(args)) {
381           val paramBoundsByName = paramNames.zip(pt.paramInfos).toMap
382 
383           // Type arguments which are specified by name (immutable after this first loop)
384           val namedArgMap = new mutable.HashMap[Name, Type]
385           for (NamedArg(name, arg) <- args)
386             if (namedArgMap.contains(name))
387               ctx.error(DuplicateNamedTypeParameter(name), arg.pos)
388             else if (!paramNames.contains(name))
389               ctx.error(UndefinedNamedTypeParameter(name, paramNames), arg.pos)
390             else
391               namedArgMap(name) = arg.tpe
392 
393           // Holds indexes of non-named typed arguments in paramNames
394           val gapBuf = new mutable.ListBuffer[Int]
395           def nextPoly(idx: Int) = {
396             val newIndex = gapBuf.length
397             gapBuf += idx
398             // Re-index unassigned type arguments that remain after transformation
399             pt.paramRefs(newIndex)
400           }
401 
402           // Type parameters after naming assignment, conserving paramNames order
403           val normArgs: List[Type] = paramNames.zipWithIndex.map { case (pname, idx) =>
404             namedArgMap.getOrElse(pname, nextPoly(idx))
405           }
406 
407           val transform = new TypeMap {
408             def apply(t: Type) = t match {
409               case TypeParamRef(`pt`, idx) => normArgs(idx)
410               case _ => mapOver(t)
411             }
412           }
413           val resultType1 = transform(pt.resultType)
414           if (gapBuf.isEmpty) resultType1
415           else {
416             val gaps = gapBuf.toList
417             pt.derivedLambdaType(
418               gaps.map(paramNames),
419               gaps.map(idx => transform(pt.paramInfos(idx)).bounds),
420               resultType1)
421           }
422         }
423         else {
424           val argTypes = args.tpes
425           if (sameLength(argTypes, paramNames)) pt.instantiate(argTypes)
426           else wrongNumberOfTypeArgs(fn.tpe, pt.typeParams, args, tree.pos)
427         }
428       case err: ErrorType =>
429         err
430       case _ =>
431         //println(i"bad type: $fn: ${fn.symbol} / ${fn.symbol.isType} / ${fn.symbol.info}") // DEBUG
432         errorType(err.takesNoParamsStr(fn, "type "), tree.pos)
433     }
434 
435     tree.withType(ownType)
436   }
