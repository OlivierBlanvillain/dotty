78         def discardForView(tpw: Type, argType: Type): Boolean = tpw match {
79           case mt: MethodType =>
80             mt.isImplicitMethod ||
81             mt.paramInfos.length != 1 ||
82             !ctx.test(implicit ctx => argType relaxed_<:< mt.paramInfos.head)
83           case poly: PolyType =>
84             // We do not need to call ProtoTypes#constrained on `poly` because
85             // `refMatches` is always called with mode TypevarsMissContext enabled.

87               case mt: MethodType =>
88                 mt.isImplicitMethod ||
89                 mt.paramInfos.length != 1 ||
90                 !ctx.test(implicit ctx => argType relaxed_<:< wildApprox(mt.paramInfos.head, null, Set.empty))
91               case rtp =>
92                 discardForView(wildApprox(rtp, null, Set.empty), argType)
93             }
94           case tpw: TermRef =>
95             false // can't discard overloaded refs
96           case tpw =>
97             // Only direct instances of Function1 and direct or indirect instances of <:< are eligible as views.
98             // However, Predef.$conforms is not eligible, because it is a no-op.
99             //
100             // In principle, it would be cleanest if only implicit methods qualified
101             // as implicit conversions. We could achieve that by having standard conversions like
102             // this in Predef:
103             //
104             //    implicit def convertIfConforms[A, B](x: A)(implicit ev: A <:< B): B = ev(a)
105             //    implicit def convertIfConverter[A, B](x: A)(implicit ev: ImplicitConverter[A, B]): B = ev(a)
106             //
107             // (Once `<:<` inherits from `ImplicitConverter` we only need the 2nd one.)
108             // But clauses like this currently slow down implicit search a lot, because
109             // they are eligible for all pairs of types, and therefore are tried too often.
110             // We emulate instead these conversions directly in the search.
111             // The reason for leaving out `Predef_conforms` is that we know it adds
112             // nothing since it only relates subtype with supertype.
113             //
114             // We keep the old behavior under -language:Scala2.
115             val isFunctionInS2 =
116               ctx.scala2Mode && tpw.derivesFrom(defn.FunctionClass(1)) && ref.symbol != defn.Predef_conforms
117             val isImplicitConverter = tpw.derivesFrom(defn.Predef_ImplicitConverter)
118             val isConforms = // An implementation of <:< counts as a view, except that $conforms is always omitted
119                 tpw.derivesFrom(defn.Predef_Conforms) && ref.symbol != defn.Predef_conforms
120             !(isFunctionInS2 || isImplicitConverter || isConforms)
121         }

124           case tpw: MethodType => !tpw.isImplicitMethod
125           case _ => false
126         }

128         def discard = pt match {
129           case pt: ViewProto => discardForView(ref.widen, pt.argType)
130           case _: ValueTypeOrProto => !defn.isFunctionType(pt) && discardForValueType(ref.widen)

132         }

138           }

142           }
143         }
144       }

147       else {

150           .filter(ref => nestedCtx.test(implicit ctx => refMatches(ref.underlyingRef)))

385   def implicitScope(rootTp: Type, liftingCtx: Context): OfTypeImplicits = {
386 
387     val seen: mutable.Set[Type] = mutable.Set()
388     val incomplete: mutable.Set[Type] = mutable.Set()
389 
390     /** Replace every typeref that does not refer to a class by a conjunction of class types
391      *  that has the same implicit scope as the original typeref. The motivation for applying
392      *  this map is that it reduces the total number of types for which we need to
393      *  compute and cache the implicit scope; all variations wrt type parameters or
394      *  abstract types are eliminated.
395      */
396     object liftToClasses extends TypeMap {
397       override implicit protected val ctx: Context = liftingCtx
398       override def stopAtStatic = true
399       def apply(tp: Type) = tp match {
400         case tp: TypeRef if tp.symbol.isAbstractOrAliasType =>
401           val pre = tp.prefix
402           def joinClass(tp: Type, cls: ClassSymbol) =
403             AndType.make(tp, cls.typeRef.asSeenFrom(pre, cls.owner))
404           val lead = if (tp.prefix eq NoPrefix) defn.AnyType else apply(tp.prefix)
405           (lead /: tp.classSymbols)(joinClass)
406         case tp: TypeVar =>
407           apply(tp.underlying)
408         case tp: AppliedType if !tp.tycon.typeSymbol.isClass =>
409           def applyArg(arg: Type) = arg match {
410             case TypeBounds(lo, hi) => AndType.make(lo, hi)
411             case WildcardType(TypeBounds(lo, hi)) => AndType.make(lo, hi)
412             case _ => arg
413           }
414           (apply(tp.tycon) /: tp.args)((tc, arg) => AndType.make(tc, applyArg(arg)))
415         case tp: TypeLambda =>
416           apply(tp.resType)
417         case _ =>
418           mapOver(tp)
419       }
420     }

468    /** The implicit scope of type `tp`
469      *  @param isLifted    Type `tp` is the result of a `liftToClasses` application
470      */
471     def iscope(tp: Type, isLifted: Boolean = false): OfTypeImplicits = {
472       val canCache = Config.cacheImplicitScopes && tp.hash != NotCached
473       def computeIScope() = {
474         val savedEphemeral = ctx.typerState.ephemeral
475         ctx.typerState.ephemeral = false
476         try {
477           val liftedTp = if (isLifted) tp else liftToClasses(tp)
478           val refs =
479             if (liftedTp ne tp)
480               iscope(liftedTp, isLifted = true).companionRefs
481             else
482               collectCompanions(tp)
483           val result = new OfTypeImplicits(tp, refs)(ctx)
484           if (ctx.typerState.ephemeral)
485             record("ephemeral cache miss: implicitScope")
486           else if (canCache &&
487                    ((tp eq rootTp) ||          // first type traversed is always cached
488                     !incomplete.contains(tp))) // other types are cached if they are not incomplete
489             implicitScopeCache(tp) = result
490           result
491         }
492         finally ctx.typerState.ephemeral |= savedEphemeral
493       }
494       if (canCache) implicitScopeCache.getOrElse(tp, computeIScope())
495       else computeIScope()
496     }
497 
498     iscope(rootTp)
499   }

945       /** Try to find a best matching implicit term among all the candidates in `pending`.
946        *  @param pending   The list of candidates that remain to be tested
947        *  @param found     The result obtained from previously tried candidates
948        *  @param rfailures A list of all failures from previously tried candidates in reverse order
949        *
950        *  The scheme is to try candidates one-by-one. If a trial is successful:
951        *   - if the query term is a `Not[T]` treat it a failure,
952        *   - otherwise, if a previous search was also successful, handle the ambiguity
953        *     in `disambiguate`,
954        *   - otherwise, continue the search with all candidates that are not strictly
955        *     worse than the succesful candidate.
956        *  If a trial failed:
957        *    - if the query term is a `Not[T]` treat it as a success,
958        *    - otherwise, if the failure is an ambiguity, try to heal it (see @healAmbiguous)
959        *      and return an ambiguous error otherwise. However, under Scala2 mode this is
960        *      treated as a simple failure, with a warning that semantics will change.
961        *    - otherwise add the failure to `rfailures` and continue testing the other candidates.

964         pending match  {
965           case cand :: remaining =>
966             negateIfNot(tryImplicit(cand)) match {
967               case fail: SearchFailure =>
968                 if (fail.isAmbiguous)

971                     if (result.isSuccess)
972                       warnAmbiguousNegation(fail.reason.asInstanceOf[AmbiguousImplicits])
973                     result
974                   }
975                   else healAmbiguous(remaining, fail)
976                 else rank(remaining, found, fail :: rfailures)
977               case best: SearchSuccess =>
978                 if (ctx.mode.is(Mode.ImplicitExploration) || isCoherent)
979                   best
980                 else disambiguate(found, best) match {

1017        *  We prefer (in order of importance)
1018        *   1. more deeply nested definitions
1019        *   2. definitions in subclasses
1020        *   3. definitions with fewer implicit parameters
1021        *  The reason for (3) is that we want to fail fast if the search type
1022        *  is underconstrained. So we look for "small" goals first, because that
1023        *  will give an ambiguity quickly.

1026         val level1 = cand1.level
1027         val level2 = cand2.level
1028         if (level1 > level2) return true
1029         if (level1 < level2) return false
1030         val sym1 = cand1.ref.symbol
1031         val sym2 = cand2.ref.symbol
1032         val ownerScore = compareOwner(sym1.maybeOwner, sym2.maybeOwner)
1033         if (ownerScore > 0) return true
1034         if (ownerScore < 0) return false
1035         val arity1 = sym1.info.firstParamTypes.length
1036         val arity2 = sym2.info.firstParamTypes.length
1037         if (arity1 < arity2) return true
1038         if (arity1 > arity2) return false
1039         false

1094   /** The number of applications and refinements in this type, after all aliases are expanded */
1095   private def typeSize(tp: Type)(implicit ctx: Context): Int = {
1096     val accu = new TypeAccumulator[Int] {
1097       def apply(n: Int, tp: Type): Int = tp match {
1098         case tp: AppliedType =>
1099           foldOver(n + 1, tp)
1100         case tp: RefinedType =>
1101           foldOver(n + 1, tp)
1102         case tp: TypeRef if tp.info.isAlias =>
1103           apply(n, tp.superType)
1104         case _ =>
1105           foldOver(n, tp)
1106       }
1107     }
1108     accu.apply(0, tp)
1109   }

1111   /** Check for possible divergence. If one is detected return the current search history
1112    *  (this will be used as a criterion to abandon the implicit search in rankImplicits).
1113    *  If no divergence is detected, produce a new search history nested in the current one
1114    *  which records that we are now also looking for type `proto`.
1115    *
1116    *  As long as `searchDepth` is lower than the `XminImplicitSearchDepth` value
1117    *  in settings, a new history is always produced, so the implicit search is always
1118    *  undertaken. If `searchDepth` matches or exceeds the `XminImplicitSearchDepth` value,
1119    *  we test that the new search is for a class that is either not yet in the set of
1120    *  `seen` classes, or the complexity of the type `proto` being searched for is strictly
1121    *  lower than the complexity of the type that was previously encountered and that had
1122    *  the same class symbol as `proto`. A possible divergence is detected if that test fails.
1123    */
1124   def nest(proto: Type)(implicit ctx: Context): SearchHistory = {
1125     if (searchDepth < ctx.settings.XminImplicitSearchDepth.value)
1126       new SearchHistory(searchDepth + 1, seen)
1127     else {
1128       val size = typeSize(proto)
1129       def updateMap(csyms: List[ClassSymbol], seen: Map[ClassSymbol, Int]): SearchHistory = csyms match {
1130         case csym :: csyms1 =>
1131           seen get csym match {
1132             // proto complexity is >= than the last time it was seen  diverge
1133             case Some(prevSize) if size >= prevSize => this
1134             case _ => updateMap(csyms1, seen.updated(csym, size))
1135           }
1136         case _ =>
1137           new SearchHistory(searchDepth + 1, seen)
1138       }
