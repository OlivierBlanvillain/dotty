10 /** Handles rewriting of Scala2 files to Dotty */
11 object Rewrites {
12   private class PatchedFiles extends mutable.HashMap[SourceFile, Patches]
13 
14   private case class Patch(pos: Position, replacement: String) {
15     def delta = replacement.length - (pos.end - pos.start)
16   }
17 
18   private class Patches(source: SourceFile) {
19     private val pbuf = new mutable.ListBuffer[Patch]()
20 
21     def addPatch(pos: Position, replacement: String): Unit =
22       pbuf += Patch(pos, replacement)
23 
24     def apply(cs: Array[Char]): Array[Char] = {
25       val delta = pbuf.map(_.delta).sum
26       val patches = pbuf.toList.sortBy(_.pos.start)
27       if (patches.nonEmpty)
28         patches reduceLeft {(p1, p2) =>
29           assert(p1.pos.end <= p2.pos.start, s"overlapping patches: $p1 and $p2")
30           p2
31         }
32       val ds = new Array[Char](cs.length + delta)
33       @tailrec def loop(ps: List[Patch], inIdx: Int, outIdx: Int): Unit = {
34         def copy(upTo: Int): Int = {
35           val untouched = upTo - inIdx
36           Array.copy(cs, inIdx, ds, outIdx, untouched)
37           outIdx + untouched
38         }
39         ps match {
40           case patch @ Patch(pos, replacement) :: ps1 =>
41             val outNew = copy(pos.start)
42             replacement.copyToArray(ds, outNew)
43             loop(ps1, pos.end, outNew + replacement.length)
44           case Nil =>
45             val outNew = copy(cs.length)
46             assert(outNew == ds.length, s"$outNew != ${ds.length}")
47         }
48       }
49       loop(patches, 0, 0)
50       ds
51     }
52 
53     def writeBack(): Unit = {
54       val out = source.file.output
55       val chars = apply(source.underlying.content)
56       val bytes = new String(chars).getBytes
57       out.write(bytes)
58       out.close()
59     }
60   }
61 
62   /** If -rewrite is set, record a patch that replaces the range
63    *  given by `pos` in `source` by `replacement`
64    */
65   def patch(source: SourceFile, pos: Position, replacement: String)(implicit ctx: Context): Unit =
66     for (rewrites <- ctx.settings.rewrite.value)
67       rewrites.patched
68         .getOrElseUpdate(source, new Patches(source))
69         .addPatch(pos, replacement)
70 
71   /** Patch position in `ctx.compilationUnit.source`. */
72   def patch(pos: Position, replacement: String)(implicit ctx: Context): Unit =
73     patch(ctx.compilationUnit.source, pos, replacement)
74 
75   /** If -rewrite is set, apply all patches and overwrite patched source files.
76    */
77   def writeBack()(implicit ctx: Context) =
78     for (rewrites <- ctx.settings.rewrite.value; source <- rewrites.patched.keys) {
79       ctx.echo(s"[patched file ${source.file.path}]")
80       rewrites.patched(source).writeBack()
81     }
82 }
