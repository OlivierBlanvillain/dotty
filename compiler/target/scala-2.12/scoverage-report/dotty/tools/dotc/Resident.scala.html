9 /** A compiler which stays resident between runs. This is more of a PoC than
10  *  something that's expected to be used often
11  *
12  *  Usage:
13  *
14  *  > scala dotty.tools.dotc.Resident <options> <initial files>
15  *
16  *  dotc> "more options and files to compile"
17  *
18  *  ...
19  *
20  *  dotc> :reset  // reset all options to the ones passed on the command line
21  *
22  *  ...
23  *
24  *  dotc> :q     // quit
25  */
26 class Resident extends Driver {
27 
28   object residentCompiler extends Compiler
29 
30   override def sourcesRequired = false
31 
32   private val quit = ":q"
33   private val reset = ":reset"
34   private val prompt = "dotc> "
35 
36   private def getLine() = {
37     Console.print(prompt)
38     try scala.io.StdIn.readLine() catch { case _: EOFException => quit }
39   }
40 
41   final override def process(args: Array[String], rootCtx: Context): Reporter = {
42     @tailrec def loop(args: Array[String], prevCtx: Context): Reporter = {
43       var (fileNames, ctx) = setup(args, prevCtx)
44       doCompile(residentCompiler, fileNames)(ctx)
45       var nextCtx = ctx
46       var line = getLine()
47       while (line == reset) {
48         nextCtx = rootCtx
49         line = getLine()
50       }
51       if (line.startsWith(quit)) ctx.reporter
52       else loop(line split "\\s+", nextCtx)
53     }
54     loop(args, rootCtx)
55   }
