15 /** Provides methods to produce fully parameterized versions of instance methods,
16  *  where the `this` of the enclosing class is abstracted out in an extra leading
17  *  `$this` parameter and type parameters of the class become additional type
18  *  parameters of the fully parameterized method.
19  *
20  *  Example usage scenarios are:
21  *
22  *    - extension methods of value classes
23  *    - implementations of trait methods
24  *    - static protected accessors
25  *    - local methods produced by tailrec transform
26  *
27  *  Note that the methods lift out type parameters of the class containing
28  *  the instance method, but not type parameters of enclosing classes. The
29  *  fully instantiated method therefore needs to be put in a scope "close"
30  *  to the original method, i.e. they need to share the same outer pointer.
31  *  Examples of legal positions are: in the companion object, or as a local
32  *  method inside the original method.
33  *
34  *  Note: The scheme does not handle yet methods where type parameter bounds
35  *  depend on value parameters of the enclosing class, as in:
36  *
37  *      class C(val a: String) extends AnyVal {
38  *        def foo[U <: a.type]: Unit = ...
39  *      }
40  *
41  *  The expansion of method `foo` would lead to
42  *
43  *      def foo$extension[U <: $this.a.type]($this: C): Unit = ...
44  *
45  *  which is not typable. Not clear yet what to do. Maybe allow PolyTypes
46  *  to follow method parameters and translate to the following:
47  *
48  *      def foo$extension($this: C)[U <: $this.a.type]: Unit = ...
49  *
50  *  @see class-dependent-extension-method.scala in pending/pos.
51  */

74   /** Converts the type `info` of a member of class `clazz` to a method type that
75    *  takes the `this` of the class and any type parameters of the class
76    *  as additional parameters. Example:
77    *
78    *    class Foo[+A <: AnyRef](val xs: List[A]) extends AnyVal {
79    *      def baz[B >: A](x: B): List[B] = ...
80    *    }
81    *
82    *  leads to:
83    *
84    *    object Foo {
85    *      def extension$baz[B >: A <: Any, A >: Nothing <: AnyRef]($this: Foo[A])(x: B): List[B]
86    *    }
87    *
88    *  If a self type is present, $this has this self type as its type.
89    *
90    *  @param abstractOverClass  if true, include the type parameters of the class in the method's list of type parameters.
91    *  @param liftThisType       if true, require created $this to be $this: (Foo[A] & Foo,this).
92    *                            This is needed if created member stays inside scope of Foo(as in tailrec)
93    */
94   def fullyParameterizedType(info: Type, clazz: ClassSymbol, abstractOverClass: Boolean = true, liftThisType: Boolean = false)(implicit ctx: Context): Type = {
95     val (mtparamCount, origResult) = info match {
96       case info: PolyType => (info.paramNames.length, info.resultType)
97       case info: ExprType => (0, info.resultType)
98       case _ => (0, info)
99     }
100     val ctparams = if (abstractOverClass) clazz.typeParams else Nil
101     val ctnames = ctparams.map(_.name)
102     val ctvariances = ctparams.map(_.variance)
103 
104     /** The method result type */
105     def resultType(mapClassParams: Type => Type) = {
106       val thisParamType = mapClassParams(clazz.classInfo.selfType)
107       val firstArgType = if (liftThisType) thisParamType & clazz.thisType else thisParamType
108       MethodType(nme.SELF :: Nil)(
109           mt => firstArgType :: Nil,
110           mt => mapClassParams(origResult).substThisUnlessStatic(clazz, mt.newParamRef(0)))
111     }
112 
113     /** Replace class type parameters by the added type parameters of the polytype `pt` */
114     def mapClassParams(tp: Type, pt: PolyType): Type = {
115       val classParamsRange = (mtparamCount until mtparamCount + ctparams.length).toList
116       tp.substDealias(ctparams, classParamsRange map (pt.paramRefs(_)))
117     }
118 
119     /** The bounds for the added type parameters of the polytype `pt` */
120     def mappedClassBounds(pt: PolyType): List[TypeBounds] =
121       ctparams.map(tparam => mapClassParams(tparam.info, pt).bounds)
122 
123     info match {
124       case info: PolyType =>
125         PolyType(info.paramNames ++ ctnames)(
126           pt =>
127             (info.paramInfos.map(mapClassParams(_, pt).bounds) ++
128              mappedClassBounds(pt)).mapConserve(_.subst(info, pt).bounds),
129           pt => resultType(mapClassParams(_, pt)).subst(info, pt))
130       case _ =>
131         if (ctparams.isEmpty) resultType(identity)
132         else PolyType(ctnames)(mappedClassBounds, pt => resultType(mapClassParams(_, pt)))
133     }
134   }

136   /** The type parameters (skolems) of the method definition `originalDef`,
137    *  followed by the class parameters of its enclosing class.
138    */
139   private def allInstanceTypeParams(originalDef: DefDef, abstractOverClass: Boolean)(implicit ctx: Context): List[Symbol] =
140     if (abstractOverClass)
141       originalDef.tparams.map(_.symbol) ::: originalDef.symbol.enclosingClass.typeParams
142     else originalDef.tparams.map(_.symbol)

144   /** Given an instance method definition `originalDef`, return a
145    *  fully parameterized method definition derived from `originalDef`, which
146    *  has `derived` as symbol and `fullyParameterizedType(originalDef.symbol.info)`
147    *  as info.
148    *  `abstractOverClass` defines weather the DefDef should abstract over type parameters
149    *  of class that contained original defDef
150    */
151   def fullyParameterizedDef(derived: TermSymbol, originalDef: DefDef, abstractOverClass: Boolean = true)(implicit ctx: Context): Tree =
152     polyDefDef(derived, trefs => vrefss => {
153       val origMeth = originalDef.symbol
154       val origClass = origMeth.enclosingClass.asClass
155       val origTParams = allInstanceTypeParams(originalDef, abstractOverClass)
156       val origVParams = originalDef.vparamss.flatten map (_.symbol)
157       val thisRef :: argRefs = vrefss.flatten
158 
159       /** If tree should be rewired, the rewired tree, otherwise EmptyTree.
160        *  @param   targs  Any type arguments passed to the rewired tree.
161        */
162       def rewireTree(tree: Tree, targs: List[Tree])(implicit ctx: Context): Tree = {
163         def rewireCall(thisArg: Tree): Tree = {
164           val rewired = rewiredTarget(tree, derived)
165           if (rewired.exists) {
166             val base = thisArg.tpe.baseType(origClass)
167             assert(base.exists)
168             ref(rewired.termRef)
169               .appliedToTypeTrees(targs ++ base.argInfos.map(TypeTree(_)))
170               .appliedTo(thisArg)
171           } else EmptyTree
172         }
173         tree match {
174           case Return(expr, from) if !from.isEmpty =>
175             val rewired = rewiredTarget(from, derived)
176             if (rewired.exists)
177               tpd.cpy.Return(tree)(expr, Ident(rewired.termRef))
178             else
179               EmptyTree
180           case Ident(_) => rewireCall(thisRef)
181           case Select(qual, _) => rewireCall(qual)
182           case tree @ TypeApply(fn, targs1) =>
183             assert(targs.isEmpty)
184             rewireTree(fn, targs1)
185           case _ => EmptyTree
186         }
187       }
188 
189       /** Type rewiring is needed because a previous reference to an instance
190        *  method might still persist in the types of enclosing nodes. Example:
191        *
192        *     if (true) this.imeth else this.imeth
193        *
194        *  is rewritten to
195        *
196        *      if (true) xmeth($this) else xmeth($this)
197        *
198        *  but the type `this.imeth` still persists as the result type of the `if`,
199        *  because it is kept by the `cpy` operation of the tree transformer.
200        *  It needs to be rewritten to the common result type of `imeth` and `xmeth`.
201        */
202       def rewireType(tpe: Type) = tpe match {
203         case tpe: TermRef if rewiredTarget(tpe.symbol, derived).exists => tpe.widen
204         case _ => tpe
205       }
206 
207       new TreeTypeMap(
208         typeMap = rewireType(_)
209           .substDealias(origTParams, trefs)
210           .subst(origVParams, argRefs.map(_.tpe))
211           .substThisUnlessStatic(origClass, thisRef.tpe),
212         treeMap = {
213           case tree: This if tree.symbol == origClass => thisRef
214           case tree => rewireTree(tree, Nil) orElse tree
215         },
216         oldOwners = origMeth :: Nil,
217         newOwners = derived :: Nil
218       ).transform(originalDef.rhs)
219     })
220 
221   /** A forwarder expression which calls `derived`, passing along
222    *  - if `abstractOverClass` the type parameters and enclosing class parameters of originalDef`,
223    *  - the `this` of the enclosing class,
224    *  - the value parameters of the original method `originalDef`.
225    */
226   def forwarder(derived: TermSymbol, originalDef: DefDef, abstractOverClass: Boolean = true, liftThisType: Boolean = false)(implicit ctx: Context): Tree = {
227     val fun =
228       ref(derived.termRef)
229         .appliedToTypes(allInstanceTypeParams(originalDef, abstractOverClass).map(_.typeRef))
230         .appliedTo(This(originalDef.symbol.enclosingClass.asClass))
231 
232     (if (!liftThisType)
233       fun.appliedToArgss(originalDef.vparamss.nestedMap(vparam => ref(vparam.symbol)))
234     else {
235       // this type could have changed on forwarding. Need to insert a cast.
236       val args = (originalDef.vparamss, fun.tpe.paramInfoss).zipped.map((vparams, paramTypes) =>
237         (vparams, paramTypes).zipped.map((vparam, paramType) => {
238           assert(vparam.tpe <:< paramType.widen) // type should still conform to widened type
239           ref(vparam.symbol).ensureConforms(paramType)
240         })
241       )
242       fun.appliedToArgss(args)
243 
244     }).withPos(originalDef.rhs.pos)
245   }
