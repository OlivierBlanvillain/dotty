9 /** If a block has a statement that evaluates to Nothing:
10  *   - Every pure statement dirrectly preceding an expression that returns Nothing can be removed,
11  *   - as every statement after an expression that returns Nothing can be removed
12  *
13  *  If an If condition evalutates to Nothing, the entire If can be replaced by condition
14  *  If an argument evaluates to Nothing, the entire call can be replaced by evaluation of arguments.
15  *
16  *  This optimisation makes it rather tricky to write meaningful examples
17  *  since the compiler will often be able to reduce them to a single main
18  *  method with body = ???.
19  *
20  *  @author DarkDimius, OlivierBlanvillain
21  */
22 class BubbleUpNothing extends Optimisation {

28   def transformer(implicit ctx: Context): Tree => Tree = {
29     case t @ Apply(Select(Notathing(qual), _), args) =>
30       Typed(qual, TypeTree(t.tpe))
31     // This case leads to complications with multiple argument lists,
32     // how to do you rewrites tree.witType(???)(ctx).withType(???)(ctx)
33     // using Ycheckable steps?
34 
35     // Solution: only transform when having a complete application,
36     // steal code from tailRec
37 
38     // case t @ Apply(Select(qual, _), args) if args.exists(notathing) =>
39     //   val (keep, noth :: other) = args.span(x => !notathing(x))
40     //   Block(qual :: keep, Typed(noth, TypeTree(t.tpe)))
41     case Assign(_, rhs) if notathing(rhs) =>
42       rhs
43     case t @ If(Notathing(cond), _, _) =>
44       Typed(cond, TypeTree(t.tpe))
45     case b: Block if b.stats.exists(x => !x.isDef && notathing(x)) =>
46       val (keep, noth :: other) = b.stats.span(x => x.isDef || !notathing(x))
47       val keepDefs = other.filter(x => x.isDef)
48       val body = keep ::: keepDefs
49       Typed(Block(body, noth), TypeTree(b.tpe))
50     case t => t
51   }
52 
53   object Notathing {
54     def unapply(t: Tree)(implicit ctx: Context): Option[Tree] = Option(lookup(t))
55     def lookup(t: Tree)(implicit ctx: Context): Tree = t match {
56       case x if x.tpe.derivesFrom(defn.NothingClass) => t
57       case Typed(x, _) => lookup(x)
58       case Block(_, x) => lookup(x)
59       case _ => null
60     }
61   }
62 
63   def notathing(t: Tree)(implicit ctx: Context): Boolean = t match {
64     case Notathing(_) => true
65     case _ => false
66   }
