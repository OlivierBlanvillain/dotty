10 /** Various constant folding.
11  *
12  *  - Starts/ends with the constant folding implemented in typer (ConstFold).
13  *
14  *  - Join branches if they are "similar"
15  *
16  *  - regularize arithmetic and boolean expressions to have constants on the
17  *    left, ie. 6 * 2 * a * 5 => 60 * a
18  *
19  *  - (if) specific optimisation that propagate booleans, negation, and factor
20  *    out (nested) if with equivalent branches wrt to isSimilar. For example:
21  *      - if (b) exp else exp → b; exp
22  *      - if (b1) e1 else if (b2) e1 else e2 → if (b1 || b2) e1 else e2
23  *      - if(!b) e1 else e2 → if(b) e2 else e1
24  *
25  *  - Constant propagation over pattern matching.
26  *
27  *  @author DarkDimius, OlivierBlanvillain, gan74
28  */

174   def isSimilar(t1: Tree, t2: Tree)(implicit ctx: Context): Boolean = t1 match {
175     case t1: Apply =>
176       t2 match {
177         case t2: Apply =>
178           (t1.symbol == t2.symbol) &&
179           (t1.args zip t2.args).forall(x => isSimilar(x._1, x._2)) &&
180           isSimilar(t1.fun, t2.fun)
181         case _ => false
182       }
183     case t1: Ident =>
184       t2 match {
185         case t2: Ident =>
186           t1.symbol eq t2.symbol
187         case _ => // Select
188           isSimilar(t2, t1)
189       }
190     case t1: Select => t2 match {
191       case t2: Select =>
192         (t1.symbol eq t2.symbol) &&
193         isSimilar(t1.qualifier, t2.qualifier)
194       case t2: Ident => desugarIdent(t2) match {
195         case t2: Select => isSimilar(t1, t2)
196         case _ => false
197       }
198       case _ => false
199     }
200     case t1: Literal => t2 match {
201       case t2: Literal =>
202         t1.const.tag   == t2.const.tag &&
203         t1.const.value == t2.const.value
204       case _ => false
205     }
206     case _ => false
207   }
