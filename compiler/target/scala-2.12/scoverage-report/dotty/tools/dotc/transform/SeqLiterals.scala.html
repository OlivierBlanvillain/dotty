12 /** A transformer that eliminates SeqLiteral's, transforming `SeqLiteral(elems)` to an operation
13  *  equivalent to
14  *
15  *      JavaSeqLiteral(elems).toSeq
16  *
17  *  Instead of `toSeq`, which takes an implicit, the appropriate "wrapArray" method
18  *  is called directly. The reason for this step is that JavaSeqLiterals, being arrays
19  *  keep a precise type after erasure, whereas SeqLiterals only get the erased type `Seq`,
20  */
21 class SeqLiterals extends MiniPhase {
22   import ast.tpd._
23 
24   override def phaseName = "seqLiterals"
25   override def runsAfter: Set[Class[_ <: Phase]] = Set(classOf[PatternMatcher])
26 
27   override def checkPostCondition(tree: Tree)(implicit ctx: Context): Unit = tree match {
28     case tpd: SeqLiteral => assert(tpd.isInstanceOf[JavaSeqLiteral])
29     case _ =>
30   }
31 
32   override def transformSeqLiteral(tree: SeqLiteral)(implicit ctx: Context): Tree = tree match {
33     case tree: JavaSeqLiteral => tree
34     case _ =>
35       val arr = JavaSeqLiteral(tree.elems, tree.elemtpt)
36       //println(i"trans seq $tree, arr = $arr: ${arr.tpe} ${arr.tpe.elemType}")
37       val elemtp = tree.elemtpt.tpe
38       val elemCls = elemtp.classSymbol
39       val (wrapMethStr, targs) =
40         if (elemCls.isPrimitiveValueClass) (s"wrap${elemCls.name}Array", Nil)
41         else if (elemtp derivesFrom defn.ObjectClass) ("wrapRefArray", elemtp :: Nil)
42         else ("genericWrapArray", elemtp :: Nil)
43       ref(defn.ScalaPredefModule)
44         .select(wrapMethStr.toTermName)
45         .appliedToTypes(targs)
46         .appliedTo(arr)
47   }
48 }
