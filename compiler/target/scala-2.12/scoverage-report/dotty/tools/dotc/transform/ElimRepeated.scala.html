26  *  them with Seq types.

33   override def changesMembers = true // the phase adds vararg bridges

35   def transformInfo(tp: Type, sym: Symbol)(implicit ctx: Context): Type =

38   override def transform(ref: SingleDenotation)(implicit ctx: Context): SingleDenotation =
39     super.transform(ref) match {
40       case ref1: SymDenotation if (ref1 ne ref) && overridesJava(ref1.symbol) =>
41         // This method won't override the corresponding Java method at the end of this phase,
42         // only the bridge added by `addVarArgsBridge` will.

44       case ref1 =>
45         ref1
46     }

50   private def overridesJava(sym: Symbol)(implicit ctx: Context) = sym.allOverriddenSymbols.exists(_ is JavaDefined)

52   private def elimRepeated(tp: Type)(implicit ctx: Context): Type = tp.stripTypeVar match {
53     case tp @ MethodTpe(paramNames, paramTypes, resultType) =>
54       val resultType1 = elimRepeated(resultType)
55       val paramTypes1 =
56         if (paramTypes.nonEmpty && paramTypes.last.isRepeatedParam) {
57           val last = paramTypes.last.underlyingIfRepeated(tp.isJavaMethod)
58           paramTypes.init :+ last
59         } else paramTypes
60       tp.derivedLambdaType(paramNames, paramTypes1, resultType1)
61     case tp: PolyType =>
62       tp.derivedLambdaType(tp.paramNames, tp.paramInfos, elimRepeated(tp.resultType))
63     case tp =>
64       tp
65   }

93   /** Convert sequence argument to Java array of type `pt` */
94   private def seqToArray(tree: Tree, pt: Type)(implicit ctx: Context): Tree = tree match {
95     case SeqLiteral(elems, elemtpt) =>
96       JavaSeqLiteral(elems, elemtpt)
97     case app@Apply(fun, args) if defn.Predef_wrapArray().contains(fun.symbol) => // rewrite a call to `wrapXArray(arr)` to `arr`
98       args.head
99     case _ =>
100       val elemType = tree.tpe.elemType
101       var elemClass = elemType.classSymbol
102       if (defn.NotRuntimeClasses contains elemClass) elemClass = defn.ObjectClass
103       ref(defn.DottyArraysModule)
104         .select(nme.seqToArray)
105         .appliedToType(elemType)
106         .appliedTo(tree, Literal(Constant(elemClass.typeRef)))
107         .ensureConforms(pt)
108           // Because of phantomclasses, the Java array's type might not conform to the return type
109   }

114   /** If method overrides a Java varargs method, add a varargs bridge.
115    *  Also transform trees inside method annotation
116    */
117   override def transformDefDef(tree: DefDef)(implicit ctx: Context): Tree =
118     ctx.atPhase(thisPhase) { implicit ctx =>
119       if (tree.symbol.info.isVarArgsMethod && overridesJava(tree.symbol))
120         addVarArgsBridge(tree)
121       else
122         tree
123     }

125   /** Add a Java varargs bridge
126    *  @param  ddef  the original method definition which is assumed to override
127    *                a Java varargs method JM up to this phase.
128    *  @return  a thicket consisting of `ddef` and a varargs bridge method
129    *           which overrides the Java varargs method JM from this phase on
130    *           and forwards to `ddef`.
131    */
132   private def addVarArgsBridge(ddef: DefDef)(implicit ctx: Context): Tree = {
133     val original = ddef.symbol.asTerm
134     val bridge = original.copy(
135       flags = ddef.symbol.flags &~ Private | Artifact,
136       info = toJavaVarArgs(ddef.symbol.info)).enteredAfter(thisPhase).asTerm
137     val bridgeDef = polyDefDef(bridge, trefs => vrefss => {
138       val (vrefs :+ varArgRef) :: vrefss1 = vrefss
139       val elemtp = varArgRef.tpe.widen.argTypes.head
140       ref(original.termRef)
141         .appliedToTypes(trefs)
142         .appliedToArgs(vrefs :+ TreeGen.wrapArray(varArgRef, elemtp))
143         .appliedToArgss(vrefss1)
144     })
145 
146     Thicket(ddef, bridgeDef)
147   }

149   /** Convert type from Scala to Java varargs method */
150   private def toJavaVarArgs(tp: Type)(implicit ctx: Context): Type = tp match {
151     case tp: PolyType =>
152       tp.derivedLambdaType(tp.paramNames, tp.paramInfos, toJavaVarArgs(tp.resultType))
153     case tp: MethodType =>
154       val inits :+ last = tp.paramInfos
155       val last1 = last.underlyingIfRepeated(isJava = true)
156       tp.derivedLambdaType(tp.paramNames, inits :+ last1, tp.resultType)
157   }
