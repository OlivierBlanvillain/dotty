96     /** Assuming importedTags = `Type1 -> tag1, ..., TypeN -> tagN`, the expression
97      *
98      *      { type <Type1> = <tag1>.unary_~
99      *        ...
100      *        type <TypeN> = <tagN>.unary.~
101      *        <expr>
102      *      }
103      *
104      *  references to `TypeI` in `expr` are rewired to point to the locally
105      *  defined versions. As a side effect, prepend the expressions `tag1, ..., `tagN`
106      *  as splices to `embedded`.
107      */
108     def addTags(expr: Tree)(implicit ctx: Context): Tree =
109       if (importedTags.isEmpty) expr
110       else {
111         val itags = importedTags.toList
112         val typeDefs = for ((tref, tag) <- itags) yield {
113           val rhs = transform(tag.select(tpnme.UNARY_~))
114           val alias = ctx.typeAssigner.assignType(untpd.TypeBoundsTree(rhs, rhs), rhs, rhs)
115           val original = tref.symbol.asType
116           val local = original.copy(
117             owner = ctx.owner,
118             flags = Synthetic,
119             info = TypeAlias(tag.tpe.select(tpnme.UNARY_~)))
120           ctx.typeAssigner.assignType(untpd.TypeDef(original.name, alias), local)
121         }
122         importedTags.clear()
123         Block(typeDefs,
124           new TreeTypeMap(substFrom = itags.map(_._1.symbol), substTo = typeDefs.map(_.symbol))
125             .apply(expr))
126       }

139     /** Does the level of `sym` match the current level?
140      *  An exception is made for inline vals in macros. These are also OK if their level
141      *  is one higher than the current level, because on execution such values
142      *  are constant expression trees and we can pull out the constant from the tree.

156     /** Try to heal phase-inconsistent reference to type `T` using a local type definition.
157      *  @return None      if successful

179     /** Check reference to `sym` for phase consistency, where `tp` is the underlying type
180      *  by which we refer to `sym`.
181      */
182     def check(sym: Symbol, tp: Type, pos: Position)(implicit ctx: Context): Unit = {
183       val isThis = tp.isInstanceOf[ThisType]
184       def symStr =
185         if (!isThis) sym.show
186         else if (sym.is(ModuleClass)) sym.sourceModule.show
187         else i"${sym.name}.this"
188       if (!isThis && sym.maybeOwner.isType)
189         check(sym.owner, sym.owner.thisType, pos)
190       else if (sym.exists && !sym.isStaticOwner && !levelOK(sym))
191         for (errMsg <- tryHeal(tp, pos))
192           ctx.error(em"""access to $symStr from wrong staging level:
193                         | - the definition is at level ${levelOf(sym)},
194                         | - but the access is at level $level.$errMsg""", pos)

197     /** Check all named types and this-types in a given type for phase consistency. */

199       def apply(acc: Unit, tp: Type): Unit = reporting.trace(i"check type level $tp at $level") {

201           case tp: NamedType if tp.symbol.isSplice =>

213           case _ =>

219     /** If `tree` refers to a locally defined symbol (either directly, or in a pickled type),
220      *  check that its staging level matches the current level. References to types
221      *  that are phase-incorrect can still be healed as follows:
222      *
223      *  If `T` is a reference to a type at the wrong level, heal it by setting things up
224      *  so that we later add a type definition
225      *
226      *     type T' = ~quoted.Type[T]
227      *
228      *  to the quoted text and rename T to T' in it. This is done later in `reify` via
229      *  `addTags`. `checkLevel` itself only records what needs to be done in the
230      *  `typeTagOfRef` field of the current `Splice` structure.

233       tree match {
234         case (_: Ident) | (_: This) =>
235           check(tree.symbol, tree.tpe, tree.pos)
236         case (_: UnApply)  | (_: TypeTree) =>
237           checkType(tree.pos).apply((), tree.tpe)
238         case Select(qual, OuterSelectName(_, levels)) =>
239           checkType(tree.pos).apply((), tree.tpe.widen)
240         case _: Bind =>
241           checkType(tree.pos).apply((), tree.symbol.info)
242         case _: Template =>
243           checkType(tree.pos).apply((), tree.symbol.owner.asClass.givenSelfType)
244         case _ =>
245       }
246       tree

249     /** Split `body` into a core and a list of embedded splices.
250      *  Then if inside a splice, make a hole from these parts.
251      *  If outside a splice, generate a call tp `scala.quoted.Unpickler.unpickleType` or
252      *  `scala.quoted.Unpickler.unpickleExpr` that matches `tpe` with
253      *  core and splices as arguments.
254      */
255     private def quotation(body: Tree, quote: Tree)(implicit ctx: Context) = {
256       val (body1, splices) = nested(isQuote = true).split(body)
257       if (inSplice)
258         makeHole(body1, splices, quote.tpe)
259       else {
260         def liftList(list: List[Tree], tpe: Type): Tree = {
261           list.foldRight[Tree](ref(defn.NilModule)) { (x, acc) =>
262             acc.select("::".toTermName).appliedToType(tpe).appliedTo(x)
263           }
264         }
265         val isType = quote.tpe.isRef(defn.QuotedTypeClass)
266         ref(if (isType) defn.Unpickler_unpickleType else defn.Unpickler_unpickleExpr)
267           .appliedToType(if (isType) body1.tpe else body1.tpe.widen)
268           .appliedTo(
269             liftList(PickledQuotes.pickleQuote(body1).map(x => Literal(Constant(x))), defn.StringType),
270             liftList(splices, defn.AnyType))
271       }
272     }.withPos(quote.pos)

297     /** Register `body` as an `embedded` quote or splice

306     override def transform(tree: Tree)(implicit ctx: Context): Tree =
307       reporting.trace(i"reify $tree at $level", show = true) {
308         def mapOverTree(lastEntered: List[Symbol]) =
309           try super.transform(tree)
310           finally
311             while (enteredSyms ne lastEntered) {
312               levelOf -= enteredSyms.head
313               enteredSyms = enteredSyms.tail
314             }
315         tree match {
316           case Quoted(quotedTree) =>
317             quotation(quotedTree, tree)
318           case Select(body, _) if tree.symbol.isSplice =>
319             splice(body, tree)
320           case Block(stats, _) =>
321             val last = enteredSyms
322             stats.foreach(markDef)
323             mapOverTree(last)
324           case Inlined(call, bindings, expansion @ Select(body, name)) if expansion.symbol.isSplice =>
325             // To maintain phase consistency, convert inlined expressions of the form
326             // `{ bindings; ~expansion }` to `~{ bindings; expansion }`
327             if (level == 0) transform(Splicer.splice(cpy.Inlined(tree)(call, bindings, body)))
328             else transform(cpy.Select(expansion)(cpy.Inlined(tree)(call, bindings, body), name))
329           case _: Import =>
330             tree
331           case tree: DefDef if tree.symbol.is(Macro) && level == 0 =>
332             markDef(tree)
333             val tree1 = nested(isQuote = true).transform(tree)
334               // check macro code as it if appeared in a quoted context
335             cpy.DefDef(tree)(rhs = EmptyTree)
336           case _ =>
337             markDef(tree)
338             checkLevel(mapOverTree(enteredSyms))
339         }
340       }
