13   /** Splice the Tree for a Quoted expression. `~'(xyz)` becomes `xyz`
14    *  and for `~xyz` the tree of `xyz` is interpreted for which the
15    *  resulting expression is return as a `Tree`
16    */
17   def splice(tree: Tree)(implicit ctx: Context): Tree = tree match {
18     case Quoted(quotedTree) => quotedTree
19     case tree: RefTree => reflectiveSplice(tree)
20     case tree: Apply => reflectiveSplice(tree)
21     case tree: Inlined => reflectiveSplice(tree)
22   }
23 
24   /** Splice the Tree for a Quoted expression which is constructed via a reflective call to the given method */
25   private def reflectiveSplice(tree: Tree)(implicit ctx: Context): Tree = {
26     val interpreter = new Interpreter
27     interpreter.interpretTree[scala.quoted.Expr[_]](tree).map(PickledQuotes.quotedToTree(_)).getOrElse(tree)
28   }
