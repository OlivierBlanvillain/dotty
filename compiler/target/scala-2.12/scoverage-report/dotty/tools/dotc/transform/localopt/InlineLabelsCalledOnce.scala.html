11 /** Inlines LabelDef which are used exactly once.

13  *  @author DarkDimius, OlivierBlanvillain

15 class InlineLabelsCalledOnce extends Optimisation {
16   import ast.tpd._

18   val timesUsed = newMutableSymbolMap[Int]
19   val defined   = newMutableSymbolMap[DefDef]
20 
21   def clear(): Unit = {
22     timesUsed.clear()
23     defined.clear()
24   }
25 
26   def visitor(implicit ctx: Context): Tree => Unit = {
27     case d: DefDef if d.symbol.is(Label)  =>
28       var isRecursive = false
29       d.rhs.foreachSubTree { x =>
30         if (x.symbol == d.symbol)
31           isRecursive = true
32       }
33       if (!isRecursive)
34         defined.update(d.symbol, d)
35 
36     case t: Apply if t.symbol.is(Label) =>
37       val b4 = timesUsed.getOrElseUpdate(t.symbol, 0)
38       timesUsed.update(t.symbol, b4 + 1)
39 
40     case _ =>
41   }
42 
43   def transformer(implicit ctx: Context): Tree => Tree = {
44     case a: Apply =>
45       defined.get(a.symbol) match {
46         case Some(defDef) if usedOnce(a) && a.symbol.info.paramInfoss == List(Nil) =>
47           simplify.println(s"Inlining labeldef ${defDef.name}")
48           defDef.rhs.changeOwner(defDef.symbol, ctx.owner)
49 
50         case Some(defDef) if defDef.rhs.isInstanceOf[Literal] =>
51           defDef.rhs
52 
53         case _ => a
54       }
55 
56     case d: DefDef if usedOnce(d) =>
57       simplify.println(s"Dropping labeldef (used once) ${d.name} ${timesUsed.get(d.symbol)}")
58       defined.update(d.symbol, d)
59       EmptyTree
60 
61     case d: DefDef if neverUsed(d) =>
62       simplify.println(s"Dropping labeldef (never used) ${d.name} ${timesUsed.get(d.symbol)}")
63       EmptyTree
64 
65     case t => t
66   }
67 
68   def usedN(t: Tree, n: Int)(implicit ctx: Context): Boolean =
69     t.symbol.is(Label)                    &&
70     timesUsed.getOrElse(t.symbol, 0) == n &&
71     defined.contains(t.symbol)
72 
73   def usedOnce(t: Tree)(implicit ctx: Context): Boolean  = usedN(t, 1)
74   def neverUsed(t: Tree)(implicit ctx: Context): Boolean = usedN(t, 0)
75 }
