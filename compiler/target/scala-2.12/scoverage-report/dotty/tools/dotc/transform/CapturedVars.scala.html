37   private class RefInfo(implicit ctx: Context) {
38     /** The classes for which a Ref type exists. */
39     val refClassKeys: collection.Set[Symbol] =
40       defn.ScalaNumericValueClasses() + defn.BooleanClass + defn.ObjectClass
41 
42     val refClass: Map[Symbol, Symbol] =
43       refClassKeys.map(rc => rc -> ctx.requiredClass(s"scala.runtime.${rc.name}Ref")).toMap
44 
45     val volatileRefClass: Map[Symbol, Symbol] =
46       refClassKeys.map(rc => rc -> ctx.requiredClass(s"scala.runtime.Volatile${rc.name}Ref")).toMap
47 
48     val boxedRefClasses: collection.Set[Symbol] =
49       refClassKeys.flatMap(k => Set(refClass(k), volatileRefClass(k)))
50   }

58   private class CollectCaptured extends TreeTraverser {
59     private val captured = mutable.HashSet[Symbol]()
60     def traverse(tree: Tree)(implicit ctx: Context) = tree match {
61       case id: Ident =>
62         val sym = id.symbol
63         if (sym.is(Mutable, butNot = Method) && sym.owner.isTerm) {
64           val enclMeth = ctx.owner.enclosingMethod
65           if (sym.enclosingMethod != enclMeth) {
66             ctx.log(i"capturing $sym in ${sym.enclosingMethod}, referenced from $enclMeth")
67             captured += sym
68           }
69         }
70       case _ =>
71         traverseChildren(tree)
72     }
73     def runOver(tree: Tree)(implicit ctx: Context): collection.Set[Symbol] = {
74       traverse(tree)
75       captured
76     }
77   }

85   /** The {Volatile|}{Int|Double|...|Object}Ref class corresponding to the class `cls`,
86     *  depending on whether the reference should be @volatile
87     */
88   def refClass(cls: Symbol, isVolatile: Boolean)(implicit ctx: Context): Symbol = {
89     val refMap = if (isVolatile) refInfo.volatileRefClass else refInfo.refClass
90     if (cls.isClass)  {
91       refMap.getOrElse(cls, refMap(defn.ObjectClass))
92     }
93     else refMap(defn.ObjectClass)
94   }

96   override def prepareForValDef(vdef: ValDef)(implicit ctx: Context) = {
97     val sym = vdef.symbol(ctx.withPhase(thisPhase))
98     if (captured contains sym) {
99       val newd = sym.denot(ctx.withPhase(thisPhase)).copySymDenotation(
100         info = refClass(sym.info.classSymbol, sym.hasAnnotation(defn.VolatileAnnot)).typeRef,
101         initFlags = sym.flags &~ Mutable)
102       newd.removeAnnotation(defn.VolatileAnnot)
103       newd.installAfter(thisPhase)
104     }
105     ctx
106   }

108   override def transformValDef(vdef: ValDef)(implicit ctx: Context): Tree = {
109     val vble = vdef.symbol
110     if (captured.contains(vble)) {
111       def boxMethod(name: TermName): Tree =
112         ref(vble.info.classSymbol.companionModule.info.member(name).symbol)
113       cpy.ValDef(vdef)(
114         rhs = vdef.rhs match {
115           case EmptyTree => boxMethod(nme.zero).appliedToNone.withPos(vdef.pos)
116           case arg => boxMethod(nme.create).appliedTo(arg)
117         },
118         tpt = TypeTree(vble.info).withPos(vdef.tpt.pos))
119     } else vdef
120   }
121 
122   override def transformIdent(id: Ident)(implicit ctx: Context): Tree = {
123     val vble = id.symbol
124     if (captured.contains(vble))
125       id.select(nme.elem).ensureConforms(vble.denot(ctx.withPhase(thisPhase)).info)
126     else id
127   }

129   /** If assignment is to a boxed ref type, e.g.
130     *
131     *      intRef.elem = expr
132     *
133     *  rewrite using a temporary var to
134     *
135     *      val ev$n = expr
136     *      intRef.elem = ev$n
137     *
138     *  That way, we avoid the problem that `expr` might contain a `try` that would
139     *  run on a non-empty stack (which is illegal under JVM rules). Note that LiftTry
140     *  has already run before, so such `try`s would not be eliminated.
141     *
142     *  Also: If the ref type lhs is followed by a cast (can be an artifact of nested translation),
143     *  drop the cast.
144     */
145   override def transformAssign(tree: Assign)(implicit ctx: Context): Tree = {
146     def recur(lhs: Tree): Tree = lhs match {
147       case TypeApply(Select(qual, nme.asInstanceOf_), _) =>
148         val Select(_, nme.elem) = qual
149         recur(qual)
150       case Select(_, nme.elem) if refInfo.boxedRefClasses.contains(lhs.symbol.maybeOwner) =>
151         val tempDef = transformFollowing(SyntheticValDef(TempResultName.fresh(), tree.rhs))
152         transformFollowing(Block(tempDef :: Nil, cpy.Assign(tree)(lhs, ref(tempDef.symbol))))
153       case _ =>
154         tree
155     }
156     recur(tree.lhs)
157   }
