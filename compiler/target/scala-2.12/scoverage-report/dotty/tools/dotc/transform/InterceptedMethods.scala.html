30 /** Replace member references as follows:
31   *
32   * - `x != y` for != in class Any becomes `!(x == y)` with == in class Any.
33   * - `x.##` for ## in NullClass becomes `0`
34   * - `x.##` for ## in Any becomes calls to ScalaRunTime.hash,
35   *     using the most precise overload available
36   * - `x.getClass` for getClass in primitives becomes `x.getClass` with getClass in class Object.
37   */
38 class InterceptedMethods extends MiniPhase {
39   import tpd._
40 
41   override def phaseName: String = "intercepted"
42 
43   private[this] var primitiveGetClassMethods: Set[Symbol] = _
44 
45   var Any_## : Symbol = _ // cached for performance reason
46 
47   /** perform context-dependant initialization */
48   override def prepareForUnit(tree: Tree)(implicit ctx: Context) = {
49     this.Any_## = defn.Any_##
50     primitiveGetClassMethods = Set[Symbol]() ++ defn.ScalaValueClasses().map(x => x.requiredMethod(nme.getClass_))
51     ctx
52   }
53 
54   // this should be removed if we have guarantee that ## will get Apply node
55   override def transformSelect(tree: tpd.Select)(implicit ctx: Context): Tree = {
56     if (tree.symbol.isTerm && (Any_## eq tree.symbol.asTerm)) {
57       val rewrite = poundPoundValue(tree.qualifier)
58       ctx.log(s"$phaseName rewrote $tree to $rewrite")
59       rewrite
60     }
61     else tree
62   }
63 
64   // TODO: add missing cases from scalac
65   private def poundPoundValue(tree: Tree)(implicit ctx: Context) = {
66     val s = tree.tpe.widen.typeSymbol
67 
68     def staticsCall(methodName: TermName): Tree =
69       ref(defn.staticsMethodRef(methodName)).appliedTo(tree)
70 
71     if (s == defn.NullClass) Literal(Constant(0))
72     else if (s == defn.DoubleClass) staticsCall(nme.doubleHash)
73     else if (s == defn.LongClass) staticsCall(nme.longHash)
74     else if (s == defn.FloatClass) staticsCall(nme.floatHash)
75     else staticsCall(nme.anyHash)
76   }
77 
78   override def transformApply(tree: Apply)(implicit ctx: Context): Tree = {
79     def unknown = {
80       assert(false, s"The symbol '${tree.fun.symbol.showLocated}' was intercepted but didn't match any cases, " +
81         s"that means the intercepted methods set doesn't match the code")
82       tree
83     }
84     lazy val qual = tree.fun match {
85       case Select(qual, _) => qual
86       case ident @ Ident(_) =>
87         ident.tpe match {
88           case TermRef(prefix: TermRef, _) =>
89             tpd.ref(prefix)
90           case TermRef(prefix: ThisType, _) =>
91             tpd.This(prefix.cls)
92         }
93 
94     }
95     val Any_## = this.Any_##
96     val Any_!= = defn.Any_!=
97     val rewrite: Tree = tree.fun.symbol match {
98       case Any_## =>
99         poundPoundValue(qual)
100       case Any_!= =>
101         qual.select(defn.Any_==).appliedToArgs(tree.args).select(defn.Boolean_!)
102         /*
103         /* else if (isPrimitiveValueClass(qual.tpe.typeSymbol)) {
104             // todo: this is needed to support value classes
105             // Rewrite 5.getClass to ScalaRunTime.anyValClass(5)
106             global.typer.typed(gen.mkRuntimeCall(nme.anyValClass,
107               List(qual, typer.resolveClassTag(tree.pos, qual.tpe.widen))))
108           }*/
109          */
110       case t if primitiveGetClassMethods.contains(t) =>
111           // if we got here then we're trying to send a primitive getClass method to either
112           // a) an Any, in which cage Object_getClass works because Any erases to object. Or
113           //
114           // b) a non-primitive, e.g. because the qualifier's type is a refinement type where one parent
115           //    of the refinement is a primitive and another is AnyRef. In that case
116           //    we get a primitive form of _getClass trying to target a boxed value
117           //    so we need replace that method name with Object_getClass to get correct behavior.
118           //    See SI-5568.
119         qual.selectWithSig(defn.Any_getClass).appliedToNone
120       case _ =>
121         tree
122     }
123     ctx.log(s"$phaseName rewrote $tree to $rewrite")
124     rewrite
125   }
126 }
