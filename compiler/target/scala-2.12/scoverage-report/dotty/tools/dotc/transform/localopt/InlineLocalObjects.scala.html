20 /** Rewrite fields of local instances as vals.
21  *
22  *  If a local instance does not escape the local scope, it will be removed
23  *  later by DropNoEffects, thus implementing the equivalent of (local) multi
24  *  parameter value classes. The main motivation for this transformation is to
25  *  get ride of the intermediate tuples object somes created when pattern
26  *  matching on Scala2 case classes.
27  */
28 class InlineLocalObjects(val simplifyPhase: Simplify) extends Optimisation {
29   import ast.tpd._
30 
31   // ValDefs whose rhs is a case class instantiation: potential candidates.
32   val candidates = mutable.HashSet[Symbol]()
33 
34   // ValDefs whose lhs is used with `._1` (or any getter call).
35   val gettersCalled = mutable.HashSet[Symbol]()
36 
37   // Immutable sorted map from class to new fields, initialized between visitor and transformer.
38   var newFieldsMapping: Map[Symbol, LinkedHashMap[Symbol, Symbol]] = null
39   //                          |                     |       |
40   //                          |                     |       New fields, replacements these getters
41   //                          |                     Usages of getters of these classes
42   //                          ValDefs of the classes that are being torn apart; = candidates.intersect(gettersCalled)
43 
44   def clear(): Unit = {
45     candidates.clear()
46     gettersCalled.clear()
47     newFieldsMapping = null
48   }
49 
50   def initNewFieldsMapping()(implicit ctx: Context): Unit =
51     if (newFieldsMapping == null) {
52       newFieldsMapping = candidates.intersect(gettersCalled).map { refVal =>
53         val accessors = refVal.info.classSymbol.caseAccessors.filter(_.isGetter)
54         val newLocals = accessors.map { x =>
55           val owner: Symbol  = refVal.owner
56           val name:  Name    = LocalOptInlineLocalObj.fresh()
57           val flags: FlagSet = Synthetic
58           val info:  Type    = x.asSeenFrom(refVal.info).info.finalResultType.widenDealias
59           ctx.newSymbol(owner, name, flags, info)
60         }
61         (refVal, LinkedHashMap[Symbol, Symbol](accessors.zip(newLocals): _*))
62       }.toMap
63     }
64 
65   // Pattern for candidates to this optimisation: ValDefs where the rhs is an
66   // immutable case class instantiation.
67   object NewCaseClassValDef {
68     def unapply(t: ValDef)(implicit ctx: Context): Option[(Tree, List[Tree])] =
69       t.rhs match {
70         case Apply(fun, args)
71           if t.symbol.info.classSymbol.is(CaseClass)                          && // is rhs a case class?
72              !t.symbol.is(Lazy | Mutable)                                     && // is lhs a val?
73              !t.symbol.info.classSymbol.caseAccessors.exists(_.is(Mutable))   && // is the case class immutable?
74              fun.symbol.isConstructor                                         && // is rhs a new?
75              t.tpe.widenDealias == t.symbol.info.finalResultType.widenDealias => // no case class inheritance or enums
76           Some((fun, args))
77         case _ => None
78       }
79   }
80 
81   def visitor(implicit ctx: Context): Tree => Unit = {
82     case t @ NewCaseClassValDef(fun, args) =>
83       candidates += t.symbol
84     case t @ Select(qual, _) if isImmutableAccessor(t) =>
85       gettersCalled += qual.symbol
86     case _ =>
87   }
88 
89   def transformer(implicit ctx: Context): Tree => Tree = {
90     initNewFieldsMapping();
91     {
92       case t @ NewCaseClassValDef(fun, args) if newFieldsMapping.contains(t.symbol) =>
93         val newFields     = newFieldsMapping(t.symbol).values.toList
94         val newFieldsDefs = newFields.zip(args).map { case (nf, arg) =>
95           val rhs = arg.changeOwnerAfter(t.symbol, nf.symbol, simplifyPhase)
96           ValDef(nf.asTerm, rhs)
97         }
98         val recreate      = cpy.ValDef(t)(rhs = fun.appliedToArgs(newFields.map(x => ref(x))))
99         simplify.println(s"Replacing ${t.symbol.fullName} with stack-allocated fields ($newFields)")
100         Thicket(newFieldsDefs :+ recreate)
101 
102       case t @ Select(rec, _) if isImmutableAccessor(t) =>
103         newFieldsMapping.getOrElse(rec.symbol, Map.empty[Symbol, Symbol]).get(t.symbol) match {
104           case None         => t
105           case Some(newSym) => ref(newSym)
106         }
107 
108       case t => t
109     }
110   }
111 }
