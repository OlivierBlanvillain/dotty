13 /** A helper class for generating bridge methods in class `root`. */
14 class Bridges(root: ClassSymbol)(implicit ctx: Context) {
15   import ast.tpd._
16 
17   assert(ctx.phase == ctx.erasurePhase.next)
18   private val preErasureCtx = ctx.withPhase(ctx.erasurePhase)
19   private val elimErasedCtx = ctx.withPhase(ctx.elimErasedValueTypePhase.next)
20 
21   private class BridgesCursor(implicit ctx: Context) extends OverridingPairs.Cursor(root) {
22 
23     /** Only use the superclass of `root` as a parent class. This means
24      *  overriding pairs that have a common implementation in a trait parent
25      *  are also counted. This is necessary because we generate bridge methods
26      *  only in classes, never in traits.
27      */
28     override def parents = Array(root.superClass)
29     override def exclude(sym: Symbol) = !sym.is(Method) || super.exclude(sym)
30   }
31 
32   //val site = root.thisType
33 
34   private[this] var toBeRemoved = immutable.Set[Symbol]()
35   private val bridges = mutable.ListBuffer[Tree]()
36   private val bridgesScope = newScope
37   private val bridgeTarget = newMutableSymbolMap[Symbol]
38 
39   def bridgePosFor(member: Symbol) =
40     if (member.owner == root && member.pos.exists) member.pos else root.pos
41 
42   /** Add a bridge between `member` and `other`, where `member` overrides `other`
43    *  before erasure, if the following conditions are satisfied.
44    *
45    *   - `member` and `other` have different signatures
46    *   - there is not yet a bridge with the same name and signature in `root`.
47    *
48    *  The bridge has the erased info of `other` and forwards to `member`.
49    *  Additionally, if `member` and `other` do have the same signature,
50    *  but not the same type after erasure and before elimErasedValueTypes
51    *  issue an error: A bridge would be needed yet it would clash with the member itself.
52    *  See neg/i1905.scala
53    */
54   private def addBridgeIfNeeded(member: Symbol, other: Symbol) = {
55     def bridgeExists =
56       bridgesScope.lookupAll(member.name).exists(bridge =>
57         bridgeTarget(bridge) == member && bridge.signature == other.signature)
58     def info(sym: Symbol)(implicit ctx: Context) = sym.info
59     def desc(sym: Symbol)= {
60       val infoStr = info(sym)(preErasureCtx) match {
61         case ExprType(info) => i": $info"
62         case info => info.show
63       }
64       i"$sym$infoStr in ${sym.owner}"
65     }
66     if (member.signature == other.signature) {
67       if (!member.info.matches(other.info))
68         ctx.error(em"""bridge generated for member ${desc(member)}
69                       |which overrides ${desc(other)}
70                       |clashes with definition of the member itself; both have erased type ${info(member)(elimErasedCtx)}."""",
71                   bridgePosFor(member))
72     }
73     else if (!bridgeExists)
74       addBridge(member, other)
75   }
76 
77   /** Generate bridge between `member` and `other`
78    */
79   private def addBridge(member: Symbol, other: Symbol) = {
80     val bridge = other.copy(
81       owner = root,
82       flags = (member.flags | Method | Bridge | Artifact) &~
83         (Accessor | ParamAccessor | CaseAccessor | Deferred | Lazy | Module),
84       coord = bridgePosFor(member))
85       .enteredAfter(ctx.erasurePhase.asInstanceOf[DenotTransformer]).asTerm
86 
87     ctx.debuglog(
88       i"""generating bridge from ${other.showLocated}: ${other.info}
89              |to ${member.showLocated}: ${member.info} @ ${member.pos}
90              |bridge: ${bridge.showLocated} with flags: ${bridge.flags}""")
91 
92     bridgeTarget(bridge) = member
93     bridgesScope.enter(bridge)
94 
95     if (other.owner == root) {
96       root.delete(other)
97       toBeRemoved += other
98     }
99 
100     bridges +=
101       DefDef(bridge, This(root).select(member).appliedToArgss(_)).withPos(bridge.pos)
102   }
103 
104   /** Add all necessary bridges to template statements `stats`, and remove at the same
105    *  time deferred methods in `stats` that are replaced by a bridge with the same signature.
106    */
107   def add(stats: List[untpd.Tree]): List[untpd.Tree] =
108     if (root.is(Trait)) stats
109     else {
110       val opc = new BridgesCursor()(preErasureCtx)
111       while (opc.hasNext) {
112         if (!opc.overriding.is(Deferred)) addBridgeIfNeeded(opc.overriding, opc.overridden)
113         opc.next()
114       }
115       if (bridges.isEmpty) stats
116       else stats.filterNot(stat => toBeRemoved contains stat.symbol) ::: bridges.toList
117     }
118 }
