15 /** Inline case class specific methods using desugarings assumptions.
16  *
17  *  Note: to run this optimisation after erasure one would need to specialize
18  *  it for constructor with outer pointer and values classes. There is
19  *  probably no need to run this more than once.
20  *
21  *  @author DarkDimius, OlivierBlanvillain
22  */
23 class InlineCaseIntrinsics(val simplifyPhase: Simplify) extends Optimisation {
24   import ast.tpd._
25 
26   def visitor(implicit ctx: Context): Tree => Unit = NoVisitor
27   def clear(): Unit = ()
28 
29   def transformer(implicit ctx: Context): Tree => Tree = {
30     // For synthetic applies on case classes (both dotty/scalac)
31     // - CC.apply(args) → new CC(args)
32     case a: Apply
33       if !a.tpe.isInstanceOf[MethodicType]           &&
34          a.symbol.is(Synthetic)                      &&
35          a.symbol.owner.is(Module)                   &&
36          a.symbol.name == nme.apply                  &&
37          a.symbol.owner.companionClass.is(CaseClass) &&
38          !a.tpe.derivesFrom(defn.EnumClass)          &&
39          (isPureExpr(a.fun) || a.fun.symbol.is(Synthetic)) =>
40 
41       def unrollArgs(t: Tree, l: List[List[Tree]]): List[List[Tree]] = t match {
42         case Apply(t, args) => unrollArgs(t, args :: l)
43         case _ => l
44       }
45       val argss = unrollArgs(a.fun, a.args :: Nil)
46       def rollInArgs(l: List[List[Tree]], fun: Tree): Tree = l match {
47         case head :: tail => rollInArgs(tail, fun.appliedToArgs(head))
48         case _ => fun
49       }
50       val constructor = a.symbol.owner.companionClass.primaryConstructor.asTerm
51       evalreceiver(a, rollInArgs(argss.tail, New(a.tpe.widenDealias.simplified, constructor, argss.head)))
52 
53     // For synthetic dotty unapplies on case classes:
54     // - CC.unapply(arg): CC → arg
55     // - CC.unapply(arg): Boolean → true, dotty only
56     // - CC.unapply(arg): Option[CC] → new Some(new scala.TupleN(arg._1, ..., arg._N))
57     case a: Apply
58       if a.symbol.is(Synthetic)                       &&
59          a.symbol.owner.is(Module)                    &&
60          a.symbol.name == nme.unapply                 &&
61          a.symbol.owner.companionClass.is(CaseClass)  &&
62          !a.tpe.derivesFrom(defn.EnumClass)           &&
63          (isPureExpr(a.fun) || a.fun.symbol.is(Synthetic)) =>
64 
65       val args = a.args.head
66       val isDottyUnapply = !a.symbol.owner.is(Scala2x)
67       val isScalaOptionUnapply =
68         a.tpe.derivesFrom(defn.OptionClass) &&
69         a.args.head.tpe.derivesFrom(a.symbol.owner.companionClass)
70 
71       if (isDottyUnapply) { // dotty only
72         if (a.tpe.derivesFrom(defn.BooleanClass))
73           // CC.unapply(arg): Boolean → true
74           evalreceiver(a, Literal(Constant(true)))
75         else
76           // CC.unapply(arg): CC → arg
77           evalreceiver(a, a.args.head)
78       }
79       else if (isScalaOptionUnapply) {
80         // CC.unapply(arg): Option[CC] → new Some(new scala.TupleN(arg._1, ..., arg._N))
81         // The output is defined as a Tree => Tree to go thought tpd.evalOnce.
82         def some(e: Tree) = {
83           val accessors = e.tpe.widenDealias.classSymbol.caseAccessors.filter(_.is(Method))
84           val fields    = accessors.map(x => e.select(x).ensureApplied)
85           val tplType   = noBounds(a.tpe.baseType(defn.OptionClass).argInfos.head)
86           val someTpe   = a.tpe.translateParameterized(defn.OptionClass, defn.SomeClass)
87 
88           if (fields.tail.nonEmpty)
89             New(someTpe, New(tplType, fields) :: Nil)
90           else // scalac does not have Tuple1
91             New(someTpe, fields.head :: Nil)
92         }
93         val none = ref(defn.NoneModuleRef)
94         def isNull(e: Tree) = e.select(defn.Object_eq).appliedTo(Literal(Constant(null)))
95         def fi(e: Tree) = If(isNull(e), none, some(e))
96         evalreceiver(a, evalOnce(a.args.head)(fi))
97       }
98       else a
99 
100     // Seq.unapplySeq(arg) → new Some(arg)
101     // Where Seq is any companion of type <: SeqFactoryClass
102     case a: Apply
103       if a.symbol.name == nme.unapplySeq                  &&
104          a.symbol.owner.derivesFrom(simplifyPhase.SeqFactoryClass) &&
105          a.symbol.extendedOverriddenSymbols.isEmpty       &&
106          (isPureExpr(a.fun) || a.fun.symbol.is(Synthetic)) =>
107 
108       def receiver(t: Tree): Type = t match {
109         case t: Apply     => receiver(t.fun)
110         case t: TypeApply => receiver(t.fun)
111         case t: Ident     =>
112           val prefix = desugarIdentPrefix(t)
113           prefix.tpe.widenDealias
114         case t: Select => t.qualifier.tpe.widenDealias
115       }
116 
117       val recv = receiver(a)
118       if (recv.typeSymbol.is(Module)) {
119         val someTpe  = a.tpe.translateParameterized(defn.OptionClass, defn.SomeClass)
120         evalreceiver(a, New(someTpe, a.args.head :: Nil))
121       }
122       else a
123     case t => t
124   }
125 
126   // Apply fun may be a side-effectful function. E.g. a block, see tests/run/t4859.scala
127   // we need to maintain expressions that were in this block
128   def evalreceiver(a: Apply, res: Tree)(implicit ctx: Context) = {
129     def receiver(t: Tree): Tree = t match {
130       case TypeApply(fun, targs) if fun.symbol eq t.symbol => receiver(fun)
131       case Apply(fn, args) if fn.symbol == t.symbol => receiver(fn)
132       case Select(qual, _) => qual
133       case x => x
134     }
135     val recv = receiver(a)
136     if (recv.isEmpty || isPureRef(recv))
137       res
138     else
139       Block(recv :: Nil, res)
140   }
141 }
