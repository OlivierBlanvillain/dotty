18 /** Move static methods from companion to the class itself */
19 class MoveStatics extends MiniPhase with SymTransformer {
20 
21   import tpd._
22   override def phaseName = "moveStatic"
23 
24   def transformSym(sym: SymDenotation)(implicit ctx: Context): SymDenotation = {
25     if (sym.hasAnnotation(defn.ScalaStaticAnnot) && sym.owner.is(Flags.Module) && sym.owner.companionClass.exists) {
26       sym.owner.asClass.delete(sym.symbol)
27       sym.owner.companionClass.asClass.enter(sym.symbol)
28       val flags = if (sym.is(Flags.Method)) sym.flags else sym.flags | Flags.Mutable
29       sym.copySymDenotation(owner = sym.owner.companionClass, initFlags = flags)
30     }
31     else sym
32   }
33 
34   override def transformStats(trees: List[Tree])(implicit ctx: Context): List[Tree] = {
35     if (ctx.owner.is(Flags.Package)) {
36       val (classes, others) = trees.partition(x => x.isInstanceOf[TypeDef] && x.symbol.isClass)
37       val pairs = classes.groupBy(_.symbol.name.stripModuleClassSuffix).asInstanceOf[Map[Name, List[TypeDef]]]
38 
39       def rebuild(orig: TypeDef, newBody: List[Tree]): Tree = {
40         if (orig eq null) return EmptyTree
41 
42         val staticFields = newBody.filter(x => x.isInstanceOf[ValDef] && x.symbol.hasAnnotation(defn.ScalaStaticAnnot)).asInstanceOf[List[ValDef]]
43         val newBodyWithStaticConstr =
44           if (staticFields.nonEmpty) {
45             /* do NOT put Flags.JavaStatic here. It breaks .enclosingClass */
46             val staticCostructor = ctx.newSymbol(orig.symbol, nme.STATIC_CONSTRUCTOR, Flags.Synthetic | Flags.Method | Flags.Private, MethodType(Nil, defn.UnitType))
47             staticCostructor.addAnnotation(Annotation(defn.ScalaStaticAnnot))
48             staticCostructor.entered
49 
50             val staticAssigns = staticFields.map(x => Assign(ref(x.symbol), x.rhs.changeOwner(x.symbol, staticCostructor)))
51             tpd.DefDef(staticCostructor, Block(staticAssigns, tpd.unitLiteral)) :: newBody
52           } else newBody
53 
54         val oldTemplate = orig.rhs.asInstanceOf[Template]
55         cpy.TypeDef(orig)(rhs = cpy.Template(orig.rhs)(oldTemplate.constr, oldTemplate.parents, oldTemplate.self, newBodyWithStaticConstr))
56       }
57 
58       def move(module: TypeDef, companion: TypeDef): List[Tree] = {
59         assert(companion ne module)
60         if (!module.symbol.is(Flags.Module)) move(companion, module)
61         else {
62           val allMembers =
63             (if(companion ne null) {companion.rhs.asInstanceOf[Template].body} else Nil) ++
64             module.rhs.asInstanceOf[Template].body
65           val (newModuleBody, newCompanionBody) = allMembers.partition(x => {assert(x.symbol.exists); x.symbol.owner == module.symbol})
66           Trees.flatten(rebuild(companion, newCompanionBody) :: rebuild(module, newModuleBody) :: Nil)
67         }
68       }
69       val newPairs =
70         for ((name, classes) <- pairs)
71           yield
72             if (classes.tail.isEmpty)
73               if (classes.head.symbol.is(Flags.Module)) move(classes.head, null)
74               else List(rebuild(classes.head, classes.head.rhs.asInstanceOf[Template].body))
75             else move(classes.head, classes.tail.head)
76       Trees.flatten(newPairs.toList.flatten ++ others)
77     } else trees
78   }
