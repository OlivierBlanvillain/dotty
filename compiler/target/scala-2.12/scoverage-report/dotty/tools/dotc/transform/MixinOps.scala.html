21   def implementation(member: TermSymbol): TermSymbol = {
22     val res = member.copy(
23       owner = cls,
24       name = member.name.stripScala2LocalSuffix,
25       flags = member.flags &~ Deferred,
26       info = cls.thisType.memberInfo(member)).enteredAfter(thisPhase).asTerm

31   def superRef(target: Symbol, pos: Position = cls.pos): Tree = {
32     val sup = if (target.isConstructor && !target.owner.is(Trait))
33       Super(This(cls), tpnme.EMPTY, true)
34     else
35       Super(This(cls), target.owner.name.asTypeName, false, target.owner)
36     //println(i"super ref $target on $sup")
37     ast.untpd.Select(sup.withPos(pos), target.name)
38       .withType(NamedType(sup.tpe, target))
39     //sup.select(target)
40   }

42   /** Is `sym` a member of implementing class `cls`?
43    *  The test is performed at phase `thisPhase`.
44    */
45   def isCurrent(sym: Symbol) =
46     ctx.atPhase(thisPhase) { implicit ctx =>
47       cls.info.member(sym.name).hasAltWith(_.symbol == sym)
48       // this is a hot spot, where we spend several seconds while compiling stdlib
49       // unfortunately it will discard and recompute all the member chaches,
50       // both making itself slow and slowing down anything that runs after it
51       // because resolveSuper uses hacks with explicit adding to scopes through .enter
52       // this cannot be fixed by a smarter caching strategy. With current implementation
53       // we HAVE to discard caches here for correctness
54     }

56   /** Does `method` need a forwarder to in  class `cls`
57    *  Method needs a forwarder in those cases:
58    *   - there's a class defining a method with same signature
59    *   - there are multiple traits defining method with same signature
60    */
61   def needsForwarder(meth: Symbol): Boolean = {
62     lazy val competingMethods = competingMethodsIterator(meth).toList
63 
64     def needsDisambiguation = competingMethods.exists(x=> !(x is Deferred)) // multiple implementations are available
65     def hasNonInterfaceDefinition = competingMethods.exists(!_.owner.is(Trait)) // there is a definition originating from class
66     meth.is(Method, butNot = PrivateOrAccessorOrDeferred) &&
67     (meth.owner.is(Scala2x) || needsDisambiguation || hasNonInterfaceDefinition || needsJUnit4Fix(meth) ) &&
68     isCurrent(meth)
69   }
70 
71   private def needsJUnit4Fix(meth: Symbol): Boolean = {
72     meth.annotations.nonEmpty && JUnit4Annotations.exists(annot => meth.hasAnnotation(annot))
73   }
74 
75   /** Get `sym` of the method that needs a forwarder
76     *  Method needs a forwarder in those cases:
77     *   - there is a trait that defines a primitive version of implemented polymorphic method.
78     *   - there is a trait that defines a polymorphic version of implemented primitive method.
79     */
80   def needsPrimitiveForwarderTo(meth: Symbol): Option[Symbol] = {
81     def hasPrimitiveMissMatch(tp1: Type, tp2: Type): Boolean = (tp1, tp2) match {
82       case (tp1: MethodicType, tp2: MethodicType) =>
83         hasPrimitiveMissMatch(tp1.resultType, tp2.resultType) ||
84         tp1.paramInfoss.flatten.zip(tp1.paramInfoss.flatten).exists(args => hasPrimitiveMissMatch(args._1, args._2))
85       case _ =>
86         def isPrimitiveOrValueClass(sym: Symbol): Boolean = sym.isPrimitiveValueClass || sym.isValueClass
87         isPrimitiveOrValueClass(tp1.typeSymbol) ^ isPrimitiveOrValueClass(tp2.typeSymbol)
88     }
89 
90     def needsPrimitiveForwarder(m: Symbol): Boolean =
91       m.owner != cls && !m.is(Deferred) && hasPrimitiveMissMatch(meth.info, m.info)
92 
93     if (!meth.is(Method | Deferred, butNot = PrivateOrAccessor) || meth.overriddenSymbol(cls).exists || needsForwarder(meth)) None
94     else competingMethodsIterator(meth).find(needsPrimitiveForwarder)
95   }
96 
97   final val PrivateOrAccessor = Private | Accessor
98   final val PrivateOrAccessorOrDeferred = Private | Accessor | Deferred
99 
100   def forwarder(target: Symbol) = (targs: List[Type]) => (vrefss: List[List[Tree]]) =>
101     superRef(target).appliedToTypes(targs).appliedToArgss(vrefss)
102 
103   private def competingMethodsIterator(meth: Symbol): Iterator[Symbol] = {
104     cls.baseClasses.iterator
105       .filter(_ ne meth.owner)
106       .map(meth.overriddenSymbol)
107       .filter(_.exists)
108   }
