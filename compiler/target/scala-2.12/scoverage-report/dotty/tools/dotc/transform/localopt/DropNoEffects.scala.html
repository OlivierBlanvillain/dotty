12 /** Removes side effect free statements in blocks and Defdef.
13  *  Flattens blocks (except Closure-blocks)
14  *  Note: BoxedUnit currently messes up this phase when run after erasure
15  *
16  *  @author DarkDimius, OlivierBlanvillain
17  */

24   def transformer(implicit ctx: Context): Tree => Tree = {
25     // Remove empty blocks
26     case Block(Nil, expr) => expr
27 
28     // Keep only side effect free statements in blocks
29     case a: Block  =>
30       val newStats0 = a.stats.mapConserve(keepOnlySideEffects)
31 
32       // Flatten nested blocks
33       val newStats1 = if (newStats0 eq a.stats) newStats0 else newStats0.flatMap {
34         case x: Block  => x.stats ::: List(x.expr)
35         case EmptyTree => Nil
36         case t => t :: Nil
37       }
38       val (newStats2, newExpr) = a.expr match {
39         case Block(stats2, expr) => (newStats1 ++ stats2, expr)
40         case _ => (newStats1, a.expr)
41       }
42 
43       if (newStats2.nonEmpty)
44         cpy.Block(a)(stats = newStats2, newExpr)
45       else newExpr
46 
47     // Keep only side effect free statements unit returning functions
48     case a: DefDef
49       if a.symbol.info.finalResultType.derivesFrom(defn.UnitClass) &&
50         !a.rhs.tpe.derivesFrom(defn.UnitClass)                     &&
51         !a.rhs.tpe.derivesFrom(defn.NothingClass)                  =>
52 
53       def insertUnit(t: Tree) = {
54         if (!t.tpe.derivesFrom(defn.UnitClass)) Block(t :: Nil, unitLiteral)
55         else t
56       }
57       cpy.DefDef(a)(rhs = insertUnit(keepOnlySideEffects(a.rhs)), tpt = TypeTree(defn.UnitType))
58 
59     case t => t
60   }

62   def keepOnlySideEffects(t: Tree)(implicit ctx: Context): Tree = t match {
63     case l: Literal =>
64       EmptyTree
65 
66     case t: This =>
67       EmptyTree
68 
69     case Typed(exp, tpe) =>
70       keepOnlySideEffects(exp)
71 
72     // If is pure, propagade the simplification
73     case t @ If(cond, thenp, elsep) =>
74       val nthenp = keepOnlySideEffects(thenp)
75       val nelsep = keepOnlySideEffects(elsep)
76       if (thenp.isEmpty && elsep.isEmpty) keepOnlySideEffects(cond)
77       else cpy.If(t)(
78         thenp = nthenp.orElse(if (thenp.isInstanceOf[Literal]) thenp else unitLiteral),
79         elsep = nelsep.orElse(if (elsep.isInstanceOf[Literal]) elsep else unitLiteral))
80 
81     // Accessing a field of a product
82     case t @ Select(rec, _) if isImmutableAccessor(t) =>
83       keepOnlySideEffects(rec)
84 
85     // !name.eq(nme.TYPE_) && // Keep the .TYPE added by ClassOf, would be needed for AfterErasure
86     // Without is(JavaStatic), { System.out } becomes { System }, but "Java class can't be used as value"
87     case s @ Select(qual, name) if !s.symbol.is(Mutable | Lazy | Method | JavaStatic) =>
88       keepOnlySideEffects(qual)
89 
90     case Block(List(t: DefDef), s: Closure) =>
91       EmptyTree
92 
93     case bl @ Block(stats, expr) =>
94       val stats1 = stats.mapConserve(keepOnlySideEffects)
95       val stats2 = if (stats1 ne stats) stats1.filter(_ ne EmptyTree) else stats1
96       val expr2: Tree = expr match {
97         case t: Literal if t.tpe.derivesFrom(defn.UnitClass) => expr
98         case _ => keepOnlySideEffects(expr).orElse(unitLiteral)
99       }
100       cpy.Block(bl)(stats2, expr2)
101 
102     case t: Ident if !t.symbol.is(Method | Lazy) && !t.symbol.info.isInstanceOf[ExprType] || effectsDontEscape(t) =>
103       val prefix = desugarIdentPrefix(t)
104       if (!prefix.isEmpty && !prefix.symbol.is(allOf(JavaDefined, Package))) t
105       else EmptyTree
106 
107     case app: Apply if app.fun.symbol.is(Label) && !app.tpe.finalResultType.derivesFrom(defn.UnitClass) =>
108       // This is "the scary hack". It changes the return type to Unit, then
109       // invalidates the denotation cache. Because this optimisation only
110       // operates locally, this should be fine.
111       val denot = app.fun.symbol.denot
112       if (!denot.info.finalResultType.derivesFrom(defn.UnitClass)) {
113         val newLabelType = app.symbol.info match {
114           case mt: MethodType =>
115             mt.derivedLambdaType(mt.paramNames, mt.paramInfos, defn.UnitType)
116           case et: ExprType =>
117             et.derivedExprType(defn.UnitType)
118         }
119         val newD = app.symbol.asSymDenotation.copySymDenotation(info = newLabelType)
120         newD.installAfter(simplifyPhase)
121       }
122 
123       ref(app.symbol).appliedToArgs(app.args)
124 
125     case t @ Apply(fun, _) if effectsDontEscape(t) =>
126       def getArgsss(a: Tree): List[Tree] = a match {
127         case a: Apply => getArgsss(a.fun) ::: a.args
128         case _ => Nil
129       }
130       def getSel(t: Tree): Tree = {t match {
131         case t: Apply => getSel(t.fun)
132         case t: Select => t.qualifier
133         case t: TypeApply => getSel(t.fun)
134         case _ => t
135       }}
136       val args = getArgsss(t)
137       val rec = getSel(t)
138       val prefix = rec match {
139         case t: New =>
140           args.map(keepOnlySideEffects)
141         case _ =>
142           rec :: args.map(keepOnlySideEffects)
143       }
144       seq(prefix, unitLiteral)
145 
146     case t => t
147   }
