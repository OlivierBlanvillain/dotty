199     def localContext(implicit ctx: Context) = {
200       val sym = tree.symbol
201       val owner = if (sym is PackageVal) sym.moduleClass else sym

205     def transformNamed(tree: Tree, start: Int, outerCtx: Context): Tree = tree match {
206       case tree: Ident =>
207         implicit val ctx = prepIdent(tree, start)(outerCtx)
208         goIdent(tree, start)
209       case tree: Select =>
210         implicit val ctx = prepSelect(tree, start)(outerCtx)
211         val qual = transformTree(tree.qualifier, start)
212         goSelect(cpy.Select(tree)(qual, tree.name), start)
213       case tree: ValDef =>
214         implicit val ctx = prepValDef(tree, start)(outerCtx)
215         def mapValDef(implicit ctx: Context) = {
216           val tpt = transformTree(tree.tpt, start)
217           val rhs = transformTree(tree.rhs, start)
218           cpy.ValDef(tree)(tree.name, tpt, rhs)
219         }
220         if (tree.isEmpty) tree
221         else goValDef(mapValDef(if (tree.symbol.exists) localContext else ctx), start)
222       case tree: DefDef =>
223         implicit val ctx = prepDefDef(tree, start)(outerCtx)
224         def mapDefDef(implicit ctx: Context) = {
225           val tparams = transformSpecificTrees(tree.tparams, start)
226           val vparamss = tree.vparamss.mapConserve(transformSpecificTrees(_, start))
227           val tpt = transformTree(tree.tpt, start)
228           val rhs = transformTree(tree.rhs, start)
229           cpy.DefDef(tree)(tree.name, tparams, vparamss, tpt, rhs)
230         }
231         goDefDef(mapDefDef(localContext), start)
232       case tree: TypeDef =>
233         implicit val ctx = prepTypeDef(tree, start)(outerCtx)
234         val rhs = transformTree(tree.rhs, start)(localContext)
235         goTypeDef(cpy.TypeDef(tree)(tree.name, rhs), start)
236       case tree: Bind =>
237         implicit val ctx = prepBind(tree, start)(outerCtx)
238         val body = transformTree(tree.body, start)
239         goBind(cpy.Bind(tree)(tree.name, body), start)
240       case _ =>
241         implicit val ctx = prepOther(tree, start)(outerCtx)
242         goOther(tree, start)
243     }

245     def transformUnnamed(tree: Tree, start: Int, outerCtx: Context): Tree = tree match {
246       case tree: Apply =>
247         implicit val ctx = prepApply(tree, start)(outerCtx)
248         val fun = transformTree(tree.fun, start)
249         val args = transformTrees(tree.args, start)
250         goApply(cpy.Apply(tree)(fun, args), start)
251       case tree: TypeTree =>
252         implicit val ctx = prepTypeTree(tree, start)(outerCtx)
253         goTypeTree(tree, start)
254       case tree: Thicket =>
255         cpy.Thicket(tree)(transformTrees(tree.trees, start))
256       case tree: This =>
257         implicit val ctx = prepThis(tree, start)(outerCtx)

259       case tree: Literal =>

262       case tree: Block =>

273         implicit val ctx = prepIf(tree, start)(outerCtx)

275         val thenp = transformTree(tree.thenp, start)
276         val elsep = transformTree(tree.elsep, start)
277         goIf(cpy.If(tree)(cond, thenp, elsep), start)
278       case tree: New =>
279         implicit val ctx = prepNew(tree, start)(outerCtx)
280         val tpt = transformTree(tree.tpt, start)
281         goNew(cpy.New(tree)(tpt), start)
282       case tree: Typed =>
283         implicit val ctx = prepTyped(tree, start)(outerCtx)
284         val expr = transformTree(tree.expr, start)
285         val tpt = transformTree(tree.tpt, start)
286         goTyped(cpy.Typed(tree)(expr, tpt), start)
287       case tree: CaseDef =>
288         implicit val ctx = prepCaseDef(tree, start)(outerCtx)

290         val guard = transformTree(tree.guard, start)
291         val body = transformTree(tree.body, start)
292         goCaseDef(cpy.CaseDef(tree)(pat, guard, body), start)
293       case tree: Closure =>
294         implicit val ctx = prepClosure(tree, start)(outerCtx)
295         val env = transformTrees(tree.env, start)
296         val meth = transformTree(tree.meth, start)
297         val tpt = transformTree(tree.tpt, start)
298         goClosure(cpy.Closure(tree)(env, meth, tpt), start)
299       case tree: Assign =>
300         implicit val ctx = prepAssign(tree, start)(outerCtx)
301         val lhs = transformTree(tree.lhs, start)
302         val rhs = transformTree(tree.rhs, start)
303         goAssign(cpy.Assign(tree)(lhs, rhs), start)
304       case tree: SeqLiteral =>
305         implicit val ctx = prepSeqLiteral(tree, start)(outerCtx)
306         val elems = transformTrees(tree.elems, start)
307         val elemtpt = transformTree(tree.elemtpt, start)
308         goSeqLiteral(cpy.SeqLiteral(tree)(elems, elemtpt), start)
309       case tree: Super =>
310         implicit val ctx = prepSuper(tree, start)(outerCtx)
311         goSuper(tree, start)
312       case tree: Template =>
313         implicit val ctx = prepTemplate(tree, start)(outerCtx)
314         val constr = transformSpecificTree(tree.constr, start)
315         val parents = transformTrees(tree.parents, start)(ctx.superCallContext)
316         val self = transformSpecificTree(tree.self, start)
317         val body = transformStats(tree.body, tree.symbol, start)
318         goTemplate(cpy.Template(tree)(constr, parents, self, body), start)
319       case tree: Match =>
320         implicit val ctx = prepMatch(tree, start)(outerCtx)
321         val selector = transformTree(tree.selector, start)
322         val cases = transformSpecificTrees(tree.cases, start)
323         goMatch(cpy.Match(tree)(selector, cases), start)
324       case tree: UnApply =>
325         implicit val ctx = prepUnApply(tree, start)(outerCtx)
326         val fun = transformTree(tree.fun, start)
327         val implicits = transformTrees(tree.implicits, start)
328         val patterns = transformTrees(tree.patterns, start)
329         goUnApply(cpy.UnApply(tree)(fun, implicits, patterns), start)
330       case tree: PackageDef =>
331         implicit val ctx = prepPackageDef(tree, start)(outerCtx)
332         def mapPackage(implicit ctx: Context) = {
333           val pid = transformSpecificTree(tree.pid, start)
334           val stats = transformStats(tree.stats, tree.symbol, start)
335           cpy.PackageDef(tree)(pid, stats)
336         }
337         goPackageDef(mapPackage(localContext), start)
338       case tree: Try =>
339         implicit val ctx = prepTry(tree, start)(outerCtx)
340         val expr = transformTree(tree.expr, start)
341         val cases = transformSpecificTrees(tree.cases, start)
342         val finalizer = transformTree(tree.finalizer, start)
343         goTry(cpy.Try(tree)(expr, cases, finalizer), start)
344       case tree: Inlined =>
345         implicit val ctx = prepInlined(tree, start)(outerCtx)
346         val bindings = transformSpecificTrees(tree.bindings, start)
347         val expansion = transformTree(tree.expansion, start)
348         goInlined(cpy.Inlined(tree)(tree.call, bindings, expansion), start)
349       case tree: Return =>
350         implicit val ctx = prepReturn(tree, start)(outerCtx)
351         val expr = transformTree(tree.expr, start)
352         goReturn(cpy.Return(tree)(expr, tree.from), start)
353           // don't transform `tree.from`, as this is not a normal ident, but
354           // a pointer to the enclosing method.
355       case tree: Alternative =>
356         implicit val ctx = prepAlternative(tree, start)(outerCtx)
357         val trees = transformTrees(tree.trees, start)
358         goAlternative(cpy.Alternative(tree)(trees), start)
359       case tree =>
360         implicit val ctx = prepOther(tree, start)(outerCtx)
361         goOther(tree, start)
362     }
363 
364     if (tree.isInstanceOf[NameTree]) transformNamed(tree, start, ctx)
365     else transformUnnamed(tree, start, ctx)
366   }

406   private val clsMethodsCache = new java.util.IdentityHashMap[Class[_], Array[java.lang.reflect.Method]]

408   /** Does `phase` contain a redefinition of method `name`?
409    *  (which is a method of MiniPhase)
410    */
411   private def defines(phase: MiniPhase, name: String) = {
412     def hasRedefinedMethod(cls: Class[_]): Boolean =
413       if (cls.eq(classOf[MiniPhase])) false
414       else {
415         var clsMethods = clsMethodsCache.get(cls)
416         if (clsMethods eq null) {
417           clsMethods = cls.getDeclaredMethods
418           clsMethodsCache.put(cls, clsMethods)
419         }
420         clsMethods.exists(_.getName == name) ||
421         hasRedefinedMethod(cls.getSuperclass)
422       }
423     hasRedefinedMethod(phase.getClass)
424   }

429   private def init(methName: String): Array[MiniPhase] = {
430     var nx: Array[MiniPhase] = emptyNxArray
431     for (idx <- miniPhases.length - 1 to 0 by -1) {
432       val subPhase = miniPhases(idx)
433       if (defines(subPhase, methName)) {
434         if (nx eq emptyNxArray) nx = newNxArray
435         nx(idx) = subPhase
436       }
437       else if (nx ne emptyNxArray) nx(idx) = nx(idx + 1)
438     }
439     nx
