106     /** Mark symbol `sym` as being free in `enclosure`, unless `sym` is defined
107      *  in `enclosure` or there is an intermediate class properly containing `enclosure`
108      *  in which `sym` is also free. Also, update `liftedOwner` of `enclosure` so
109      *  that `enclosure` can access `sym`, or its proxy in an intermediate class.
110      *  This means:
111      *
112      *    1. If there is an intermediate class in which `sym` is free, `enclosure`
113      *       must be contained in that class (in order to access the `sym proxy stored
114      *       in the class).
115      *
116      *    2. If there is no intermediate class, `enclosure` must be contained
117      *       in the class enclosing `sym`.
118      *
119      *  @return  If there is a non-trait class between `enclosure` and
120      *           the owner of `sym`, the largest such class.
121      *           Otherwise, if there is a trait between `enclosure` and
122      *           the owner of `sym`, the largest such trait.
123      *           Otherwise, NoSymbol.
124      *
125      *  @pre sym.owner.isTerm, (enclosure.isMethod || enclosure.isClass)
126      *
127      *  The idea of `markFree` is illustrated with an example:
128      *
129      *  def f(x: int) = {
130      *    class C {
131      *      class D {
132      *        val y = x
133      *      }
134      *    }
135      *  }
136      *
137      *  In this case `x` is free in the primary constructor of class `C`.
138      *  but it is not free in `D`, because after lambda lift the code would be transformed
139      *  as follows:
140      *
141      *  def f(x$0: int) {
142      *    class C(x$0: int) {
143      *      val x$1 = x$0
144      *      class D {
145      *        val y = outer.x$1
146      *      }
147      *    }
148      *  }
149      */
150     private def markFree(sym: Symbol, enclosure: Symbol)(implicit ctx: Context): Symbol = try {
151       if (!enclosure.exists) throw new NoPath
152       if (enclosure == sym.enclosure) NoSymbol
153       else {
154         ctx.debuglog(i"mark free: ${sym.showLocated} with owner ${sym.maybeOwner} marked free in $enclosure")
155         val intermediate =
156           if (enclosure.is(PackageClass)) enclosure
157           else if (enclosure.isConstructor) markFree(sym, enclosure.owner.enclosure)
158           else markFree(sym, enclosure.enclosure)
159         if (intermediate.exists) narrowLiftedOwner(enclosure, intermediate)
160         if (!intermediate.isRealClass || enclosure.isConstructor) {
161           // Constructors and methods nested inside traits get the free variables
162           // of the enclosing trait or class.
163           // Conversely, local traits do not get free variables.
164           if (!enclosure.is(Trait))
165             if (symSet(free, enclosure).add(sym)) {
166               changedFreeVars = true
167               ctx.log(i"$sym is free in $enclosure")
168             }
169         }
170         if (intermediate.isRealClass) intermediate
171         else if (enclosure.isRealClass) enclosure
172         else if (intermediate.isClass) intermediate
173         else if (enclosure.isClass) enclosure
174         else NoSymbol
175       }
176     } catch {
177       case ex: NoPath =>
178         println(i"error lambda lifting ${ctx.compilationUnit}: $sym is not visible from $enclosure")
179         throw ex
180     }

271     private def computeLiftedOwners()(implicit ctx: Context): Unit =
272       do {
273         changedLiftedOwner = false
274         for {
275           caller <- called.keys
276           callee <- called(caller)

279           val calleeOwner = normalizedCallee.owner

324       for ((local, lOwner) <- liftedOwner) {

326           if (lOwner is Package)  {

455 /** This phase performs the necessary rewritings to eliminate classes and methods
456  *  nested in other methods. In detail:
457  *   1. It adds all free variables of local functions as additional parameters (proxies).
458  *   2. It rebinds references to free variables to the corresponding proxies,
459  *   3. It lifts all local functions and classes out as far as possible, but at least
460  *      to the enclosing class.
461  *   4. It stores free variables of non-trait classes as additional fields of the class.
462  *      The fields serve as proxies for methods in the class, which avoids the need
463  *      of passing additional parameters to these methods.
464  *
465  *  A particularly tricky case are local traits. These cannot store free variables
466  *  as field proxies, because LambdaLift runs after Mixin, so the fields cannot be
467  *  expanded anymore. Instead, methods of local traits get free variables of
468  *  the trait as additional proxy parameters. The difference between local classes
469  *  and local traits is illustrated by the two rewritings below.
470  *
471  *     def f(x: Int) = {           def f(x: Int) = new C(x).f2
472  *       class C {          ==>    class C(x$1: Int) {
473  *         def f2 = x                def f2 = x$1
474  *       }                         }
475  *       new C().f2
476  *     }
477  *
478  *     def f(x: Int) = {           def f(x: Int) = new C().f2(x)
479  *       trait T {          ==>    trait T
480  *         def f2 = x                def f2(x$1: Int) = x$1
481  *       }                         }
482  *       class C extends T         class C extends T
483  *       new C().f2
484  *     }
485  */
