22 /** This phase adds outer accessors to classes and traits that need them.
23  *  Compared to Scala 2.x, it tries to minimize the set of classes
24  *  that take outer accessors by scanning class implementations for
25  *  outer references.
26  *
27  *  The following things are delayed until erasure and are performed
28  *  by class OuterOps:
29  *
30  *   - add outer parameters to constructors
31  *   - pass outer arguments in constructor calls
32  *
33  *   replacement of outer this by outer paths is done in Erasure.
34  *   needs to run after pattern matcher as it can add outer checks and force creation of $outer
35  */

61   /** First, add outer accessors if a class does not have them yet and it references an outer this.
62    *  If the class has outer accessors, implement them.
63    *  Furthermore, if a parent trait might have an outer accessor,
64    *  provide an implementation for the outer accessor by computing the parent's
65    *  outer from the parent type prefix. If the trait ends up not having an outer accessor
66    *  after all, the implementation is redundant, but does not harm.
67    *  The same logic is not done for non-trait parent classes because for them the outer
68    *  pointer is passed in the super constructor, which will be implemented later in
69    *  a separate phase which needs to run after erasure. However, we make sure here
70    *  that the super class constructor is indeed a New, and not just a type.
71    */
72   override def transformTemplate(impl: Template)(implicit ctx: Context): Tree = {
73     val cls = ctx.owner.asClass
74     val isTrait = cls.is(Trait)
75     if (needsOuterIfReferenced(cls) &&
76         !needsOuterAlways(cls) &&
77         impl.existsSubTree(referencesOuter(cls, _)))
78       ensureOuterAccessors(cls)
79 
80     val clsHasOuter = hasOuter(cls)
81     if (clsHasOuter || cls.mixins.exists(needsOuterIfReferenced)) {
82       val newDefs = new mutable.ListBuffer[Tree]
83 
84       if (clsHasOuter) {
85         if (isTrait)
86           newDefs += DefDef(outerAccessor(cls).asTerm, EmptyTree)
87         else {
88           val outerParamAcc = outerParamAccessor(cls)
89           newDefs += ValDef(outerParamAcc, EmptyTree)
90           newDefs += DefDef(outerAccessor(cls).asTerm, ref(outerParamAcc))
91         }
92       }
93 
94       for (parentTrait <- cls.mixins) {
95         if (needsOuterIfReferenced(parentTrait)) {
96           val parentTp = cls.denot.thisType.baseType(parentTrait)
97           val outerAccImpl = newOuterAccessor(cls, parentTrait).enteredAfter(thisPhase)
98           newDefs += DefDef(outerAccImpl, singleton(fixThis(outerPrefix(parentTp))))
99         }
100       }
101 
102       val parents1 =
103         for (parent <- impl.parents) yield {
104           val parentCls = parent.tpe.classSymbol.asClass
105           if (parentCls.is(Trait)) {
106             parent
107           }
108           else parent match { // ensure class parent is a constructor
109             case parent: TypeTree => New(parent.tpe, Nil).withPos(impl.pos)
110             case _ => parent
111           }
112         }
113       cpy.Template(impl)(parents = parents1, body = impl.body ++ newDefs)
114     }
115     else impl
116   }

142   /** A new outer accessor or param accessor.
143    *  @param  owner  The class where the outer accessor is located
144    *  @param  cls    The class relative to which the outer is computed (can be a base class of owner)
145    *  @param  name   The name of the outer access
146    *  @param  flags  The flags of the outer accessor
147    *
148    * The type of the outer accessor is computed as follows:
149    * Let O[X1, .., Xn] be the class enclosing `cls`.
150    *  - if owner == cls, O[X1, ..., Xn]
151    *  - otherwise, if the class P enclosing `owner` derives from O, the
152    *    base type of P.this wrt class O
153    *  - otherwise O[_, ..., _]
154    */
155   private def newOuterSym(owner: ClassSymbol, cls: ClassSymbol, name: TermName, flags: FlagSet)(implicit ctx: Context) = {
156     val outerThis = owner.owner.enclosingClass.thisType
157     val outerCls = cls.owner.enclosingClass
158     val target =
159       if (owner == cls)
160         outerCls.appliedRef
161       else
162         outerThis.baseType(outerCls).orElse(
163   		    outerCls.typeRef.appliedTo(outerCls.typeParams.map(_ => TypeBounds.empty)))
164     val info = if (flags.is(Method)) ExprType(target) else target
165     ctx.withPhaseNoEarlier(ctx.explicitOuterPhase.next) // outer accessors are entered at explicitOuter + 1, should not be defined before.
166        .newSymbol(owner, name, Synthetic | flags, info, coord = cls.coord)
167   }

213   /** The outer accessor of class `cls`. To find it is a bit tricky. The
214    *  class might have been moved with new owners between ExplicitOuter and Erasure,
215    *  where the method is also called. For instance, it might have been part
216    *  of a by-name argument, and therefore be moved under a closure method
217    *  by ElimByName. In that case looking up the method again at Erasure with the
218    *  fully qualified name `outerAccName` will fail, because the `outerAccName`'s
219    *  result is phase dependent. In that case we use a backup strategy where we search all
220    *  definitions in the class to find the one with the OuterAccessor flag.

235   /** Tree references an outer class of `cls` which is not a static owner.
236    */
237   def referencesOuter(cls: Symbol, tree: Tree)(implicit ctx: Context): Boolean = {
238     def isOuterSym(sym: Symbol) =
239       !sym.isStaticOwner && cls.isProperlyContainedIn(sym)
240     def isOuterRef(ref: Type): Boolean = ref match {
241       case ref: ThisType =>
242         isOuterSym(ref.cls)
243       case ref: TermRef =>
244         if (ref.prefix ne NoPrefix)
245           !ref.symbol.isStatic && isOuterRef(ref.prefix)
246         else (
247           (ref.symbol is Hoistable) &&
248             // ref.symbol will be placed in enclosing class scope by LambdaLift, so it might need
249             // an outer path then.
250             isOuterSym(ref.symbol.owner.enclosingClass)
251           ||
252             // If not hoistable, ref.symbol will get a proxy in immediately enclosing class. If this properly
253             // contains the current class, it needs an outer path.
254             // If the symbol is hoistable, it might have free variables for which the same
255             // reasoning applies. See pos/i1664.scala
256             ctx.owner.enclosingClass.owner.enclosingClass.isContainedIn(ref.symbol.owner)
257           )
258       case _ => false
259     }
260     def hasOuterPrefix(tp: Type) = tp match {
261       case TypeRef(prefix, _) => isOuterRef(prefix)
262       case _ => false
263     }
264     tree match {
265       case _: This | _: Ident => isOuterRef(tree.tpe)
266       case nw: New =>
267         val newCls = nw.tpe.classSymbol
268         isOuterSym(newCls.owner.enclosingClass) ||
269         hasOuterPrefix(nw.tpe) ||
270         newCls.owner.isTerm && cls.isProperlyContainedIn(newCls)
271           // newCls might get proxies for free variables. If current class is
272           // properly contained in newCls, it needs an outer path to newCls access the
273           // proxies and forward them to the new instance.
274       case _ =>
275         false
276     }
277   }

317   /** The operations in this class
318    *   - add outer parameters
319    *   - pass outer arguments to these parameters
320    *   - replace outer this references by outer paths.
321    *  They are called from erasure. There are two constraints which
322    *  suggest these operations should be done in erasure.
323    *   - Replacing this references with outer paths loses aliasing information,
324    *     so programs will not typecheck with unerased types unless a lot of type
325    *     refinements are added. Therefore, outer paths should be computed no
326    *     earlier than erasure.
327    *   - outer parameters should not show up in signatures, so again
328    *     they cannot be added before erasure.
329    *   - outer arguments need access to outer parameters as well as to the
330    *     original type prefixes of types in New expressions. These prefixes
331    *     get erased during erasure. Therefore, outer arguments have to be passed
332    *     no later than erasure.

367     /** A path of outer accessors starting from node `start`. `start` defaults to the
368      *  context owner's this node. There are two alternative conditions that determine
369      *  where the path ends:
370      *
371      *   - if the initial `count` parameter is non-negative: where the number of
372      *     outer accessors reaches count.
373      *   - if the initial `count` parameter is negative: where the class symbol of
374      *     the type of the reached tree matches `toCls`.
375      */
376     def path(start: Tree = This(ctx.owner.lexicallyEnclosingClass.asClass),
377              toCls: Symbol = NoSymbol,
378              count: Int = -1): Tree = try {
379       @tailrec def loop(tree: Tree, count: Int): Tree = {
380         val treeCls = tree.tpe.widen.classSymbol
381         val outerAccessorCtx = ctx.withPhaseNoLater(ctx.lambdaLiftPhase) // lambdalift mangles local class names, which means we cannot reliably find outer acessors anymore
382         ctx.log(i"outer to $toCls of $tree: ${tree.tpe}, looking for ${outerAccName(treeCls.asClass)(outerAccessorCtx)} in $treeCls")
383         if (count == 0 || count < 0 && treeCls == toCls) tree
384         else {
385           val acc = outerAccessor(treeCls.asClass)(outerAccessorCtx)
386           assert(acc.exists,
387               i"failure to construct path from ${ctx.owner.ownersIterator.toList}%/% to `this` of ${toCls.showLocated};\n${treeCls.showLocated} does not have an outer accessor")
388           loop(tree.select(acc).ensureApplied, count - 1)
389         }
390       }
391       ctx.log(i"computing outerpath to $toCls from ${ctx.outersIterator.map(_.owner).toList}")
392       loop(start, count)
393     } catch {
394       case ex: ClassCastException =>
395         throw new ClassCastException(i"no path exists from ${ctx.owner.enclosingClass} to $toCls")
396     }
