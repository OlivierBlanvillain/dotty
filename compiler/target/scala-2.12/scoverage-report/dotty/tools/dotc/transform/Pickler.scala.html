33   /** Drop any elements of this list that are linked module classes of other elements in the list */
34   private def dropCompanionModuleClasses(clss: List[ClassSymbol])(implicit ctx: Context): List[ClassSymbol] = {
35     val companionModuleClasses =
36       clss.filterNot(_ is Module).map(_.linkedClass).filterNot(_.unforcedIsAbsent)
37     clss.filterNot(companionModuleClasses.contains)
38   }

40   override def run(implicit ctx: Context): Unit = {
41     val unit = ctx.compilationUnit
42     pickling.println(i"unpickling in run ${ctx.runId}")
43 
44     for { cls <- dropCompanionModuleClasses(topLevelClasses(unit.tpdTree))
45           tree <- sliceTopLevel(unit.tpdTree, cls) } {
46       val pickler = new TastyPickler(cls)
47       if (ctx.settings.YtestPickler.value) {
48         beforePickling(cls) = tree.show
49         picklers(cls) = pickler
50       }
51       val treePkl = pickler.treePkl
52       treePkl.pickle(tree :: Nil)
53       treePkl.compactify()
54       pickler.addrOfTree = treePkl.buf.addrOfTree
55       pickler.addrOfSym = treePkl.addrOfSym
56       if (tree.pos.exists)
57         new PositionPickler(pickler, treePkl.buf.addrOfTree).picklePositions(tree :: Nil)
58 
59       // other pickle sections go here.
60       val pickled = pickler.assembleParts()
61       unit.pickled += (cls -> pickled)
62 
63       def rawBytes = // not needed right now, but useful to print raw format.
64         pickled.iterator.grouped(10).toList.zipWithIndex.map {
65           case (row, i) => s"${i}0: ${row.mkString(" ")}"
66         }
67       // println(i"rawBytes = \n$rawBytes%\n%") // DEBUG
68       if (pickling ne noPrinter) {
69         println(i"**** pickled info of $cls")
70         new TastyPrinter(pickler.assembleParts()).printContents()
71       }
72     }
73   }

103   private def testSame(unpickled: String, previous: String, cls: ClassSymbol)(implicit ctx: Context) =
104     if (previous != unpickled) {
