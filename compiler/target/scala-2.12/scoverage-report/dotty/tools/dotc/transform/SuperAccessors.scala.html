17 /** This class performs the following functions:
18  *
19  *  (1) Adds super accessors for all super calls that either
20  *  appear in a trait or have as a target a member of some outer class.
21  *
22  *  (2) Adds protected accessors if the access to the protected member happens
23  *  in a class which is not a subclass of the member's owner.
24  *
25  *  It also checks that:
26  *
27  *  (1) Symbols accessed from super are not abstract, or are overridden by
28  *  an abstract override.
29  *
30  *  (2) If a symbol accessed from super is defined in a real class (not a trait),
31  *  there are no abstract members which override this member in Java's rules
32  *  (see SI-4989; such an access would lead to illegal bytecode)
33  *
34  *  (3) Super calls do not go to some synthetic members of Any (see isDisallowed)
35  *
36  *  (4) Super calls do not go to synthetic field accessors
37  */

70     /** A super accessor call corresponding to `sel` */
71     private def superAccessorCall(sel: Select)(implicit ctx: Context) = {
72       val Select(qual, name) = sel
73       val sym = sel.symbol
74       val clazz = qual.symbol.asClass
75       var superName = SuperAccessorName(name.asTermName)
76       if (clazz is Trait) superName = superName.expandedName(clazz)
77       val superInfo = sel.tpe.widenSingleton.ensureMethodic
78 
79       val accPos = sel.pos.focus
80       val superAcc = clazz.info.decl(superName)
81         .suchThat(_.signature == superInfo.signature).symbol
82         .orElse {
83           ctx.debuglog(s"add super acc ${sym.showLocated} to $clazz")
84           val maybeDeferred = if (clazz is Trait) Deferred else EmptyFlags
85           val acc = ctx.newSymbol(
86               clazz, superName, Artifact | Method | maybeDeferred,
87               superInfo, coord = accPos).enteredAfter(thisPhase)
88           // Diagnostic for SI-7091
89           if (!accDefs.contains(clazz))
90             ctx.error(s"Internal error: unable to store accessor definition in ${clazz}. clazz.hasPackageFlag=${clazz is Package}. Accessor required for ${sel} (${sel.show})", sel.pos)
91           else accDefs(clazz) += DefDef(acc, EmptyTree).withPos(accPos)
92           acc
93         }
94 
95       This(clazz).select(superAcc).withPos(sel.pos)
96     }

98     /** Check selection `super.f` for conforming to rules. If necessary,
99      *  replace by a super accessor call.
100      */
101     private def transformSuperSelect(sel: Select)(implicit ctx: Context): Tree = {
102       val Select(sup @ Super(_, mix), name) = sel
103       val sym   = sel.symbol
104       assert(sup.symbol.exists, s"missing symbol in $sel: ${sup.tpe}")
105       val clazz = sup.symbol
106 
107       if (sym.isTerm && !sym.is(Method, butNot = Accessor) && !ctx.owner.is(ParamForwarder))
108         // ParamForwaders as installed ParamForwarding.scala do use super calls to vals
109         ctx.error(s"super may be not be used on ${sym.underlyingSymbol}", sel.pos)
110       else if (isDisallowed(sym))
111         ctx.error(s"super not allowed here: use this.${sel.name} instead", sel.pos)
112       else if (sym is Deferred) {
113         val member = sym.overridingSymbol(clazz.asClass)
114         if (!mix.name.isEmpty ||
115             !member.exists ||
116             !((member is AbsOverride) && member.isIncompleteIn(clazz)))
117           ctx.error(
118               i"${sym.showLocated} is accessed from super. It may not be abstract unless it is overridden by a member declared `abstract' and `override'",
119               sel.pos)
120         else ctx.log(i"ok super $sel ${sym.showLocated} $member $clazz ${member.isIncompleteIn(clazz)}")
121       }
122       else if (mix.name.isEmpty && !(sym.owner is Trait))
123         // SI-4989 Check if an intermediate class between `clazz` and `sym.owner` redeclares the method as abstract.
124         for (intermediateClass <- clazz.info.baseClasses.tail.takeWhile(_ != sym.owner)) {
125           val overriding = sym.overridingSymbol(intermediateClass)
126           if ((overriding is (Deferred, butNot = AbsOverride)) && !(overriding.owner is Trait))
127             ctx.error(
128                 s"${sym.showLocated} cannot be directly accessed from ${clazz} because ${overriding.owner} redeclares it as abstract",
129                 sel.pos)
130 
131         }
132       if (name.isTermName && mix.name.isEmpty &&
133           ((clazz is Trait) || clazz != ctx.owner.enclosingClass || !validCurrentClass))
134         superAccessorCall(sel)(ctx.withPhase(thisPhase.next))
135       else sel
136     }

159      *  the accessor and returns the same member. The result is already
160      *  typed.
161      */
162     private def protectedAccessorCall(sel: Select, targs: List[Tree])(implicit ctx: Context): Tree = {
163       val Select(qual, _) = sel
164       val sym = sel.symbol.asTerm
165       val clazz = hostForAccessorOf(sym, currentClass)
166       assert(clazz.exists, sym)
167       ctx.debuglog("Decided for host class: " + clazz)
168 
169       val accName = ProtectedAccessorName(sym.name)
170 
171       // if the result type depends on the this type of an enclosing class, the accessor
172       // has to take an object of exactly this type, otherwise it's more general
173       val receiverType =
174         if (isThisType(sym.info.finalResultType)) clazz.thisType
175         else clazz.classInfo.selfType
176       val accType = {
177         def accTypeOf(tpe: Type): Type = tpe match {
178           case tpe: PolyType =>
179             tpe.derivedLambdaType(tpe.paramNames, tpe.paramInfos, accTypeOf(tpe.resultType))
180           case _ =>
181             MethodType(receiverType :: Nil)(mt => tpe.substThis(sym.owner.asClass, mt.newParamRef(0)))
182         }
183         accTypeOf(sym.info)
184       }
185       val accPos = sel.pos.focus
186       val protectedAccessor = clazz.info.decl(accName).suchThat(_.signature == accType.signature).symbol orElse {
187         val newAcc = ctx.newSymbol(
188           clazz, accName, Artifact | Method, accType, coord = accPos).enteredAfter(thisPhase)
189         val code = polyDefDef(newAcc, trefs => vrefss => {
190           val (receiver :: _) :: tail = vrefss
191           val base = receiver.select(sym).appliedToTypes(trefs)
192           (base /: tail)(Apply(_, _)).withPos(accPos)
193         })
194         ctx.debuglog("created protected accessor: " + code)
195         accDefs(clazz) += code
196         newAcc
197       }
198       val res = This(clazz)
199         .select(protectedAccessor)
200         .appliedToTypeTrees(targs)
201         .appliedTo(qual)
202         .withPos(sel.pos)
203       ctx.debuglog(s"Replaced $sel with $res")
204       res

286     /** Does `sym` need an accessor when accessed from `currentClass`?
287      *  A special case arises for classes with explicit self-types. If the
288      *  self type is a Java class, and a protected accessor is needed, we issue
289      *  an error. If the self type is a Scala class, we don't add an accessor.
290      *  An accessor is not needed if the access boundary is larger than the
291      *  enclosing package, since that translates to 'public' on the host sys.
292      *  (as Java has no real package nesting).
293      *
294      * If the access happens inside a 'trait', access is more problematic since
295      * the implementation code is moved to an '$class' class which does not
296      * inherit anything. Since we can't (yet) add accessors for 'required'
297      * classes, this has to be signaled as error.
298      * FIXME Need to better understand this logic
299      */
300     private def needsProtectedAccessor(sym: Symbol, pos: Position)(implicit ctx: Context): Boolean = {
301       val clazz = currentClass
302       val host = hostForAccessorOf(sym, clazz)
303       def accessibleThroughSubclassing =
304         validCurrentClass && clazz.classInfo.selfType.derivesFrom(sym.owner) && !clazz.is(Trait)
305 
306       val isCandidate = (
307            sym.is(Protected)
308         && sym.is(JavaDefined)
309         && !sym.effectiveOwner.is(Package)
310         && !accessibleThroughSubclassing
311         && (sym.enclosingPackageClass != currentClass.enclosingPackageClass)
312         && (sym.enclosingPackageClass == sym.accessBoundary(sym.enclosingPackageClass))
313       )
314       val hostSelfType = host.classInfo.selfType
315       def isSelfType = !(host.appliedRef <:< hostSelfType) && {
316         if (hostSelfType.typeSymbol.is(JavaDefined))
317           ctx.restrictionError(
318             s"cannot accesses protected $sym from within $clazz with host self type $hostSelfType", pos)
319         true
320       }
321       def isJavaProtected = host.is(Trait) && sym.is(JavaDefined) && {
322         ctx.restrictionError(
323           s"""$clazz accesses protected $sym inside a concrete trait method.
324              |Add an accessor in a class extending ${sym.enclosingClass} as a workaround.""".stripMargin,
325           pos
326         )
327         true
328       }
329       isCandidate && !host.is(Package) && !isSelfType && !isJavaProtected
330     }

356     /** Transform select node, adding super and protected accessors as needed */
357     def transformSelect(tree: Tree, targs: List[Tree])(implicit ctx: Context) = {
358       val sel @ Select(qual, name) = tree
359       val sym = sel.symbol
360       qual match {
361         case _: This =>
362           /*
363            * A trait which extends a class and accesses a protected member
364            *  of that class cannot implement the necessary accessor method
365            *  because its implementation is in an implementation class (e.g.
366            *  Foo$class) which inherits nothing, and jvm access restrictions
367            *  require the call site to be in an actual subclass. So non-trait
368            *  classes inspect their ancestors for any such situations and
369            *  generate the accessors.  See SI-2296.
370            */
371           // FIXME (from scalac's SuperAccessors)
372           // - this should be unified with needsProtectedAccessor, but some
373           // subtlety which presently eludes me is foiling my attempts.
374           val shouldEnsureAccessor = (
375             (currentClass is Trait)
376             && (sym is Protected)
377             && sym.enclosingClass != currentClass
378             && !(sym.owner is PackageClass) // SI-7091 no accessor needed package owned (ie, top level) symbols
379             && !(sym.owner is Trait)
380             && sym.owner.enclosingPackageClass != currentClass.enclosingPackageClass
381             && qual.symbol.info.member(sym.name).exists
382             && !needsProtectedAccessor(sym, sel.pos))
383           if (shouldEnsureAccessor) {
384             ctx.log("Ensuring accessor for call to protected " + sym.showLocated + " from " + currentClass)
385             superAccessorCall(sel)
386           } else
387             ensureProtectedAccessOK(sel, targs)
388 
389         case Super(_, mix) =>
390           transformSuperSelect(sel)
391 
392         case _ =>
393           ensureProtectedAccessOK(sel, targs)
394       }
395     }
