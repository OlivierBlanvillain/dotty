18 /** This phase consists of a series of small, simple, local optimisations
19  *  applied as a fix point transformation over Dotty Trees.
20  *
21  *  The termination condition uses referential equality on Trees. Furthermore,
22  *  termination relies of every optimisation to be shrinking transformations.
23  *
24  *  This phase is intended to be run multiple times in the compilation pipeline.
25  *  This is due to several reasons:
26  *   - running this phase early allows to reduce size of compilation unit, speeding up subsequent transformations.
27  *   - running this phase late allows to eliminate inefficiencies created by previous phase
28  *   - different patters are easier to optimize at different moments of pipeline
29  */
30 class Simplify extends MiniPhase with IdentityDenotTransformer {
31   import tpd._
32   override def phaseName: String = "simplify"
33   override val cpy = tpd.cpy

38   /** The original intention is to run most optimizations both before and after erasure.
39    *  Erasure creates new inefficiencies as well as new optimization opportunities.
40    *
41    *  The order of optimizations is tuned to converge faster.
42    *  Reordering them may require quadratically more rounds to finish.
43    */
44   private def beforeErasure: List[Optimisation] =
45     new InlineCaseIntrinsics(this)  ::
46     new RemoveUnnecessaryNullChecks ::
47     new InlineOptions               ::
48     //new InlineLabelsCalledOnce    :: // not needed: new pattern matcher does this already
49     new Valify(this)                ::
50     new Devalify                    ::
51     new Jumpjump                    ::
52     new DropGoodCasts               ::
53     new DropNoEffects(this)         ::
54     new InlineLocalObjects(this)    ::
55     // new Varify                      :: // varify could stop other transformations from being applied. postponed.
56     // new BubbleUpNothing             ::
57     new ConstantFold(this)          ::
58     Nil

63     new Devalify                    ::
64     new Jumpjump                    ::

66     new DropNoEffects(this)         ::

72   /** Optimisation fuel, for debugging. Decremented every time Simplify
73    *  applies an optimisation until fuel == 0. Original idea from Automatic
74    *  Isolation of Compiler Errors by David Whalley. Unable with -Yopt-fuel.
75    *
76    *  The fuel can be used to do a bisection on large test cases that fail
77    *  -optimise. See compiler/test/bisect.sh for a shell script to automates
78    *  the bisection search.
79    */
80   var fuel: Int = -1

82   override def prepareForUnit(tree: Tree)(implicit ctx: Context) = {
83     SeqFactoryClass = ctx.requiredClass("scala.collection.generic.SeqFactory")
84     CommutativePrimitiveOperations = Set(defn.Boolean_&&, defn.Boolean_||, defn.Int_+, defn.Int_*, defn.Long_+, defn.Long_*)
85 
86     val maxFuel = ctx.settings.YoptFuel.value
87     if (fuel < 0 && maxFuel > 0) // Both defaults are at -1
88       fuel = maxFuel
89 
90     optimisations = {
91       val o = if (ctx.erasedTypes) afterErasure else beforeErasure
92       val p = ctx.settings.YoptPhases.value
93       if (p.isEmpty) o else o.filter(x => p.contains(x.name))
94     }
95 
96     ctx
97   }

99   // The entry point of local optimisation: DefDefs
100   override def transformDefDef(tree: DefDef)(implicit ctx: Context): Tree = {
101     val ctx0 = ctx
102     if (ctx.settings.optimise.value && !tree.symbol.is(Label)) {
103       implicit val ctx: Context = ctx0.withOwner(tree.symbol(ctx0))
104       var rhs0 = tree.rhs
105       var rhs1: Tree = null
106       while (rhs1 ne rhs0) {
107         rhs1 = rhs0
108         optimisations.foreach { optimisation =>
109           // Visit
110           rhs0.foreachSubTree(optimisation.visitor)
111 
112           // Transform
113           rhs0 = new TreeMap() {
114             override def transform(tree: Tree)(implicit ctx: Context): Tree = {
115               val innerCtx = if (tree.isDef && tree.symbol.exists) ctx.withOwner(tree.symbol) else ctx
116               val childOptimizedTree = super.transform(tree)(innerCtx)
117               printIfDifferent(childOptimizedTree, optimisation.transformer(ctx)(childOptimizedTree), optimisation)
118             }
119           }.transform(rhs0)
120 
121           // Clean
122           optimisation.clear()
123         }
124       }
125       if (rhs0 ne tree.rhs) tpd.cpy.DefDef(tree)(rhs = rhs0)
126       else tree
127     } else tree
128   }
129 
130   private def printIfDifferent(tree1: Tree, tree2: => Tree, opt: Optimisation)(implicit ctx: Context): Tree = {
131     if (fuel == -1)
132       tree2 // Does nothing when fuel is disabled.
133     else if (fuel == 0)
134       tree1 // No more fuel? No more transformations for you!
135     else {  // Print the trees if different and consume fuel accordingly.
136       val t2 = tree2
137       if (tree1 ne t2) {
138         if (fuel > 0)
139           fuel -= 1
140         if (fuel != -1 && fuel < 5) {
141           println(s"${tree1.symbol} was simplified by ${opt.name} (fuel=$fuel): ${tree1.show}")
142           println(s"became after ${opt.name}: (fuel=$fuel) ${t2.show}")
143         }
144       }
145       t2
146     }
147   }

153   /** Is this tree mutable, or java.lang.System.{in, out, err}? These three
154    *  System members are the only static final fields that are mutable.
155    *  See https://docs.oracle.com/javase/specs/jls/se8/html/jls-17.html#jls-17.5.4
156    */
157   def isEffectivelyMutable(t: Tree)(implicit ctx: Context): Boolean = t match {
158     case _ if t.symbol.is(Mutable) => true
159     case _: Select | _: Ident =>
160       t.symbol.owner == defn.SystemModule
161     case _ => false
162   }

164   def isImmutableAccessor(t: Tree)(implicit ctx: Context): Boolean = {
165     val sym = t.symbol
166     val isImmutableGetter = sym.isGetter && !sym.is(Mutable | Lazy)
167     val isCaseAccessor    = sym.is(CaseAccessor) && !sym.is(Mutable | Lazy)
168     val isProductAccessor = sym.exists                               &&
169                             sym.owner.derivesFrom(defn.ProductClass) &&
170                             sym.owner.is(CaseClass)                  &&
171                             sym.name.isSelectorName                  &&
172                             !sym.info.decls.exists(_.is(Mutable | Lazy)) // Conservatively covers case class A(var x: Int)
173     val isErasedPhantom   = PhantomErasure.isErasedPhantom(sym)
174 
175     isImmutableGetter || isCaseAccessor || isProductAccessor || isErasedPhantom
176   }
