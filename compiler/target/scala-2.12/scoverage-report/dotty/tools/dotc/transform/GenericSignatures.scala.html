299   /** The intersection dominator (SLS 3.7) of a list of types is computed as follows.
300     *
301     *  - If the list contains one or more occurrences of scala.Array with
302     *    type parameters El1, El2, ... then the dominator is scala.Array with
303     *    type parameter of intersectionDominator(List(El1, El2, ...)).           <--- @PP: not yet in spec.
304     *  - Otherwise, the list is reduced to a subsequence containing only types
305     *    which are not subtypes of other listed types (the span.)
306     *  - If the span is empty, the dominator is Object.
307     *  - If the span contains a class Tc which is not a trait and which is
308     *    not Object, the dominator is Tc.                                        <--- @PP: "which is not Object" not in spec.
309     *  - Otherwise, the dominator is the first element of the span.
310     */
311   private def intersectionDominator(parents: List[Type])(implicit ctx: Context): Type = {
312     if (parents.isEmpty) defn.ObjectType
313     else {
314       val psyms = parents map (_.typeSymbol)
315       if (psyms contains defn.ArrayClass) {
316         // treat arrays specially
317         defn.ArrayType.appliedTo(intersectionDominator(parents.filter(_.typeSymbol == defn.ArrayClass).map(t => t.typeParams.head.paramInfo)))
318       } else {
319         // implement new spec for erasure of refined types.
320         def isUnshadowed(psym: Symbol) =
321           !(psyms exists (qsym => (psym ne qsym) && (qsym isSubClass psym)))
322         val cs = parents.iterator.filter { p => // isUnshadowed is a bit expensive, so try classes first
323           val psym = p.typeSymbol
324           psym.ensureCompleted()
325           psym.isClass && !psym.is(Trait) && isUnshadowed(psym)
326         }
327         (if (cs.hasNext) cs else parents.iterator.filter(p => isUnshadowed(p.typeSymbol))).next()
328       }
329     }
330   }

485     override def apply(x: Boolean, tp: Type): Boolean =
486       if (!x) {
487         tp match {
488           case RefinedType(parent, refinedName, refinedInfo) =>
