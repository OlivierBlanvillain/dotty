12 /** Transform references of the form
13  *
14  *     C.this.m
15  *
16  *  where `C` is a class with explicit self type and `C` is not a
17  *  subclass of the owner of `m` to
18  *
19  *     C.this.asInstanceOf[S & C.this.type].m
20  *
21  *  where `S` is the self type of `C`.
22  *  See run/i789.scala for a test case why this is needed.
23  *
24  *  Also replaces idents referring to the self type with ThisTypes.
25  */
26 class ExplicitSelf extends MiniPhase {
27   import ast.tpd._
28 
29   override def phaseName = "explicitSelf"
30 
31   override def transformIdent(tree: Ident)(implicit ctx: Context) = tree.tpe match {
32     case tp: ThisType =>
33       ctx.debuglog(s"owner = ${ctx.owner}, context = ${ctx}")
34       This(tp.cls) withPos tree.pos
35     case _ => tree
36   }
37 
38   override def transformSelect(tree: Select)(implicit ctx: Context): Tree = tree match {
39     case Select(thiz: This, name) if name.isTermName =>
40       val cls = thiz.symbol.asClass
41       if (cls.givenSelfType.exists && !cls.derivesFrom(tree.symbol.owner))
42         cpy.Select(tree)(thiz.asInstance(AndType(cls.classInfo.selfType, thiz.tpe)), name)
43       else tree
44     case _ => tree
45   }
46 }
