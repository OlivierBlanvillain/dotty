47   def isJavaEntryPoint(sym: Symbol)(implicit ctx: Context): Boolean = {
48     def fail(msg: String, pos: Position = sym.pos) = {
49       ctx.warning(sym.name +
50         s" has a main method with parameter type Array[String], but ${sym.fullName} will not be a runnable program.\n  Reason: $msg",
51         sourcePos(sym.pos)
52         // TODO: make this next claim true, if possible
53         //   by generating valid main methods as static in module classes
54         //   not sure what the jvm allows here
55         // + "  You can still run the program by calling it as " + javaName(sym) + " instead."
56       )
57       false
58     }
59     def failNoForwarder(msg: String) = {
60       fail(s"$msg, which means no static forwarder can be generated.\n")
61     }
62     val possibles = if (sym.flags is Flags.Module) (sym.info nonPrivateMember nme.main).alternatives else Nil
63     val hasApproximate = possibles exists {
64       m =>
65         m.info match {
66           case MethodTpe(_, p :: Nil, _) =>
67             p.typeSymbol == defn.ArrayClass
68           case _ => false
69         }
70     }
71     def precise(implicit ctx: Context) = {
72       val companion = sym.companionClass //sym.asClass.linkedClassOfClass
73       val javaPlatform = ctx.platform.asInstanceOf[JavaPlatform]
74       if (javaPlatform.hasJavaMainMethod(companion))
75         failNoForwarder("companion contains its own main method")
76       else if (companion.exists && companion.info.member(nme.main).exists)
77       // this is only because forwarders aren't smart enough yet
78         failNoForwarder("companion contains its own main method (implementation restriction: no main is allowed, regardless of signature)")
79       else if (companion.flags is Flags.Trait)
80         failNoForwarder("companion is a trait")
81       // Now either succeed, or issue some additional warnings for things which look like
82       // attempts to be java main methods.
83       else (possibles exists (x => javaPlatform.isJavaMainMethod(x.symbol))) || {
84         possibles exists {
85           m =>
86             m.symbol.info match {
87               case t: PolyType =>
88                 fail("main methods cannot be generic.")
89               case t: MethodType =>
90                 if (t.resultType :: t.paramInfos exists (_.typeSymbol.isAbstractType))
91                   fail("main methods cannot refer to type parameters or abstract types.", m.symbol.pos)
92                 else
93                   javaPlatform.isJavaMainMethod(m.symbol) || fail("main method must have exact signature (Array[String])Unit", m.symbol.pos)
94               case tp =>
95                 fail(s"don't know what this is: $tp", m.symbol.pos)
96             }
97         }
98       }
99     }
100 
101   // At this point it's a module with a main-looking method, so either succeed or warn that it isn't.
102   hasApproximate && precise(ctx.withPhase(ctx.erasurePhase))
103     // Before erasure so we can identify generic mains.
104

106 }
