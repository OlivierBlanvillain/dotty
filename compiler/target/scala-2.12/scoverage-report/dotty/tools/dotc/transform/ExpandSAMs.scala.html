13 /** Expand SAM closures that cannot be represented by the JVM as lambdas to anonymous classes.
14  *  These fall into five categories
15  *
16  *   1. Partial function closures, we need to generate a isDefinedAt method for these.
17  *   2. Closures implementing non-trait classes.
18  *   3. Closures implementing classes that inherit from a class other than Object
19  *      (a lambda cannot not be a run-time subtype of such a class)
20  *   4. Closures that implement traits which run initialization code.
21  *   5. Closures that get synthesized abstract methods in the transformation pipeline. These methods can be
22  *      (1) superaccessors, (2) outer references, (3) accessors for fields.
23  */
24 class ExpandSAMs extends MiniPhase {
25   override def phaseName = "expandSAMs"
26 
27   import ast.tpd._

29   /** Is the SAMType `cls` also a SAM under the rules of the platform? */
30   def isPlatformSam(cls: ClassSymbol)(implicit ctx: Context): Boolean =
31     ctx.platform.isSam(cls)

33   override def transformBlock(tree: Block)(implicit ctx: Context): Tree = tree match {
34     case Block(stats @ (fn: DefDef) :: Nil, Closure(_, fnRef, tpt)) if fnRef.symbol == fn.symbol =>
35       tpt.tpe match {
36         case NoType => tree // it's a plain function
37         case tpe @ SAMType(_) if tpe.isRef(defn.PartialFunctionClass) =>
38           checkRefinements(tpe, fn.pos)
39           toPartialFunction(tree)
40         case tpe @ SAMType(_) if isPlatformSam(tpe.classSymbol.asClass) =>
41           checkRefinements(tpe, fn.pos)
42           tree
43         case tpe =>
44           val tpe1 = checkRefinements(tpe, fn.pos)
45           val Seq(samDenot) = tpe1.abstractTermMembers.filter(!_.symbol.isSuperAccessor)
46           cpy.Block(tree)(stats,
47               AnonClass(tpe1 :: Nil, fn.symbol.asTerm :: Nil, samDenot.symbol.asTerm.name :: Nil))
48       }
49     case _ =>
50       tree
51   }

53   private def toPartialFunction(tree: Block)(implicit ctx: Context): Tree = {
54     val Block(
55           (applyDef @ DefDef(nme.ANON_FUN, Nil, List(List(param)), _, _)) :: Nil,
56           Closure(_, _, tpt)) = tree
57     val applyRhs: Tree = applyDef.rhs
58     val applyFn = applyDef.symbol.asTerm
59 
60     val MethodTpe(paramNames, paramTypes, _) = applyFn.info
61     val isDefinedAtFn = applyFn.copy(
62         name  = nme.isDefinedAt,
63         flags = Synthetic | Method,
64         info = MethodType(paramNames, paramTypes, defn.BooleanType)).asTerm
65     val tru = Literal(Constant(true))
66     def isDefinedAtRhs(paramRefss: List[List[Tree]]) = applyRhs match {
67       case Match(selector, cases) =>
68         assert(selector.symbol == param.symbol)
69         val paramRef = paramRefss.head.head
70         // Again, the alternative
71         //     val List(List(paramRef)) = paramRefs
72         // fails with a similar self instantiation error
73         def translateCase(cdef: CaseDef): CaseDef =
74           cpy.CaseDef(cdef)(body = tru).changeOwner(applyFn, isDefinedAtFn)
75         val defaultSym = ctx.newSymbol(isDefinedAtFn, nme.WILDCARD, Synthetic, selector.tpe.widen)
76         val defaultCase =
77           CaseDef(
78             Bind(defaultSym, Underscore(selector.tpe.widen)),
79             EmptyTree,
80             Literal(Constant(false)))
81         val annotated = Annotated(paramRef, New(ref(defn.UncheckedAnnotType)))
82         cpy.Match(applyRhs)(annotated, cases.map(translateCase) :+ defaultCase)
83       case _ =>
84         tru
85     }
86     val isDefinedAtDef = transformFollowingDeep(DefDef(isDefinedAtFn, isDefinedAtRhs(_)))
87     val anonCls = AnonClass(tpt.tpe :: Nil, List(applyFn, isDefinedAtFn), List(nme.apply, nme.isDefinedAt))
88     cpy.Block(tree)(List(applyDef, isDefinedAtDef), anonCls)
89   }
90 
91   private def checkRefinements(tpe: Type, pos: Position)(implicit ctx: Context): Type = tpe match {
92     case RefinedType(parent, name, _) =>
93       if (name.isTermName && tpe.member(name).symbol.ownersIterator.isEmpty) // if member defined in the refinement
94         ctx.error("Lambda does not define " + name, pos)
95       checkRefinements(parent, pos)
96     case _ =>
97       tpe
98   }
