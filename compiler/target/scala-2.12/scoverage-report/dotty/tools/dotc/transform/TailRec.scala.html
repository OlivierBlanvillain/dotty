17 /**
18  * A Tail Rec Transformer
19  * @author     Erik Stenman, Iulian Dragos,
20  *             ported and heavily modified for dotty by Dmitry Petrashko
21  * @version    1.1
22  *
23  *             What it does:
24  *             <p>
25  *             Finds method calls in tail-position and replaces them with jumps.
26  *             A call is in a tail-position if it is the last instruction to be
27  *             executed in the body of a method.  This is done by recursing over
28  *             the trees that may contain calls in tail-position (trees that can't
29  *             contain such calls are not transformed). However, they are not that
30  *             many.
31  *             </p>
32  *             <p>
33  *             Self-recursive calls in tail-position are replaced by jumps to a
34  *             label at the beginning of the method. As the JVM provides no way to
35  *             jump from a method to another one, non-recursive calls in
36  *             tail-position are not optimized.
37  *             </p>
38  *             <p>
39  *             A method call is self-recursive if it calls the current method and
40  *             the method is final (otherwise, it could
41  *             be a call to an overridden method in a subclass).
42  *
43  *             Recursive calls on a different instance
44  *             are optimized. Since 'this' is not a local variable it s added as
45  *             a label parameter.
46  *             </p>
47  *             <p>
48  *             This phase has been moved before pattern matching to catch more
49  *             of the common cases of tail recursive functions. This means that
50  *             more cases should be taken into account (like nested function, and
51  *             pattern cases).
52  *             </p>
53  *             <p>
54  *             If a method contains self-recursive calls, a label is added to at
55  *             the beginning of its body and the calls are replaced by jumps to
56  *             that label.
57  *             </p>
58  *             <p>
59  *
60  *             In scalac, If the method had type parameters, the call must contain same
61  *             parameters as type arguments. This is no longer case in dotc.
62  *             In scalac, this is named tailCall but it does only provide optimization for
63  *             self recursive functions, that's why it's renamed to tailrec
64  *             </p>
65  */

97   override def transformDefDef(tree: tpd.DefDef)(implicit ctx: Context): tpd.Tree = {
98     val sym = tree.symbol
99     tree match {
100       case dd@DefDef(name, tparams, vparamss0, tpt, _)
101         if (sym.isEffectivelyFinal) && !((sym is Flags.Accessor) || (dd.rhs eq EmptyTree) || (sym is Flags.Label)) =>
102         val mandatory = sym.hasAnnotation(defn.TailrecAnnot)
103         cpy.DefDef(dd)(rhs = {
104 
105           val defIsTopLevel = sym.owner.isClass
106           val origMeth = sym
107           val label = mkLabel(sym, abstractOverClass = defIsTopLevel)
108           val owner = ctx.owner.enclosingClass.asClass
109           val thisTpe = owner.thisType.widen
110 
111           var rewrote = false
112 
113           // Note: this can be split in two separate transforms(in different groups),
114           // than first one will collect info about which transformations and rewritings should be applied
115           // and second one will actually apply,
116           // now this speculatively transforms tree and throws away result in many cases
117           val rhsSemiTransformed = {
118             val transformer = new TailRecElimination(origMeth, dd.tparams, owner, thisTpe, mandatory, label, abstractOverClass = defIsTopLevel)
119             val rhs = transformer.transform(dd.rhs)
120             rewrote = transformer.rewrote
121             rhs
122           }
123 
124           if (rewrote) {
125             val dummyDefDef = cpy.DefDef(tree)(rhs = rhsSemiTransformed)
126             if (tree.symbol.owner.isClass) {
127               val labelDef = fullyParameterizedDef(label, dummyDefDef, abstractOverClass = defIsTopLevel)
128               val call = forwarder(label, dd, abstractOverClass = defIsTopLevel, liftThisType = true)
129               Block(List(labelDef), call)
130             } else { // inner method. Tail recursion does not change `this`
131               val labelDef = polyDefDef(label, trefs => vrefss => {
132                 val origMeth = tree.symbol
133                 val origTParams = tree.tparams.map(_.symbol)
134                 val origVParams = tree.vparamss.flatten map (_.symbol)
135                 new TreeTypeMap(
136                   typeMap = identity(_)
137                     .substDealias(origTParams, trefs)
138                     .subst(origVParams, vrefss.flatten.map(_.tpe)),
139                     oldOwners = origMeth :: Nil,
140                   newOwners = label :: Nil
141                 ).transform(rhsSemiTransformed)
142               })
143               val callIntoLabel = (
144                   if (dd.tparams.isEmpty) ref(label)
145                   else ref(label).appliedToTypes(dd.tparams.map(_.tpe))
146                 ).appliedToArgss(vparamss0.map(_.map(x=> ref(x.symbol))))
147               Block(List(labelDef), callIntoLabel)
148           }} else {
149             if (mandatory) ctx.error(
150               "TailRec optimisation not applicable, method not tail recursive",
151               // FIXME: want to report this error on `dd.namePos`, but
152               // because of extension method getting a weird pos, it is
153               // better to report on symbol so there's no overlap
154               sym.pos
155             )
156             dd.rhs
157           }
158         })
159       case d: DefDef if d.symbol.hasAnnotation(defn.TailrecAnnot) || methodsWithInnerAnnots.contains(d.symbol) =>
160         ctx.error(TailrecNotApplicable(sym), sym.pos)
161         d
162       case d if d.symbol.hasAnnotation(defn.TailrecAnnot) || methodsWithInnerAnnots.contains(d.symbol) =>
163         ctx.error("TailRec optimisation not applicable, not a method", sym.pos)
164         d
165       case _ => tree
166     }

180     /** Rewrite this tree to contain no tail recursive calls */
181     def transform(tree: Tree, nctx: TailContext)(implicit c: Context): Tree = {
182       if (ctx == nctx) transform(tree)
183       else {
184         val saved = ctx
185         ctx = nctx
186         try transform(tree)
187         finally this.ctx = saved

200     override def transform(tree: Tree)(implicit c: Context): Tree = {
201       /* A possibly polymorphic apply to be considered for tail call transformation. */
202       def rewriteApply(tree: Tree, sym: Symbol, required: Boolean = false): Tree = {
203         def receiverArgumentsAndSymbol(t: Tree, accArgs: List[List[Tree]] = Nil, accT: List[Tree] = Nil):
204             (Tree, Tree, List[List[Tree]], List[Tree], Symbol) = t match {
205           case TypeApply(fun, targs) if fun.symbol eq t.symbol => receiverArgumentsAndSymbol(fun, accArgs, targs)
206           case Apply(fn, args) if fn.symbol == t.symbol => receiverArgumentsAndSymbol(fn, args :: accArgs, accT)
207           case Select(qual, _) => (qual, t, accArgs, accT, t.symbol)
208           case x: This => (x, x, accArgs, accT, x.symbol)
209           case x: Ident if x.symbol eq method => (EmptyTree, x, accArgs, accT, x.symbol)
210           case x => (x, x, accArgs, accT, x.symbol)
211         }
212 
213         val (prefix, call, arguments, typeArguments, symbol) = receiverArgumentsAndSymbol(tree)
214         val hasConformingTargs = (typeArguments zip methTparams).forall{x => x._1.tpe <:< x._2.tpe}
215 
216         val targs = typeArguments.map(noTailTransform)
217         val argumentss = arguments.map(noTailTransforms)
218 
219         val isRecursiveCall = (method eq sym)
220         val recvWiden = prefix.tpe.widenDealias
221 
222         def continue = {
223           val method = noTailTransform(call)
224           val methodWithTargs = if (targs.nonEmpty) TypeApply(method, targs) else method
225           if (methodWithTargs.tpe.widen.isParameterless) methodWithTargs
226           else argumentss.foldLeft(methodWithTargs) {
227             // case (method, args) => Apply(method, args) // Dotty deviation no auto-detupling yet. Interesting that one can do it in Scala2!
228             (method, args) => Apply(method, args)
229           }
230         }
231         def fail(reason: String) = {
232           if (isMandatory || required) c.error(s"Cannot rewrite recursive call: $reason", tree.pos)
233           else c.debuglog("Cannot rewrite recursive call at: " + tree.pos + " because: " + reason)
234           continue
235         }
236 
237         if (isRecursiveCall) {
238           if (ctx.tailPos) {
239             val receiverIsSame =
240               recvWiden <:< enclosingClass.appliedRef &&
241               (sym.isEffectivelyFinal || enclosingClass.appliedRef <:< recvWiden)
242             val receiverIsThis = prefix.tpe =:= thisType || prefix.tpe.widen =:= thisType
243 
244             def rewriteTailCall(recv: Tree): Tree = {
245               c.debuglog("Rewriting tail recursive call:  " + tree.pos)
246               rewrote = true
247               val receiver = noTailTransform(recv)
248 
249               val callTargs: List[tpd.Tree] =
250                 if (abstractOverClass) {
251                   val classTypeArgs = recv.tpe.baseType(enclosingClass).argInfos
252                   targs ::: classTypeArgs.map(x => ref(x.typeSymbol))
253                 } else targs
254 
255               val method = if (callTargs.nonEmpty) TypeApply(Ident(label.termRef), callTargs) else Ident(label.termRef)
256               val thisPassed =
257                 if (this.method.owner.isClass)
258                   method.appliedTo(receiver.ensureConforms(method.tpe.widen.firstParamTypes.head))
259                 else method
260 
261               val res =
262                 if (thisPassed.tpe.widen.isParameterless) thisPassed
263                 else argumentss.foldLeft(thisPassed) {
264                   (met, ar) => Apply(met, ar) // Dotty deviation no auto-detupling yet.
265                 }
266               res
267             }
268 
269             if (!hasConformingTargs) fail("it changes type arguments on a polymorphic recursive call")
270             else {
271               val recv = noTailTransform(prefix)
272               if (recv eq EmptyTree) rewriteTailCall(This(enclosingClass.asClass))
273               else if (receiverIsSame || receiverIsThis) rewriteTailCall(recv)
274               else fail("it changes type of 'this' on a polymorphic recursive call")
275             }
276           }
277           else fail(defaultReason)
278         } else {
279           val receiverIsSuper = (method.name eq sym) && enclosingClass.appliedRef.widen <:< recvWiden
280 
281           if (receiverIsSuper) fail("it contains a recursive call targeting a supertype")
282           else continue
283         }
284       }
285 
286       def rewriteTry(tree: Try): Try = {
287         if (tree.finalizer eq EmptyTree) {
288           // SI-1672 Catches are in tail position when there is no finalizer
289           tpd.cpy.Try(tree)(
290             noTailTransform(tree.expr),
291             transformSub(tree.cases),
292             EmptyTree
293           )
294         }
295         else {
296           tpd.cpy.Try(tree)(
297             noTailTransform(tree.expr),
298             noTailTransforms(tree.cases),
299             noTailTransform(tree.finalizer)
300           )
301         }
302       }
303 
304       val res: Tree = tree match {
305 
306         case Ident(qual) =>
307           val sym = tree.symbol
308           if (sym == method && ctx.tailPos) rewriteApply(tree, sym)
309           else tree
310 
311         case tree: Select =>
312           val sym = tree.symbol
313           if (sym == method && ctx.tailPos) rewriteApply(tree, sym)
314           else tpd.cpy.Select(tree)(noTailTransform(tree.qualifier), tree.name)
315 
316         case Apply(fun, args) =>
317           val meth = fun.symbol
318           if (meth == defn.Boolean_|| || meth == defn.Boolean_&&)
319             tpd.cpy.Apply(tree)(fun, transform(args))
320           else
321             rewriteApply(tree, meth)
322 
323         case TypeApply(fun, targs) =>
324           val meth = fun.symbol
325           rewriteApply(tree, meth)
326 
327         case tree@Block(stats, expr) =>
328           tpd.cpy.Block(tree)(
329             noTailTransforms(stats),
330             transform(expr)
331           )
332         case tree @ Typed(t: Apply, tpt) if tpt.tpe.hasAnnotation(defn.TailrecAnnot) =>
333           tpd.Typed(rewriteApply(t, t.fun.symbol, required = true), tpt)
334         case tree@If(cond, thenp, elsep) =>
335           tpd.cpy.If(tree)(
336             noTailTransform(cond),
337             transform(thenp),
338             transform(elsep)
339           )
340 
341         case tree@CaseDef(_, _, body) =>
342           cpy.CaseDef(tree)(body = transform(body))
343 
344         case tree@Match(selector, cases) =>
345           tpd.cpy.Match(tree)(
346             noTailTransform(selector),
347             transformSub(cases)
348           )
349 
350         case tree: Try =>
351           rewriteTry(tree)
352 
353         case Alternative(_) | Bind(_, _) =>
354           assert(false, "We should never have gotten inside a pattern")
355           tree
356 
357         case t @ DefDef(_, _, _, _, _) =>
358           t // todo: could improve to handle DefDef's with a label flag calls to which are in tail position
359 
360         case ValDef(_, _, _) | EmptyTree | Super(_, _) | This(_) |
361              Literal(_) | TypeTree() | TypeDef(_, _) =>
362           tree
363 
364         case Return(expr, from) =>
365           tpd.cpy.Return(tree)(noTailTransform(expr), from)
366 
367         case _ =>
368           super.transform(tree)
369       }
370 
371       res
372     }
373   }
