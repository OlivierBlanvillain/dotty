27 /** A transformer that check that requirements of Static fields\methods are implemented:
28   *  1. Only objects can have members annotated with `@static`
29   *  2. The fields annotated with `@static` should preceed any non-`@static` fields.
30   *     This ensures that we do not introduce surprises for users in initialization order.
31   *  3. If a member `foo` of an `object C` is annotated with `@static`,
32   *     the companion class `C` is not allowed to define term members with name `foo`.
33   *  4. If a member `foo` of an `object C` is annotated with `@static`, the companion class `C`
34   *     is not allowed to inherit classes that define a term member with name `foo`.
35   *  5. Only `@static` methods and vals are supported in companions of traits.
36   *     Java8 supports those, but not vars, and JavaScript does not have interfaces at all.
37   *  6. `@static` Lazy vals are currently unsupported.
38   */
39 class CheckStatic extends MiniPhase {
40   import ast.tpd._
41 
42   override def phaseName = "checkStatic"
43 
44   override def transformTemplate(tree: tpd.Template)(implicit ctx: Context): tpd.Tree = {
45     val defns = tree.body.collect{case t: ValOrDefDef => t}
46     var hadNonStaticField = false
47     for(defn <- defns) {
48       if (defn.symbol.hasAnnotation(ctx.definitions.ScalaStaticAnnot)) {
49         if(!ctx.owner.is(Module)) {
50           ctx.error(StaticFieldsOnlyAllowedInObjects(defn.symbol), defn.pos)
51         }
52 
53         if (defn.isInstanceOf[ValDef] && hadNonStaticField) {
54           ctx.error("@static fields should preceed non-static ones", defn.pos)
55         }
56 
57         val companion = ctx.owner.companionClass
58         def clashes = companion.asClass.membersNamed(defn.name)
59 
60         if (!companion.exists) {
61           ctx.error(MissingCompanionForStatic(defn.symbol), defn.pos)
62         } else if (clashes.exists) {
63           ctx.error("companion classes cannot define members with same name as @static member", defn.pos)
64          } else if (defn.symbol.is(Flags.Mutable) && companion.is(Flags.Trait)) {
65           ctx.error("Companions of traits cannot define mutable @static fields", defn.pos)
66         } else if (defn.symbol.is(Flags.Lazy)) {
67           ctx.error("Lazy @static fields are not supported", defn.pos)
68         } else if (defn.symbol.allOverriddenSymbols.nonEmpty) {
69           ctx.error("@static members cannot override or implement non-static ones", defn.pos)
70         }
71       } else hadNonStaticField = hadNonStaticField || defn.isInstanceOf[ValDef]
72 
73     }
74     tree
75   }
76 
77   override def transformSelect(tree: tpd.Select)(implicit ctx: Context): tpd.Tree = {
78     if (tree.symbol.hasAnnotation(defn.ScalaStaticAnnot)) {
79       val symbolWhitelist = tree.symbol.ownersIterator.flatMap(x => if (x.is(Flags.Module)) List(x, x.companionModule) else List(x)).toSet
80       def isSafeQual(t: Tree): Boolean = { // follow the desugared paths created by typer
81         t match {
82           case t: This => true
83           case t: Select => isSafeQual(t.qualifier) && symbolWhitelist.contains(t.symbol)
84           case t: Ident => symbolWhitelist.contains(t.symbol)
85           case t: Block => t.stats.forall(tpd.isPureExpr) && isSafeQual(t.expr)
86         }
87       }
88       if (isSafeQual(tree.qualifier))
89         ref(tree.symbol)
90       else tree
91     } else tree
92   }
