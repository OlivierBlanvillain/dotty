23     if (needsRefresh(ref.symbol)) ref.copySymDenotation(name = refreshedName(ref.symbol))
24     else ref

26   /** If the name of the symbol with a unique name needs to be refreshed
27    *    - if it is a lifted class
28    *    - if it is a lifted method
29    */
30   private def needsRefresh(sym: Symbol)(implicit ctx: Context): Boolean =
31     (sym.isClass || sym.is(Private | Method | JavaStatic)) && sym.name.is(UniqueName)
32 
33   /** Refreshes the number of the name based on the full name of the symbol */
34   private def refreshedName(sym: Symbol)(implicit ctx: Context): Name = {
35     def rewriteUnique: PartialFunction[Name, Name] = {
36       case name: DerivedName if name.info.kind == UniqueName =>
37         val fullName = (sym.owner.fullName.toString + name.underlying).toTermName
38         val freshName = UniqueName.fresh(fullName)
39         val info = freshName.asInstanceOf[DerivedName].info
40         DerivedName(name.underlying.rewrite(rewriteUnique), info)
41       case DerivedName(underlying, info: QualifiedInfo) =>
42         underlying.rewrite(rewriteUnique).derived(info)
43     }
44 
45     sym.name.rewrite(rewriteUnique)
46   }
