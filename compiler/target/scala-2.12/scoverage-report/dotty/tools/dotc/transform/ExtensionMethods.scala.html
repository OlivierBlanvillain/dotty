24 /**
25  * Perform Step 1 in the inline classes SIP: Creates extension methods for all
26  * methods in a value class, except parameter or super accessors, or constructors.
27  *
28  * Additionally, for a value class V, let U be the underlying type after erasure. We add
29  * to the companion module of V two cast methods:
30  *   def u2evt$(x0: U): ErasedValueType(V, U)
31  *   def evt2u$(x0: ErasedValueType(V, U)): U
32  * The casts are used in [[Erasure]] to make it typecheck, they are then removed
33  * in [[ElimErasedValueType]].
34  * This is different from the implementation of value classes in Scala 2
35  * (see SIP-15) which uses `asInstanceOf` which does not typecheck.
36  *
37  * Finally, if the constructor of a value class is private pr protected
38  * it is widened to public.
39  */

54   override def transform(ref: SingleDenotation)(implicit ctx: Context): SingleDenotation = ref match {
55     case moduleClassSym: ClassDenotation if moduleClassSym is ModuleClass =>
56       moduleClassSym.linkedClass match {
57         case valueClass: ClassSymbol if isDerivedValueClass(valueClass) =>
58           val cinfo = moduleClassSym.classInfo
59           val decls1 = cinfo.decls.cloneScope
60           val moduleSym = moduleClassSym.symbol.asClass
61 
62           var newSuperClass: Type = null
63 
64           ctx.atPhase(thisPhase.next) { implicit ctx =>
65             // In Scala 2, extension methods are added before pickling so we should
66             // not generate them again.
67             if (!(valueClass is Scala2x)) ctx.atPhase(thisPhase) { implicit ctx =>
68               for (decl <- valueClass.classInfo.decls) {
69                 if (isMethodWithExtension(decl)) {
70                   val meth = createExtensionMethod(decl, moduleClassSym.symbol)

72                   // Workaround #1895: force denotation of `meth` to be
73                   // at phase where `meth` is entered into the decls of a class

82               MethodType(List(nme.x_0), List(underlying), evt))
83             val evt2uSym = ctx.newSymbol(moduleSym, nme.EVT2U, Synthetic | Method,
84               MethodType(List(nme.x_0), List(evt), underlying))

86             val defn = ctx.definitions
87 
88             val underlyingCls = underlying.classSymbol

90               if (underlyingCls.isNumericValueClass || underlyingCls == defn.BooleanClass) underlyingCls.name
91               else nme.Object
92 
93             val syp = ctx.requiredClass(s"dotty.runtime.vc.VC${underlyingClsName}Companion").asClass
94 
95             newSuperClass = tpd.ref(syp).select(nme.CONSTRUCTOR).appliedToType(valueClass.typeRef).tpe.resultType

97             decls1.enter(u2evtSym)
98             decls1.enter(evt2uSym)
99           }
100 
101           // Add the extension methods, the cast methods u2evt$ and evt2u$, and a VC*Companion superclass
102           moduleClassSym.copySymDenotation(info =
103             cinfo.derivedClassInfo(
104               // FIXME: use of VC*Companion superclasses is disabled until the conflicts with SyntheticMethods are solved.
105               //classParents = List(newSuperClass)
106               decls = decls1))
107         case _ =>
108           moduleClassSym
109       }
110     case ref: SymDenotation =>
111       if (isMethodWithExtension(ref) && ref.hasAnnotation(defn.TailrecAnnot)) {
112         val ref1 = ref.copySymDenotation()
113         ref1.removeAnnotation(defn.TailrecAnnot)
114         ref1
115       }
116       else if (ref.isConstructor && isDerivedValueClass(ref.owner) && ref.is(AccessFlags)) {
117         val ref1 = ref.copySymDenotation()
118         ref1.resetFlag(AccessFlags)
119         ref1
120       }
121       else ref
122     case _ =>
123       ref
124   }

163   override def transformDefDef(tree: tpd.DefDef)(implicit ctx: Context): tpd.Tree = {
164     if (isMethodWithExtension(tree.symbol)) {
165       val origMeth = tree.symbol
166       val origClass = ctx.owner.asClass

168       assert(staticClass.exists, s"$origClass lacks companion, ${origClass.owner.definedPeriodsString} ${origClass.owner.info.decls} ${origClass.owner.info.decls}")

171       val store: ListBuffer[Tree] = extensionDefs.get(staticClass) match {
172         case Some(x) => x
173         case None =>
174           val newC = new ListBuffer[Tree]()
175           extensionDefs(staticClass) = newC
176           newC
177       }
178       store += fullyParameterizedDef(extensionMeth, tree)
179       cpy.DefDef(tree)(rhs = forwarder(extensionMeth, tree))

184 object ExtensionMethods {
185   /** Generate stream of possible names for the extension version of given instance method `imeth`.
186    *  If the method is not overloaded, this stream consists of just "imeth$extension".
187    *  If the method is overloaded, the stream has as first element "imeth$extenionX", where X is the
188    *  index of imeth in the sequence of overloaded alternatives with the same name. This choice will
189    *  always be picked as the name of the generated extension method.
190    *  After this first choice, all other possible indices in the range of 0 until the number
191    *  of overloaded alternatives are returned. The secondary choices are used to find a matching method
192    *  in `extensionMethod` if the first name has the wrong type. We thereby gain a level of insensitivity
193    *  of how overloaded types are ordered between phases and picklings.
194    */
195   private def extensionNames(imeth: Symbol)(implicit ctx: Context): Stream[Name] = {
196     val decl = imeth.owner.info.decl(imeth.name)
197 
198     /** No longer needed for Dotty, as we are more disciplined with scopes now.
199     // Bridge generation is done at phase `erasure`, but new scopes are only generated
200     // for the phase after that. So bridges are visible in earlier phases.
201     //
202     // `info.member(imeth.name)` filters these out, but we need to use `decl`
203     // to restrict ourselves to members defined in the current class, so we
204     // must do the filtering here.
205     val declTypeNoBridge = decl.filter(sym => !sym.isBridge).tpe
206     */
207     decl match {
208       case decl: MultiDenotation =>
209         val alts = decl.alternatives
210         val index = alts indexOf imeth.denot
211         assert(index >= 0, alts + " does not contain " + imeth)
212         def altName(index: Int) = UniqueExtMethName(imeth.name.asTermName, index)
213         altName(index) #:: ((0 until alts.length).toStream filter (index != _) map altName)
214       case decl =>
215         assert(decl.exists, imeth.name + " not found in " + imeth.owner + "'s decls: " + imeth.owner.info.decls)
216         Stream(ExtMethName(imeth.name.asTermName))
217     }
218   }
219 
220   /** Return the extension method that corresponds to given instance method `meth`. */
221   def extensionMethod(imeth: Symbol)(implicit ctx: Context): TermSymbol =
222     ctx.atPhase(ctx.extensionMethodsPhase.next) { implicit ctx =>
223       // FIXME use toStatic instead?
224       val companionInfo = imeth.owner.companionModule.info
225       val candidates = extensionNames(imeth) map (companionInfo.decl(_).symbol) filter (_.exists)
226       val matching = candidates filter (c => FullParameterization.memberSignature(c.info) == imeth.signature)
227       assert(matching.nonEmpty,
228        i"""no extension method found for:
229           |
230           |  $imeth:${imeth.info.show} with signature ${imeth.signature}
231           |
232           | Candidates:
233           |
234           | ${candidates.map(c => c.name + ":" + c.info.show).mkString("\n")}
235           |
236           | Candidates (signatures normalized):
237           |
238           | ${candidates.map(c => c.name + ":" + c.info.signature + ":" + FullParameterization.memberSignature(c.info)).mkString("\n")}
239           |
240           | Eligible Names: ${extensionNames(imeth).mkString(",")}""")
241       matching.head.asTerm
242     }
