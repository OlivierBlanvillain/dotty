31   override def transformMatch(tree: Match)(implicit ctx: Context): Tree = {
32     val translated = new Translator(tree.tpe, this).translateMatch(tree)
33 
34     // check exhaustivity and unreachability
35     val engine = new patmat.SpaceEngine
36 
37     if (engine.checkable(tree)) {

60    *  Its general structure is a pipeline:
61    *

215     private def patternPlan(scrutinee: Symbol, tree: Tree, onSuccess: Plan, onFailure: Plan): Plan = {

217       /** Plan for matching `selectors` against argument patterns `args` */
218       def matchArgsPlan(selectors: List[Tree], args: List[Tree], onSuccess: Plan): Plan = {
219         /* For a case with arguments that have some test on them such as
220          * ```
221          * case Foo(1, 2) => someCode
222          * ```
223          * all arguments values are extracted before the checks are performed. This shape is expected by `emit`
224          * to avoid generating deep trees.
225          * ```
226          * val x1: Foo = ...
227          * val x2: Int = x1._1
228          * val x3: Int = x1._2
229          * if (x2 == 1) {
230          *   if (x3 == 2) someCode
231          *   else label$1()
232          * } else label$1()
233          * ```
234          */
235         def matchArgsSelectorsPlan(selectors: List[Tree], syms: List[Symbol]): Plan =
236           selectors match {
237             case selector :: selectors1 => letAbstract(selector)(sym => matchArgsSelectorsPlan(selectors1, sym :: syms))
238             case Nil => matchArgsPatternPlan(args, syms.reverse)
239           }
240         def matchArgsPatternPlan(args: List[Tree], syms: List[Symbol]): Plan =
241           args match {
242             case arg :: args1 =>
243               val sym :: syms1 = syms
244               patternPlan(sym, arg, matchArgsPatternPlan(args1, syms1), onFailure)
245             case Nil =>
246               assert(syms.isEmpty)
247               onSuccess
248           }
249         matchArgsSelectorsPlan(selectors, Nil)
250       }

252       /** Plan for matching the sequence in `seqSym` against sequence elements `args`.
253        *  If `exact` is true, the sequence is not permitted to have any elements following `args`.
254        */
255       def matchElemsPlan(seqSym: Symbol, args: List[Tree], exact: Boolean, onSuccess: Plan) = {
256         val selectors = args.indices.toList.map(idx =>
257           ref(seqSym).select(nme.apply).appliedTo(Literal(Constant(idx))))
258         TestPlan(LengthTest(args.length, exact), seqSym, seqSym.pos,
259           matchArgsPlan(selectors, args, onSuccess), onFailure)
260       }

262       /** Plan for matching the sequence in `getResult` against sequence elements
263        *  and a possible last varargs argument `args`.
264        */
265       def unapplySeqPlan(getResult: Symbol, args: List[Tree]): Plan = args.lastOption match {
266         case Some(VarArgPattern(arg)) =>
267           val matchRemaining =
268             if (args.length == 1)
269               patternPlan(getResult, arg, onSuccess, onFailure)
270             else {
271               val dropped = ref(getResult)
272                 .select(defn.Seq_drop.matchingMember(getResult.info))
273                 .appliedTo(Literal(Constant(args.length - 1)))
274               letAbstract(dropped) { droppedResult =>
275                 patternPlan(droppedResult, arg, onSuccess, onFailure)
276               }
277             }
278           matchElemsPlan(getResult, args.init, exact = false, matchRemaining)
279         case _ =>
280           matchElemsPlan(getResult, args, exact = true, onSuccess)
281       }

283       /** Plan for matching the result of an unapply against argument patterns `args` */
284       def unapplyPlan(unapp: Tree, args: List[Tree]): Plan = {
285         def caseClass = unapp.symbol.owner.linkedClass
286         lazy val caseAccessors = caseClass.caseAccessors.filter(_.is(Method))
287 
288         def isSyntheticScala2Unapply(sym: Symbol) =
289           sym.is(SyntheticCase) && sym.owner.is(Scala2x)
290 
291         if (isSyntheticScala2Unapply(unapp.symbol) && caseAccessors.length == args.length)
292           matchArgsPlan(caseAccessors.map(ref(scrutinee).select(_)), args, onSuccess)
293         else if (unapp.tpe.isRef(defn.BooleanClass))
294           TestPlan(GuardTest, unapp, unapp.pos, onSuccess, onFailure)
295         else {
296           letAbstract(unapp) { unappResult =>
297             val isUnapplySeq = unapp.symbol.name == nme.unapplySeq
298             if (isProductMatch(unapp.tpe.widen, args.length) && !isUnapplySeq) {
299               val selectors = productSelectors(unapp.tpe).take(args.length)
300                 .map(ref(unappResult).select(_))
301               matchArgsPlan(selectors, args, onSuccess)
302             }
303             else {
304               assert(isGetMatch(unapp.tpe))
305               val argsPlan = {
306                 val get = ref(unappResult).select(nme.get, _.info.isParameterless)
307                 if (isUnapplySeq)
308                   letAbstract(get)(unapplySeqPlan(_, args))
309                 else
310                   letAbstract(get) { getResult =>
311                     val selectors =
312                       if (args.tail.isEmpty) ref(getResult) :: Nil
313                       else productSelectors(get.tpe).map(ref(getResult).select(_))
314                     matchArgsPlan(selectors, args, onSuccess)
315                   }
316               }
317               TestPlan(NonEmptyTest, unappResult, unapp.pos, argsPlan, onFailure)
318             }
319           }
320         }
321       }
322 
323       // begin patternPlan
324       swapBind(tree) match {
325         case Typed(pat, tpt) =>
326           TestPlan(TypeTest(tpt), scrutinee, tree.pos,
327             letAbstract(ref(scrutinee).asInstance(tpt.tpe)) { casted =>
328               nonNull += casted
329               patternPlan(casted, pat, onSuccess, onFailure)
330             },
331             onFailure)
332         case UnApply(extractor, implicits, args) =>
333           val mt @ MethodType(_) = extractor.tpe.widen
334           var unapp = extractor.appliedTo(ref(scrutinee).ensureConforms(mt.paramInfos.head))
335           if (implicits.nonEmpty) unapp = unapp.appliedToArgs(implicits)
336           val unappPlan = unapplyPlan(unapp, args)
337           if (scrutinee.info.isNotNull || nonNull(scrutinee)) unappPlan
338           else TestPlan(NonNullTest, scrutinee, tree.pos, unappPlan, onFailure)
339         case Bind(name, body) =>
340           if (name == nme.WILDCARD) patternPlan(scrutinee, body, onSuccess, onFailure)
341           else {
342             // The type of `name` may refer to val in `body`, therefore should come after `body`
343             val bound = tree.symbol.asTerm
344             initializer(bound) = ref(scrutinee)
345             patternPlan(scrutinee, body, LetPlan(bound, onSuccess), onFailure)
346           }
347         case Alternative(alts) =>
348           labelAbstract(onSuccess) { ons =>
349             (alts :\ onFailure) { (alt, onf) =>
350               labelAbstract(onf) { onf1 =>
351                 patternPlan(scrutinee, alt, ons, onf1)
352               }
353             }
354           }
355         case WildcardPattern() =>
356           onSuccess
357         case SeqLiteral(pats, _) =>
358           matchElemsPlan(scrutinee, pats, exact = true, onSuccess)
359         case _ =>
360           TestPlan(EqualTest(tree), scrutinee, tree.pos, onSuccess, onFailure)
361       }
362     }

510     /** Eliminate tests that are redundant (known to be true or false).
511      *  Two parts:
512      *
513      *   - If we know at some point that a test is true or false skip it and continue
514      *     diretcly with the test's onSuccess or onFailure continuation.
515      *   - If a label of a call points to a test that is known to be true or false
516      *     at the point of call, let the label point instead to the test's onSuccess
517      *     or onFailure continuation.
518      *
519      *  We use some tricks to identify a let pointing to an unapply and the
520      *  NonEmptyTest that follows it as a single `UnappTest` test.
521      */
522     def elimRedundantTests(plan: Plan): Plan = {
523       type SeenTests = Map[TestPlan, Boolean] // Map from tests to their outcomes
524 
525       def isUnapply(sym: Symbol) = sym.name == nme.unapply || sym.name == nme.unapplySeq
526 
527       /** A locally used test value that represents combos of
528        *
529        *   let x = X.unapply(...) in if !x.isEmpty then ... else ...
530        */
531       case object UnappTest extends Test
532 
533       /** If `plan` is the NonEmptyTest part of an unapply, the corresponding UnappTest
534        *  otherwise the original plan
535        */
536       def normalize(plan: TestPlan): TestPlan = plan.scrutinee match {
537         case id: Ident
538         if plan.test == NonEmptyTest &&
539            isPatmatGenerated(id.symbol) &&
540            isUnapply(initializer(id.symbol).symbol) =>
541           TestPlan(UnappTest, initializer(id.symbol), plan.pos, plan.onSuccess, plan.onFailure)
542         case _ =>
543           plan
544       }
545 
546       /** Extractor for Let/NonEmptyTest combos that represent unapplies */
547       object UnappTestPlan {
548         def unapply(plan: Plan): Option[TestPlan] = plan match {
549           case LetPlan(sym, body: TestPlan) =>
550             val RHS = initializer(sym)
551             normalize(body) match {
552               case normPlan @ TestPlan(UnappTest, RHS, _, _, _) => Some(normPlan)
553               case _ => None
554             }
555           case _ => None
556         }
557       }
558 
559       def intersect(tests1: SeenTests, tests2: SeenTests) =
560         tests1.filter { case(test, outcome) => tests2.get(test) == Some(outcome) }
561 
562       /** The tests with known outcomes valid at entry to label */
563       val seenAtLabel = newMutableSymbolMap[SeenTests]
564 
565       class ElimRedundant(seenTests: SeenTests) extends PlanTransform {
566         override def apply(plan: TestPlan): Plan = {
567           val normPlan = normalize(plan)
568           seenTests.get(normPlan) match {
569             case Some(outcome) =>
570               apply(if (outcome) plan.onSuccess else plan.onFailure)
571             case None =>
572               plan.onSuccess = new ElimRedundant(seenTests + (normPlan -> true))(plan.onSuccess)
573               plan.onFailure = new ElimRedundant(seenTests + (normPlan -> false))(plan.onFailure)
574               plan
575           }
576         }
577         override def apply(plan: LabelledPlan): Plan = {
578           plan.body = apply(plan.body)
579           for (seenTests1 <- seenAtLabel.get(plan.sym))
580             labelled(plan.sym) = new ElimRedundant(seenTests1)(labelled(plan.sym))
581           plan
582         }
583         override def apply(plan: CallPlan): Plan = {
584           val label = plan.label
585           def redirect(target: Plan): Plan = {
586             def forward(tst: TestPlan) = seenTests.get(tst) match {
587               case Some(true) => redirect(tst.onSuccess)
588               case Some(false) => redirect(tst.onFailure)
589               case none => target
590             }
591             target match {
592               case tst: TestPlan => forward(tst)
593               case UnappTestPlan(tst) => forward(tst)
594               case _ => target
595             }
596           }
597           redirect(labelled(label)) match {
598             case target: CallPlan =>
599               apply(target)
600             case _ =>
601               seenAtLabel(label) = seenAtLabel.get(label) match {
602                 case Some(seenTests1) => intersect(seenTests1, seenTests)
603                 case none => seenTests
604               }
605               plan
606           }
607         }
608       }
609       new ElimRedundant(Map())(plan)
610     }

860     /** Translate plan to tree */
861     private def emit(plan: Plan): Tree = {
862       if (selfCheck) {

870             Match(plan.scrutinee, emitSwitchCases(switchCases))

873              *  This optimization targets calls to label defs for case failure jumps to next case.
874              *
875              *  Plan for
876              *  ```
877              *  val x1: Int = ...
878              *  val x2: Int = ...
879              *  if (x1 == y1) {

881              *    else label$1()
882              *  } else label$1()
883              *  ```
884              *  is emitted as
885              *  ```
886              *  val x1: Int = ...
887              *  val x2: Int = ...
888              *  if (x1 == y1 && x2 == y2) someCode
889              *  else label$1()
890              *  ```
891              */

893               val plan = plans.head
894               plan.onSuccess match {
895                 case plan2: TestPlan if plan.onFailure == plan2.onFailure =>
896                   emitWithMashedConditions(plan2 :: plans)
897                 case _ =>
898                   def emitCondWithPos(plan: TestPlan) = emitCondition(plan).withPos(plan.pos)
899                   val conditions =
900                     plans.foldRight[Tree](EmptyTree) { (otherPlan, acc) =>
901                       if (acc.isEmpty) emitCondWithPos(otherPlan)
902                       else acc.select(nme.ZAND).appliedTo(emitCondWithPos(otherPlan))
903                     }
904                   If(conditions, emit(plan.onSuccess), emit(plan.onFailure))
905               }
906 
907             }
908             emitWithMashedConditions(plan :: Nil)
909           }
910         case LetPlan(sym, body) =>
911           seq(ValDef(sym, initializer(sym).ensureConforms(sym.info)) :: Nil, emit(body))
912         case LabelledPlan(label, body, params) =>
913           label.info = MethodType.fromSymbols(params, resultType)
914           val labelDef = DefDef(label, Nil, params :: Nil, resultType, emit(labelled(label)))
915           seq(labelDef :: Nil, emit(body))
916         case CodePlan(tree) =>
917           tree
918         case CallPlan(label, args) =>
919           ref(label).appliedToArgs(args.map { case (_, actual) => ref(actual) })
920       }
