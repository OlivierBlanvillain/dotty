27   def clear(): Unit = {
28     defined.clear()

34   def visitor(implicit ctx: Context): Tree => Unit = {
35     case t: ValDef if t.symbol.is(Mutable, Lazy) && !t.symbol.is(Method) && !t.symbol.owner.isClass =>
36       if (isPureExpr(t.rhs))
37         defined(t.symbol) = t
38 
39     case t: RefTree if t.symbol.exists && !t.symbol.is(Method) && !t.symbol.owner.isClass =>
40       if (!firstWrite.contains(t.symbol)) firstRead(t.symbol) = t
41 
42     case t @ Assign(l, expr) if !l.symbol.is(Method) && !l.symbol.owner.isClass =>
43       if (!firstRead.contains(l.symbol)) {
44         if (firstWrite.contains(l.symbol)) {
45           if (!secondWrite.contains(l.symbol))
46             secondWrite(l.symbol) = t
47         } else if (!expr.existsSubTree(x => x match {
48           case tree: RefTree if x.symbol == l.symbol => firstRead(l.symbol) = tree; true
49           case _ => false
50         })) {
51           firstWrite(l.symbol) = t
52         }
53       }
54     case _ =>
55   }
56 
57   def transformer(implicit ctx: Context): Tree => Tree = {
58     case t: Block => // Drop non-side-effecting stats
59       val valdefs = t.stats.collect {
60         case t: ValDef if defined.contains(t.symbol) => t
61       }
62 
63       val assigns = t.stats.filter {
64         case t @ Assign(lhs, r) =>
65           firstWrite.contains(lhs.symbol) && !secondWrite.contains(lhs.symbol)
66         case _ => false
67       }
68 
69       val pairs = valdefs.flatMap(x => assigns.find(y => y.asInstanceOf[Assign].lhs.symbol == x.symbol) match {
70         case Some(y: Assign) => List((x, y))
71         case _ => Nil
72       })
73 
74       val valsToDrop = pairs.map(_._1).toSet
75       val assignsToReplace: Map[Assign, ValDef] = pairs.map(_.swap).toMap
76 
77       val newStats = t.stats.mapConserve {
78         case x: ValDef if valsToDrop.contains(x) => EmptyTree
79         case t: Assign => assignsToReplace.get(t) match {
80           case Some(vd) =>
81             val newD = vd.symbol.asSymDenotation.copySymDenotation(initFlags = vd.symbol.flags.&~(Mutable))
82             newD.installAfter(simplifyPhase)
83             ValDef(vd.symbol.asTerm, t.rhs)
84           case None => t
85         }
86         case x => x
87       }
88 
89       if (newStats eq t.stats) t
90       else cpy.Block(t)(newStats, t.expr)
91     case tree => tree
92   }
