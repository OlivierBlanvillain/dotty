15 /** Compiles the cases that can not be handled by primitive catch cases as a common pattern match.
16  *
17  *  The following code:
18  *    ```
19  *    try { <code> }
20  *    catch {
21  *      <tryCases> // Cases that can be handled by catch
22  *      <patternMatchCases> // Cases starting with first one that can't be handled by catch
23  *    }
24  *    ```
25  *  will become:
26  *    ```
27  *    try { <code> }
28  *    catch {
29  *      <tryCases>
30  *      case e => e match {
31  *        <patternMatchCases>
32  *      }
33  *    }
34  *    ```
35  *
36  *  Cases that are not supported include:
37  *   - Applies and unapplies
38  *   - Idents
39  *   - Alternatives
40  *   - `case _: T =>` where `T` is not `Throwable`
41  *
42  */
43 class TryCatchPatterns extends MiniPhase {
44   import dotty.tools.dotc.ast.tpd._
45 
46   def phaseName: String = "tryCatchPatterns"
47 
48   override def runsAfter = Set(classOf[ElimRepeated])
49 
50   override def checkPostCondition(tree: Tree)(implicit ctx: Context): Unit = tree match {
51     case Try(_, cases, _) =>
52       cases.foreach {
53         case CaseDef(Typed(_, _), guard, _) => assert(guard.isEmpty, "Try case should not contain a guard.")
54         case CaseDef(Bind(_, _), guard, _) => assert(guard.isEmpty, "Try case should not contain a guard.")
55         case c =>
56           assert(isDefaultCase(c), "Pattern in Try should be Bind, Typed or default case.")
57       }
58     case _ =>
59   }
60 
61   override def transformTry(tree: Try)(implicit ctx: Context): Tree = {
62     val (tryCases, patternMatchCases) = tree.cases.span(isCatchCase)
63     val fallbackCase = mkFallbackPatterMatchCase(patternMatchCases, tree.pos)
64     cpy.Try(tree)(cases = tryCases ++ fallbackCase)
65   }
66 
67   /** Is this pattern node a catch-all or type-test pattern? */
68   private def isCatchCase(cdef: CaseDef)(implicit ctx: Context): Boolean = cdef match {
69     case CaseDef(Typed(Ident(nme.WILDCARD), tpt), EmptyTree, _)          => isSimpleThrowable(tpt.tpe)
70     case CaseDef(Bind(_, Typed(Ident(nme.WILDCARD), tpt)), EmptyTree, _) => isSimpleThrowable(tpt.tpe)
71     case _                                                               => isDefaultCase(cdef)
72   }
73 
74   private def isSimpleThrowable(tp: Type)(implicit ctx: Context): Boolean = tp.stripAnnots match {
75     case tp @ TypeRef(pre, _) =>
76       (pre == NoPrefix || pre.widen.typeSymbol.isStatic) && // Does not require outer class check
77       !tp.symbol.is(Flags.Trait) && // Traits not supported by JVM
78       tp.derivesFrom(defn.ThrowableClass)
79     case tp: AppliedType =>
80       isSimpleThrowable(tp.tycon)
81     case _ =>
82       false
83   }
84 
85   private def mkFallbackPatterMatchCase(patternMatchCases: List[CaseDef], pos: Position)(
86       implicit ctx: Context): Option[CaseDef] = {
87     if (patternMatchCases.isEmpty) None
88     else {
89       val exName = ExceptionBinderName.fresh()
90       val fallbackSelector =
91         ctx.newSymbol(ctx.owner, exName, Flags.Synthetic | Flags.Case, defn.ThrowableType, coord = pos)
92       val sel = Ident(fallbackSelector.termRef).withPos(pos)
93       val rethrow = CaseDef(EmptyTree, EmptyTree, Throw(ref(fallbackSelector)))
94       Some(CaseDef(
95           Bind(fallbackSelector, Underscore(fallbackSelector.info).withPos(pos)),
96           EmptyTree,
97           transformFollowing(Match(sel, patternMatchCases ::: rethrow :: Nil)))
98       )
99     }
100   }
