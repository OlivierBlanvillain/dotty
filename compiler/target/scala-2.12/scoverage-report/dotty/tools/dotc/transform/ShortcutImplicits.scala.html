20 /** This phase optimizes code using implicit function types, by applying two rewrite rules.
21  *  Let IF be the implicit function type
22  *
23  *      implicit Us => R
24  *
25  *  (1) A method definition
26  *
27  *      def m(xs: Ts): IF = implicit (ys: Us) => E
28  *
29  *  is expanded to two methods:
30  *
31  *      def m(xs: Ts): IF = implicit (ys: Us) => m$direct(xs)(ys)
32  *      def m$direct(xs: Ts)(ys: Us): R = E
33  *
34  *  (and equivalently for methods with type parameters or a different number of value parameter lists).
35  *  An abstract method definition
36  *
37  *     def m(xs: Ts): IF
38  *
39  *  is expanded to:
40  *
41  *     def m(xs: Ts): IF
42  *     def m$direct(xs: Ts)(ys: Us): R
43  *
44  *  (2) A reference `qual.apply` where `qual` has implicit function type and
45  *  `qual` refers to a method `m` is rewritten to a reference to `m$direct`,
46  *  keeping the same type and value arguments as they are found in `qual`.
47  */
48 class ShortcutImplicits extends MiniPhase with IdentityDenotTransformer { thisPhase =>
49   import tpd._
50 
51   override def phaseName: String = "shortcutImplicits"
52 
53   override def changesMembers = true  // the phase adds "direct" methods

55   /** A map to cache mapping local methods to their direct counterparts.
56    *  A fresh map is created for each unit.
57    */
58   private var DirectMeth: Store.Location[MutableSymbolMap[Symbol]] = _
59   private def directMeth(implicit ctx: Context) = ctx.store(DirectMeth)
60 
61   override def initContext(ctx: FreshContext) =
62     DirectMeth = ctx.addLocation[MutableSymbolMap[Symbol]]()
63 
64   /** If this option is true, we don't specialize symbols that are known to be only
65    *  targets of monomorphic calls.
66    *  The reason for this option is that benchmarks show that on the JVM for monomorphic dispatch
67    *  scenarios inlining and escape analysis can often remove all calling overhead, so we might as
68    *  well not duplicate the code. We need more experience to decide on the best setting of this option.
69    */
70   final val specializeMonoTargets = true
71 
72   override def prepareForUnit(tree: Tree)(implicit ctx: Context) =
73     ctx.fresh.updateStore(DirectMeth, newMutableSymbolMap[Symbol])

75   /** Should `sym` get a ..$direct companion?
76     *  This is the case if (1) `sym` is a method with an implicit function type as final result type.
77     *  However if `specializeMonoTargets` is false, we exclude symbols that are known
78     *  to be only targets of monomorphic calls because they are effectively
79     *  final and don't override anything.
80     */
81   private def shouldBeSpecialized(sym: Symbol)(implicit ctx: Context) =
82     sym.is(Method, butNot = Accessor) &&
83     defn.isImplicitFunctionType(sym.info.finalResultType) &&
84     !sym.isAnonymousFunction &&
85     (specializeMonoTargets || !sym.isEffectivelyFinal || sym.allOverriddenSymbols.nonEmpty)
86 
87   /** @pre    The type's final result type is an implicit function type `implicit Ts => R`.
88     *  @return The type of the `apply` member of `implicit Ts => R`.
89     */
90   private def directInfo(info: Type)(implicit ctx: Context): Type = info match {
91     case info: PolyType   => info.derivedLambdaType(resType = directInfo(info.resultType))
92     case info: MethodType => info.derivedLambdaType(resType = directInfo(info.resultType))
93     case info: ExprType   => directInfo(info.resultType)
94     case info             => info.member(nme.apply).info
95   }
96 
97   /** A new `m$direct` method to accompany the given method `m` */
98   private def newDirectMethod(sym: Symbol)(implicit ctx: Context): Symbol = {
99     val direct = sym.copy(
100       name = DirectMethodName(sym.name.asTermName).asInstanceOf[sym.ThisName],
101       flags = sym.flags | Synthetic,
102       info = directInfo(sym.info))
103     if (direct.allOverriddenSymbols.isEmpty) direct.resetFlag(Override)
104     direct
105   }
106 
107   /** The direct method `m$direct` that accompanies the given method `m`.
108     *  Create one if it does not exist already.
109     */
110   private def directMethod(sym: Symbol)(implicit ctx: Context): Symbol =
111     if (sym.owner.isClass) {
112       val direct = sym.owner.info.member(DirectMethodName(sym.name.asTermName))
113         .suchThat(_.info matches directInfo(sym.info)).symbol
114       if (direct.maybeOwner == sym.owner) direct
115       else newDirectMethod(sym).enteredAfter(thisPhase)
116     }
117     else directMeth.getOrElseUpdate(sym, newDirectMethod(sym))
118 
119   /** Transform `qual.apply` occurrences according to rewrite rule (2) above */
120   override def transformSelect(tree: Select)(implicit ctx: Context) =
121     if (tree.name == nme.apply &&
122         defn.isImplicitFunctionType(tree.qualifier.tpe.widen) &&
123         shouldBeSpecialized(tree.qualifier.symbol)) {
124       def directQual(tree: Tree): Tree = tree match {
125         case Apply(fn, args)     => cpy.Apply(tree)(directQual(fn), args)
126         case TypeApply(fn, args) => cpy.TypeApply(tree)(directQual(fn), args)
127         case Block(stats, expr)  => cpy.Block(tree)(stats, directQual(expr))
128         case tree: RefTree =>
129           cpy.Ref(tree)(DirectMethodName(tree.name.asTermName))
130             .withType(directMethod(tree.symbol).termRef)
131       }
132       directQual(tree.qualifier)
133     } else tree
134 
135   /** Transform methods with implicit function type result according to rewrite rule (1) above */
136   override def transformDefDef(mdef: DefDef)(implicit ctx: Context): Tree = {
137     val original = mdef.symbol
138     if (shouldBeSpecialized(original)) {
139       val direct = directMethod(original)
140 
141       def splitClosure(tree: Tree): (List[Type] => List[List[Tree]] => Tree, Tree) = tree match {
142         case Block(Nil, expr) => splitClosure(expr)
143         case Block((meth @ DefDef(nme.ANON_FUN, Nil, clparams :: Nil, _, _)) :: Nil, cl: Closure) =>
144           val tparamSyms = mdef.tparams.map(_.symbol)
145           val vparamSymss = mdef.vparamss.map(_.map(_.symbol))
146           val clparamSyms = clparams.map(_.symbol)
147           val remappedCore = (ts: List[Type]) => (prefss: List[List[Tree]]) =>
148             meth.rhs
149               .subst(tparamSyms ::: (vparamSymss.flatten ++ clparamSyms),
150                       ts.map(_.typeSymbol) ::: prefss.flatten.map(_.symbol))
151               .changeOwnerAfter(original, direct, thisPhase)
152               .changeOwnerAfter(meth.symbol, direct, thisPhase)
153           val forwarder = ref(direct)
154             .appliedToTypeTrees(tparamSyms.map(ref(_)))
155             .appliedToArgss(vparamSymss.map(_.map(ref(_))) :+ clparamSyms.map(ref(_)))
156           val fwdClosure = cpy.Block(tree)(cpy.DefDef(meth)(rhs = forwarder) :: Nil, cl)
157           (remappedCore, fwdClosure)
158         case EmptyTree =>
159           (_ => _ => EmptyTree, EmptyTree)
160       }
161 
162       val (remappedCore, fwdClosure) = splitClosure(mdef.rhs)
163       val originalDef = cpy.DefDef(mdef)(rhs = fwdClosure)
164       val directDef = transformDefDef(polyDefDef(direct.asTerm, remappedCore))
165       flatTree(List(originalDef, directDef))
166     }
167     else mdef
168   }
