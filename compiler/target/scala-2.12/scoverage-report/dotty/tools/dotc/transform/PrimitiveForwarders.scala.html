22 /** This phase adds forwarder where mixedin generic and primitive typed methods have a missmatch.
23   *  In particular for every method that is declared both as generic with a primitive type and with a primitive type
24   *    `<mods> def f[Ts](ps1)...(psN): U` in trait M` and
25   *    `<mods> def f[Ts](ps1)...(psN): V = ...` in implemented in N`
26   *    where U is a primitive and V a polymorphic type (or vice versa) needs:
27   *
28   *    <mods> def f[Ts](ps1)...(psN): U = super[N].f[Ts](ps1)...(psN)
29   *
30   *  IMPORTANT: When\If Valhalla happens, we'll need to move mixin before erasure and than this code will need to be rewritten
31   *  as it will instead change super-class.
32   */
33 class PrimitiveForwarders extends MiniPhase with IdentityDenotTransformer { thisPhase =>
34   import ast.tpd._
35 
36   override def phaseName: String = "primitiveForwarders"
37 
38   override def runsAfter = Set(classOf[ResolveSuper])
39 
40   override def changesMembers = true   // the phase adds primitive forwarders
41 
42   override def transformTemplate(impl: Template)(implicit ctx: Context) = {
43     val cls = impl.symbol.owner.asClass
44     val ops = new MixinOps(cls, thisPhase)
45     import ops._
46 
47     def methodPrimitiveForwarders: List[Tree] =
48       for (meth <- mixins.flatMap(_.info.decls.toList.flatMap(needsPrimitiveForwarderTo)).distinct)
49         yield polyDefDef(implementation(meth.asTerm), forwarder(meth))
50 
51     cpy.Template(impl)(body = methodPrimitiveForwarders ::: impl.body)
52   }
53 }
