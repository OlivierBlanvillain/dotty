18 /** This phase eliminates ExprTypes `=> T` as types of method parameter references, and replaces them b
19  *  nullary function types.  More precisely:
20  *
21  *  For the types of parameter symbols:
22  *
23  *         => T       ==>    () => T
24  *
25  *  For cbn parameter values
26  *
27  *         x          ==>    x()
28  *
29  *  Note: This scheme to have inconsistent types between method types (whose formal types are still
30  *  ExprTypes and parameter valdefs (which are now FunctionTypes) is not pretty. There are two
31  *  other options which have been abandoned or not yet pursued.
32  *
33  *  Option 1: Transform => T to () => T also in method and function types. The problem with this is
34  *  that is that it requires to look at every type, and this forces too much, causing
35  *  Cyclic Reference errors. Abandoned for this reason.
36  *
37  *  Option 2: Merge ElimByName with erasure, or have it run immediately before. This has not been
38  *  tried yet.
39  */

49   private def applyIfFunction(tree: Tree, ftree: Tree)(implicit ctx: Context) =

60   override def transformTypeApply(tree: TypeApply)(implicit ctx: Context): Tree = tree match {

66   }

78   }
