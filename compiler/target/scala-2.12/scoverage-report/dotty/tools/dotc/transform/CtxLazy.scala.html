6 /** Utility class for lazy values whose evaluation depends on a context.
7  *  This should be used whenever the evaluation of a lazy expression
8  *  depends on some context, but the value can be re-used afterwards
9  *  with a different context.
10  *
11  *  A typical use case is a lazy val in a phase object which exists once per root context where
12  *  the expression intiializing the lazy val depends only on the root context, but not any changes afterwards.
13  */

17   def apply()(implicit ctx: Context): T = {

21     }

23   }
