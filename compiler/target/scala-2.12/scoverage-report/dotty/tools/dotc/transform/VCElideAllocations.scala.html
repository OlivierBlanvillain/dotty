10 /** This phase elides unnecessary value class allocations
11  *
12  *  For a value class V defined as:
13  *    class V(val underlying: U) extends AnyVal
14  *  we avoid unnecessary allocations:
15  *     new V(u1) == new V(u2) => u1 == u2
16  *    (new V(u)).underlying() => u
17  */
18 class VCElideAllocations extends MiniPhase with IdentityDenotTransformer {
19   import tpd._
20 
21   override def phaseName: String = "vcElideAllocations"
22 
23   override def runsAfter: Set[Class[_ <: Phase]] = Set(classOf[ElimErasedValueType])
24 
25   override def transformApply(tree: Apply)(implicit ctx: Context): Tree =
26     tree match {
27       // new V(u1) == new V(u2) => u1 == u2
28       // (We don't handle != because it has been eliminated by InterceptedMethods)
29       case BinaryOp(NewWithArgs(tp1, List(u1)), op, NewWithArgs(tp2, List(u2)))
30       if (tp1 eq tp2) && (op eq defn.Any_==) && isDerivedValueClass(tp1.typeSymbol) =>
31         // == is overloaded in primitive classes
32         applyOverloaded(u1, nme.EQ, List(u2), Nil, defn.BooleanType)
33 
34       // (new V(u)).underlying() => u
35       case ValueClassUnbox(NewWithArgs(_, List(u))) =>
36         u
37 
38       case _ =>
39         tree
40     }
41 }
