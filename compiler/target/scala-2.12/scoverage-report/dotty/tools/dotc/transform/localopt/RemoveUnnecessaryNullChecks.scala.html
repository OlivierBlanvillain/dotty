12 /** Eliminated null checks based on the following observations:
13  *
14  *  - (this)  cannot be null
15  *  - (new C) cannot be null
16  *  - literal is either null itself or non null
17  *  - fallsback to `tpe.isNotNull`, which will eventually be true for non nullable types.
18  *  - in (a.call; a == null), the first call throws a NPE if a is null; the test can be removed.
19  *
20  *  @author DarkDimius, Jvican, OlivierBlanvillain
21  */
22  class RemoveUnnecessaryNullChecks extends Optimisation {
23   import ast.tpd._
24 
25   val initializedVals = mutable.HashSet[Symbol]()
26 
27   val checkGood = newMutableSymbolMap[Set[Symbol]]
28 
29   def clear(): Unit = {
30     initializedVals.clear()
31     checkGood.clear()
32   }

34   def isGood(t: Symbol)(implicit ctx: Context): Boolean = {
35     t.exists && initializedVals.contains(t) && {
36       var changed = true
37       var set = Set(t)
38       while (changed) {
39         val oldSet = set
40         set = set ++ set.flatMap(x => checkGood.getOrElse(x, Nil))
41         changed = set != oldSet
42       }
43       !set.exists(x => !initializedVals.contains(x))
44     }
45   }
46 
47   def visitor(implicit ctx: Context): Tree => Unit = {
48     case vd: ValDef =>
49       val rhs = vd.rhs
50       if (!vd.symbol.is(Mutable) && !rhs.isEmpty) {
51         def checkNonNull(t: Tree, target: Symbol): Boolean = t match {
52           case Block(_ , expr) =>
53             checkNonNull(expr, target)
54 
55           case If(_, thenp, elsep) =>
56             checkNonNull(thenp, target) && checkNonNull(elsep, target)
57 
58           case _: New | _: This => true
59 
60           case t: Apply if t.symbol.isPrimaryConstructor => true
61 
62           case t: Literal => t.const.value != null
63 
64           case t: Ident if !t.symbol.owner.isClass =>
65             checkGood.update(target, checkGood.getOrElse(target, Set.empty) + t.symbol)
66             true
67 
68           case t: Apply if !t.symbol.owner.isClass =>
69             checkGood.update(target, checkGood.getOrElse(target, Set.empty) + t.symbol)
70             true
71 
72           case t: Typed =>
73             checkNonNull(t.expr, target)
74 
75           case _ => t.tpe.isNotNull
76 
77         }
78         if (checkNonNull(vd.rhs, vd.symbol))
79           initializedVals += vd.symbol
80       }
81     case t: Tree =>
82   }

85   def transformer(implicit ctx: Context): Tree => Tree = {
86     def isNullLiteral(tree: Tree) = tree match {
87       case literal: Literal =>
88         literal.const.tag == NullTag
89       case _ => false
90     }
91     val transformation: Tree => Tree = {
92       case check @ Apply(Select(lhs, _), List(rhs)) =>
93         val sym = check.symbol
94         val eqOrNe  = sym == defn.Object_eq || sym == defn.Object_ne
95         val nullLhs = isNullLiteral(lhs) && isGood(rhs.symbol)
96         val nullRhs = isNullLiteral(rhs) && isGood(lhs.symbol)
97 
98         if (eqOrNe && (nullLhs || nullRhs)) {
99           def block(b: Boolean) = Block(List(lhs, rhs), Literal(Constant(b)))
100           if (sym == defn.Object_eq) block(false)
101           else if (sym == defn.Object_ne) block(true)
102           else check
103         } else check
104 
105       case t => t
106     }
107     transformation
108   }
