25 /** A no-op transform that checks whether the compiled sources are re-entrant.
26  *  If -Ycheck:reentrant is set, the phase makes sure that there are no variables
27  *  that are accessible from a global object. It excludes from checking paths that
28  *  are labeled with one of the annotations
29  *
30  *      @sharable   Indicating a class or val can be safely shared
31  *      @unshared   Indicating an object will not be accessed from multiple threads
32  *
33  *  Currently the analysis is only intended to check the dotty compiler itself. To make
34  *  it generally useful we'd need to add at least the following:
35  *
36  *   - Handle polymorphic instantiation: We might instantiate a generic class
37  *     with a type that contains vars. If the class contains fields of the generic
38  *     type, this may constitute a path to a shared var, which currently goes undetected.
39  *   - Handle arrays: Array elements are currently ignored because they are often used
40  *     in an immutable way anyway. To do better, it would be helpful to have a type
41  *     for immutable array.
42  */
43 class CheckReentrant extends MiniPhase {
44   import ast.tpd._
45 
46   override def phaseName = "checkReentrant"
47 
48   private[this] var shared: Set[Symbol] = Set()
49   private[this] var seen: Set[ClassSymbol] = Set()
50   private[this] var indent: Int = 0
51 
52   private val sharableAnnot = new CtxLazy(implicit ctx =>
53     ctx.requiredClass("dotty.tools.sharable"))
54   private val unsharedAnnot = new CtxLazy(implicit ctx =>
55     ctx.requiredClass("dotty.tools.unshared"))
56 
57   def isIgnored(sym: Symbol)(implicit ctx: Context) =
58     sym.hasAnnotation(sharableAnnot()) ||
59     sym.hasAnnotation(unsharedAnnot())
60 
61   def scanning(sym: Symbol)(op: => Unit)(implicit ctx: Context): Unit = {
62     ctx.log(i"${"  " * indent}scanning $sym")
63     indent += 1
64     try op
65     finally indent -= 1
66   }
67 
68   def addVars(cls: ClassSymbol)(implicit ctx: Context): Unit = {
69     if (!seen.contains(cls) && !isIgnored(cls)) {
70       seen += cls
71       scanning(cls) {
72         for (sym <- cls.classInfo.decls)
73           if (sym.isTerm && !sym.isSetter && !isIgnored(sym))
74             if (sym.is(Mutable)) {
75               ctx.error(
76                 i"""possible data race involving globally reachable ${sym.showLocated}: ${sym.info}
77                    |  use -Ylog:checkReentrant+ to find out more about why the variable is reachable.""")
78               shared += sym
79             } else if (!sym.is(Method) || sym.is(Accessor | ParamAccessor)) {
80               scanning(sym) {
81                 sym.info.widenExpr.classSymbols.foreach(addVars)
82               }
83             }
84         for (parent <- cls.classInfo.classParents)
85           addVars(parent.typeSymbol.asClass)
86       }
87     }
88   }
89 
90   override def transformTemplate(tree: Template)(implicit ctx: Context): Tree = {
91     if (ctx.settings.YcheckReentrant.value && tree.symbol.owner.isStaticOwner)
92       addVars(tree.symbol.owner.asClass)
93     tree
94   }
