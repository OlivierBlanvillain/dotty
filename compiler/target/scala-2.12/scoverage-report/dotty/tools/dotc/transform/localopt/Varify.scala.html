10 /** Inline val with exactly one assignment to a var. For example:
11  *
12  *  {
13  *    val l = <expr>
14  *    // code that may use l
15  *    var r = l
16  *    // code not using l
17  *  }
18  *
19  *  becomes:
20  *

39   def visitor(implicit ctx: Context): Tree => Unit = {
40     case t: ValDef if t.symbol.is(Param) =>
41       paramsTimesUsed.update(t.symbol, 0)

43     case t: ValDef if t.symbol.is(Mutable) =>
44       t.rhs.foreachSubTree { subtree =>

49         }
50       }
51

53       val param = t.symbol

56 
57     case _ =>
58   }

60   def transformer(implicit ctx: Context): Tree => Tree = {
61     val paramCandidates = paramsTimesUsed.filter(kv => kv._2 == 1).keySet
62     val renames: Map[Symbol, Symbol] = possibleRenames.iterator
63       .map(kv => (kv._1, kv._2.intersect(paramCandidates)))
64       .filter(x => x._2.nonEmpty)
65       .map(x => (x._1, x._2.head))
66       .toMap
67 
68     val transformation: Tree => Tree = {
69       case t: RefTree if renames.contains(t.symbol) =>
70         ref(renames(t.symbol))
71 
72       case t: ValDef if renames.contains(t.symbol) =>
73         val replaced = renames(t.symbol)
74         if (t.rhs.symbol == replaced) EmptyTree
75         else ref(replaced).becomes(t.rhs)
76 
77       case t: ValDef if paramCandidates.contains(t.symbol) =>
78         t.symbol.flags = Mutable
79         t
80 
81       case t => t
82     }
83     transformation
84   }
