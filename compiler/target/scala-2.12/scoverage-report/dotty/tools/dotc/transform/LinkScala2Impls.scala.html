22 /** Rewrite calls
23  *
24  *    super[M].f(args)
25  *
26  *  where M is a Scala 2.11 trait implemented by the current class to
27  *
28  *    M$class.f(this, args)
29  *
30  *  provided the implementation class M$class defines a corresponding function `f`.
31  *  If M is a Scala 2.12 or newer trait, rewrite to
32  *
33  *    M.f(this, args)
34  *
35  *  where f is a static member of M.
36  */
37 class LinkScala2Impls extends MiniPhase with IdentityDenotTransformer { thisPhase =>
38   import ast.tpd._

40   override def phaseName: String = "linkScala2Impls"
41   override def changesMembers = true

43   override def runsAfterGroupsOf: Set[Class[_ <: Phase]] = Set(classOf[Mixin])
44     // Adds as a side effect static members to traits which can confuse Mixin,
45     // that's why it is runsAfterGroupOf

47   /** Copy definitions from implementation class to trait itself */
48   private def augmentScala_2_12_Trait(mixin: ClassSymbol)(implicit ctx: Context): Unit = {
49     def info_2_12(sym: Symbol) = sym.info match {
50       case mt @ MethodType(paramNames @ nme.SELF :: _) =>
51         // 2.12 seems to always assume the enclsing mixin class as self type parameter,
52         // whereas 2.11 used the self type of this class instead.
53         val selfType :: otherParamTypes = mt.paramInfos
54         MethodType(paramNames, mixin.typeRef :: otherParamTypes, mt.resType)
55       case info => info
56     }
57     def newImpl(sym: TermSymbol): Symbol = sym.copy(
58       owner = mixin,
59       name = if (sym.isConstructor) sym.name else ImplMethName(sym.name),
60       info = info_2_12(sym)
61     )
62     for (sym <- mixin.implClass.info.decls)
63       newImpl(sym.asTerm).enteredAfter(thisPhase)
64   }
65 
66   override def prepareForTemplate(impl: Template)(implicit ctx: Context) = {
67     val cls = impl.symbol.owner.asClass
68     for (mixin <- cls.mixins)
69       if (mixin.is(Scala_2_12_Trait, butNot = Scala_2_12_Augmented)) {
70         augmentScala_2_12_Trait(mixin)
71         mixin.setFlag(Scala_2_12_Augmented)
72       }
73     ctx
74   }
75 
76   override def transformApply(app: Apply)(implicit ctx: Context) = {
77     def currentClass = ctx.owner.enclosingClass.asClass
78     app match {
79       case Apply(sel @ Select(Super(_, _), _), args)
80       if sel.symbol.owner.is(Scala2xTrait) && currentClass.mixins.contains(sel.symbol.owner) =>
81         val impl = implMethod(sel.symbol)
82         if (impl.exists) Apply(ref(impl), This(currentClass) :: args).withPos(app.pos)
83         else app // could have been an abstract method in a trait linked to from a super constructor
84       case Apply(sel, args)
85       if sel.symbol.maybeOwner.is(ImplClass) && sel.symbol.owner.traitOfImplClass.is(Scala_2_12_Trait) =>
86         val impl = implMethod(sel.symbol)
87         cpy.Apply(app)(ref(impl), args)
88       case _ =>
89         app
90     }
91   }
92 
93   /** The 2.12 implementation method of a super call or implementation class target */
94   private def implMethod(meth: Symbol)(implicit ctx: Context): Symbol = {
95     val implName = ImplMethName(meth.name.asTermName)
96     val cls = meth.owner
97     if (cls.is(ImplClass))
98       cls.traitOfImplClass.info.decl(implName).atSignature(meth.signature).symbol
99     else if (cls.is(Scala_2_12_Trait))
100       if (meth.isConstructor)
101         cls.info.decl(nme.TRAIT_CONSTRUCTOR).symbol
102       else
103         cls.info.decl(implName)
104           .suchThat(c => FullParameterization.memberSignature(c.info) == meth.signature)
105           .symbol
106     else throw new AssertionError(i"no impl method for $meth")
107   }

109   private val Scala2xTrait = allOf(Scala2x, Trait)
