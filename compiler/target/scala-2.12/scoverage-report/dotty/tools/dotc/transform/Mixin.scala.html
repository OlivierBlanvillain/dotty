22 /** This phase performs the following transformations:
23  *
24  *   1. (done in `traitDefs` and `transformSym`) Map every concrete trait getter
25  *
26  *       <mods> def x(): T = expr
27  *
28  *   to the pair of definitions:
29  *
30  *       <mods> def x(): T
31  *       protected def initial$x(): T = { stats; expr }
32  *
33  *   where `stats` comprises all statements between either the start of the trait
34  *   or the previous field definition which are not definitions (i.e. are executed for
35  *   their side effects).
36  *
37  *   2. (done in `traitDefs`) Make every concrete trait setter
38  *
39  *      <mods> def x_=(y: T) = ()
40  *
41  *     deferred by mapping it to
42  *
43  *      <mods> def x_=(y: T)
44  *
45  *   3. For a non-trait class C:
46  *
47  *        For every trait M directly implemented by the class (see SymUtils.mixin), in
48  *        reverse linearization order, add the following definitions to C:
49  *
50  *          3.1 (done in `traitInits`) For every parameter accessor `<mods> def x(): T` in M,
51  *              in order of textual occurrence, add
52  *
53  *               <mods> def x() = e
54  *
55  *              where `e` is the constructor argument in C that corresponds to `x`. Issue
56  *              an error if no such argument exists.
57  *
58  *          3.2 (done in `traitInits`) For every concrete trait getter `<mods> def x(): T` in M
59  *              which is not a parameter accessor, in order of textual occurrence, produce the following:
60  *
61  *              3.2.1 If `x` is also a member of `C`, and M is a Dotty trait:
62  *
63  *                <mods> def x(): T = super[M].initial$x()
64  *
65  *              3.2.2 If `x` is also a member of `C`, and M is a Scala 2.x trait:
66  *
67  *                <mods> def x(): T = _
68  *
69  *              3.2.3 If `x` is not a member of `C`, and M is a Dotty trait:
70  *
71  *                super[M].initial$x()
72  *
73  *              3.2.4 If `x` is not a member of `C`, and M is a Scala2.x trait, nothing gets added.
74  *
75  *
76  *          3.3 (done in `superCallOpt`) The call:
77  *
78  *                super[M].<init>
79  *
80  *          3.4 (done in `setters`) For every concrete setter `<mods> def x_=(y: T)` in M:
81  *
82  *                <mods> def x_=(y: T) = ()
83  *
84  *   4. (done in `transformTemplate` and `transformSym`) Drop all parameters from trait
85  *      constructors.
86  *
87  *   5. (done in `transformSym`) Drop ParamAccessor flag from all parameter accessors in traits.
88  *
89  *  Conceptually, this is the second half of the previous mixin phase. It needs to run
90  *  after erasure because it copies references to possibly private inner classes and objects
91  *  into enclosing classes where they are not visible. This can only be done if all references
92  *  are symbolic.
93  */

106   override def transformSym(sym: SymDenotation)(implicit ctx: Context): SymDenotation =
107     if (sym.is(Accessor, butNot = Deferred) && sym.owner.is(Trait)) {
108       val sym1 =

135   override def transformTemplate(impl: Template)(implicit ctx: Context) = {
136     val cls = impl.symbol.owner.asClass
137     val ops = new MixinOps(cls, thisPhase)
138     import ops._

140     def traitDefs(stats: List[Tree]): List[Tree] = {
141       val initBuf = new mutable.ListBuffer[Tree]
142       stats.flatMap({
143         case stat: DefDef if stat.symbol.isGetter && !stat.rhs.isEmpty && !stat.symbol.is(Flags.Lazy) =>
144           // make initializer that has all effects of previous getter,
145           // replace getter rhs with empty tree.
146           val vsym = stat.symbol
147           val isym = initializer(vsym)
148           val rhs = Block(
149             initBuf.toList.map(_.changeOwnerAfter(impl.symbol, isym, thisPhase)),
150             stat.rhs.changeOwnerAfter(vsym, isym, thisPhase).wildcardToDefault)
151           initBuf.clear()
152           cpy.DefDef(stat)(rhs = EmptyTree) :: DefDef(isym, rhs) :: Nil
153         case stat: DefDef if stat.symbol.isSetter =>
154           cpy.DefDef(stat)(rhs = EmptyTree) :: Nil
155         case stat: DefTree =>
156           stat :: Nil
157         case stat =>
158           initBuf += stat
159           Nil
160       }) ++ initBuf
161     }

163     /** Map constructor call to a pair of a supercall and a list of arguments
164      *  to be used as initializers of trait parameters if the target of the call
165      *  is a trait.
166      */
167     def transformConstructor(tree: Tree): (Tree, List[Tree]) = {
168       val Apply(sel @ Select(New(_), nme.CONSTRUCTOR), args) = tree
169       val (callArgs, initArgs) = if (tree.symbol.owner.is(Trait)) (Nil, args) else (args, Nil)
170       (superRef(tree.symbol, tree.pos).appliedToArgs(callArgs), initArgs)
171     }
172 
173     val superCallsAndArgs = (
174       for (p <- impl.parents if p.symbol.isConstructor)
175       yield p.symbol.owner -> transformConstructor(p)
176     ).toMap
177     val superCalls = superCallsAndArgs.mapValues(_._1)
178     val initArgs = superCallsAndArgs.mapValues(_._2)
179 
180     def superCallOpt(baseCls: Symbol): List[Tree] = superCalls.get(baseCls) match {
181       case Some(call) =>
182         if (defn.NotRuntimeClasses.contains(baseCls) || baseCls.is(NoInitsTrait)) Nil
183         else call :: Nil
184       case None =>
185         if (baseCls.is(NoInitsTrait) || defn.NoInitClasses.contains(baseCls)) Nil
186         else {
187           //println(i"synth super call ${baseCls.primaryConstructor}: ${baseCls.primaryConstructor.info}")

190     }

192     def was(sym: Symbol, flags: FlagSet) =
193       ctx.atPhase(thisPhase) { implicit ctx => sym is flags }
194 
195     def traitInits(mixin: ClassSymbol): List[Tree] = {
196       var argNum = 0
197       def nextArgument() = initArgs.get(mixin) match {
198         case Some(arguments) =>
199           val result = arguments(argNum)
200           argNum += 1
201           result
202         case None =>
203           assert(
204               impl.parents.forall(_.tpe.typeSymbol != mixin),
205               i"missing parameters for $mixin from $impl should have been caught in typer")
206           ctx.error(
207               em"""parameterized $mixin is indirectly implemented,
208                   |needs to be implemented directly so that arguments can be passed""",
209               cls.pos)
210           EmptyTree
211       }
212 
213       for (getter <- mixin.info.decls.toList if getter.isGetter && !was(getter, Deferred)) yield {
214         val isScala2x = mixin.is(Scala2x)
215         def default = Underscore(getter.info.resultType)
216         def initial = transformFollowing(superRef(initializer(getter)).appliedToNone)
217 
218         /** A call to the implementation of `getter` in `mixin`'s implementation class */
219         def lazyGetterCall = {
220           def canbeImplClassGetter(sym: Symbol) = sym.info.firstParamTypes match {
221             case t :: Nil => t.isDirectRef(mixin)
222             case _ => false
223           }
224           val implClassGetter = mixin.implClass.info.nonPrivateDecl(getter.name)
225             .suchThat(canbeImplClassGetter).symbol
226           ref(mixin.implClass).select(implClassGetter).appliedTo(This(cls))
227         }
228 
229         if (isCurrent(getter) || getter.name.is(ExpandedName)) {
230           val rhs =
231             if (was(getter, ParamAccessor)) nextArgument()
232             else if (isScala2x)
233               if (getter.is(Lazy, butNot = Module)) lazyGetterCall
234               else if (getter.is(Module))
235                 New(getter.info.resultType, List(This(cls)))
236               else Underscore(getter.info.resultType)
237             else initial
238           // transformFollowing call is needed to make memoize & lazy vals run
239           transformFollowing(DefDef(implementation(getter.asTerm), rhs))
240         }
241         else if (isScala2x || was(getter, ParamAccessor)) EmptyTree
242         else initial
243       }
244     }
245 
246     def setters(mixin: ClassSymbol): List[Tree] =
247       for (setter <- mixin.info.decls.filter(setr => setr.isSetter && !was(setr, Deferred)))
248         yield transformFollowing(DefDef(implementation(setter.asTerm), unitLiteral.withPos(cls.pos)))
249 
250     cpy.Template(impl)(
251       constr =
252         if (cls.is(Trait)) cpy.DefDef(impl.constr)(vparamss = Nil :: Nil)
253         else impl.constr,
254       parents = impl.parents.map(p => TypeTree(p.tpe).withPos(p.pos)),
255       body =
256         if (cls is Trait) traitDefs(impl.body)
257         else {
258           val mixInits = mixins.flatMap { mixin =>
259             flatten(traitInits(mixin)) ::: superCallOpt(mixin) ::: setters(mixin)
260           }
261           superCallOpt(superCls) ::: mixInits ::: impl.body
262         })
263   }
