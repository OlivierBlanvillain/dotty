48   // Collect all private parameter accessors and value definitions that need
49   // to be retained. There are several reasons why a parameter accessor or
50   // definition might need to be retained:
51   // 1. It is accessed after the constructor has finished
52   // 2. It is accessed before it is defined
53   // 3. It is accessed on an object other than `this`
54   // 4. It is a mutable parameter accessor
55   // 5. It is has a wildcard initializer `_`
56   private def markUsedPrivateSymbols(tree: RefTree)(implicit ctx: Context): Unit = {
57 
58     val sym = tree.symbol
59     def retain() = retainedPrivateVals.add(sym)
60 
61     if (sym.exists && sym.owner.isClass && mightBeDropped(sym)) {
62       val owner = sym.owner.asClass
63 
64         tree match {
65           case Ident(_) | Select(This(_), _) =>
66             def inConstructor = {
67               val method = ctx.owner.enclosingMethod
68               method.isPrimaryConstructor && ctx.owner.enclosingClass == owner
69             }
70             if (inConstructor &&
71                 (sym.is(ParamAccessor) || seenPrivateVals.contains(sym))) {
72               // used inside constructor, accessed on this,
73               // could use constructor argument instead, no need to retain field
74             }
75             else retain()
76           case _ => retain()
77         }
78     }
79   }

126   override def transformTemplate(tree: Template)(implicit ctx: Context): Tree = {
127     val cls = ctx.owner.asClass
128 
129     val constr @ DefDef(nme.CONSTRUCTOR, Nil, vparams :: Nil, _, EmptyTree) = tree.constr
130 
131     // Produce aligned accessors and constructor parameters. We have to adjust
132     // for any outer parameters, which are last in the sequence of original
133     // parameter accessors but come first in the constructor parameter list.
134     val accessors = cls.paramAccessors.filterNot(x => x.isSetter || x.info.resultType.classSymbol == defn.ErasedPhantomClass)
135     val vparamsWithOuterLast = vparams match {
136       case vparam :: rest if vparam.name == nme.OUTER => rest ::: vparam :: Nil
137       case _ => vparams
138     }
139     val paramSyms = vparamsWithOuterLast map (_.symbol)
140 
141     // Adjustments performed when moving code into the constructor:
142     //  (1) Replace references to param accessors by constructor parameters
143     //      except possibly references to mutable variables, if `excluded = Mutable`.
144     //      (Mutable parameters should be replaced only during the super call)
145     //  (2) If the parameter accessor reference was to an alias getter,
146     //      drop the () when replacing by the parameter.
147     object intoConstr extends TreeMap {
148       override def transform(tree: Tree)(implicit ctx: Context): Tree = tree match {
149         case Ident(_) | Select(This(_), _) =>
150           var sym = tree.symbol
151           if (sym is (ParamAccessor, butNot = Mutable)) sym = sym.subst(accessors, paramSyms)
152           if (sym.owner.isConstructor) ref(sym).withPos(tree.pos) else tree
153         case Apply(fn, Nil) =>
154           val fn1 = transform(fn)
155           if ((fn1 ne fn) && fn1.symbol.is(Param) && fn1.symbol.owner.isPrimaryConstructor)
156             fn1 // in this case, fn1.symbol was an alias for a parameter in a superclass
157           else cpy.Apply(tree)(fn1, Nil)
158         case _ =>
159           if (noDirectRefsFrom(tree)) tree else super.transform(tree)
160       }
161 
162       def apply(tree: Tree, prevOwner: Symbol)(implicit ctx: Context): Tree = {
163         transform(tree).changeOwnerAfter(prevOwner, constr.symbol, thisPhase)
164       }
165     }

167     def isRetained(acc: Symbol) = {
168       !mightBeDropped(acc) || retainedPrivateVals(acc)
169     }
170 
171     val constrStats, clsStats = new mutable.ListBuffer[Tree]
172 
173     /** Map outer getters $outer and outer accessors $A$B$$$outer to the given outer parameter. */
174     def mapOuter(outerParam: Symbol) = new TreeMap {
175       override def transform(tree: Tree)(implicit ctx: Context) = tree match {
176         case Apply(fn, Nil)
177           if (fn.symbol.is(OuterAccessor)
178              || fn.symbol.isGetter && fn.symbol.name == nme.OUTER
179              ) &&
180              fn.symbol.info.resultType.classSymbol == outerParam.info.classSymbol =>
181           ref(outerParam)
182         case _ =>
183           super.transform(tree)
184       }
185     }
186 
187     val dropped = mutable.Set[Symbol]()
188 
189     // Split class body into statements that go into constructor and
190     // definitions that are kept as members of the class.
191     def splitStats(stats: List[Tree]): Unit = stats match {
192       case stat :: stats1 =>
193         stat match {
194           case stat @ ValDef(name, tpt, _) if !stat.symbol.is(Lazy) && !stat.symbol.hasAnnotation(defn.ScalaStaticAnnot) =>
195             val sym = stat.symbol
196             if (isRetained(sym)) {
197               if (!stat.rhs.isEmpty && !isWildcardArg(stat.rhs))
198                 constrStats += Assign(ref(sym), intoConstr(stat.rhs, sym)).withPos(stat.pos)
199               clsStats += cpy.ValDef(stat)(rhs = EmptyTree)
200             }
201             else if (!stat.rhs.isEmpty) {
202               dropped += sym
203               sym.copySymDenotation(
204                 initFlags = sym.flags &~ Private,
205                 owner = constr.symbol).installAfter(thisPhase)
206               constrStats += intoConstr(stat, sym)
207             }
208           case DefDef(nme.CONSTRUCTOR, _, ((outerParam @ ValDef(nme.OUTER, _, _)) :: _) :: Nil, _, _) =>
209             clsStats += mapOuter(outerParam.symbol).transform(stat)
210           case _: DefTree =>
211             clsStats += stat
212           case _ =>
213             constrStats += intoConstr(stat, tree.symbol)
214         }
215         splitStats(stats1)
216       case Nil =>
217         (Nil, Nil)
218     }
219     splitStats(tree.body)
220 
221     // The initializers for the retained accessors */
222     val copyParams = accessors flatMap { acc =>
223       if (!isRetained(acc)) {
224         dropped += acc
225         Nil
226       } else {
227         val target = if (acc.is(Method)) acc.field else acc
228         if (!target.exists) Nil // this case arises when the parameter accessor is an alias
229         else {
230           val param = acc.subst(accessors, paramSyms)
231           val assigns = Assign(ref(target), ref(param)).withPos(tree.pos) :: Nil
232           if (acc.name != nme.OUTER) assigns
233           else {
234             // insert test: if ($outer eq null) throw new NullPointerException
235             val nullTest =
236               If(ref(param).select(defn.Object_eq).appliedTo(Literal(Constant(null))),
237                  Throw(New(defn.NullPointerExceptionClass.typeRef, Nil)),
238                  unitLiteral)
239             nullTest :: assigns
240           }
241         }
242       }
243     }
244 
245     // Drop accessors that are not retained from class scope
246     if (dropped.nonEmpty) {
247       val clsInfo = cls.classInfo
248       cls.copy(
249         info = clsInfo.derivedClassInfo(
250           decls = clsInfo.decls.filteredScope(!dropped.contains(_))))
251 
252       // TODO: this happens to work only because Constructors is the last phase in group
253     }
254 
255     val (superCalls, followConstrStats) = constrStats.toList match {
256       case (sc: Apply) :: rest if sc.symbol.isConstructor => (sc :: Nil, rest)
257       case stats => (Nil, stats)
258     }
259 
260     val mappedSuperCalls = vparams match {
261       case (outerParam @ ValDef(nme.OUTER, _, _)) :: _ =>
262         superCalls.map(mapOuter(outerParam.symbol).transform)
263       case _ => superCalls
264     }
265 
266     // Lazy Vals may decide to create an eager val instead of a lazy val
267     // this val should be assigned before constructor body code starts running
268 
269     val (lazyAssignments, stats) = followConstrStats.partition {
270       case Assign(l, r) if l.symbol.name.is(NameKinds.LazyLocalName) => true
271       case _ => false
272     }
273 
274     val finalConstrStats = copyParams ::: mappedSuperCalls ::: lazyAssignments ::: stats
275     val expandedConstr =
276       if (cls.is(NoInitsTrait)) {
277         assert(finalConstrStats.isEmpty)
278         constr
279       }
280       else cpy.DefDef(constr)(rhs = Block(finalConstrStats, unitLiteral))
281 
282     cpy.Template(tree)(
283       constr = expandedConstr,
284       body = clsStats.toList)
285   }
