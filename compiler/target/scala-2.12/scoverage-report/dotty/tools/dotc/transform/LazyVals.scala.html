62   def transformLazyVal(tree: ValOrDefDef)(implicit ctx: Context): Tree = {
63     val sym = tree.symbol
64     if (!(sym is Flags.Lazy) ||
65         sym.owner.is(Flags.Trait) || // val is accessor, lazy field will be implemented in subclass
66         (sym.isStatic && sym.is(Flags.Module, butNot = Flags.Method))) // static module vals are implemented in the JVM by lazy loading
67       tree
68     else {
69       val isField = sym.owner.isClass
70       if (isField) {
71         if (sym.isVolatile ||
72            (sym.is(Flags.Module)/* || ctx.scala2Mode*/) &&
73             // TODO assume @volatile once LazyVals uses static helper constructs instead of
74             // ones in the companion object.
75            !sym.is(Flags.Synthetic))
76             // module class is user-defined.
77             // Should be threadsafe, to mimic safety guaranteed by global object
78           transformMemberDefVolatile(tree)
79         else if (sym.is(Flags.Module)) // synthetic module
80           transformSyntheticModule(tree)
81         else
82           transformMemberDefNonVolatile(tree)
83       }
84       else transformLocalDef(tree)
85     }
86   }

238     /** Create a threadsafe lazy accessor equivalent to such code
239       *
240       * def methodSymbol(): Int = {
241       *   val result: Int = 0
242       *   val retry: Boolean = true
243       *   var flag: Long = 0L

245       *     flag = dotty.runtime.LazyVals.get(this, $claz.$OFFSET)
246       *     dotty.runtime.LazyVals.STATE(flag, 0) match {
247       *       case 0 =>
248       *         if dotty.runtime.LazyVals.CAS(this, $claz.$OFFSET, flag, 1, $ord) {
249       *           try {result = rhs} catch {
250       *             case x: Throwable =>
251       *               dotty.runtime.LazyVals.setFlag(this, $claz.$OFFSET, 0, $ord)
252       *               throw x
253       *           }
254       *           $target = result
255       *           dotty.runtime.LazyVals.setFlag(this, $claz.$OFFSET, 3, $ord)
256       *           retry = false
257       *           }
258       *       case 1 =>
259       *         dotty.runtime.LazyVals.wait4Notification(this, $claz.$OFFSET, flag, $ord)
260       *       case 2 =>
261       *         dotty.runtime.LazyVals.wait4Notification(this, $claz.$OFFSET, flag, $ord)
262       *       case 3 =>
263       *         retry = false
264       *         result = $target
265       *       }
266       *     }
267       *   result
268       * }
269       */

272       val computeState = Literal(Constants.Constant(1))
273       val notifyState = Literal(Constants.Constant(2))
274       val computedState = Literal(Constants.Constant(3))
275       val flagSymbol = ctx.newSymbol(methodSymbol, lazyNme.flag, containerFlags, defn.LongType)
276       val flagDef = ValDef(flagSymbol, Literal(Constant(0L)))
277 
278       val thiz = This(claz)(ctx.fresh.setOwner(claz))
279 
280       val resultSymbol = ctx.newSymbol(methodSymbol, lazyNme.result, containerFlags, tp)
281       val resultDef = ValDef(resultSymbol, defaultValue(tp))
282 
283       val retrySymbol = ctx.newSymbol(methodSymbol, lazyNme.retry, containerFlags, defn.BooleanType)
284       val retryDef = ValDef(retrySymbol, Literal(Constants.Constant(true)))
285 
286       val whileCond = ref(retrySymbol)
287 
288       val compute = {
289         val handlerSymbol = ctx.newSymbol(methodSymbol, nme.ANON_FUN, Flags.Synthetic,
290           MethodType(List(nme.x_1), List(defn.ThrowableType), defn.IntType))
291         val caseSymbol = ctx.newSymbol(methodSymbol, nme.DEFAULT_EXCEPTION_NAME, Flags.Synthetic, defn.ThrowableType)
292         val triggerRetry = setFlagState.appliedTo(thiz, offset, initState, Literal(Constant(ord)))
293         val complete = setFlagState.appliedTo(thiz, offset, computedState, Literal(Constant(ord)))
294 
295         val handler = CaseDef(Bind(caseSymbol, ref(caseSymbol)), EmptyTree,
296           Block(List(triggerRetry), Throw(ref(caseSymbol))
297         ))
298 
299         val compute = ref(resultSymbol).becomes(rhs)
300         val tr = Try(compute, List(handler), EmptyTree)
301         val assign = ref(target).becomes(ref(resultSymbol))
302         val noRetry = ref(retrySymbol).becomes(Literal(Constants.Constant(false)))
303         val body = If(casFlag.appliedTo(thiz, offset, ref(flagSymbol), computeState, Literal(Constant(ord))),
304           Block(tr :: assign :: complete :: noRetry :: Nil, Literal(Constant(()))),
305           Literal(Constant(())))
306 
307         CaseDef(initState, EmptyTree, body)
308       }
309 
310       val waitFirst = {
311         val wait = waitOnLock.appliedTo(thiz, offset, ref(flagSymbol), Literal(Constant(ord)))
312         CaseDef(computeState, EmptyTree, wait)
313       }
314 
315       val waitSecond = {
316         val wait = waitOnLock.appliedTo(thiz, offset, ref(flagSymbol), Literal(Constant(ord)))
317         CaseDef(notifyState, EmptyTree, wait)
318       }
319 
320       val computed = {
321         val noRetry = ref(retrySymbol).becomes(Literal(Constants.Constant(false)))
322         val result = ref(resultSymbol).becomes(ref(target))
323         val body = Block(noRetry :: result :: Nil, Literal(Constant(())))
324         CaseDef(computedState, EmptyTree, body)
325       }
326 
327       val default = CaseDef(Underscore(defn.LongType), EmptyTree, Literal(Constant(())))
328 
329       val cases = Match(stateMask.appliedTo(ref(flagSymbol), Literal(Constant(ord))),
330         List(compute, waitFirst, waitSecond, computed, default)) //todo: annotate with @switch
331 
332       val whileBody = List(ref(flagSymbol).becomes(getFlag.appliedTo(thiz, offset)), cases)
333       val cycle = WhileDo(methodSymbol, whileCond, whileBody)
334       DefDef(methodSymbol, Block(resultDef :: retryDef :: flagDef :: cycle :: Nil, ref(resultSymbol)))
335     }

337     def transformMemberDefVolatile(x: ValOrDefDef)(implicit ctx: Context) = {
338         assert(!(x.symbol is Flags.Mutable))
339 
340         val tpe = x.tpe.widen.resultType.widen
341         val claz = x.symbol.owner.asClass
342         val thizClass = Literal(Constant(claz.info))
343         val helperModule = ctx.requiredModule("dotty.runtime.LazyVals")
344         val getOffset = Select(ref(helperModule), lazyNme.RLazyVals.getOffset)
345         var offsetSymbol: TermSymbol = null
346         var flag: Tree = EmptyTree
347         var ord = 0
348 
349         def offsetName(id: Int) = (StdNames.nme.LAZY_FIELD_OFFSET + (if(x.symbol.owner.is(Flags.Module)) "_m_" else "") + id.toString).toTermName
350 
351         // compute or create appropriate offsetSymol, bitmap and bits used by current ValDef
352         appendOffsetDefs.get(claz) match {
353           case Some(info) =>
354             val flagsPerLong = (64 / dotty.runtime.LazyVals.BITS_PER_LAZY_VAL).toInt
355             info.ord += 1
356             ord = info.ord % flagsPerLong
357             val id = info.ord / flagsPerLong
358             val offsetById = offsetName(id)
359             if (ord != 0) { // there are unused bits in already existing flag
360               offsetSymbol = claz.info.decl(offsetById)
361                 .suchThat(sym => (sym is Flags.Synthetic) && sym.isTerm)
362                  .symbol.asTerm
363             } else { // need to create a new flag
364               offsetSymbol = ctx.newSymbol(claz, offsetById, Flags.Synthetic, defn.LongType).enteredAfter(this)
365               offsetSymbol.addAnnotation(Annotation(defn.ScalaStaticAnnot))
366               val flagName = (StdNames.nme.BITMAP_PREFIX + id.toString).toTermName
367               val flagSymbol = ctx.newSymbol(claz, flagName, containerFlags, defn.LongType).enteredAfter(this)
368               flag = ValDef(flagSymbol, Literal(Constants.Constant(0L)))
369               val offsetTree = ValDef(offsetSymbol, getOffset.appliedTo(thizClass, Literal(Constant(flagName.toString))))
370               info.defs = offsetTree :: info.defs
371             }
372 
373           case None =>
374             offsetSymbol = ctx.newSymbol(claz, offsetName(0), Flags.Synthetic, defn.LongType).enteredAfter(this)
375             offsetSymbol.addAnnotation(Annotation(defn.ScalaStaticAnnot))
376             val flagName = (StdNames.nme.BITMAP_PREFIX + "0").toTermName
377             val flagSymbol = ctx.newSymbol(claz, flagName, containerFlags, defn.LongType).enteredAfter(this)
378             flag = ValDef(flagSymbol, Literal(Constants.Constant(0L)))
379             val offsetTree = ValDef(offsetSymbol, getOffset.appliedTo(thizClass, Literal(Constant(flagName.toString))))
380             appendOffsetDefs += (claz -> new OffsetInfo(List(offsetTree), ord))
381         }
382 
383         val containerName = LazyLocalName.fresh(x.name.asTermName)
384         val containerSymbol = ctx.newSymbol(claz, containerName, x.symbol.flags &~ containerFlagsMask | containerFlags, tpe, coord = x.symbol.coord).enteredAfter(this)
385 
386         val containerTree = ValDef(containerSymbol, defaultValue(tpe))
387 
388         val offset =  ref(offsetSymbol)
389         val getFlag = Select(ref(helperModule), lazyNme.RLazyVals.get)
390         val setFlag = Select(ref(helperModule), lazyNme.RLazyVals.setFlag)
391         val wait =    Select(ref(helperModule), lazyNme.RLazyVals.wait4Notification)
392         val state =   Select(ref(helperModule), lazyNme.RLazyVals.state)
393         val cas =     Select(ref(helperModule), lazyNme.RLazyVals.cas)
394 
395         val accessor = mkThreadSafeDef(x.symbol.asTerm, claz, ord, containerSymbol, x.rhs, tpe, offset, getFlag, state, cas, setFlag, wait)
396         if (flag eq EmptyTree)
397           Thicket(containerTree, accessor)
398         else Thicket(containerTree, flag, accessor)
399     }
