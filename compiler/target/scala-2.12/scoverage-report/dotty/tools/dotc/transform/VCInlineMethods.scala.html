12 /** This phase inlines calls to methods of value classes.
13  *
14  *  A value class V after [[ExtensionMethods]] will look like:
15  *    class V[A, B, ...](val underlying: U) extends AnyVal {
16  *      def foo[T, S, ...](arg1: A1, arg2: A2, ...) =
17  *        V.foo$extension[T, S, ..., A, B, ...](this)(arg1, arg2, ...)
18  *
19  *       ...
20  *    }
21  *
22  *  Let e have type V, if e is a stable prefix or if V does not have any class
23  *  type parameter, then we can rewrite:
24  *    e.foo[X, Y, ...](args)
25  *  as:
26  *    V.foo$extension[X, Y, ..., A', B', ...](e)(args)
27  *    where A', B', ... are the class type parameters A, B, ... as seen from `e`.
28  *  Otherwise, we need to evaluate e first:
29  *    {
30  *      val ev = e
31  *      V.foo$extension[X, Y, ..., A', B', ...](ev)(args)
32  *    }
33  *
34  *  This phase needs to be placed after phases which may introduce calls to
35  *  value class methods (like [[PatternMatcher]]). This phase uses name mangling
36  *  to find the correct extension method corresponding to a value class method
37  *  (see [[ExtensionMethods.extensionMethod]]), therefore we choose to place it
38  *  before phases which may perform their own name mangling on value class
39  *  methods (like [[TypeSpecializer]]), this way [[VCInlineMethods]] does not
40  *  need to have any knowledge of the name mangling done by other phases.
41  */

50   /** Replace a value class method call by a call to the corresponding extension method.
51    *
52    *  @param tree   The tree corresponding to the method call
53    *  @param mtArgs Type arguments for the method call not present in `tree`
54    *  @param mArgss Arguments for the method call not present in `tree`
55    *  @return       A tree for the extension method call
56    */
57   private def rewire(tree: Tree, mtArgs: List[Tree] = Nil, mArgss: List[List[Tree]] = Nil)
58     (implicit ctx: Context): Tree =
59     tree match {
60       case Apply(qual, mArgs) =>
61         rewire(qual, mtArgs, mArgs :: mArgss)
62       case TypeApply(qual, mtArgs2) =>
63         assert(mtArgs == Nil)
64         rewire(qual, mtArgs2, mArgss)
65       case sel @ Select(qual, _) =>
66         val origMeth = sel.symbol
67         val origCls = origMeth.enclosingClass
68         val ctParams = origCls.typeParams
69         val extensionMeth = extensionMethod(origMeth)
70 
71         if (!ctParams.isEmpty) {
72           evalOnce(qual) { ev =>
73             val ctArgs = ctParams.map(tparam =>
74               TypeTree(tparam.typeRef.asSeenFrom(ev.tpe, origCls)))
75             ref(extensionMeth)
76               .appliedToTypeTrees(mtArgs ++ ctArgs)
77               .appliedTo(ev)
78               .appliedToArgss(mArgss)
79           }
80         } else {
81           ref(extensionMeth)
82             .appliedToTypeTrees(mtArgs)
83             .appliedTo(qual)
84             .appliedToArgss(mArgss)
85         }
86     }
87 
88   /** If this tree corresponds to a fully-applied value class method call, replace it
89    *  by a call to the corresponding extension method, otherwise return it as is.
90    */
91   private def rewireIfNeeded(tree: Tree)(implicit ctx: Context) = tree.tpe.widen match {
92     case tp: LambdaType =>
93       tree // The rewiring will be handled by a fully-applied parent node
94     case _ =>
95       if (isMethodWithExtension(tree.symbol))
96         rewire(tree).ensureConforms(tree.tpe)
97       else
98         tree
99   }
100 
101   override def transformSelect(tree: Select)(implicit ctx: Context): Tree =

103   override def transformTypeApply(tree: TypeApply)(implicit ctx: Context): Tree =

105   override def transformApply(tree: Apply)(implicit ctx: Context): Tree =
