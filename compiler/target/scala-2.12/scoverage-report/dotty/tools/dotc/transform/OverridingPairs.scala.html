12 /** A module that can produce a kind of iterator (`Cursor`),
13  *  which yields all pairs of overriding/overridden symbols
14  *  that are visible in some baseclass, unless there's a parent class
15  *  that already contains the same pairs.
16  *
17  *  Adapted from the 2.9 version of OverridingPairs. The 2.10 version is IMO
18  *  way too unwieldy to be maintained.
19  */
20 object OverridingPairs {
21 
22   /** The cursor class
23    *  @param base   the base class that contains the overriding pairs
24    */
25   class Cursor(base: Symbol)(implicit ctx: Context) {
26 
27     private val self = base.thisType
28 
29     /** Symbols to exclude: Here these are constructors and private locals.
30      *  But it may be refined in subclasses.
31      */
32     protected def exclude(sym: Symbol): Boolean = !sym.memberCanMatchInheritedSymbols
33 
34     /** The parents of base that are checked when deciding whether an overriding
35      *  pair has already been treated in a parent class.
36      *  This may be refined in subclasses. @see Bridges for a use case.
37      */
38     protected def parents: Array[Symbol] = base.info.parents.toArray map (_.typeSymbol)
39 
40     /** Does `sym1` match `sym2` so that it qualifies as overriding.
41      *  Types always match. Term symbols match if their membertypes
42      *  relative to <base>.this do
43      */
44     protected def matches(sym1: Symbol, sym2: Symbol): Boolean =
45       sym1.isType || self.memberInfo(sym1).matches(self.memberInfo(sym2))
46 
47     /** The symbols that can take part in an overriding pair */
48     private val decls = {
49       val decls = newScope
50       // fill `decls` with overriding shadowing overridden */
51       def fillDecls(bcs: List[Symbol], deferred: Boolean): Unit = bcs match {
52         case bc :: bcs1 =>
53           fillDecls(bcs1, deferred)
54           var e = bc.info.decls.lastEntry
55           while (e != null) {
56             if (e.sym.is(Deferred) == deferred && !exclude(e.sym))
57               decls.enter(e.sym)
58             e = e.prev
59           }
60         case nil =>
61       }
62       // first, deferred (this will need to change if we change lookup rules!
63       fillDecls(base.info.baseClasses, deferred = true)
64       // then, concrete.
65       fillDecls(base.info.baseClasses, deferred = false)
66       decls
67     }
68 
69     private val subParents = {
70       val subParents = newMutableSymbolMap[BitSet]
71       for (bc <- base.info.baseClasses)
72         subParents(bc) = BitSet(parents.indices.filter(parents(_).derivesFrom(bc)): _*)
73       subParents
74     }
75 
76     private def hasCommonParentAsSubclass(cls1: Symbol, cls2: Symbol): Boolean =
77       (subParents(cls1) intersect subParents(cls2)).nonEmpty
78 
79     /** The scope entries that have already been visited as overridden
80      *  (maybe excluded because of hasCommonParentAsSubclass).
81      *  These will not appear as overriding
82      */
83     private val visited = new mutable.HashSet[Symbol]
84 
85     /** The current entry candidate for overriding
86      */
87     private[this] var curEntry = decls.lastEntry
88 
89     /** The current entry candidate for overridden */
90     private[this] var nextEntry = curEntry
91 
92     /** The current candidate symbol for overriding */
93     var overriding: Symbol = _
94 
95     /** If not null: The symbol overridden by overriding */
96     var overridden: Symbol = _
97 
98     //@M: note that next is called once during object initialization
99     final def hasNext: Boolean = nextEntry ne null
100 
101     /**  @post
102      *     curEntry   = the next candidate that may override something else
103      *     nextEntry  = curEntry
104      *     overriding = curEntry.sym
105      */
106     private def nextOverriding(): Unit = {
107       @tailrec def loop(): Unit =
108         if (curEntry ne null) {
109           overriding = curEntry.sym
110           if (visited.contains(overriding)) {
111             curEntry = curEntry.prev
112             loop()
113           }
114         }
115       loop()
116       nextEntry = curEntry
117     }
118 
119     /** @post
120      *    hasNext    = there is another overriding pair
121      *    overriding = overriding member of the pair, provided hasNext is true
122      *    overridden = overridden member of the pair, provided hasNext is true
123      */
124     @tailrec final def next(): Unit =
125       if (nextEntry ne null) {
126         nextEntry = decls.lookupNextEntry(nextEntry)
127         if (nextEntry ne null) {
128           try {
129             overridden = nextEntry.sym
130             if (overriding.owner != overridden.owner && matches(overriding, overridden)) {
131               visited += overridden
132               if (!hasCommonParentAsSubclass(overriding.owner, overridden.owner)) return
133             }
134           }
135           catch {
136             case ex: CyclicReference =>
137               // See neg/i1750a for an example where a cyclic error can arise.
138               // The root cause in this example is an illegal "override" of an inner trait
139               ctx.error(cyclicErrorMsg(ex), base.pos)
140           }
141         } else {
142           curEntry = curEntry.prev
143           nextOverriding()
144         }
145         next()
146       }
147 
148     nextOverriding()
149     next()
150   }
151 }
