29   def transformer(implicit ctx: Context): Tree => Tree = {
30     case t @ If(cond, thenp, elsep) =>
31       val newTypeTested = collectTypeTests(cond)
32       val nullTested = collectNullTests(cond).toSet
33       val testedMap = newTypeTested.foldRight[Map[Symbol, List[Type]]](Map.empty) { case (x, y) =>
34         y + ((x._1, x._2 :: y.getOrElse(x._1, Nil)))
35       }
36       val dropGoodCastsInStats = new TreeMap() {
37         override def transform(tree: Tree)(implicit ctx: Context): Tree = {
38           def applyCondition(fun: Select, tree: Tree, const: Constant): Boolean =
39             const.tag == Constants.NullTag &&
40             (fun.symbol == defn.Object_eq || fun.symbol == defn.Object_ne) &&
41             (nullTested.contains(tree.symbol))
42 
43           def applyBody(fun: Select): Tree =
44             if (fun.symbol == defn.Object_eq) Literal(Constant(false))
45             else Literal(Constant(true))
46 
47           super.transform(tree) match {
48             case t: Block =>
49               val nstats = t.stats.filterConserve({
50                 case TypeApply(fun @ Select(rec, _), List(tp))
51                   if fun.symbol == defn.Any_asInstanceOf =>
52                     !testedMap.getOrElse(rec.symbol, Nil).exists(x => x <:< tp.tpe)
53                 case _ => true
54               })
55               if (nstats eq t.stats) t
56               else Block(nstats, t.expr)
57             case Apply(fun @ Select(lhs, _), List(Literal(const))) if applyCondition(fun, lhs, const) =>
58               applyBody(fun)
59             case Apply(fun @ Select(Literal(const), _), List(rhs)) if applyCondition(fun, rhs, const) =>
60               applyBody(fun)
61             case t => t
62           }
63         }
64       }
65       val nthenp = dropGoodCastsInStats.transform(thenp)
66 
67       cpy.If(t)(thenp = nthenp, elsep = elsep)
68     case t => t
69   }

71   def collectTypeTests(t: Tree)(implicit ctx: Context): List[(Symbol, Type)] = {
72     def recur(t: Tree): List[(Symbol, Type)] =
73       t match {
