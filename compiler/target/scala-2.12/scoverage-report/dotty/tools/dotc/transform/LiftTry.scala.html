15 /** Lifts try's that might be executed on non-empty expression stacks
16  *  to their own methods. I.e.
17  *
18  *      try body catch handler
19  *
20  *  is lifted to
21  *
22  *      { def liftedTree$n() = try body catch handler; liftedTree$n() }
23  */
24 class LiftTry extends MiniPhase with IdentityDenotTransformer { thisPhase =>
25   import ast.tpd._
26 
27   /** the following two members override abstract members in Transform */
28   val phaseName: String = "liftTry"
29 
30   private var NeedLift: Store.Location[Boolean] = _
31   private def needLift(implicit ctx: Context): Boolean = ctx.store(NeedLift)
32 
33   override def initContext(ctx: FreshContext) =
34     NeedLift = ctx.addLocation(false)
35 
36   private def liftingCtx(p: Boolean)(implicit ctx: Context) =
37     if (needLift == p) ctx else ctx.fresh.updateStore(NeedLift, p)
38 
39   override def prepareForApply(tree: Apply)(implicit ctx: Context) =
40     if (tree.fun.symbol.is(Label)) ctx
41     else liftingCtx(true)
42 
43   override def prepareForValDef(tree: ValDef)(implicit ctx: Context) =
44     if (!tree.symbol.exists  ||
45         tree.symbol.isSelfSym ||
46         tree.symbol.owner == ctx.owner.enclosingMethod) ctx
47     else liftingCtx(true)
48 
49   override def prepareForAssign(tree: Assign)(implicit ctx: Context) =
50     if (tree.lhs.symbol.maybeOwner == ctx.owner.enclosingMethod) ctx
51     else liftingCtx(true)
52 
53   override def prepareForReturn(tree: Return)(implicit ctx: Context) =
54     if (!isNonLocalReturn(tree)) ctx
55     else liftingCtx(true)
56 
57   override def prepareForTemplate(tree: Template)(implicit ctx: Context) =
58     liftingCtx(false)
59 
60   override def transformTry(tree: Try)(implicit ctx: Context): Tree =
61     if (needLift) {
62       ctx.debuglog(i"lifting tree at ${tree.pos}, current owner = ${ctx.owner}")
63       val fn = ctx.newSymbol(
64         ctx.owner, LiftedTreeName.fresh(), Synthetic | Method,
65         MethodType(Nil, tree.tpe.widenIfUnstable), coord = tree.pos)
66       tree.changeOwnerAfter(ctx.owner, fn, thisPhase)
67       Block(DefDef(fn, tree) :: Nil, ref(fn).appliedToNone)
68     }
69     else tree
70 }
