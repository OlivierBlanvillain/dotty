16 /** This transform normalizes type tests and type casts,
17  *  also replacing type tests with singleton argument type with reference equality check
18  *  Any remaining type tests
19  *   - use the object methods $isInstanceOf and $asInstanceOf
20  *   - have a reference type as receiver
21  *   - can be translated directly to machine instructions
22  *
23  *
24  * Unfortunately this phase ended up being not Y-checkable unless types are erased. A cast to an ConstantType(3) or x.type
25  * cannot be rewritten before erasure.
26  */
27 object TypeTestsCasts {
28   import ast.tpd._
29 
30   def interceptTypeApply(tree: TypeApply)(implicit ctx: Context): Tree = trace(s"transforming ${tree.show}", show = true) {
31     tree.fun match {
32       case fun @ Select(expr, selector) =>
33         val sym = tree.symbol
34 
35         def isPrimitive(tp: Type) = tp.classSymbol.isPrimitiveValueClass
36 
37         def derivedTree(expr1: Tree, sym: Symbol, tp: Type) =
38           cpy.TypeApply(tree)(expr1.select(sym).withPos(expr.pos), List(TypeTree(tp)))
39 
40         def foundCls = expr.tpe.widen.classSymbol
41         // println(i"ta $tree, found = $foundCls")
42 
43         def inMatch =
44           fun.symbol == defn.Any_typeTest ||  // new scheme
45           expr.symbol.is(Case)                // old scheme
46 
47         def transformIsInstanceOf(expr:Tree, testType: Type, flagUnrelated: Boolean): Tree = {
48           def testCls = testType.classSymbol
49 
50           def unreachable(why: => String) =
51             if (flagUnrelated)
52               if (inMatch) ctx.error(em"this case is unreachable since $why", expr.pos)
53               else ctx.warning(em"this will always yield false since $why", expr.pos)
54 
55           /** Are `foundCls` and `testCls` classes that allow checks
56            *  whether a test would be always false?
57            */
58           def isCheckable =
59             foundCls.isClass && testCls.isClass &&
60             !(testCls.isPrimitiveValueClass && !foundCls.isPrimitiveValueClass) &&
61                // if `test` is primitive but `found` is not, we might have a case like
62                // found = java.lang.Integer, test = Int, which could be true
63                // (not sure why that is so, but scalac behaves the same way)
64             !isDerivedValueClass(foundCls) && !isDerivedValueClass(testCls)
65                // we don't have the logic to handle derived value classes
66 
67           /** Check whether a runtime test that a value of `foundCls` can be a `testCls`
68            *  can be true in some cases. Issure a warning or an error if that's not the case.
69            */
70           def checkSensical: Boolean =
71             if (!isCheckable) true
72             else if (foundCls.isPrimitiveValueClass && !testCls.isPrimitiveValueClass) {
73                 ctx.error("cannot test if value types are references", tree.pos)
74                 false
75               }
76             else if (!foundCls.derivesFrom(testCls)) {
77               if (foundCls.is(Final)) {
78                 unreachable(i"$foundCls is not a subclass of $testCls")
79                 false
80               }
81               else if (!testCls.derivesFrom(foundCls) &&
82                        (testCls.is(Final) ||
83                         !testCls.is(Trait) && !foundCls.is(Trait))) {
84                 unreachable(i"$foundCls and $testCls are unrelated")
85                 false
86               }
87               else true
88             }
89             else true
90 
91           if (expr.tpe <:< testType)
92             if (expr.tpe.isNotNull) {
93               ctx.warning(
94                 em"this will always yield true, since `$foundCls` is a subclass of `$testCls`",
95                 expr.pos)
96               constant(expr, Literal(Constant(true)))
97             }
98             else expr.testNotNull
99           else if (!checkSensical)
100             constant(expr, Literal(Constant(false)))
101           else if (testCls.isPrimitiveValueClass)
102             if (foundCls.isPrimitiveValueClass)
103               constant(expr, Literal(Constant(foundCls == testCls)))
104             else
105               transformIsInstanceOf(expr, defn.boxedType(testCls.typeRef), flagUnrelated)
106           else
107             derivedTree(expr, defn.Any_isInstanceOf, testType)
108         }
109 
110         def transformAsInstanceOf(testType: Type): Tree = {
111           def testCls = testType.widen.classSymbol
112           if (expr.tpe <:< testType)
113             Typed(expr, tree.args.head)
114           else if (foundCls.isPrimitiveValueClass) {
115             if (testCls.isPrimitiveValueClass) primitiveConversion(expr, testCls)
116             else derivedTree(box(expr), defn.Any_asInstanceOf, testType)
117           }
118           else if (testCls.isPrimitiveValueClass)
119             unbox(expr.ensureConforms(defn.ObjectType), testType)
120           else if (isDerivedValueClass(testCls)) {
121             expr // adaptToType in Erasure will do the necessary type adaptation
122           }
123           else
124             derivedTree(expr, defn.Any_asInstanceOf, testType)
125         }
126 
127         /** Transform isInstanceOf OrType
128          *
129          *    expr.isInstanceOf[A | B]  ~~>  expr.isInstanceOf[A] | expr.isInstanceOf[B]
130          *    expr.isInstanceOf[A & B]  ~~>  expr.isInstanceOf[A] & expr.isInstanceOf[B]
131          *
132          *  The transform happens before erasure of `testType`, thus cannot be merged
133          *  with `transformIsInstanceOf`, which depends on erased type of `testType`.
134          */
135         def transformTypeTest(expr: Tree, testType: Type, flagUnrelated: Boolean): Tree = testType.dealias match {
136           case _: SingletonType =>
137             expr.isInstance(testType).withPos(tree.pos)
138           case OrType(tp1, tp2) =>
139             evalOnce(expr) { e =>
140               transformTypeTest(e, tp1, flagUnrelated = false)
141                 .or(transformTypeTest(e, tp2, flagUnrelated = false))
142             }
143           case AndType(tp1, tp2) =>
144             evalOnce(expr) { e =>
145               transformTypeTest(e, tp1, flagUnrelated)
146                 .and(transformTypeTest(e, tp2, flagUnrelated))
147             }
148           case defn.MultiArrayOf(elem, ndims) if isUnboundedGeneric(elem) =>
149             def isArrayTest(arg: Tree) =
150               ref(defn.runtimeMethodRef(nme.isArray)).appliedTo(arg, Literal(Constant(ndims)))
151             if (ndims == 1) isArrayTest(expr)
152             else evalOnce(expr) { e =>
153               derivedTree(e, defn.Any_isInstanceOf, e.tpe)
154                 .and(isArrayTest(e))
155             }
156           case _ =>
157             transformIsInstanceOf(expr, erasure(testType), flagUnrelated)
158         }
159 
160         if (sym.isTypeTest)
161           transformTypeTest(expr, tree.args.head.tpe, flagUnrelated = true)
162         else if (sym eq defn.Any_asInstanceOf)
163           transformAsInstanceOf(erasure(tree.args.head.tpe))
164         else tree
165 
166       case _ =>
167         tree
168     }
169   }
170 }
