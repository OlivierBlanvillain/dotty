20   override def transformThis(tree: This)(implicit ctx: Context): Tree =

24     }

27   override def transformIdent(tree: tpd.Ident)(implicit ctx: Context): tpd.Tree = {
28     if (ctx.owner.enclosingMethod.is(JavaStatic)) {

31           ref(thiz.cls.sourceModule).select(tree.symbol)
32         case TermRef(thiz: ThisType, _) =>
33           assert(tree.symbol.is(Flags.JavaStatic) || thiz.cls.is(JavaDefined))
34           tree
35         case _ => tree
36       }
37     }
38     else tree
39   }
