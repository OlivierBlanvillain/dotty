23 /** This phase adds super accessors and method overrides where
24  *  linearization differs from Java's rule for default methods in interfaces.
25  *  In particular:
26  *
27  *        For every trait M directly implemented by the class (see SymUtils.mixin), in
28  *        reverse linearization order, add the following definitions to C:
29  *
30  *          3.1 (done in `superAccessors`) For every superAccessor
31  *              `<mods> def super$f[Ts](ps1)...(psN): U` in M:
32  *
33  *                <mods> def super$f[Ts](ps1)...(psN): U = super[S].f[Ts](ps1)...(psN)
34  *
35  *              where `S` is the superclass of `M` in the linearization of `C`.
36  *
37  *          3.2 (done in `methodOverrides`) For every method
38  *              `<mods> def f[Ts](ps1)...(psN): U` in M` that needs to be disambiguated:
39  *
40  *                <mods> def f[Ts](ps1)...(psN): U = super[M].f[Ts](ps1)...(psN)
41  *
42  *        A method in M needs to be disambiguated if it is concrete, not overridden in C,
43  *        and if it overrides another concrete method.
44  *
45  *  This is the first part of what was the mixin phase. It is complemented by
46  *  Mixin, which runs after erasure.
47  */
48 class ResolveSuper extends MiniPhase with IdentityDenotTransformer { thisPhase =>
49   import ast.tpd._
50 
51   override def phaseName: String = "resolveSuper"
52 
53   override def runsAfter = Set(classOf[ElimByName], // verified empirically, need to figure out what the reason is.
54                                classOf[AugmentScala2Traits])
55 
56   override def changesMembers = true // the phase adds super accessors and method forwarders
57 
58   override def transformTemplate(impl: Template)(implicit ctx: Context) = {
59     val cls = impl.symbol.owner.asClass
60     val ops = new MixinOps(cls, thisPhase)
61     import ops._
62 
63     def superAccessors(mixin: ClassSymbol): List[Tree] =
64       for (superAcc <- mixin.info.decls.filter(_.isSuperAccessor))
65         yield {
66           util.Stats.record("super accessors")
67           polyDefDef(implementation(superAcc.asTerm), forwarder(rebindSuper(cls, superAcc)))
68         }
69 
70     def methodOverrides(mixin: ClassSymbol): List[Tree] =
71       for (meth <- mixin.info.decls.toList if needsForwarder(meth))
72         yield {
73           util.Stats.record("method forwarders")
74           polyDefDef(implementation(meth.asTerm), forwarder(meth))
75         }
76 
77     val overrides = mixins.flatMap(mixin => superAccessors(mixin) ::: methodOverrides(mixin))
78 
79     cpy.Template(impl)(body = overrides ::: impl.body)
80   }
81 
82   override def transformDefDef(ddef: DefDef)(implicit ctx: Context) = {
83     val meth = ddef.symbol.asTerm
84     if (meth.isSuperAccessor && !meth.is(Deferred)) {
85       assert(ddef.rhs.isEmpty)
86       val cls = meth.owner.asClass
87       val ops = new MixinOps(cls, thisPhase)
88       import ops._
89       polyDefDef(meth, forwarder(rebindSuper(cls, meth)))
90     }
91     else ddef
92   }
93 
94   private val PrivateOrAccessorOrDeferred = Private | Accessor | Deferred
95 }

97 object ResolveSuper {
98   /** Returns the symbol that is accessed by a super-accessor in a mixin composition.
99    *
100    *  @param base       The class in which everything is mixed together
101    *  @param acc        The symbol statically referred to by the superaccessor in the trait
102    */
103   def rebindSuper(base: Symbol, acc: Symbol)(implicit ctx: Context): Symbol = {
104     var bcs = base.info.baseClasses.dropWhile(acc.owner != _).tail
105     var sym: Symbol = NoSymbol
106     val SuperAccessorName(memberName) = acc.name.unexpandedName
107     ctx.debuglog(i"starting rebindsuper from $base of ${acc.showLocated}: ${acc.info} in $bcs, name = $memberName")
108     while (bcs.nonEmpty && sym == NoSymbol) {
109       val other = bcs.head.info.nonPrivateDecl(memberName)
110       if (ctx.settings.Ydebug.value)
111         ctx.log(i"rebindsuper ${bcs.head} $other deferred = ${other.symbol.is(Deferred)}")
112       sym = other.matchingDenotation(base.thisType, base.thisType.memberInfo(acc)).symbol
113       bcs = bcs.tail
114     }
115     assert(sym.exists)
116     sym
117   }
