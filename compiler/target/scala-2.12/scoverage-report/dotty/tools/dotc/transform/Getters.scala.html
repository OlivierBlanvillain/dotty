17 /** Performs the following rewritings for fields of a class:
18  *
19  *    <mods> val x: T = e
20  *      -->  <mods> <stable> <accessor> def x: T = e
21  *    <mods> var x: T = e
22  *      -->  <mods> <accessor> def x: T = e
23  *
24  *    <mods> val x: T
25  *      -->  <mods> <stable> <accessor> def x: T
26  *
27  *    <mods> lazy val x: T = e
28  *      -->  <mods> <accessor> lazy def x: T =e
29  *
30  *    <mods> var x: T
31  *      -->  <mods> <accessor> def x: T
32  *
33  *    <mods> non-static <module> val x$ = e
34  *      -->  <mods> <module> <accessor> def x$ = e
35  *
36  *  Omitted from the rewritings are
37  *
38  *   - private[this] fields in classes (excluding traits, value classes)
39  *   - fields generated for static modules (TODO: needed?)
40  *   - parameters, static fields, and fields coming from Java
41  *
42  *  Furthermore, assignments to mutable vars are replaced by setter calls
43  *
44  *     p.x = e
45  *      -->  p.x_=(e)
46  *
47  *  No fields are generated yet. This is done later in phase Memoize.
48  */

54   override def transformSym(d: SymDenotation)(implicit ctx: Context): SymDenotation = {
55     def noGetterNeeded =
56       d.is(NoGetterNeeded) ||
57       d.initial.asInstanceOf[SymDenotation].is(PrivateLocal) && !d.owner.is(Trait) && !isDerivedValueClass(d.owner) && !d.is(Flags.Lazy) ||
58       d.is(Module) && d.isStatic ||
59       d.hasAnnotation(defn.ScalaStaticAnnot) ||
60       d.isSelfSym
61     if (d.isTerm && (d.is(Lazy) || d.owner.isClass) && d.info.isValueType && !noGetterNeeded) {
62       val maybeStable = if (d.isStable) Stable else EmptyFlags
63       d.copySymDenotation(
64         initFlags = d.flags | maybeStable | AccessorCreationFlags,
65         info = ExprType(d.info))
66     }
67     else d
68   }
69   private val NoGetterNeeded = Method | Param | JavaDefined | JavaStatic
70 
71   override def transformValDef(tree: ValDef)(implicit ctx: Context): Tree =
72     if (tree.symbol is Method) DefDef(tree.symbol.asTerm, tree.rhs).withPos(tree.pos) else tree
73 
74   override def transformAssign(tree: Assign)(implicit ctx: Context): Tree =
75     if (tree.lhs.symbol is Method) tree.lhs.becomes(tree.rhs).withPos(tree.pos) else tree
