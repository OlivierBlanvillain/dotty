59   def syntheticMethods(clazz: ClassSymbol)(implicit ctx: Context): List[Tree] = {
60     val clazzType = clazz.appliedRef
61     lazy val accessors =
62       if (isDerivedValueClass(clazz)) clazz.paramAccessors
63       else clazz.caseAccessors

65     val symbolsToSynthesize: List[Symbol] =
66       if (clazz.is(Case)) {
67         if (clazz.is(Module)) caseModuleSymbols
68         else caseSymbols
69       }

79     def syntheticDef(sym: Symbol): Tree = {
80       val synthetic = sym.copy(
81         owner = clazz,
82         flags = sym.flags &~ Deferred | Synthetic | Override,
83         coord = clazz.coord).enteredAfter(thisPhase).asTerm
84 
85       def forwardToRuntime(vrefss: List[List[Tree]]): Tree =
86         ref(defn.runtimeMethodRef("_" + sym.name.toString)).appliedToArgs(This(clazz) :: vrefss.head)
87 
88       def ownName(vrefss: List[List[Tree]]): Tree =
89         Literal(Constant(clazz.name.stripModuleClassSuffix.toString))
90 
91       def syntheticRHS(implicit ctx: Context): List[List[Tree]] => Tree = synthetic.name match {
92         case nme.hashCode_ if isDerivedValueClass(clazz) => vrefss => valueHashCodeBody
93         case nme.hashCode_ => vrefss => caseHashCodeBody
94         case nme.toString_ => if (clazz.is(ModuleClass)) ownName else forwardToRuntime
95         case nme.equals_ => vrefss => equalsBody(vrefss.head.head)
96         case nme.canEqual_ => vrefss => canEqualBody(vrefss.head.head)
97         case nme.productArity => vrefss => Literal(Constant(accessors.length))
98         case nme.productPrefix => ownName
99         case nme.productElement => vrefss => productElementBody(accessors.length, vrefss.head.head)
100       }
101       ctx.log(s"adding $synthetic to $clazz at ${ctx.phase}")
102       DefDef(synthetic, syntheticRHS(ctx.withOwner(synthetic))).withPos(ctx.owner.pos.focus)
103     }

105     /** The class
106      *
107      *  ```
108      *  case class C(x: T, y: T)
109      *  ```
110      *
111      *  gets the `productElement` method:
112      *
113      *  ```
114      *  def productElement(index: Int): Any = index match {
115      *    case 0 => this._1
116      *    case 1 => this._2
117      *    case _ => throw new IndexOutOfBoundsException(index.toString)
118      *  }
119      *  ```
120      */
121     def productElementBody(arity: Int, index: Tree)(implicit ctx: Context): Tree = {
122       val ioob = defn.IndexOutOfBoundsException.typeRef
123       // Second constructor of ioob that takes a String argument
124       def filterStringConstructor(s: Symbol): Boolean = s.info match {
125         case m: MethodType if s.isConstructor => m.paramInfos == List(defn.StringType)
126         case _ => false
127       }
128       val constructor = ioob.typeSymbol.info.decls.find(filterStringConstructor _).asTerm
129       val stringIndex = Apply(Select(index, nme.toString_), Nil)
130       val error = Throw(New(ioob, constructor, List(stringIndex)))
131 
132       // case _ => throw new IndexOutOfBoundsException(i.toString)
133       val defaultCase = CaseDef(Underscore(defn.IntType), EmptyTree, error)
134 
135       // case N => _${N + 1}
136       val cases = 0.until(arity).map { i =>
137         CaseDef(Literal(Constant(i)), EmptyTree, Select(This(clazz), nme.selectorName(i)))
138       }
139 
140       Match(index, (cases :+ defaultCase).toList)
141     }

143     /** The class
144      *
145      *  ```
146      *  case class C(x: T, y: U)
147      *  ```
148      *
149      *  gets the `equals` method:
150      *
151      *  ```
152      *  def equals(that: Any): Boolean =
153      *    (this eq that) || {
154      *      that match {
155      *        case x$0 @ (_: C) => this.x == this$0.x && this.y == x$0.y
156      *        case _ => false
157      *     }
158      *  ```
159      *
160      *  If `C` is a value class the initial `eq` test is omitted.
161      */
162     def equalsBody(that: Tree)(implicit ctx: Context): Tree = {
163       val thatAsClazz = ctx.newSymbol(ctx.owner, nme.x_0, Synthetic, clazzType, coord = ctx.owner.pos) // x$0
164       def wildcardAscription(tp: Type) = Typed(Underscore(tp), TypeTree(tp))
165       val pattern = Bind(thatAsClazz, wildcardAscription(clazzType)) // x$0 @ (_: C)
166       val comparisons = accessors collect { case accessor if !accessor.info.isPhantom =>
167         This(clazz).select(accessor).select(defn.Any_==).appliedTo(ref(thatAsClazz).select(accessor)) }
168       val rhs = // this.x == this$0.x && this.y == x$0.y
169         if (comparisons.isEmpty) Literal(Constant(true)) else comparisons.reduceLeft(_ and _)
170       val matchingCase = CaseDef(pattern, EmptyTree, rhs) // case x$0 @ (_: C) => this.x == this$0.x && this.y == x$0.y
171       val defaultCase = CaseDef(wildcardAscription(defn.AnyType), EmptyTree, Literal(Constant(false))) // case _ => false
172       val matchExpr = Match(that, List(matchingCase, defaultCase))
173       if (isDerivedValueClass(clazz)) matchExpr
174       else {
175         val eqCompare = This(clazz).select(defn.Object_eq).appliedTo(that.asInstance(defn.ObjectType))
176         eqCompare or matchExpr
177       }
178     }

198     /** The class
199      *
200      *  ```
201      *  package p
202      *  case class C(x: T, y: T)
203      *  ```
204      *
205      *  gets the `hashCode` method:
206      *
207      *  ```
208      *  def hashCode: Int = {
209      *    <synthetic> var acc: Int = "p.C".hashCode // constant folded
210      *    acc = Statics.mix(acc, x);
211      *    acc = Statics.mix(acc, Statics.this.anyHash(y));
212      *    Statics.finalizeHash(acc, 2)
213      *  }
214      *  ```
215      */
216     def caseHashCodeBody(implicit ctx: Context): Tree = {
217       val seed = clazz.fullName.toString.hashCode
218       if (accessors.nonEmpty) {
219         val acc = ctx.newSymbol(ctx.owner, "acc".toTermName, Mutable | Synthetic, defn.IntType, coord = ctx.owner.pos)
220         val accDef = ValDef(acc, Literal(Constant(seed)))
221         val mixes = for (accessor <- accessors) yield
222           Assign(ref(acc), ref(defn.staticsMethod("mix")).appliedTo(ref(acc), hashImpl(accessor)))
223         val finish = ref(defn.staticsMethod("finalizeHash")).appliedTo(ref(acc), Literal(Constant(accessors.size)))
224         Block(accDef :: mixes, finish)
225       } else {
226         // Pre-compute the hash code
227         val hash = scala.runtime.Statics.finalizeHash(seed, 0)
228         Literal(Constant(hash))
229       }
230     }

232     /** The `hashCode` implementation for given symbol `sym`. */
233     def hashImpl(sym: Symbol)(implicit ctx: Context): Tree =
234       defn.scalaClassName(sym.info.finalResultType) match {
235         case tpnme.Unit | tpnme.Null               => Literal(Constant(0))
236         case tpnme.Boolean                         => If(ref(sym), Literal(Constant(1231)), Literal(Constant(1237)))
237         case tpnme.Int                             => ref(sym)
238         case tpnme.Short | tpnme.Byte | tpnme.Char => ref(sym).select(nme.toInt)
239         case tpnme.Long                            => ref(defn.staticsMethod("longHash")).appliedTo(ref(sym))
240         case tpnme.Double                          => ref(defn.staticsMethod("doubleHash")).appliedTo(ref(sym))
241         case tpnme.Float                           => ref(defn.staticsMethod("floatHash")).appliedTo(ref(sym))
242         case _                                     => ref(defn.staticsMethod("anyHash")).appliedTo(ref(sym))
243       }
