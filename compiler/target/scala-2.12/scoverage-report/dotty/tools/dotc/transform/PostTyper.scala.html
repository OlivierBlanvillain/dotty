19 /** A macro transform that runs immediately after typer and that performs the following functions:
20  *
21  *  (1) Add super accessors and protected accessors (@see SuperAccessors)
22  *
23  *  (2) Convert parameter fields that have the same name as a corresponding
24  *      public parameter field in a superclass to a forwarder to the superclass
25  *      field (corresponding = super class field is initialized with subclass field)
26  *      (@see ForwardParamAccessors)
27  *
28  *  (3) Add synthetic methods (@see SyntheticMethods)
29  *
30  *  (4) Check that `New` nodes can be instantiated, and that annotations are valid
31  *
32  *  (5) Convert all trees representing types to TypeTrees.
33  *
34  *  (6) Check the bounds of AppliedTypeTrees
35  *
36  *  (7) Insert `.package` for selections of package object members
37  *
38  *  (8) Replaces self references by name with `this`
39  *
40  *  (9) Adds SourceFile annotations to all top-level classes and objects
41  *
42  *  (10) Adds Child annotations to all sealed classes
43  *
44  *  (11) Minimizes `call` fields of `Inline` nodes to just point to the toplevel
45  *       class from which code was inlined.
46  *
47  *  (12) Converts GADT bounds into normal type bounds
48  *
49  *  The reason for making this a macro transform is that some functions (in particular
50  *  super and protected accessors and instantiation checks) are naturally top-down and
51  *  don't lend themselves to the bottom-up approach of a mini phase. The other two functions
52  *  (forwarding param accessors and synthetic methods) only apply to templates and fit
53  *  mini-phase or subfunction of a macro phase equally well. But taken by themselves
54  *  they do not warrant their own group of miniphases before pickling.
55  */

180     override def transform(tree: Tree)(implicit ctx: Context): Tree =
181       try tree match {
182         case tree: Ident if !tree.isType =>
183           handleMeta(tree.symbol)
184           tree.tpe match {
185             case tpe: ThisType => This(tpe.cls).withPos(tree.pos)
186             case _ => tree
187           }
188         case tree @ Select(qual, name) =>
189           handleMeta(tree.symbol)
190           if (name.isTypeName) {
191             Checking.checkRealizable(qual.tpe, qual.pos.focus)
192             super.transform(tree)
193           }
194           else
195             transformSelect(tree, Nil)
196         case tree: Super =>
197           if (ctx.owner.enclosingMethod.isInlineMethod)
198             ctx.error(SuperCallsNotAllowedInline(ctx.owner), tree.pos)
199           super.transform(tree)
200         case tree: Apply =>
201           methPart(tree) match {
202             case Select(nu: New, nme.CONSTRUCTOR) if isCheckable(nu) =>
203               // need to check instantiability here, because the type of the New itself
204               // might be a type constructor.
205               Checking.checkInstantiable(tree.tpe, nu.pos)
206               withNoCheckNews(nu :: Nil)(super.transform(tree))
207             case _ =>
208               super.transform(tree)
209           }
210         case tree: TypeApply =>
211           val tree1 @ TypeApply(fn, args) = normalizeTypeArgs(tree)
212           Checking.checkBounds(args, fn.tpe.widen.asInstanceOf[PolyType])
213           fn match {
214             case sel: Select =>
215               val args1 = transform(args)
216               val sel1 = transformSelect(sel, args1)
217               if (superAcc.isProtectedAccessor(sel1)) sel1 else cpy.TypeApply(tree1)(sel1, args1)
218             case _ =>
219               super.transform(tree1)
220           }
221         case tree @ Assign(sel: Select, _) =>
222           super.transform(superAcc.transformAssign(tree))
223         case Inlined(call, bindings, expansion) =>
224           // Leave only a call trace consisting of
225           //  - a reference to the top-level class from which the call was inlined,
226           //  - the call's position
227           // in the call field of an Inlined node.
228           // The trace has enough info to completely reconstruct positions.
229           // The minimization is done for two reasons:
230           //  1. To save space (calls might contain large inline arguments, which would otherwise
231           //     be duplicated
232           //  2. To enable correct pickling (calls can share symbols with the inlined code, which
233           //     would trigger an assertion when pickling).
234           val callTrace = Ident(call.symbol.topLevelClass.typeRef).withPos(call.pos)
235           cpy.Inlined(tree)(callTrace, transformSub(bindings), transform(expansion))
236         case tree: Template =>
237           withNoCheckNews(tree.parents.flatMap(newPart)) {
238             val templ1 = paramFwd.forwardParamAccessors(tree)
239             synthMth.addSyntheticMethods(
240                 superAcc.wrapTemplate(templ1)(
241                   super.transform(_).asInstanceOf[Template]))
242           }
243         case tree: DefDef =>
244           transformMemberDef(tree)
245           superAcc.wrapDefDef(tree)(super.transform(tree).asInstanceOf[DefDef])
246         case tree: TypeDef =>
247           transformMemberDef(tree)
248           val sym = tree.symbol
249           if (sym.isClass) {
250             // Add SourceFile annotation to top-level classes
251             if (sym.owner.is(Package) &&
252               ctx.compilationUnit.source.exists &&
253               sym != defn.SourceFileAnnot)
254               sym.addAnnotation(Annotation.makeSourceFile(ctx.compilationUnit.source.file.path))
255             tree
256           }
257           super.transform(tree)
258         case tree: MemberDef =>
259           transformMemberDef(tree)
260           super.transform(tree)
261         case tree: New if isCheckable(tree) =>
262           Checking.checkInstantiable(tree.tpe, tree.pos)
263           super.transform(tree)
264         case tree @ Annotated(annotated, annot) =>
265           cpy.Annotated(tree)(transform(annotated), transformAnnot(annot))
266         case tree: AppliedTypeTree =>
267           Checking.checkAppliedType(tree, boundsCheck = !ctx.mode.is(Mode.Pattern))
268           super.transform(tree)
269         case SingletonTypeTree(ref) =>
270           Checking.checkRealizable(ref.tpe, ref.pos.focus)
271           super.transform(tree)
272         case tree: TypeTree =>
273           tree.withType(
274             tree.tpe match {
275               case AnnotatedType(tpe, annot) => AnnotatedType(tpe, transformAnnot(annot))
276               case tpe => tpe
277             }
278           )
279         case Import(expr, selectors) =>
280           val exprTpe = expr.tpe
281           val seen = mutable.Set.empty[Name]
282           def checkIdent(ident: untpd.Ident): Unit = {
283             val name = ident.name.asTermName
284             if (name != nme.WILDCARD && !exprTpe.member(name).exists && !exprTpe.member(name.toTypeName).exists)
285               ctx.error(NotAMember(exprTpe, name, "value"), ident.pos)
286             if (seen(ident.name))
287               ctx.error(s"${ident.show} is renamed twice", ident.pos)
288             seen += ident.name
289           }
290           selectors.foreach {
291             case ident: untpd.Ident                 => checkIdent(ident)
292             case Thicket((ident: untpd.Ident) :: _) => checkIdent(ident)
293             case _                                  =>
294           }
295           super.transform(tree)
296         case Typed(Ident(nme.WILDCARD), _) =>
297           super.transform(tree)(ctx.addMode(Mode.Pattern))
298             // The added mode signals that bounds in a pattern need not
299             // conform to selector bounds. I.e. assume
300             //     type Tree[T >: Null <: Type]
301             // One is still allowed to write
302             //     case x: Tree[_]
303             // (which translates to)
304             //     case x: (_: Tree[_])
305         case tree =>
306           super.transform(tree)
307       }
308       catch {
309         case ex : AssertionError =>
310           println(i"error while transforming $tree")
311           throw ex
312       }
313   }
