24 /** Make private term members that are accessed from another class
25  *  non-private by resetting the Private flag and expanding their name.
26  *
27  *  Make private accessor in value class not-private. This is necessary to unbox
28  *  the value class when accessing it from separate compilation units
29  *
30  *  Also, make non-private any private parameter forwarders that forward to an inherited
31  *  public or protected parameter accessor with the same name as the forwarder.
32  *  This is necessary since private methods are not allowed to have the same name
33  *  as inherited public ones.
34  *
35  *  See discussion in https://github.com/lampepfl/dotty/pull/784
36  *  and https://github.com/lampepfl/dotty/issues/783
37  */
38 class ExpandPrivate extends MiniPhase with IdentityDenotTransformer { thisPhase =>
39   import ast.tpd._
40 
41   override def phaseName: String = "expandPrivate"
42 
43   // This phase moves methods around (in infotransform) so it may need to make other methods public
44   override def runsAfter: Set[Class[_ <: Phase]] = Set(classOf[MoveStatics])
45 
46   override def changesMembers = true // the phase introduces new members with mangled names
47 
48   override def checkPostCondition(tree: Tree)(implicit ctx: Context): Unit = {
49     tree match {
50       case t: DefDef =>
51         val sym = t.symbol
52         def hasWeakerAccess(other: Symbol) = {
53           // public > protected > /* default */ > private
54           if (sym.is(Private)) other.is(Private)
55           else if (sym.is(Protected)) other.is(Protected | Private)
56           else true // sym is public
57         }
58         val fail = sym.allOverriddenSymbols.findSymbol(x => !hasWeakerAccess(x))
59         if (fail.exists) {
60           assert(false, i"${sym.showFullName}: ${sym.info} has weaker access than superclass method ${fail.showFullName}: ${fail.info}")
61         }
62       case _ =>
63     }
64   }
65 
66   private def isVCPrivateParamAccessor(d: SymDenotation)(implicit ctx: Context) =
67     d.isTerm && d.is(PrivateParamAccessor) && isDerivedValueClass(d.owner)
68 
69   /** Make private terms accessed from different classes non-private.
70    *  Note: this happens also for accesses between class and linked module class.
71    *  If we change the scheme at one point to make static module class computations
72    *  static members of the companion class, we should tighten the condition below.
73    */
74   private def ensurePrivateAccessible(d: SymDenotation)(implicit ctx: Context) =
75     if (isVCPrivateParamAccessor(d))
76       d.ensureNotPrivate.installAfter(thisPhase)
77     else if (d.is(PrivateTerm) && !d.owner.is(Package) && d.owner != ctx.owner.lexicallyEnclosingClass) {
78       // Paths `p1` and `p2` are similar if they have a common suffix that follows
79       // possibly different directory paths. That is, their common suffix extends
80       // in both cases either to the start of the path or to a file separator character.
81       def isSimilar(p1: String, p2: String): Boolean = {
82         var i = p1.length - 1
83         var j = p2.length - 1
84         while (i >= 0 && j >= 0 && p1(i) == p2(j) && p1(i) != separatorChar) {
85           i -= 1
86           j -= 1
87         }
88         (i < 0 || p1(i) == separatorChar) &&
89         (j < 0 || p1(j) == separatorChar)
90       }
91 
92       assert(d.symbol.sourceFile != null &&
93              isSimilar(d.symbol.sourceFile.path, ctx.owner.sourceFile.path),
94           s"private ${d.symbol.showLocated} in ${d.symbol.sourceFile} accessed from ${ctx.owner.showLocated} in ${ctx.owner.sourceFile}")
95       d.ensureNotPrivate.installAfter(thisPhase)
96     }
97 
98   override def transformIdent(tree: Ident)(implicit ctx: Context) = {
99     ensurePrivateAccessible(tree.symbol)
100     tree
101   }
102 
103   override def transformSelect(tree: Select)(implicit ctx: Context) = {
104     ensurePrivateAccessible(tree.symbol)
105     tree
106   }
107 
108   override def transformDefDef(tree: DefDef)(implicit ctx: Context) = {
109     val sym = tree.symbol
110     tree.rhs match {
111       case Apply(sel @ Select(_: Super, _), _)
112       if sym.is(PrivateParamAccessor) && sel.symbol.is(ParamAccessor) && sym.name == sel.symbol.name =>
113         sym.ensureNotPrivate.installAfter(thisPhase)
114       case _ =>
115         if (isVCPrivateParamAccessor(sym))
116           sym.ensureNotPrivate.installAfter(thisPhase)
117     }
118     tree
119   }
