44   def transform(ref: SingleDenotation)(implicit ctx: Context): SingleDenotation = ref match {
45     case ref: SymDenotation =>
46       def isCompacted(sym: Symbol) =
47         sym.isAnonymousFunction && {
48           sym.info(ctx.withPhase(ctx.phase.next)) match {
49             case MethodType(nme.ALLARGS :: Nil) => true
50             case _                              => false
51           }
52         }
53 
54       assert(ctx.phase == this, s"transforming $ref at ${ctx.phase}")
55       if (ref.symbol eq defn.ObjectClass) {
56         // After erasure, all former Any members are now Object members
57         val ClassInfo(pre, _, ps, decls, selfInfo) = ref.info
58         val extendedScope = decls.cloneScope
59         for (decl <- defn.AnyClass.classInfo.decls)
60           if (!decl.isConstructor) extendedScope.enter(decl)
61         ref.copySymDenotation(
62           info = transformInfo(ref.symbol,
63               ClassInfo(pre, defn.ObjectClass, ps, extendedScope, selfInfo))
64         )
65       }
66       else {
67         val oldSymbol = ref.symbol
68         val newSymbol =
69           if ((oldSymbol.owner eq defn.AnyClass) && oldSymbol.isConstructor)
70             defn.ObjectClass.primaryConstructor
71         else oldSymbol
72         val oldOwner = ref.owner
73         val newOwner = if (oldOwner eq defn.AnyClass) defn.ObjectClass else oldOwner
74         val oldInfo = ref.info
75         val newInfo = transformInfo(oldSymbol, oldInfo)
76         val oldFlags = ref.flags
77         val newFlags =
78           if (oldSymbol.is(Flags.TermParam) && isCompacted(oldSymbol.owner)) oldFlags &~ Flags.Param
79           else oldFlags &~ Flags.HasDefaultParams // HasDefaultParams needs to be dropped because overriding might become overloading
80 
81         // TODO: define derivedSymDenotation?
82         if ((oldSymbol eq newSymbol) && (oldOwner eq newOwner) && (oldInfo eq newInfo) && (oldFlags == newFlags)) ref
83         else {
84           assert(!ref.is(Flags.PackageClass), s"trans $ref @ ${ctx.phase} oldOwner = $oldOwner, newOwner = $newOwner, oldInfo = $oldInfo, newInfo = $newInfo ${oldOwner eq newOwner} ${oldInfo eq newInfo}")
85           ref.copySymDenotation(symbol = newSymbol, owner = newOwner, initFlags = newFlags, info = newInfo)
86         }
87       }
88     case ref: JointRefDenotation =>
89       new UniqueRefDenotation(
90         ref.symbol, transformInfo(ref.symbol, ref.symbol.info), ref.validFor)
91     case _ =>
92       ref.derivedSingleDenotation(ref.symbol, transformInfo(ref.symbol, ref.symbol.info))
93   }

202     def unbox(tree: Tree, pt: Type)(implicit ctx: Context): Tree = trace(i"unboxing ${tree.showSummary}: ${tree.tpe} as a $pt") {
203       pt match {
204         case ErasedValueType(tycon, underlying) =>
205           def unboxedTree(t: Tree) =
206             adaptToType(t, tycon)
207             .select(valueClassUnbox(tycon.symbol.asClass))
208             .appliedToNone
209 
210           // Null unboxing needs to be treated separately since we cannot call a method on null.
211           // "Unboxing" null to underlying is equivalent to doing null.asInstanceOf[underlying]
212           // See tests/pos/valueclasses/nullAsInstanceOfVC.scala for cases where this might happen.
213           val tree1 =
214             if (tree.tpe isRef defn.NullClass)
215               adaptToType(tree, underlying)
216             else if (wasPhantom(underlying))
217               PhantomErasure.erasedParameterRef
218             else if (!(tree.tpe <:< tycon)) {
219               assert(!(tree.tpe.typeSymbol.isPrimitiveValueClass))

221               val unboxedNull = adaptToType(nullTree, underlying)

224                 If(t.select(defn.Object_eq).appliedTo(nullTree),
225                   unboxedNull,

227               }
228             } else unboxedTree(tree)

234           else {

241     /** Generate a synthetic cast operation from tree.tpe to pt.
242      *  Does not do any boxing/unboxing (this is handled upstream).
243      *  Casts from and to ErasedValueType are special, see the explanation
244      *  in ExtensionMethods#transform.

246     def cast(tree: Tree, pt: Type)(implicit ctx: Context): Tree = trace(i"cast ${tree.tpe.widen} --> $pt", show = true) {
247 
248       def wrap(tycon: TypeRef) =
249         ref(u2evt(tycon.typeSymbol.asClass)).appliedTo(tree)
250       def unwrap(tycon: TypeRef) =
251         ref(evt2u(tycon.typeSymbol.asClass)).appliedTo(tree)
252 
253       assert(!pt.isInstanceOf[SingletonType], pt)
254       if (pt isRef defn.UnitClass) unbox(tree, pt)
255       else (tree.tpe.widen, pt) match {
256         case (JavaArrayType(treeElem), JavaArrayType(ptElem))
257         if treeElem.widen.isPrimitiveValueType && !ptElem.isPrimitiveValueType =>
258           // See SI-2386 for one example of when this might be necessary.
259           cast(ref(defn.runtimeMethodRef(nme.toObjectArray)).appliedTo(tree), pt)
260 
261         // When casting between two EVTs, we need to check which one underlies the other to determine
262         // wheter u2evt or evt2u should be used.
263         case (tp1 @ ErasedValueType(tycon1, underlying1), tp2 @ ErasedValueType(tycon2, underlying2)) =>
264           if (tp1 <:< underlying2)
265             // Cast EVT(tycon1, underlying1) to EVT(tycon2, EVT(tycon1, underlying1))
266             wrap(tycon2)
267           else {
268             assert(underlying1 <:< tp2, i"Non-sensical cast between unrelated types $tp1 and $tp2")
269             // Cast EVT(tycon1, EVT(tycon2, underlying2)) to EVT(tycon2, underlying2)
270             unwrap(tycon1)
271           }
272 
273         // When only one type is an EVT then we already know that the other one is the underlying
274         case (_, ErasedValueType(tycon2, _)) =>
275           wrap(tycon2)
276         case (ErasedValueType(tycon1, _), _) =>
277           unwrap(tycon1)
278 
279         case _ =>
280           if (pt.isPrimitiveValueType)
281             primitiveConversion(tree, pt.classSymbol)
282           else
283             tree.asInstance(pt)
284       }
285     }

287     /** Adaptation of an expression `e` to an expected type `PT`, applying the following
288      *  rewritings exhaustively as long as the type of `e` is not a subtype of `PT`.
289      *
290      *    e -> e()           if `e` appears not as the function part of an application
291      *    e -> box(e)        if `e` is of erased value type
292      *    e -> unbox(e, PT)  otherwise, if `PT` is an erased value type
293      *    e -> box(e)        if `e` is of primitive type and `PT` is not a primitive type
294      *    e -> unbox(e, PT)  if `PT` is a primitive type and `e` is not of primitive type
295      *    e -> cast(e, PT)   otherwise
296      */
297     def adaptToType(tree: Tree, pt: Type)(implicit ctx: Context): Tree =
298       if (pt.isInstanceOf[FunProto]) tree
299       else tree.tpe.widen match {
300         case MethodType(Nil) if tree.isTerm =>
301           adaptToType(tree.appliedToNone, pt)
302         case tpw =>
303           if (pt.isInstanceOf[ProtoType] || tree.tpe <:< pt)
304             tree
305           else if (tpw.isErasedValueType)
306             adaptToType(box(tree), pt)
307           else if (pt.isErasedValueType)
308             adaptToType(unbox(tree, pt), pt)
309           else if (tpw.isPrimitiveValueType && !pt.isPrimitiveValueType)
310             adaptToType(box(tree), pt)
311           else if (pt.isPrimitiveValueType && !tpw.isPrimitiveValueType)
312             adaptToType(unbox(tree, pt), pt)
313           else
314             cast(tree, pt)
315       }

580     override def typedClosure(tree: untpd.Closure, pt: Type)(implicit ctx: Context) = {
581       val xxl = defn.isXXLFunctionClass(tree.typeOpt.typeSymbol)
582       var implClosure @ Closure(_, meth, _) = super.typedClosure(tree, pt)
583       if (xxl) implClosure = cpy.Closure(implClosure)(tpt = TypeTree(defn.FunctionXXLType))
584       implClosure.tpe match {
585         case SAMType(sam) =>
586           val implType = meth.tpe.widen.asInstanceOf[MethodType]
587 
588           val implParamTypes = implType.paramInfos
589           val List(samParamTypes) = sam.info.paramInfoss
590           val implResultType = implType.resultType
591           val samResultType = sam.info.resultType
592 
593           // The following code:
594           //
595           //     val f: Function1[Int, Any] = x => ...
596           //
597           // results in the creation of a closure and a method in the typer:
598           //
599           //     def $anonfun(x: Int): Any = ...
600           //     val f: Function1[Int, Any] = closure($anonfun)
601           //
602           // Notice that `$anonfun` takes a primitive as argument, but the single abstract method
603           // of `Function1` after erasure is:
604           //
605           //     def apply(x: Object): Object
606           //
607           // which takes a reference as argument. Hence, some form of adaptation is required.
608           //
609           // If we do nothing, the LambdaMetaFactory bootstrap method will
610           // automatically do the adaptation. Unfortunately, the result does not
611           // implement the expected Scala semantics: null should be "unboxed" to
612           // the default value of the value class, but LMF will throw a
613           // NullPointerException instead. LMF is also not capable of doing
614           // adaptation for derived value classes.
615           //
616           // Thus, we need to replace the closure method by a bridge method that
617           // forwards to the original closure method with appropriate
618           // boxing/unboxing. For our example above, this would be:
619           //
620           //     def $anonfun1(x: Object): Object = $anonfun(BoxesRunTime.unboxToInt(x))
621           //     val f: Function1 = closure($anonfun1)
622           //
623           // In general, a bridge is needed when, after Erasure:
624           // - one of the parameter type of the closure method is a non-reference type,
625           //   and the corresponding type in the SAM is a reference type
626           // - or the result type of the closure method is an erased value type
627           //   and the result type in the SAM isn't
628           // However, the following exception exists: If the SAM is replaced by
629           // JFunction*mc* in [[FunctionalInterfaces]], no bridge is needed: the
630           // SAM contains default methods to handle adaptation
631           //
632           // See test cases lambda-*.scala and t8017/ for concrete examples.
633 
634           def isReferenceType(tp: Type) = !tp.isPrimitiveValueType && !tp.isErasedValueType
635 
636           if (!defn.isSpecializableFunction(implClosure.tpe.widen.classSymbol.asClass, implParamTypes, implResultType)) {
637             val paramAdaptationNeeded =
638               (implParamTypes, samParamTypes).zipped.exists((implType, samType) =>
639                 !isReferenceType(implType) && isReferenceType(samType))
640             val resultAdaptationNeeded =
641               implResultType.isErasedValueType && !samResultType.isErasedValueType
642 
643             if (paramAdaptationNeeded || resultAdaptationNeeded) {
644               val bridgeType =
645                 if (paramAdaptationNeeded) {
646                   if (resultAdaptationNeeded) sam.info
647                   else implType.derivedLambdaType(paramInfos = samParamTypes)
648                 } else implType.derivedLambdaType(resType = samResultType)
649               val bridge = ctx.newSymbol(ctx.owner, AdaptedClosureName(meth.symbol.name.asTermName), Flags.Synthetic | Flags.Method, bridgeType)
650               val bridgeCtx = ctx.withOwner(bridge)
651               Closure(bridge, bridgeParamss => {
652                 implicit val ctx = bridgeCtx
653 
654                 val List(bridgeParams) = bridgeParamss
655                 val rhs = Apply(meth, (bridgeParams, implParamTypes).zipped.map(adapt(_, _)))
656                 adapt(rhs, bridgeType.resultType)
657               }, targetType = implClosure.tpt.tpe)
658             } else implClosure
659           } else implClosure
660         case _ =>
661           implClosure
662       }
663     }
