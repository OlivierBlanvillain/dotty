51   def doVisit(tree: Tree, used: MutableSymbolMap[Int])(implicit ctx: Context): Unit = tree match {
52     case valdef: ValDef if !valdef.symbol.is(Param | Mutable | Module | Lazy) &&
53                            valdef.symbol.exists && !valdef.symbol.owner.isClass =>
54       defined += valdef.symbol

56       dropCasts(valdef.rhs) match {

60       }

63       val normalized = t.tpt.tpe.normalizedPrefix
64       val symIfExists = normalized.termSymbol
65       val b4 = used.getOrElseUpdate(symIfExists, 0)
66       used(symIfExists) = b4 + 1

69     case valdef: ValDef if valdef.symbol.exists && !valdef.symbol.owner.isClass &&
70                            !valdef.symbol.is(Param | Module | Lazy) =>
71       // TODO: handle params after constructors. Start changing public signatures by eliminating unused arguments.
72       defined += valdef.symbol

74     case valdef: ValDef => visitType(valdef.symbol.info)
75     case t: DefDef      => visitType(t.symbol.info)
76     case t: Typed       => visitType(t.tpt.tpe)
77     case t: TypeApply   => t.args.foreach(x => visitType(x.tpe))

84   def visitor(implicit ctx: Context): Tree => Unit = { tree =>
85     def crossingClassBoundaries(t: Tree): Boolean = t match {
86       case _: New      => true
87       case _: Template => true
88       case _           => false
89     }
90     // We shouldn't inline `This` nodes, which we approximate by not inlining
91     // anything across class boundaries. To do so, we visit every class a
92     // second time and record what's used in the usedInInnerClass Set.
93     if (crossingClassBoundaries(tree)) {
94       // Doing a foreachSubTree(tree) here would work, but would also
95       // be exponential for deeply nested classes. Instead we do a short
96       // circuit traversal that doesn't visit further nested classes.
97       val reVisitClass = new TreeAccumulator[Unit] {
98         def apply(u: Unit, t: Tree)(implicit ctx: Context): Unit = {
99           doVisit(t, usedInInnerClass)
100           if (!crossingClassBoundaries(t))
101             foldOver((), t)
102         }
103       }
104       reVisitClass.foldOver((), tree)
105     }
106     doVisit(tree, timesUsed)
107   }

109   def transformer(implicit ctx: Context): Tree => Tree = {
110     val valsToDrop = defined -- timesUsed.keysIterator -- timesUsedAsType.keysIterator
111     val copiesToReplaceAsDuplicates = copies.filter { x =>
112       val rhs = dropCasts(x._2)
113       rhs.isInstanceOf[Literal] || (!rhs.symbol.owner.isClass && !rhs.symbol.is(Method | Mutable))
114     } -- timesUsedAsType.keysIterator
115     // TODO: if a non-synthetic val is duplicate of a synthetic one, rename a synthetic one and drop synthetic flag?
116 
117     val copiesToReplaceAsUsedOnce =
118       timesUsed.filter(x => x._2 == 1)
119         .flatMap(x => copies.get(x._1) match {
120           case Some(tr) => List((x._1, tr))
121           case None => Nil
122         }) -- timesUsedAsType.keysIterator
123 
124     val replacements = copiesToReplaceAsDuplicates ++ copiesToReplaceAsUsedOnce -- usedInInnerClass.keysIterator
125 
126     val deepReplacer = new TreeMap() {
127       override def transform(tree: Tree)(implicit ctx: Context): Tree = {
128         def loop(tree: Tree): Tree  =
129           tree match {
130             case t: RefTree if replacements.contains(t.symbol) =>
131               loop(replacements(t.symbol))
132             case _ => tree
133           }
134         super.transform(loop(tree))
135       }
136     }
137 
138     val transformation: Tree => Tree = {
139       case t: ValDef if valsToDrop.contains(t.symbol) =>
140         // TODO: Could emit a warning for non synthetic code? This valdef is
141         // probably something users would want to remove from source...
142         simplify.println(s"Dropping definition of ${t.symbol.showFullName} as not used")
143         t.rhs.changeOwner(t.symbol, t.symbol.owner)
144       case t: ValDef if replacements.contains(t.symbol) =>
145         simplify.println(s"Dropping definition of ${t.symbol.showFullName} as an alias")
146         EmptyTree
147       case t: New =>
148         val symIfExists = t.tpt.tpe.normalizedPrefix.termSymbol
149         if (replacements.contains(symIfExists)) {
150           val newPrefix = deepReplacer.transform(replacements(symIfExists))
151           val newTpt = t.tpt.tpe match {
152             case t: NamedType =>
153               t.derivedSelect(newPrefix.tpe)
154           }
155           New(newTpt)
156         }
157         else t
158       case t: RefTree if !t.symbol.is(Method | Param | Mutable) =>
159         if (replacements.contains(t.symbol))
160           deepReplacer.transform(replacements(t.symbol)).ensureConforms(t.tpe.widen)
161         else t
162       case t: DefDef if !t.symbol.owner.isClass =>
163         if (timesUsed.getOrElse(t.symbol, 0) + timesUsedAsType.getOrElse(t.symbol, 0) != 0) t
164         else {
165           simplify.println(s"Dropping definition of ${t.symbol.showFullName} as not used")
166           EmptyTree
167         }
168       case tree => tree
169     }
170 
171     transformation
172   }

180   def readingOnlyVals(t: Tree)(implicit ctx: Context): Boolean = dropCasts(t) match {
181     case Typed(exp, _) => readingOnlyVals(exp)
182 
183     case TypeApply(fun @ Select(rec, _), List(tp)) =>
184       val isAsInstanceOf = fun.symbol == defn.Any_asInstanceOf && rec.tpe.derivesFrom(tp.tpe.classSymbol)
185       isAsInstanceOf && readingOnlyVals(rec)
186 
187     case t @ Apply(Select(rec, _), Nil) =>
188       isImmutableAccessor(t) && readingOnlyVals(rec)
189 
190     case t @ Select(rec, _) if t.symbol.is(Method) =>
191       isImmutableAccessor(t) && readingOnlyVals(rec)
192 
193     case t @ Select(qual, _) if !isEffectivelyMutable(t) =>
194       readingOnlyVals(qual)
195 
196     case t: Ident if !t.symbol.is(Mutable | Method) && !t.symbol.info.dealias.isInstanceOf[ExprType] =>
197       desugarIdent(t) match { case s: Select => readingOnlyVals(s); case _ => true }
198 
199     case t: This => true
200     // null => false, or the following fails devalify:
201     // trait I {
202     //   def foo: Any = null
203     // }
204     // object Main {
205     //   def main = {
206     //     val s: I = null
207     //     s.foo
208     //   }
209     // }
210     case Literal(Constant(null)) => false
211     case t: Literal => true
212     case _ => false
213   }
