44   /** Generate a non-local return throw with given return expression from given method.

51   private def nonLocalReturnThrow(expr: Tree, meth: Symbol)(implicit ctx: Context) =
52     Throw(
53       New(
54         defn.NonLocalReturnControlType,
55         ref(nonLocalReturnKey(meth)) :: expr.ensureConforms(defn.ObjectType) :: Nil))

57   /** Transform (body, key) to:
58    *
59    *  {
60    *    val key = new Object()
61    *    try {
62    *      body
63    *    } catch {
64    *      case ex: NonLocalReturnControl =>
65    *        if (ex.key().eq(key)) ex.value().asInstanceOf[T]
66    *        else throw ex
67    *    }
68    *  }
69    */
70   private def nonLocalReturnTry(body: Tree, key: TermSymbol, meth: Symbol)(implicit ctx: Context) = {
71     val keyDef = ValDef(key, New(defn.ObjectType, Nil))
72     val nonLocalReturnControl = defn.NonLocalReturnControlType
73     val ex = ctx.newSymbol(meth, nme.ex, EmptyFlags, nonLocalReturnControl, coord = body.pos)
74     val pat = BindTyped(ex, nonLocalReturnControl)
75     val rhs = If(
76         ref(ex).select(nme.key).appliedToNone.select(nme.eq).appliedTo(ref(key)),
77         ref(ex).select(nme.value).ensureConforms(meth.info.finalResultType),
78         Throw(ref(ex)))
79     val catches = CaseDef(pat, EmptyTree, rhs) :: Nil
80     val tryCatch = Try(body, catches, EmptyTree)
81     Block(keyDef :: Nil, tryCatch)
82   }
83 
84   override def transformDefDef(tree: DefDef)(implicit ctx: Context): Tree =
85     nonLocalReturnKeys.remove(tree.symbol) match {
86       case Some(key) => cpy.DefDef(tree)(rhs = nonLocalReturnTry(tree.rhs, key, tree.symbol))
87       case _ => tree
88     }

92     else tree
