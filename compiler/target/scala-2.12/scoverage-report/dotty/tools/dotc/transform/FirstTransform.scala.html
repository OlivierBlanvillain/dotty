28 /** The first tree transform
29  *   - ensures there are companion objects for all classes except module classes
30  *   - eliminates some kinds of trees: Imports, NamedArgs
31  *   - stubs out native methods
32  *   - eliminates self tree in Template and self symbol in ClassInfo
33  *   - collapses all type trees to trees of class TypeTree
34  *   - converts idempotent expressions with constant types
35  *   - drops branches of ifs using the rules
36  *          if (true) A else B    ==> A
37  *          if (false) A else B   ==> B
38  */

78   /** Reorder statements so that module classes always come after their companion classes, add missing companion classes */
79   private def reorderAndComplete(stats: List[Tree])(implicit ctx: Context): List[Tree] = {
80     val moduleClassDefs, singleClassDefs = mutable.Map[Name, Tree]()
81 
82     /* Returns the result of reordering stats and prepending revPrefix in reverse order to it.
83      * The result of reorder is equivalent to reorder(stats, revPrefix) = revPrefix.reverse ::: reorder(stats, Nil).
84      * This implementation is tail recursive as long as the element is not a module TypeDef.
85      */
86     def reorder(stats: List[Tree], revPrefix: List[Tree]): List[Tree] = stats match {
87       case (stat: TypeDef) :: stats1 if stat.symbol.isClass =>
88         if (stat.symbol is Flags.Module) {
89           def pushOnTop(xs: List[Tree], ys: List[Tree]): List[Tree] =
90             (ys /: xs)((ys, x) => x :: ys)
91           moduleClassDefs += (stat.name -> stat)
92           singleClassDefs -= stat.name.stripModuleClassSuffix
93           val stats1r = reorder(stats1, Nil)
94           pushOnTop(revPrefix, if (moduleClassDefs contains stat.name) stat :: stats1r else stats1r)
95         } else {
96           reorder(
97             stats1,
98             moduleClassDefs remove stat.name.moduleClassName match {
99               case Some(mcdef) =>
100                 mcdef :: stat :: revPrefix
101               case None =>
102                 singleClassDefs += (stat.name -> stat)
103                 stat :: revPrefix
104             }
105           )
106         }
107       case stat :: stats1 => reorder(stats1, stat :: revPrefix)
108       case Nil => revPrefix.reverse
109     }
110 
111     def registerCompanion(name: TermName, forClass: Symbol): TermSymbol = {
112       val (modul, mcCompanion, classCompanion) = newCompanion(name, forClass)
113       if (ctx.owner.isClass) modul.enteredAfter(thisPhase)
114       mcCompanion.enteredAfter(thisPhase)
115       classCompanion.enteredAfter(thisPhase)
116       modul
117     }
118 
119     def addMissingCompanions(stats: List[Tree]): List[Tree] = stats map {
120       case stat: TypeDef if (singleClassDefs contains stat.name) && needsCompanion(stat.symbol.asClass) =>
121         val objName = stat.name.toTermName
122         val nameClash = stats.exists {
123           case other: MemberDef =>
124             other.name == objName && other.symbol.info.isParameterless
125           case _ =>
126             false
127         }
128         val uniqueName = if (nameClash) AvoidClashName(objName) else objName
129         Thicket(stat :: ModuleDef(registerCompanion(uniqueName, stat.symbol), Nil).trees)
130       case stat => stat
131     }
132 
133     addMissingCompanions(reorder(stats, Nil))
134   }

225   /** Perform one of the following simplification if applicable:
226    *
227    *      true  && y   ==>  y
228    *      false && y   ==>  false
229    *      true  || y   ==>  true
230    *      false || y   ==>  y
231    */
232   private def foldCondition(tree: Apply)(implicit ctx: Context) = tree.fun match {
233     case Select(x @ Literal(Constant(c: Boolean)), op) =>
234       tree.args match {
235         case y :: Nil if y.tpe.widen.isRef(defn.BooleanClass) =>
236           op match {
237             case nme.ZAND => if (c) y else x
238             case nme.ZOR  => if (c) x else y
239             case _ => tree
240           }
241         case _ => tree
242       }
243     case _ => tree
244   }
