20 /** This phase augments Scala2 traits with implementation classes and with additional members
21  *  needed for mixin composition.
22  *  These symbols would have been added between Unpickling and Mixin in the Scala2 pipeline.
23  *  Specifcally, it adds
24  *
25  *   - an implementation class which defines a trait constructor and trait method implementations
26  *   - trait setters for vals defined in traits
27  *
28  *  Furthermore, it expands the names of all private getters and setters as well as super accessors in the trait and makes
29  *  them not-private.
30  */
31 class AugmentScala2Traits extends MiniPhase with IdentityDenotTransformer with FullParameterization { thisPhase =>
32   import ast.tpd._
33 
34   override def changesMembers = true
35 
36   override def phaseName: String = "augmentScala2Traits"
37 
38   override def rewiredTarget(referenced: Symbol, derived: Symbol)(implicit ctx: Context) = NoSymbol
39 
40   override def transformTemplate(impl: Template)(implicit ctx: Context) = {
41     val cls = impl.symbol.owner.asClass
42     for (mixin <- cls.mixins)
43       if (mixin.is(Scala2x))
44         augmentScala2Trait(mixin, cls)
45     impl
46   }
47 
48   private def augmentScala2Trait(mixin: ClassSymbol, cls: ClassSymbol)(implicit ctx: Context): Unit = {
49     if (mixin.implClass.is(Scala2x)) () // nothing to do, mixin was already augmented
50     else {
51       //println(i"creating new implclass for $mixin ${mixin.implClass}")
52       val ops = new MixinOps(cls, thisPhase)
53       import ops._
54 
55       val implClass = ctx.newCompleteClassSymbol(
56         owner = mixin.owner,
57         name = mixin.name.implClassName,
58         flags = Abstract | Scala2x | ImplClass,
59         parents = defn.ObjectType :: Nil,
60         assocFile = mixin.assocFile).enteredAfter(thisPhase)
61 
62       def implMethod(meth: TermSymbol): Symbol = {
63         val mold =
64           if (meth.isConstructor)
65             meth.copySymDenotation(
66               name = nme.TRAIT_CONSTRUCTOR,
67               info = MethodType(Nil, defn.UnitType))
68           else meth.ensureNotPrivate
69         meth.copy(
70           owner = implClass,
71           name = mold.name.asTermName,
72           flags = Method | JavaStatic,
73           info = fullyParameterizedType(mold.info, mixin))
74       }
75 
76       def traitSetter(getter: TermSymbol) =
77         getter.copy(
78           name = getter.ensureNotPrivate.name
79                   .expandedName(getter.owner, TraitSetterName)
80                   .asTermName.setterName,
81           flags = Method | Accessor,
82           info = MethodType(getter.info.resultType :: Nil, defn.UnitType))
83 
84       for (sym <- mixin.info.decls) {
85         if (needsForwarder(sym) || sym.isConstructor || sym.isGetter && sym.is(Lazy) || sym.is(Method, butNot = Deferred))
86           implClass.enter(implMethod(sym.asTerm))
87         if (sym.isGetter)
88           if (sym.is(Lazy)) {
89             if (!sym.hasAnnotation(defn.VolatileAnnot))
90               sym.addAnnotation(Annotation(defn.VolatileAnnot, Nil))
91           }
92           else if (!sym.is(Deferred) && !sym.setter.exists &&
93                    !sym.info.resultType.isInstanceOf[ConstantType])
94             traitSetter(sym.asTerm).enteredAfter(thisPhase)
95         if ((sym.is(PrivateAccessor) && !sym.name.is(ExpandedName) &&
96           (sym.isGetter || sym.isSetter)) // strangely, Scala 2 fields are also methods that have Accessor set.
97           || sym.isSuperAccessor) // scala2 superaccessors are pickled as private, but are compiled as public expanded
98           sym.ensureNotPrivate.installAfter(thisPhase)
99       }
100       ctx.log(i"Scala2x trait decls of $mixin = ${mixin.info.decls.toList.map(_.showDcl)}%\n %")
101       ctx.log(i"Scala2x impl decls of $mixin = ${implClass.info.decls.toList.map(_.showDcl)}%\n %")
102     }
103   }
