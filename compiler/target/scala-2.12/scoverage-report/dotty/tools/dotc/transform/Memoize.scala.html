20 /** Provides the implementations of all getters and setters, introducing
21  *  fields to hold the value accessed by them.

40   /* Makes sure that, after getters and constructors gen, there doesn't
41    * exist non-deferred definitions that are not implemented. */
42   override def checkPostCondition(tree: Tree)(implicit ctx: Context): Unit = {
43     def errorLackImplementation(t: Tree) = {
44       val firstPhaseId = t.symbol.initial.validFor.firstPhaseId
45       val definingPhase = ctx.withPhase(firstPhaseId).phase.prev
46       throw new AssertionError(
47         i"Non-deferred definition introduced by $definingPhase lacks implementation: $t")
48     }
49     tree match {
50       case ddef: DefDef
51         if !ddef.symbol.is(Deferred) &&
52            !ddef.symbol.isConstructor && // constructors bodies are added later at phase Constructors
53            ddef.rhs == EmptyTree =>
54         errorLackImplementation(ddef)
55       case tdef: TypeDef
56         if tdef.symbol.isClass && !tdef.symbol.is(Deferred) && tdef.rhs == EmptyTree =>
57         errorLackImplementation(tdef)
58       case _ =>
59     }
60     super.checkPostCondition(tree)
61   }
62 
63   /** Should run after mixin so that fields get generated in the
64    *  class that contains the concrete getter rather than the trait
65    *  that defines it.
66    */
67   override def runsAfter: Set[Class[_ <: Phase]] = Set(classOf[Mixin])

69   override def transformDefDef(tree: DefDef)(implicit ctx: Context): Tree = {
70     val sym = tree.symbol
71 
72     def newField = {
73       assert(!sym.hasAnnotation(defn.ScalaStaticAnnot))
74       val fieldType =
75         if (sym.isGetter) sym.info.resultType
76         else /*sym.isSetter*/ sym.info.firstParamTypes.head
77 
78       ctx.newSymbol(
79         owner = ctx.owner,
80         name  = sym.name.asTermName.fieldName,
81         flags = Private | (if (sym is Stable) EmptyFlags else Mutable),
82         info  = fieldType,
83         coord = tree.pos
84       ).withAnnotationsCarrying(sym, defn.FieldMetaAnnot)
85        .enteredAfter(thisPhase)
86     }
87 
88     def addAnnotations(denot: Denotation): Unit =
89       denot match {
90         case fieldDenot: SymDenotation if sym.annotations.nonEmpty =>
91           val cpy = fieldDenot.copySymDenotation()
92           cpy.annotations = sym.annotations
93           cpy.installAfter(thisPhase)
94         case _ => ()
95       }
96 
97     def removeAnnotations(denot: SymDenotation): Unit =
98       if (sym.annotations.nonEmpty) {
99         val cpy = sym.copySymDenotation()
100         cpy.annotations = Nil
101         cpy.installAfter(thisPhase)
102       }
103 
104     val NoFieldNeeded = Lazy | Deferred | JavaDefined | (if (ctx.settings.YnoInline.value) EmptyFlags else Inline)
105 
106     def erasedBottomTree(sym: Symbol) = {
107       if (sym eq defn.NothingClass) Throw(Literal(Constant(null)))
108       else if (sym eq defn.NullClass) Literal(Constant(null))
109       else if (sym eq defn.BoxedUnitClass) ref(defn.BoxedUnit_UNIT)
110       else if (sym eq defn.ErasedPhantomClass) ref(defn.ErasedPhantom_UNIT)
111       else {
112         assert(false, sym + " has no erased bottom tree")
113         EmptyTree
114       }
115     }
116 
117     if (sym.is(Accessor, butNot = NoFieldNeeded)) {
118       val field = sym.field.orElse(newField).asTerm
119 
120       def adaptToField(tree: Tree): Tree =
121         if (tree.isEmpty) tree else tree.ensureConforms(field.info.widen)
122 
123       def isErasableBottomField(cls: Symbol): Boolean = {
124         // TODO: For Scala.js, return false if this field is in a js.Object unless it is an ErasedPhantomClass.
125         !field.isVolatile &&
126         ((cls eq defn.NothingClass) || (cls eq defn.NullClass) || (cls eq defn.BoxedUnitClass) || (cls eq defn.ErasedPhantomClass))
127       }
128 
129       if (sym.isGetter) {
130         var rhs = tree.rhs.changeOwnerAfter(sym, field, thisPhase)
131         if (isWildcardArg(rhs)) rhs = EmptyTree
132         val fieldDef = transformFollowing(ValDef(field, adaptToField(rhs)))
133         val rhsClass = tree.tpt.tpe.widenDealias.classSymbol
134         val getterRhs =
135           if (isErasableBottomField(rhsClass)) erasedBottomTree(rhsClass)
136           else transformFollowingDeep(ref(field))(ctx.withOwner(sym))
137         val getterDef = cpy.DefDef(tree)(rhs = getterRhs)
138         addAnnotations(fieldDef.denot)
139         removeAnnotations(sym)
140         Thicket(fieldDef, getterDef)
141       } else if (sym.isSetter) {
142         if (!sym.is(ParamAccessor)) { val Literal(Constant(())) = tree.rhs } // This is intended as an assertion
143         field.setFlag(Mutable) // Necessary for vals mixed in from Scala2 traits
144         if (isErasableBottomField(tree.vparamss.head.head.tpt.tpe.classSymbol)) tree
145         else {
146           val initializer = Assign(ref(field), adaptToField(ref(tree.vparamss.head.head.symbol)))
147           val setterDef = cpy.DefDef(tree)(rhs = transformFollowingDeep(initializer)(ctx.withOwner(sym)))
148           removeAnnotations(sym)
149           setterDef
150         }
151       }
152       else tree // curiously, some accessors from Scala2 have ' ' suffixes. They count as
153                 // neither getters nor setters
154     }
155     else tree
156   }
