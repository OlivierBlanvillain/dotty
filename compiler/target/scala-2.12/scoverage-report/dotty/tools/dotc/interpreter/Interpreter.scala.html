22 /** Tree interpreter that can interpret
23  *   * Literal constants
24  *   * Calls to static methods
25  *   * New objects with explicit `new` keyword
26  *   * Quoted code
27  *
28  *  The interpreter assumes that all calls in the trees are to code that was
29  *  previously compiled and is present in the classpath of the current context.
30  */

41   /** Returns the interpreted result of interpreting the code represented by the tree.
42    *  Return Some of the result or None if some error happen during the interpretation.
43    */
44   def interpretTree[T](tree: Tree)(implicit ct: ClassTag[T]): Option[T] = {
45     try {
46       interpretTreeImpl(tree, Map.empty) match {
47         case obj: T => Some(obj)
48         case obj =>
49           // TODO upgrade to a full type tag check or something similar
50           ctx.error(s"Interpreted tree returned a result of an unexpected type. Expected ${ct.runtimeClass} but was ${obj.getClass}", tree.pos)
51           None
52       }
53     } catch {
54       case ex: StopInterpretation =>
55         ctx.error(ex.msg, ex.pos)
56         None
57     }
58   }

60   /** Returns the interpreted result of interpreting the code represented by the tree.
61    *  Returns the result of the interpreted tree.
62    *
63    *  If some error is encountered while interpreting a ctx.error is emitted and a StopInterpretation is thrown.
64    */
65   private def interpretTreeImpl(tree: Tree, env: Env): Object = {
66     ctx.debuglog(
67       s"""Interpreting:
68         |${tree.show}
69         |$env
70       """.stripMargin)
71 
72     implicit val pos: Position = tree.pos
73 
74     tree match {
75       case Quoted(quotedTree) => RawQuoted(quotedTree)
76 
77       case Literal(Constant(c)) => c.asInstanceOf[Object]
78 
79       case Apply(fn, args) if fn.symbol.isConstructor =>
80         val clazz = loadClass(fn.symbol.owner.symbol.fullName)
81         val paramClasses = paramsSig(fn.symbol)
82         val interpretedArgs = args.map(arg => interpretTreeImpl(arg, env))
83         val constructor = getConstructor(clazz, paramClasses)
84         interpreted(constructor.newInstance(interpretedArgs: _*))
85 
86       case _: RefTree if tree.symbol.isStatic =>
87         val clazz = loadClass(tree.symbol.owner.companionModule.fullName)
88         val method = getMethod(clazz, tree.symbol.name, Nil)
89         interpreted(method.invoke(null))
90 
91       case tree: Apply =>
92         val evaluatedPrefix = if (tree.symbol.isStatic) null else interpretPrefix(tree, env)
93         val clazz =
94           if (tree.symbol.isStatic) loadClass(tree.symbol.owner.companionModule.fullName)
95           else evaluatedPrefix.getClass
96         val paramClasses = paramsSig(tree.symbol)
97         val interpretedArgs = interpretArgs(tree, env)
98         val method = getMethod(clazz, tree.symbol.name, paramClasses)
99         interpreted(method.invoke(evaluatedPrefix, interpretedArgs: _*))
100 
101       case tree: Ident if env.contains(tree.symbol) =>
102         env(tree.symbol)
103 
104       case Block(stats, expr) =>
105         val env2 = stats.foldLeft(env)((acc, x) => interpretStat(x, acc))
106         interpretTreeImpl(expr, env2)
107 
108       case tree: NamedArg =>
109         interpretTreeImpl(tree.arg, env)
110 
111       case Inlined(_, bindings, expansion) =>
112         val env2 = bindings.foldLeft(env)((acc, x) => interpretStat(x, acc))
113         interpretTreeImpl(expansion, env2)
114 
115       case Typed(expr, _) =>
116         interpretTreeImpl(expr, env)
117 
118       // Getting the underlying value of a value class. The value class is evaluated as its boxed representation
119       // as values in the interpreter are `Object`s. Therefore we just get it from the enviroment as is.
120       case Select(qualifier, _)
121           if tree.symbol.owner.isValueClass && tree.symbol.is(ParamAccessor) && env.contains(qualifier.symbol) =>
122         env(qualifier.symbol)
123 
124       case SeqLiteral(elems, _) =>
125         elems.map(elem => interpretTreeImpl(elem, env))
126 
127       case _ =>
128         // TODO Add more precise descriptions of why it could not be interpreted.
129         // This should be done after the full interpreter is implemented.
130         throw new StopInterpretation(s"Could not interpret ${tree.show}. Consider extracting logic into a helper def.", tree.pos)
131     }
132   }

134   private def interpretArgs(tree: Tree, env: Env): Seq[Object] = {
135     val b = Seq.newBuilder[Object]
136     def interpretArgs(tree: Tree): Unit = tree match {
137       case Apply(fn, args) =>
138         interpretArgs(fn)
139         args.foreach(arg => b += interpretTreeImpl(arg, env))
140       case _ =>
141     }
142     interpretArgs(tree)
143     b.result()
144   }

146   private def interpretPrefix(tree: Tree, env: Env): Object = tree match {
147     case Apply(qual, _) => interpretPrefix(qual, env)
148     case TypeApply(qual, _) => interpretPrefix(qual, env)
149     case Select(qual, _) => interpretTreeImpl(qual, env)
150   }

152   /** Interprets the statement and returns the updated environment */
153   private def interpretStat(stat: Tree, env: Env): Env = stat match {
154     case tree: ValDef =>
155       val obj = interpretTreeImpl(tree.rhs, env)
156       env.updated(tree.symbol, obj)
157 
158     case _ =>
159       interpretTreeImpl(stat, env)
160       env

163   private def loadClass(name: Name)(implicit pos: Position): Class[_] = {
164     try classLoader.loadClass(name.toString)
165     catch {
166       case _: ClassNotFoundException =>
167         val msg = s"Could not find interpreted class $name in classpath"
168         throw new StopInterpretation(msg, pos)
169     }

172   private def getMethod(clazz: Class[_], name: Name, paramClasses: List[Class[_]])(implicit pos: Position): Method = {
173     try clazz.getMethod(name.toString, paramClasses: _*)
174     catch {
175       case _: NoSuchMethodException =>
176         val msg = s"Could not find interpreted method ${clazz.getCanonicalName}.$name with parameters $paramClasses"
177         throw new StopInterpretation(msg, pos)
178     }
179   }
180 
181   private def getConstructor(clazz: Class[_], paramClasses: List[Class[_]])(implicit pos: Position): Constructor[Object] = {
182     try clazz.getConstructor(paramClasses: _*).asInstanceOf[Constructor[Object]]
183     catch {
184       case _: NoSuchMethodException =>
185         val msg = s"Could not find interpreted constructor of ${clazz.getCanonicalName} with parameters $paramClasses"
186         throw new StopInterpretation(msg, pos)
187     }
188   }

190   private def interpreted[T](thunk: => T)(implicit pos: Position): T = {
191     try thunk
192     catch {
193       case ex: RuntimeException =>
194         val sw = new StringWriter()
195         sw.write("A runtime exception occurred while interpreting\n")
196         sw.write(ex.getMessage)
197         sw.write("\n")
198         ex.printStackTrace(new PrintWriter(sw))
199         sw.write("\n")
200         throw new StopInterpretation(sw.toString, pos)
201     }
202   }

204   /** List of classes of the parameters of the signature of `sym` */
205   private def paramsSig(sym: Symbol): List[Class[_]] = {
206     sym.signature.paramsSig.map { param =>
207       defn.valueTypeNameToJavaType(param) match {
208         case Some(clazz) => clazz
209         case None => classLoader.loadClass(param.toString)
210       }
211     }
