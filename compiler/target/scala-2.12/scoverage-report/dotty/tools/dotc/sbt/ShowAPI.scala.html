24 object ShowAPI {
25   private lazy val numDecls = Try { java.lang.Integer.parseInt(sys.props.get("sbt.inc.apidiff.decls").get) } getOrElse 0
26 
27   private def truncateDecls(decls: Array[Definition]): Array[Definition] = if (numDecls <= 0) decls else decls.take(numDecls)
28   private def lines(ls: Seq[String]): String = ls.mkString("\n", "\n", "\n")
29 
30   def showApi(a: SourceAPI)(implicit nesting: Int) =
31     a.packages.map(pkg => "package " + pkg.name).mkString("\n") + lines(truncateDecls(a.definitions).map(showDefinition))
32 
33   def showDefinition(d: Definition)(implicit nesting: Int): String = d match {
34     case v: Val              => showMonoDef(v, "val") + ": " + showType(v.tpe)
35     case v: Var              => showMonoDef(v, "var") + ": " + showType(v.tpe)
36     case d: Def              => showPolyDef(d, "def") + showValueParams(d.valueParameters) + ": " + showType(d.returnType)
37     case ta: TypeAlias       => showPolyDef(ta, "type") + " = " + showType(ta.tpe)
38     case td: TypeDeclaration => showPolyDef(td, "type") + showBounds(td.lowerBound, td.upperBound)
39     case cl: ClassLike       => showPolyDef(cl, showDefinitionType(cl.definitionType)) + " extends " + showTemplate(cl)
40   }
41 
42   private def showTemplate(cl: ClassLike)(implicit nesting: Int) =
43     if (nesting <= 0) "<nesting level reached>"
44     else {
45       val showSelf = if (cl.selfType.isInstanceOf[EmptyType]) "" else " self: " + showNestedType(cl.selfType) + " =>"
46 
47       cl.structure.parents.map(showNestedType).mkString("", " with ", " {") + showSelf +
48         lines(truncateDecls(cl.structure.inherited).map(d => "^inherited^ " + showNestedDefinition(d))) +
49         lines(truncateDecls(cl.structure.declared).map(showNestedDefinition)) +
50         "}"
51     }
52 
53   def showType(t: Type)(implicit nesting: Int): String = t match {
54     case st: Projection   => showType(st.prefix) + "#" + st.id
55     case st: ParameterRef => "<" + st.id + ">"
56     case st: Singleton    => showPath(st.path)
57     case st: EmptyType    => "<empty>"
58     case p: Parameterized => showType(p.baseType) + p.typeArguments.map(showType).mkString("[", ", ", "]")
59     case c: Constant      => showType(c.baseType) + "(" + c.value + ")"
60     case a: Annotated     => showAnnotations(a.annotations) + " " + showType(a.baseType)
61     case s: Structure =>
62       s.parents.map(showType).mkString(" with ") + (
63         if (nesting <= 0) "{ <nesting level reached> }"
64         else truncateDecls(s.declared).map(showNestedDefinition).mkString(" {", "\n", "}"))
65     case e: Existential =>
66       showType(e.baseType) + (
67         if (nesting <= 0) " forSome { <nesting level reached> }"
68         else e.clause.map(t => "type " + showNestedTypeParameter(t)).mkString(" forSome { ", "; ", " }"))
69     case p: Polymorphic => showType(p.baseType) + (
70       if (nesting <= 0) " [ <nesting level reached> ]"
71       else showNestedTypeParameters(p.parameters))
72   }
73 
74   private def showPath(p: Path): String = p.components.map(showPathComponent).mkString(".")
75   private def showPathComponent(pc: PathComponent) = pc match {
76     case s: Super => "super[" + showPath(s.qualifier) + "]"
77     case _: This  => "this"
78     case i: Id    => i.id
79   }
80 
81   private def space(s: String) = if (s.isEmpty) s else s + " "
82   private def showMonoDef(d: Definition, label: String)(implicit nesting: Int): String =
83     space(showAnnotations(d.annotations)) + space(showAccess(d.access)) + space(showModifiers(d.modifiers)) + space(label) + d.name
84 
85   private def showPolyDef(d: ParameterizedDefinition, label: String)(implicit nesting: Int): String =
86     showMonoDef(d, label) + showTypeParameters(d.typeParameters)
87 
88   private def showTypeParameters(tps: Seq[TypeParameter])(implicit nesting: Int): String =
89     if (tps.isEmpty) ""
90     else tps.map(showTypeParameter).mkString("[", ", ", "]")
91 
92   private def showTypeParameter(tp: TypeParameter)(implicit nesting: Int): String =
93     showAnnotations(tp.annotations) + " " + showVariance(tp.variance) + tp.id + showTypeParameters(tp.typeParameters) + " " + showBounds(tp.lowerBound, tp.upperBound)
94 
95   private def showAnnotations(as: Seq[Annotation])(implicit nesting: Int) = as.map(showAnnotation).mkString(" ")
96   private def showAnnotation(a: Annotation)(implicit nesting: Int) =
97     "@" + showType(a.base) + (
98       if (a.arguments.isEmpty) ""
99       else a.arguments.map(a => a.name + " = " + a.value).mkString("(", ", ", ")")
100     )
101 
102   private def showBounds(lower: Type, upper: Type)(implicit nesting: Int): String = ">: " + showType(lower) + " <: " + showType(upper)
103 
104   private def showValueParams(ps: Seq[ParameterList])(implicit nesting: Int): String =
105     ps.map(pl =>
106       pl.parameters.map(mp =>
107         mp.name + ": " + showParameterModifier(showType(mp.tpe), mp.modifier) + (if (mp.hasDefault) "= ..." else "")
108       ).mkString(if (pl.isImplicit) "(implicit " else "(", ", ", ")")
109     ).mkString("")
110 
111   private def showParameterModifier(base: String, pm: ParameterModifier): String = pm match {
112     case ParameterModifier.Plain    => base
113     case ParameterModifier.Repeated => base + "*"
114     case ParameterModifier.ByName   => "=> " + base
115   }
116 
117   private def showDefinitionType(d: DefinitionType) = d match {
118     case DefinitionType.Trait         => "trait"
119     case DefinitionType.ClassDef      => "class"
120     case DefinitionType.Module        => "object"
121     case DefinitionType.PackageModule => "package object"
122   }
123 
124   private def showAccess(a: Access) = a match {
125     case p: Public    => ""
126     case p: Protected => "protected" + showQualifier(p.qualifier)
127     case p: Private   => "private" + showQualifier(p.qualifier)
128   }
129 
130   private def showQualifier(q: Qualifier) = q match {
131     case _: Unqualified   => ""
132     case _: ThisQualifier => "[this]"
133     case i: IdQualifier   => "[" + i.value + "]"
134   }
135 
136   private def showModifiers(m: Modifiers) = List(
137     (m.isOverride, "override"),
138     (m.isFinal, "final"),
139     (m.isSealed, "sealed"),
140     (m.isImplicit, "implicit"),
141     (m.isAbstract, "abstract"),
142     (m.isLazy, "lazy")
143   ).collect { case (true, mod) => mod }.mkString(" ")
144 
145   private def showVariance(v: Variance) = v match {
146     case Variance.Invariant     => ""
147     case Variance.Covariant     => "+"
148     case Variance.Contravariant => "-"
149   }
150 
151   // limit nesting to prevent cycles and generally keep output from getting humongous
152   private def showNestedType(tp: Type)(implicit nesting: Int) = showType(tp)(nesting - 1)
153   private def showNestedTypeParameter(tp: TypeParameter)(implicit nesting: Int) = showTypeParameter(tp)(nesting - 1)
154   private def showNestedTypeParameters(tps: Seq[TypeParameter])(implicit nesting: Int) = showTypeParameters(tps)(nesting - 1)
155   private def showNestedDefinition(d: Definition)(implicit nesting: Int) = showDefinition(d)(nesting - 1)
