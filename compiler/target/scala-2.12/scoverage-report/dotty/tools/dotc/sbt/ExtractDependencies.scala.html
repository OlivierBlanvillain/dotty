18 /** This phase sends information on classes' dependencies to sbt via callbacks.
19  *
20  *  This is used by sbt for incremental recompilation. Briefly, when a file
21  *  changes sbt will recompile it, if its API has changed (determined by what
22  *  `ExtractAPI` sent) then sbt will determine which reverse-dependencies
23  *  (determined by what `ExtractDependencies` sent) of the API have to be
24  *  recompiled depending on what changed.
25  *
26  *  See the documentation of `ExtractDependenciesCollector`, `ExtractAPI`,
27  *  `ExtractAPICollector` and
28  *  http://www.scala-sbt.org/0.13/docs/Understanding-Recompilation.html for more
29  *  information on how sbt incremental compilation works.
30  *
31  *  The following flags affect this phase:
32  *   -Yforce-sbt-phases
33  *   -Ydump-sbt-inc
34  *
35  *  @see ExtractAPI
36  */
37 class ExtractDependencies extends Phase {
38   override def phaseName: String = "sbt-deps"
39 
40   // This phase should be run directly after `Frontend`, if it is run after
41   // `PostTyper`, some dependencies will be lost because trees get simplified.
42   // See the scripted test `constants` for an example where this matters.
43   // TODO: Add a `Phase#runsBefore` method ?
44 
45   override def run(implicit ctx: Context): Unit = {
46     val unit = ctx.compilationUnit
47     val dumpInc = ctx.settings.YdumpSbtInc.value
48     val forceRun = dumpInc || ctx.settings.YforceSbtPhases.value
49     if ((ctx.sbtCallback != null || forceRun) && !unit.isJava) {
50       val sourceFile = unit.source.file
51       val extractDeps = new ExtractDependenciesCollector
52       extractDeps.traverse(unit.tpdTree)
53 
54       if (dumpInc) {
55         val names = extractDeps.usedNames.map(_.toString).toArray[Object]
56         val deps = extractDeps.topLevelDependencies.map(_.toString).toArray[Object]
57         val inhDeps = extractDeps.topLevelInheritanceDependencies.map(_.toString).toArray[Object]
58         Arrays.sort(names)
59         Arrays.sort(deps)
60         Arrays.sort(inhDeps)
61 
62         val pw = Path(sourceFile.jpath).changeExtension("inc").toFile.printWriter()
63         try {
64           pw.println(s"// usedNames: ${names.mkString(",")}")
65           pw.println(s"// topLevelDependencies: ${deps.mkString(",")}")
66           pw.println(s"// topLevelInheritanceDependencies: ${inhDeps.mkString(",")}")
67         } finally pw.close()
68       }
69 
70       if (ctx.sbtCallback != null) {
71         extractDeps.usedNames.foreach(name =>
72           ctx.sbtCallback.usedName(sourceFile.file, name.toString))
73         extractDeps.topLevelDependencies.foreach(dep =>
74           recordDependency(sourceFile.file, dep, DependencyContext.DependencyByMemberRef))
75         extractDeps.topLevelInheritanceDependencies.foreach(dep =>
76           recordDependency(sourceFile.file, dep, DependencyContext.DependencyByInheritance))
77       }
78     }
79   }

81   /** Record that `currentSourceFile` depends on the file where `dep` was loaded from.
82    *
83    *  @param currentSourceFile  The source file of the current unit
84    *  @param dep                The dependency
85    *  @param context            Describes how `currentSourceFile` depends on `dep`
86    */
87   def recordDependency(currentSourceFile: File, dep: Symbol, context: DependencyContext)
88       (implicit ctx: Context) = {
89     val depFile = dep.associatedFile
90     if (depFile != null) {
91       if (depFile.path.endsWith(".class")) {
92         /** Transform `List(java, lang, String.class)` into `java.lang.String` */
93         def className(classSegments: List[String]) =
94           classSegments.mkString(".").stripSuffix(".class")
95         def binaryDependency(file: File, className: String) =
96           ctx.sbtCallback.binaryDependency(file, className, currentSourceFile, context)
97 
98         depFile match {
99           case ze: ZipArchive#Entry =>
100             for (zip <- ze.underlyingSource; zipFile <- Option(zip.file)) {
101               val classSegments = Path(ze.path).segments
102               binaryDependency(zipFile, className(classSegments))
103             }
104           case pf: PlainFile =>
105             val packages = dep.ownersIterator
106               .filter(x => x.is(PackageClass) && !x.isEffectiveRoot).length
107             // We can recover the fully qualified name of a classfile from
108             // its path
109             val classSegments = pf.givenPath.segments.takeRight(packages + 1)
110             binaryDependency(pf.file, className(classSegments))
111           case _ =>
112             ctx.warning(s"sbt-deps: Ignoring dependency $depFile of class ${depFile.getClass}")
113         }
114       } else if (depFile.file != currentSourceFile) {
115         ctx.sbtCallback.sourceDependency(depFile.file, currentSourceFile, context)
116       }
117     }
118   }

121 /** Extract the dependency information of a compilation unit.
122  *
123  *  To understand why we track the used names see the section "Name hashing
124  *  algorithm" in http://www.scala-sbt.org/0.13/docs/Understanding-Recompilation.html
125  *  To understand why we need to track dependencies introduced by inheritance
126  *  specially, see the subsection "Dependencies introduced by member reference and
127  *  inheritance" in the "Name hashing algorithm" section.
128  */
129 private class ExtractDependenciesCollector(implicit val ctx: Context) extends tpd.TreeTraverser {
130   import tpd._
131 
132   private[this] val _usedNames = new mutable.HashSet[Name]
133   private[this] val _topLevelDependencies = new mutable.HashSet[Symbol]
134   private[this] val _topLevelInheritanceDependencies = new mutable.HashSet[Symbol]
135 
136   /** The names used in this class, this does not include names which are only
137    *  defined and not referenced.
138    */
139   def usedNames: Set[Name] = _usedNames
140 
141   /** The set of top-level classes that the compilation unit depends on
142    *  because it refers to these classes or something defined in them.
143    *  This is always a superset of `topLevelInheritanceDependencies` by definition.
144    */
145   def topLevelDependencies: Set[Symbol] = _topLevelDependencies
146 
147   /** The set of top-level classes that the compilation unit extends or that
148    *  contain a non-top-level class that the compilaion unit extends.
149    */
150   def topLevelInheritanceDependencies: Set[Symbol] = _topLevelInheritanceDependencies
151 
152   private def addUsedName(name: Name) =
153     _usedNames += name
154 
155   private def addDependency(sym: Symbol): Unit =
156     if (!ignoreDependency(sym)) {
157       val tlClass = sym.topLevelClass
158       if (tlClass.ne(NoSymbol)) // Some synthetic type aliases like AnyRef do not belong to any class
159         _topLevelDependencies += sym.topLevelClass
160       addUsedName(sym.name)
161     }
162 
163   private def ignoreDependency(sym: Symbol) =
164     sym.eq(NoSymbol) ||
165     sym.isEffectiveRoot ||
166     sym.isAnonymousFunction ||
167     sym.isAnonymousClass
168 
169   private def addInheritanceDependency(sym: Symbol): Unit =
170     _topLevelInheritanceDependencies += sym.topLevelClass
171 
172   /** Traverse the tree of a source file and record the dependencies which
173    *  can be retrieved using `topLevelDependencies`, `topLevelInheritanceDependencies`,
174    *  and `usedNames`
175    */
176   override def traverse(tree: Tree)(implicit ctx: Context): Unit = {
177     tree match {
178       case Import(expr, selectors) =>
179         def lookupImported(name: Name) = expr.tpe.member(name).symbol
180         def addImported(name: Name) = {
181           // importing a name means importing both a term and a type (if they exist)
182           addDependency(lookupImported(name.toTermName))
183           addDependency(lookupImported(name.toTypeName))
184         }
185         selectors foreach {
186           case Ident(name) =>
187             addImported(name)
188           case Thicket(Ident(name) :: Ident(rename) :: Nil) =>
189             addImported(name)
190             if (rename ne nme.WILDCARD)
191               addUsedName(rename)
192           case _ =>
193         }
194       case Inlined(call, _, _) =>
195         // The inlined call is normally ignored by TreeTraverser but we need to
196         // record it as a dependency
197         traverse(call)
198       case t: TypeTree =>
199         usedTypeTraverser.traverse(t.tpe)
200       case ref: RefTree =>
201         addDependency(ref.symbol)
202         usedTypeTraverser.traverse(ref.tpe)
203       case t @ Template(_, parents, _, _) =>
204         t.parents.foreach(p => addInheritanceDependency(p.tpe.classSymbol))
205       case _ =>
206     }
207     traverseChildren(tree)
208   }
209 
210   /** Traverse a used type and record all the dependencies we need to keep track
211    *  of for incremental recompilation.
212    *
213    *  As a motivating example, given a type `T` defined as:
214    *
215    *    type T >: L <: H
216    *    type L <: A1
217    *    type H <: B1
218    *    class A1 extends A0
219    *    class B1 extends B0
220    *
221    *  We need to record a dependency on `T`, `L`, `H`, `A1`, `B1`. This is
222    *  necessary because the API representation that `ExtractAPI` produces for
223    *  `T` just refers to the strings "L" and "H", it does not contain their API
224    *  representation. Therefore, the name hash of `T` does not change if for
225    *  example the definition of `L` changes.
226    *
227    *  We do not need to keep track of superclasses like `A0` and `B0` because
228    *  the API representation of a class (and therefore its name hash) already
229    *  contains all necessary information on superclasses.
230    *
231    *  A natural question to ask is: Since traversing all referenced types to
232    *  find all these names is costly, why not change the API representation
233    *  produced by `ExtractAPI` to contain that information? This way the name
234    *  hash of `T` would change if any of the types it depends on change, and we
235    *  would only need to record a dependency on `T`. Unfortunately there is no
236    *  simple answer to the question "what does T depend on?" because it depends
237    *  on the prefix and `ExtractAPI` does not compute types as seen from every
238    *  possible prefix, the documentation of `ExtractAPI` explains why.
239    *
240    *  The tests in sbt `types-in-used-names-a`, `types-in-used-names-b`,
241    *  `as-seen-from-a` and `as-seen-from-b` rely on this.
242    */
243   private object usedTypeTraverser extends TypeTraverser {
244     val seen = new mutable.HashSet[Type]
245     def traverse(tp: Type): Unit = if (!seen.contains(tp)) {
246       seen += tp
247       tp match {
248         case tp: NamedType =>
249           val sym = tp.symbol
250           if (!sym.is(Package)) {
251             addDependency(sym)
252             if (!sym.isClass)
253               traverse(tp.info)
254             traverse(tp.prefix)
255           }
256         case tp: ThisType =>
257           traverse(tp.underlying)
258         case tp: ConstantType =>
259           traverse(tp.underlying)
260         case tp: ParamRef =>
261           traverse(tp.underlying)
262         case _ =>
263           traverseChildren(tp)
264       }
265     }
266   }
