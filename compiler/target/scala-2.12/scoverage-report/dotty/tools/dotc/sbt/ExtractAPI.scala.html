18 /** This phase sends a representation of the API of classes to sbt via callbacks.
19  *
20  *  This is used by sbt for incremental recompilation.
21  *
22  *  See the documentation of `ExtractAPICollector`, `ExtractDependencies`,
23  *  `ExtractDependenciesCollector` and
24  *  http://www.scala-sbt.org/0.13/docs/Understanding-Recompilation.html for more
25  *  information on incremental recompilation.
26  *
27  *  The following flags affect this phase:
28  *   -Yforce-sbt-phases
29  *   -Ydump-sbt-inc
30  *
31  *  @see ExtractDependencies
32  */
33 class ExtractAPI extends Phase {
34   override def phaseName: String = "sbt-api"
35 
36   // SuperAccessors need to be part of the API (see the scripted test
37   // `trait-super` for an example where this matters), this is only the case
38   // after `PostTyper` (unlike `ExtractDependencies`, the simplication to trees
39   // done by `PostTyper` do not affect this phase because it only cares about
40   // definitions, and `PostTyper` does not change definitions).
41   override def runsAfter = Set(classOf[transform.PostTyper])
42 
43   override def run(implicit ctx: Context): Unit = {
44     val unit = ctx.compilationUnit
45     val dumpInc = ctx.settings.YdumpSbtInc.value
46     val forceRun = dumpInc || ctx.settings.YforceSbtPhases.value
47     if ((ctx.sbtCallback != null || forceRun) && !unit.isJava) {
48       val sourceFile = unit.source.file
49       val apiTraverser = new ExtractAPICollector
50       val source = apiTraverser.apiSource(unit.tpdTree)
51 
52       if (dumpInc) {
53         // Append to existing file that should have been created by ExtractDependencies
54         val pw = new PrintWriter(Path(sourceFile.jpath).changeExtension("inc").toFile
55           .bufferedWriter(append = true), true)
56         try {
57           pw.println(DefaultShowAPI(source))
58         } finally pw.close()
59       }
60 
61       if (ctx.sbtCallback != null)
62         ctx.sbtCallback.api(sourceFile.file, source)
63     }
64   }

67 /** Extracts full (including private members) API representation out of Symbols and Types.
68  *
69  *  The exact representation used for each type is not important: the only thing
70  *  that matters is that a binary-incompatible or source-incompatible change to
71  *  the API (for example, changing the signature of a method, or adding a parent
72  *  to a class) should result in a change to the API representation so that sbt
73  *  can recompile files that depend on this API.
74  *
75  *  Note that we only records types as they are defined and never "as seen from"
76  *  some other prefix because `Types#asSeenFrom` is a complex operation and
77  *  doing it for every inherited member would be slow, and because the number
78  *  of prefixes can be enormous in some cases:
79  *
80  *    class Outer {
81  *      type T <: S
82  *      type S
83  *      class A extends Outer { /*...*/ }
84  *      class B extends Outer { /*...*/ }
85  *      class C extends Outer { /*...*/ }
86  *      class D extends Outer { /*...*/ }
87  *      class E extends Outer { /*...*/ }
88  *    }
89  *
90  *  `S` might be refined in an arbitrary way inside `A` for example, this
91  *  affects the type of `T` as seen from `Outer#A`, so we could record that, but
92  *  the class `A` also contains itself as a member, so `Outer#A#A#A#...` is a
93  *  valid prefix for `T`. Even if we avoid loops, we still have a combinatorial
94  *  explosion of possible prefixes, like `Outer#A#B#C#D#E`.
95  *
96  *  It is much simpler to record `T` once where it is defined, but that means
97  *  that the API representation of `T` may not change even though `T` as seen
98  *  from some prefix has changed. This is why in `ExtractDependencies` we need
99  *  to traverse used types to not miss dependencies, see the documentation of
100  *  `ExtractDependencies#usedTypeTraverser`.
101  *
102  *  TODO: sbt does not store the full representation that we compute, instead it
103  *  hashes parts of it to reduce memory usage, then to see if something changed,
104  *  it compares the hashes instead of comparing the representations. We should
105  *  investigate whether we can just directly compute hashes in this phase
106  *  without going through an intermediate representation, see
107  *  http://www.scala-sbt.org/0.13/docs/Understanding-Recompilation.html#Hashing+an+API+representation
108  */

198   def apiClassStructure(csym: ClassSymbol): api.Structure = {
199     val cinfo = csym.classInfo
200 
201     val bases =
202       try linearizedAncestorTypes(cinfo)
203       catch {
204         case ex: CyclicReference =>
205           // See neg/i1750a for an example where a cyclic error can arise.
206           // The root cause in this example is an illegal "override" of an inner trait
207           ctx.error(cyclicErrorMsg(ex), csym.pos)
208           defn.ObjectType :: Nil
209        }
210 
211     val apiBases = bases.map(apiType)
212 
213     // Synthetic methods that are always present do not affect the API
214     // and can therefore be ignored.
215     def alwaysPresent(s: Symbol) =
216       s.isCompanionMethod || (csym.is(ModuleClass) && s.isConstructor)
217     val decls = cinfo.decls.filter(!alwaysPresent(_))
218     val apiDecls = apiDefinitions(decls)
219 
220     val declSet = decls.toSet
221     // TODO: We shouldn't have to compute inherited members. Instead, `Structure`
222     // should have a lazy `parentStructures` field.
223     val inherited = cinfo.baseClasses
224       // We cannot filter out `LegacyApp` because it contains the main method,
225       // see the comment about main class discovery in `computeType`.
226       .filter(bc => !bc.is(Scala2x) || bc.eq(LegacyAppClass))
227       .flatMap(_.classInfo.decls.filter(s => !(s.is(Private) || declSet.contains(s))))
228     // Inherited members need to be computed lazily because a class might contain
229     // itself as an inherited member, like in `class A { class B extends A }`,
230     // this works because of `classLikeCache`
231     val apiInherited = lzy(apiDefinitions(inherited).toArray)
232 
233     new api.Structure(strict2lzy(apiBases.toArray), strict2lzy(apiDecls.toArray), apiInherited)
234   }

242   def apiDefinitions(defs: List[Symbol]): List[api.Definition] = {
243     // The hash generated by sbt for definitions is supposed to be symmetric so
244     // we shouldn't have to sort them, but it actually isn't symmetric for
245     // definitions which are classes, therefore we need to sort classes to
246     // ensure a stable hash.
247     // Modules and classes come first and are sorted by name, all other
248     // definitions come later and are not sorted.
249     object classFirstSort extends Ordering[Symbol] {
250       override def compare(a: Symbol, b: Symbol) = {
251         val aIsClass = a.isClass
252         val bIsClass = b.isClass
253         if (aIsClass == bIsClass) {
254           if (aIsClass) {
255             if (a.is(Module) == b.is(Module))
256               a.fullName.toString.compareTo(b.fullName.toString)
257             else if (a.is(Module))
258               -1
259             else
260               1
261           } else
262             0
263         } else if (aIsClass)
264           -1
265         else
266           1
267       }
268     }
269 
270     defs.sorted(classFirstSort).map(apiDefinition)
271   }

276     } else if (sym.isType) {

356   private def computeType(tp: Type): api.Type = {
357     // TODO: Never dealias. We currently have to dealias because
358     // sbt main class discovery relies on the signature of the main
359     // method being fully dealiased. See https://github.com/sbt/zinc/issues/102
360     val tp2 = if (!tp.isHK) tp.dealiasKeepAnnots else tp
361     tp2 match {
362       case NoPrefix | NoType =>
363         Constants.emptyType
364       case tp: NamedType =>
365         val sym = tp.symbol
366         // A type can sometimes be represented by multiple different NamedTypes
367         // (they will be `=:=` to each other, but not `==`), and the compiler
368         // may choose to use any of these representation, there is no stability
369         // guarantee. We avoid this instability by always normalizing the
370         // prefix: if it's a package, if we didn't do this sbt might conclude
371         // that some API changed when it didn't, leading to overcompilation
372         // (recompiling more things than what is needed for incremental
373         // compilation to be correct).
374         val prefix = if (sym.owner.is(Package))
375           sym.owner.thisType
376         else
377           tp.prefix
378         new api.Projection(simpleType(prefix), sym.name.toString)
379       case AppliedType(tycon, args) =>
380         def processArg(arg: Type): api.Type = arg match {
381           case arg @ TypeBounds(lo, hi) => // Handle wildcard parameters
382             if (lo.isDirectRef(defn.NothingClass) && hi.isDirectRef(defn.AnyClass))
383               Constants.emptyType
384             else {
385               val name = "_"
386               val ref = new api.ParameterRef(name)
387               new api.Existential(ref,
388                 Array(apiTypeParameter(name, 0, lo, hi)))
389             }
390           case _ =>
391             apiType(arg)
392         }
393 
394         val apiTycon = simpleType(tycon)
395         val apiArgs = args.map(processArg)
396         new api.Parameterized(apiTycon, apiArgs.toArray)
397       case tl: TypeLambda =>
398         val apiTparams = tl.typeParams.map(apiTypeParameter)
399         val apiRes = apiType(tl.resType)
400         new api.Polymorphic(apiRes, apiTparams.toArray)
401       case rt: RefinedType =>
402         val name = rt.refinedName.toString
403         val parent = apiType(rt.parent)
404 
405         def typeRefinement(name: String, tp: TypeBounds): api.TypeMember = tp match {
406           case TypeAlias(alias) =>
407             new api.TypeAlias(apiType(alias),
408               Array(), name, Constants.public, Constants.emptyModifiers, Array())
409           case TypeBounds(lo, hi) =>
410             new api.TypeDeclaration(apiType(lo), apiType(hi),
411               Array(), name, Constants.public, Constants.emptyModifiers, Array())
412         }
413         val decl = rt.refinedInfo match {
414           case rinfo: TypeBounds =>
415             typeRefinement(name, rinfo)
416           case _ =>
417             ctx.debuglog(i"sbt-api: skipped structural refinement in $rt")
418             null
419         }
420 
421         // Aggressive caching for RefinedTypes: `typeCache` is enough as long as two
422         // RefinedType are `==`, but this is only the case when their `refinedInfo`
423         // are `==` and this is not always the case, consider:
424         //
425         //     val foo: { type Bla = a.b.T }
426         //     val bar: { type Bla = a.b.T }
427         //
428         // The sbt API representations of `foo` and `bar` (let's call them `apiFoo`
429         // and `apiBar`) will both be instances of `Structure`. If `typeCache` was
430         // the only cache, then in some cases we would have `apiFoo eq apiBar` and
431         // in other cases we would just have `apiFoo == apiBar` (this happens
432         // because the dotty representation of `a.b.T` is unstable, see the comment
433         // in the `NamedType` case above).
434         //
435         // The fact that we may or may not have `apiFoo eq apiBar` is more than
436         // an optimisation issue: it will determine whether the sbt name hash for
437         // `Bla` contains one or two entries (because sbt `NameHashing` will not
438         // traverse both `apiFoo` and `apiBar` if they are `eq`), therefore the
439         // name hash of `Bla` will be unstable, unless we make sure that
440         // `apiFoo == apiBar` always imply `apiFoo eq apiBar`. This is what
441         // `refinedTypeCache` is for.
442         refinedTypeCache.getOrElseUpdate((parent, decl), {
443           val adecl: Array[api.Definition] = if (decl == null) Array() else Array(decl)
444           new api.Structure(strict2lzy(Array(parent)), strict2lzy(adecl), strict2lzy(Array()))
445         })
446       case tp: RecType =>
447         apiType(tp.parent)
448       case RecThis(recType) =>
449         // `tp` must be present inside `recType`, so calling `apiType` on
450         // `recType` would lead to an infinite recursion, we avoid this by
451         //  computing the representation of `recType` lazily.
452         apiLazy(recType)
453       case tp: AndOrType =>
454         val parents = List(apiType(tp.tp1), apiType(tp.tp2))
455 
456         // TODO: Add a real representation for AndOrTypes in xsbti. The order of
457         // types in an `AndOrType` does not change the API, so the API hash should
458         // be symmetric.
459         val s = combineApiTypes(apiType(tp.tp1), apiType(tp.tp2))
460         if (tp.isAnd)
461           s
462         else
463           withMarker(s, orMarker)
464       case ExprType(resultType) =>
465         withMarker(apiType(resultType), byNameMarker)
466       case ConstantType(constant) =>
467         new api.Constant(apiType(constant.tpe), constant.stringValue)
468       case AnnotatedType(tpe, annot) =>
469         new api.Annotated(apiType(tpe), Array(apiAnnotation(annot)))
470       case tp: ThisType =>
471         apiThis(tp.cls)
472       case tp: ParamRef =>
473         // TODO: Distinguishing parameters based on their names alone is not enough,
474         // the binder is also needed (at least for type lambdas).
475         new api.ParameterRef(tp.paramName.toString)
476       case tp: LazyRef =>
477         apiType(tp.ref)
478       case tp: TypeVar =>
479         apiType(tp.underlying)
480       case _ => {
481         ctx.warning(i"sbt-api: Unhandled type ${tp.getClass} : $tp")
482         Constants.emptyType
483       }
484     }
485   }

524   def apiAccess(sym: Symbol): api.Access = {
525     // Symbols which are private[foo] do not have the flag Private set,
526     // but their `privateWithin` exists, see `Parsers#ParserCommon#normalize`.
527     if (!sym.is(Protected | Private) && !sym.privateWithin.exists)
528       Constants.public
529     else if (sym.is(PrivateLocal))
530       Constants.privateLocal
531     else if (sym.is(ProtectedLocal))
532       Constants.protectedLocal
533     else {
534       val qualifier =
535         if (sym.privateWithin eq NoSymbol)
536           Constants.unqualified
537         else
538           new api.IdQualifier(sym.privateWithin.fullName.toString)
539       if (sym.is(Protected))
540         new api.Protected(qualifier)
541       else
542         new api.Private(qualifier)
543     }
544   }

554   def apiAnnotations(s: Symbol): List[api.Annotation] = {
555     val annots = new mutable.ListBuffer[api.Annotation]
556 
557     if (Inliner.hasBodyToInline(s)) {
558       // FIXME: If the body of an inline method changes, all the reverse
559       // dependencies of this method need to be recompiled. sbt has no way
560       // of tracking method bodies, so as a hack we include the pretty-printed
561       // typed tree of the method as part of the signature we send to sbt.
562       // To do this properly we would need a way to hash trees and types in
563       // dotty itself.
564       val printTypesCtx = ctx.fresh.setSetting(ctx.settings.XprintTypes, true)
565       annots += marker(Inliner.bodyToInline(s).show(printTypesCtx).toString)
566     }
567
