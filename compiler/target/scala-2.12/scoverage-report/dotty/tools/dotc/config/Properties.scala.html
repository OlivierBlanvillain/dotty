25   /** The loaded properties */
26   @sharable protected lazy val scalaProps: java.util.Properties = {
27     val props = new java.util.Properties
28     val stream = pickJarBasedOn getResourceAsStream propFilename
29     if (stream ne null)
30       quietlyDispose(props load stream, stream.close)

32     props
33   }

35   private def quietlyDispose(action: => Unit, disposal: => Unit) =
36     try     { action }
