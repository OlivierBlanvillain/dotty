28 /**
29  * A specific Scala version, not one of the magic min/max versions. An SpecificScalaVersion
30  * may or may not be a released version - i.e. this same class is used to represent
31  * final, release candidate, milestone, and development builds. The build argument is used
32  * to segregate builds
33  */
34 case class SpecificScalaVersion(major: Int, minor: Int, rev: Int, build: ScalaBuild) extends ScalaVersion {
35   def unparse = s"${major}.${minor}.${rev}.${build.unparse}"
36 
37   def compare(that: ScalaVersion): Int =  that match {
38     case SpecificScalaVersion(thatMajor, thatMinor, thatRev, thatBuild) =>
39       // this could be done more cleanly by importing scala.math.Ordering.Implicits, but we have to do these
40       // comparisons a lot so I'm using brute force direct style code
41       if (major < thatMajor) -1
42       else if (major > thatMajor) 1
43       else if (minor < thatMinor) -1
44       else if (minor > thatMinor) 1
45       else if (rev < thatRev) -1
46       else if (rev > thatRev) 1
47       else build compare thatBuild
48     case AnyScalaVersion => 1
49     case NoScalaVersion => -1
50   }
51 }
52

65 /**
66  * Methods for parsing ScalaVersions
67  */
68 @sharable object ScalaVersion {
69   private val dot = "\\."
70   private val dash = "\\-"
71   private def not(s:String) = s"[^${s}]"
72   private val R = s"((${not(dot)}*)(${dot}(${not(dot)}*)(${dot}(${not(dash)}*)(${dash}(.*))?)?)?)".r
73 
74   def parse(versionString : String): Try[ScalaVersion] = {
75     def failure = Failure(new NumberFormatException(
76       s"There was a problem parsing ${versionString}. " +
77        "Versions should be in the form major[.minor[.revision]] " +
78        "where each part is a positive number, as in 2.10.1. " +
79        "The minor and revision parts are optional."
80     ))
81 
82     def toInt(s: String) = s match {
83       case null | "" => 0
84       case _ => s.toInt
85     }
86 
87     def isInt(s: String) = Try(toInt(s)).isSuccess
88 
89     import ScalaBuild._
90 
91     def toBuild(s: String) = s match {
92       case null | "FINAL" => Final
93       case s if (s.toUpperCase.startsWith("RC") && isInt(s.substring(2))) => RC(toInt(s.substring(2)))
94       case s if (s.toUpperCase.startsWith("M") && isInt(s.substring(1))) => Milestone(toInt(s.substring(1)))
95       case _ => Development(s)
96     }
97 
98     try versionString match {
99       case "" | "any" => Success(AnyScalaVersion)
100       case "none" => Success(NoScalaVersion)
101       case R(_, majorS, _, minorS, _, revS, _, buildS) =>
102         Success(SpecificScalaVersion(toInt(majorS), toInt(minorS), toInt(revS), toBuild(buildS)))
103       case _ => failure
104     } catch {
105       case e: NumberFormatException => failure
106     }
107   }
108 
109   /**
110    * The version of the compiler running now
111    */
112   val current = parse(util.Properties.versionNumberString).get
113 }

118 abstract class ScalaBuild extends Ordered[ScalaBuild] {
119   /**

126 object ScalaBuild {
127 
128   /** A development, test, nightly, snapshot or other "unofficial" build
129    */
130   case class Development(id: String) extends ScalaBuild {
131     def unparse = s"-${id}"
132 
133     def compare(that: ScalaBuild) = that match {
134       // sorting two development builds based on id is reasonably valid for two versions created with the same schema
135       // otherwise it's not correct, but since it's impossible to put a total ordering on development build versions
136       // this is a pragmatic compromise
137       case Development(thatId) => id compare thatId
138       // assume a development build is newer than anything else, that's not really true, but good luck
139       // mapping development build versions to other build types
140       case _ => 1
141     }
142   }
143 
144   /** A final build
145    */
146   case object Final extends ScalaBuild {
147     def unparse = ""
148 
149     def compare(that: ScalaBuild) = that match {
150       case Final => 0
151       // a final is newer than anything other than a development build or another final
152       case Development(_) => -1
153       case _ => 1
154     }
155   }
156 
157   /** A candidate for final release
158    */
159   case class RC(n: Int) extends ScalaBuild {
160     def unparse = s"-RC${n}"
161 
162     def compare(that: ScalaBuild) = that match {
163       // compare two rcs based on their RC numbers
164       case RC(thatN) => n - thatN
165       // an rc is older than anything other than a milestone or another rc
166       case Milestone(_) => 1
167       case _ => -1
168     }
169   }
170 
171   /** An intermediate release
172    */
173   case class Milestone(n: Int) extends ScalaBuild {
174     def unparse = s"-M${n}"
175 
176     def compare(that: ScalaBuild) = that match {
177       // compare two milestones based on their milestone numbers
178       case Milestone(thatN) => n - thatN
179       // a milestone is older than anything other than another milestone
180       case _ => -1
181 
182     }
183   }
