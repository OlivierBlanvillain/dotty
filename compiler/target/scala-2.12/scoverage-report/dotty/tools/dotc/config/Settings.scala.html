116     def tryToSet(state: ArgsSummary): ArgsSummary = {
117       val ArgsSummary(sstate, arg :: args, errors, warnings) = state
118       def update(value: Any, args: List[String]) = {
119         if (changed) {
120           ArgsSummary(updateIn(sstate, value), args, errors, warnings :+ s"Flag $name set repeatedly")
121         } else {
122           changed = true
123           ArgsSummary(updateIn(sstate, value), args, errors, warnings)
124         }
125       }

127         ArgsSummary(sstate, args, errors :+ msg, warnings)
128       def missingArg =
129         fail(s"missing argument for option $name", args)

131         case (BooleanTag, _) =>
132           update(true, args)
133         case (OptionTag, _) =>
134           update(Some(propertyClass.get.newInstance), args)
135         case (ListTag, _) =>
136           if (argRest.isEmpty) missingArg
137           else update((argRest split ",").toList, args)
138         case (StringTag, _) if choices.nonEmpty =>
139           if (argRest.isEmpty) missingArg
140           else if (!choices.contains(argRest))
141             fail(s"$arg is not a valid choice for $name", args)
142           else update(argRest, args)
143         case (StringTag, arg :: args) if name == "-d" =>
144           Path(arg) match {
145             case _: Directory =>
146               update(arg, args)
147             case p if p.extension == "jar" =>
148               update(arg, args)
149             case _ =>
150               fail(s"'$arg' does not exist or is not a directory", args)
151           }
152         case (StringTag, arg2 :: args2) =>
153           update(arg2, args2)
154         case (IntTag, arg2 :: args2) =>
155           try {
156             val x = arg2.toInt
157             choices match {
158               case r: Range if x < r.head || r.last < x =>
159                 fail(s"$arg2 is out of legal range $legalChoices for $name", args2)
160               case _ =>
161                 update(x, args2)
162             }
163           } catch {
164             case _: NumberFormatException =>
165               fail(s"$arg2 is not an integer argument for $name", args2)
166           }
167         case (VersionTag, _) =>
168           ScalaVersion.parse(argRest) match {
169             case Success(v) => update(v, args)
170             case Failure(ex) => fail(ex.getMessage, args)
171           }
172         case (_, Nil) =>
173           missingArg
174       }
175 
176       if (prefix != "" && arg.startsWith(prefix))
177         doSet(arg drop prefix.length)
178       else if (prefix == "" && name == arg.takeWhile(_ != ':'))
179         doSet(arg.dropWhile(_ != ':').drop(1))
180       else
181         state
182     }

216     /** Iterates over the arguments applying them to settings where applicable.
217      *  Then verifies setting dependencies are met.
218      *
219      *  This temporarily takes a boolean indicating whether to keep
220      *  processing if an argument is seen which is not a command line option.
221      *  This is an expedience for the moment so that you can say
222      *
223      *    scalac -d /tmp foo.scala -optimise
224      *
225      *  while also allowing
226      *
227      *    scala Program opt opt
228      *
229      *  to get their arguments.
230      */
231     protected def processArguments(state: ArgsSummary, processAll: Boolean, skipped: List[String]): ArgsSummary = {
232       def stateWithArgs(args: List[String]) = ArgsSummary(state.sstate, args, state.errors, state.warnings)
233       state.arguments match {
234         case Nil =>
235           checkDependencies(stateWithArgs(skipped))
236         case "--" :: args =>
237           checkDependencies(stateWithArgs(skipped ++ args))
238         case x :: _ if x startsWith "-" =>
239           @tailrec def loop(settings: List[Setting[_]]): ArgsSummary = settings match {
240             case setting :: settings1 =>
241               val state1 = setting.tryToSet(state)
242               if (state1 ne state) processArguments(state1, processAll, skipped)
243               else loop(settings1)
244             case Nil =>
245               processArguments(state.warn(s"bad option '$x' was ignored"), processAll, skipped)
246           }
247           loop(allSettings.toList)
248         case arg :: args =>
249           if (processAll) processArguments(stateWithArgs(args), processAll, skipped :+ arg)
250           else state
251       }
252     }
