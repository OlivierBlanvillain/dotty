16   private def explainAdvanced = """
17     |-- Notes on option parsing --
18     |Boolean settings are always false unless set.
19     |Where multiple values are accepted, they should be comma-separated.
20     |  example: -Xplugin:plugin1,plugin2
21     |<phases> means one or a comma-separated list of:
22     |  - (partial) phase names with an optional "+" suffix to include the next phase
23     |  - the string "all"
24     |  example: -Xprint:all prints all phases.
25     |  example: -Xprint:front,mixin prints the frontend and mixin phases.
26     |  example: -Ylog:erasure+ logs the erasure phase and the phase after the erasure phase.
27     |           This is useful because during the tree transform of phase X, we often
28     |           already are in phase X + 1.
29   """

35   /** Distill arguments into summary detailing settings, errors and files to compiler */
36   def distill(args: Array[String])(implicit ctx: Context): ArgsSummary = {
37     /**
38      * Expands all arguments starting with @ to the contents of the
39      * file named like each argument.
40      */
41     def expandArg(arg: String): List[String] = {
42       def stripComment(s: String) = s takeWhile (_ != '#')
43       val path = Paths.get(arg stripPrefix "@")
44       if (!Files.exists(path))
45         throw new java.io.FileNotFoundException("argument file %s could not be found" format path.getFileName)
46 
47       val lines = Files.readAllLines(path) // default to UTF-8 encoding
48 
49       val params = lines.asScala map stripComment mkString " "
50       CommandLineParser.tokenize(params)
51     }
52 
53     // expand out @filename to the contents of that filename
54     def expandedArguments = args.toList flatMap {
55       case x if x startsWith "@"  => expandArg(x)
56       case x                      => List(x)
57     }
58 
59     ctx.settings.processArguments(expandedArguments, processAll = true)
60   }
61 
62   /** Provide usage feedback on argument summary, assuming that all settings
63    *  are already applied in context.
64    *  @return  The list of files to compile.
65    */
66   def checkUsage(summary: ArgsSummary, sourcesRequired: Boolean)(implicit ctx: Context): List[String] = {
67     val settings = ctx.settings
68 
69     /** Creates a help message for a subset of options based on cond */
70     def availableOptionsMsg(cond: Setting[_] => Boolean): String = {
71       val ss                  = (ctx.settings.allSettings filter cond).toList sortBy (_.name)
72       val width               = (ss map (_.name.length)).max
73       def format(s: String)   = ("%-" + width + "s") format s
74       def helpStr(s: Setting[_]) = s"${format(s.name)} ${s.description}"
75       ss map helpStr mkString "\n"
76     }
77 
78     def createUsageMsg(label: String, shouldExplain: Boolean, cond: Setting[_] => Boolean): String = {
79       val prefix = List(
80         Some(shortUsage),
81         Some(explainAdvanced) filter (_ => shouldExplain),
82         Some(label + " options include:")
83       ).flatten mkString "\n"
84 
85       prefix + "\n" + availableOptionsMsg(cond)
86     }
87 
88     def isStandard(s: Setting[_]): Boolean = !isAdvanced(s) && !isPrivate(s)
89     def isAdvanced(s: Setting[_]): Boolean = s.name startsWith "-X"
90     def isPrivate(s: Setting[_]) : Boolean = s.name startsWith "-Y"
91 
92     /** Messages explaining usage and options */
93     def usageMessage    = createUsageMsg("where possible standard", shouldExplain = false, isStandard)
94     def xusageMessage   = createUsageMsg("Possible advanced", shouldExplain = true, isAdvanced)
95     def yusageMessage   = createUsageMsg("Possible private", shouldExplain = true, isPrivate)
96 
97     def shouldStopWithInfo = {
98       import settings._
99       Set(help, Xhelp, Yhelp) exists (_.value)
100     }
101 
102     def infoMessage: String = {
103       import settings._
104       if (help.value) usageMessage
105       else if (Xhelp.value) xusageMessage
106       else if (Yhelp.value) yusageMessage
107       else ""
108     }
109 
110     // Print all warnings encountered during arguments parsing
111     summary.warnings.foreach(ctx.warning(_))
112 
113     if (summary.errors.nonEmpty) {
114       summary.errors foreach (ctx.error(_))
115       ctx.echo("  dotc -help  gives more information")
116       Nil
117     }
118     else if (settings.version.value) {
119       ctx.echo(versionMsg)
120       Nil
121     } else if (!Properties.isJavaAtLeast("1.8")) {
122       ctx.error("Dotty requires Java 8 to run")
123       Nil
124     }
125     else if (shouldStopWithInfo) {
126       ctx.echo(infoMessage)
127       Nil
128     } else {
129       if (sourcesRequired && summary.arguments.isEmpty) ctx.echo(usageMessage)
130       summary.arguments
131     }
132   }
