7 /** A class for holding mappings from source directories to
8  *  their output location. This functionality can be accessed
9  *  only programmatically. The command line compiler uses a
10  *  single output location, but tools may use this functionality
11  *  to set output location per source directory.
12  */
13 class OutputDirs {
14   /** Pairs of source directory - destination directory. */
15   private[this] var outputDirs: List[(AbstractFile, AbstractFile)] = Nil

23    *  Both directories should exits.
24    */

29   /** Check that dir is exists and is a directory. */
30   private def checkDir(dir: AbstractFile, name: String, allowJar: Boolean = false): AbstractFile = (

61   /** Return the list of source-destination directory pairs. */
62   def outputs: List[(AbstractFile, AbstractFile)] = outputDirs

64   /** Return the output directory for the given file.
65    */
66   def outputDirFor(src: AbstractFile): AbstractFile = {
67     def isBelow(srcDir: AbstractFile, outDir: AbstractFile) =
68       src.path.startsWith(srcDir.path)
69 
70     singleOutDir match {
71       case Some(d) => d
72       case None =>
73         (outputs find (isBelow _).tupled) match {
74           case Some((_, d)) => d
75           case _ =>
76             throw new FatalError("Could not find an output directory for "
77               + src.path + " in " + outputs)
78         }
79     }
80   }

82   /** Return the source file path(s) which correspond to the given
83    *  classfile path and SourceFile attribute value, subject to the
84    *  condition that source files are arranged in the filesystem
85    *  according to Java package layout conventions.
86    *
87    *  The given classfile path must be contained in at least one of
88    *  the specified output directories. If it does not then this
89    *  method returns Nil.
90    *
91    *  Note that the source file is not required to exist, so assuming
92    *  a valid classfile path this method will always return a list
93    *  containing at least one element.
94    *
95    *  Also that if two or more source path elements target the same
96    *  output directory there will be two or more candidate source file
97    *  paths.
98    */
99   def srcFilesFor(classFile: AbstractFile, srcPath: String): List[AbstractFile] = {
100     def isBelow(srcDir: AbstractFile, outDir: AbstractFile) =
101       classFile.path.startsWith(outDir.path)
102 
103     singleOutDir match {
104       case Some(d) =>
105         d match {
106           case _: VirtualDirectory | _: io.ZipArchive => Nil
107           case _ => List(d.lookupPathUnchecked(srcPath, false))
108         }
109       case None =>
110         (outputs filter (isBelow _).tupled) match {
111           case Nil => Nil
112           case matches => matches.map(_._1.lookupPathUnchecked(srcPath, false))
113         }
114     }
115   }
