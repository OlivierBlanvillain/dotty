3 /** A class inheriting from Attachment.Container supports
4  *  adding, removing and lookup of attachments. Attachments are typed key/value pairs.
5  */
6 object Attachment {
7   import Property.Key

9   /** An implementation trait for attachments.
10    *  Clients should inherit from Container instead.
11    */
12   trait LinkSource {
13     private[Attachment] var next: Link[_]
14 
15     /** Optionally get attachment corresponding to `key` */
16     final def getAttachment[V](key: Key[V]): Option[V] = {
17       val nx = next
18       if (nx == null) None
19       else if (nx.key eq key) Some(nx.value.asInstanceOf[V])
20       else nx.getAttachment[V](key)
21     }
22 
23     /** The attachment corresponding to `key`.
24      *  @throws NoSuchElementException  if no attachment with key exists
25      */
26     final def attachment[V](key: Key[V]): V = {
27       val nx = next
28       if (nx == null) throw new NoSuchElementException
29       else if (nx.key eq key) nx.value.asInstanceOf[V]
30       else nx.attachment(key)
31     }
32 
33     /** The attachment corresponding to `key`, or `default`
34      *  if no attachment with `key` exists.
35      */
36     final def attachmentOrElse[V](key: Key[V], default: V): V = {
37       val nx = next
38       if (nx == null) default
39       else if (nx.key eq key) nx.value.asInstanceOf[V]
40       else nx.attachmentOrElse(key, default)
41     }
42 
43     /** Add attachment with given `key` and `value`.
44      *  @return  Optionally, the old attachment with given `key` if one existed before.
45      *  The new attachment is added at the position of the old one, or at the end
46      *  if no attachment with same `key` existed.
47      */
48     final def putAttachment[V](key: Key[V], value: V): Option[V] = {
49       val nx = next
50       if (nx == null) {
51         next = new Link(key, value, null)
52         None
53       }
54       else if (nx.key eq key) {
55         next = new Link(key, value, nx.next)
56         Some(nx.value.asInstanceOf[V])
57       }
58       else nx.putAttachment(key, value)
59     }
60 
61     /** Remove attachment with given `key`, if it exists.
62      *  @return  Optionally, the removed attachment with given `key` if one existed before.
63      */
64     final def removeAttachment[V](key: Key[V]): Option[V] = {
65       val nx = next
66       if (nx == null)
67         None
68       else if (nx.key eq key) {
69         next = nx.next
70         Some(nx.value.asInstanceOf[V])
71       }
72       else nx.removeAttachment(key)
73     }
74 
75     /** The list of all keys and values attached to this container. */
76     final def allAttachments: List[(Key[_], Any)] = {
77       val nx = next
78       if (nx == null) Nil else (nx.key, nx.value) :: nx.allAttachments
79     }
80   }
81 
82   /** A private, concrete implementation class linking attachments.
83    */
84   private[Attachment] class Link[+V](val key: Key[V], val value: V, var next: Link[_])
85       extends LinkSource
86 
87   /** A trait for objects that can contain attachments */
88   trait Container extends LinkSource {
89     private[Attachment] var next: Link[_] = null
90 
91     final def pushAttachment[V](key: Key[V], value: V): Unit = {
92       assert(!getAttachment(key).isDefined, s"duplicate attachment for key $key")
93       next = new Link(key, value, next)
94     }
95 
96     final def removeAllAttachments() =
97       next = null
98   }
