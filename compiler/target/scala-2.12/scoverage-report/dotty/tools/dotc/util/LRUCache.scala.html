6 /** A least-recently-used cache for Key -> Value computations
7  *  It currently keeps the last 8 associations, but this can be
8  *  changed to anywhere between 2 and 16 by changing `LRUCache.Retained`.
9  *
10  *  Implementation: We keep a ring of eight places, linked
11  *  with the `next` data structure. The ring models a priority queue.
12  *  `last` points to the last element of the queue, and
13  *  `next(last)` to the first one. Lookups compare keys
14  *  sequentially from first to last. Elements with successful lookup
15  *  get promoted to be first in the queue. Elements are evicted
16  *  at the `last` position.
17  */
18 class LRUCache[Key >: Null <: AnyRef : ClassTag, Value >: Null: ClassTag] {
19   import LRUCache._
20   val keys = new Array[Key](Retained)
21   val values = new Array[Value](Retained)
22   var next = new SixteenNibbles(initialRing.bits)
23   var last = Retained - 1 // value is arbitrary
24   var lastButOne: Int = last - 1
25 
26   def first = next(last)
27 
28   /** Lookup key, returning value or `null` for not found.
29    *  As a side effect, sets `lastButOne` to the element before `last`
30    *  if key was not found.
31    */
32   def lookup(key: Key): Value = {
33     @tailrec
34     def lookupNext(prev: Int, current: Int, nx: SixteenNibbles): Value = {
35       val follow = nx(current)
36       if (keys(current) eq key) {
37         // arrange so that found element is at position `first`.
38         if (current == last) last = prev
39         else if (prev != last) {
40           next = next.updated(prev, follow)
41           next = next.updated(current, first)
42           next = next.updated(last, current)
43         }
44         values(current)
45       } else if (current == last) {
46         lastButOne = prev
47         null
48       } else
49         lookupNext(current, follow, nx)
50     }
51     lookupNext(last, first, next)
52   }
53 
54   /** Enter key/value in cache at position `last`.
55    *  As a side effect, sets `last` to `lastButOne`.
56    *  If `lastButOne` was set by a preceding unsuccessful `lookup`
57    *  for the same key, this means that the new element is now the
58    *  first in the queue. If there was no preceding lookup, the element
59    *  is inserted at a random position in the queue.
60    */
61   def enter(key: Key, value: Value): Unit = {
62     keys(last) = key
63     values(last) = value
64     last = lastButOne
65   }
66 
67   /** Invalidate key. The invalidated element becomes
68    *  the last in the queue.
69    */
70   def invalidate(key: Key): Unit =
71     if (lookup(key) != null) {
72       keys(first) = null
73       last = first
74     }
75 
76   def indices: Iterator[Int] = Iterator.iterate(first)(next.apply)
77 
78   def keysIterator: Iterator[Key] =
79     indices take Retained map keys filter (_ != null)
80 
81   override def toString = {
82     val assocs = keysIterator
83       .toList  // double reverse so that lookups do not perturb order
84       .reverse
85       .map(key => s"$key -> ${lookup(key)}")
86       .reverse
87     s"LRUCache(${assocs.mkString(", ")})"
88   }
89 }
