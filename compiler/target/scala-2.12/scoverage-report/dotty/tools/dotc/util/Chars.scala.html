24   /** Convert a character digit to an Int according to given base,
25    *  -1 if no success
26    */
27   def digit2int(ch: Char, base: Int): Int = {
28     val num = (
29       if (ch <= '9') ch - '0'
30       else if ('a' <= ch && ch <= 'z') ch - 'a' + 10
31       else if ('A' <= ch && ch <= 'Z') ch - 'A' + 10
32       else -1
33     )
34     if (0 <= num && num < base) num else -1
35   }
