12   @tailrec private def splitTokens(str: String, acc: List[String] = Nil): List[String] = {
13       if (str == "") {
14         acc.reverse
15       } else {
16         val head = str.charAt(0)
17         val (token, rest) =
18           if (head == ansiColorToken) { // ansi color token
19             val splitIndex = str.indexOf('m') + 1
20             (str.substring(0, splitIndex), str.substring(splitIndex))
21           } else if (Character.isAlphabetic(head) || Character.isDigit(head)) {
22             str.span(c => Character.isAlphabetic(c) || Character.isDigit(c) && c != ansiColorToken)
23           } else if (Character.isMirrored(head) || Character.isWhitespace(head)) {
24             str.splitAt(1)
25           } else {
26             str.span { c =>
27               !Character.isAlphabetic(c) && !Character.isDigit(c) &&
28                 !Character.isMirrored(c) && !Character.isWhitespace(c) && c != ansiColorToken
29             }
30           }
31         splitTokens(rest, token :: acc)
32       }
33     }

127   private def hirschberg(a: Array[String], b: Array[String]): Array[Patch] = {
128     def build(x: Array[String], y: Array[String], builder: mutable.ArrayBuilder[Patch]): Unit = {
129       if (x.isEmpty) {
130         builder += Inserted(y.mkString)
131       } else if (y.isEmpty) {
132         builder += Deleted(x.mkString)
133       } else if (x.length == 1 || y.length == 1) {
134         needlemanWunsch(x, y, builder)
135       } else {
136         val xlen = x.length
137         val xmid = xlen / 2
138         val ylen = y.length
139 
140         val (x1, x2) = x.splitAt(xmid)
141         val leftScore = nwScore(x1, y)
142         val rightScore = nwScore(x2.reverse, y.reverse)
143         val scoreSum = (leftScore zip rightScore.reverse).map {
144           case (left, right) => left + right
145         }
146         val max = scoreSum.max
147         val ymid = scoreSum.indexOf(max)
148 
149         val (y1, y2) = y.splitAt(ymid)
150         build(x1, y1, builder)
151         build(x2, y2, builder)
152       }
153     }
154     val builder = Array.newBuilder[Patch]
155     build(a, b, builder)
156     builder.result()
157   }

159   private def nwScore(x: Array[String], y: Array[String]): Array[Int] = {
160     def ins(s: String) = -2
161     def del(s: String) = -2
162     def sub(s1: String, s2: String) = if (s1 == s2) 2 else -1
163 
164     val score = Array.fill(x.length + 1, y.length + 1)(0)
165     for (j <- 1 to y.length)
166       score(0)(j) = score(0)(j - 1) + ins(y(j - 1))
167     for (i <- 1 to x.length) {
168       score(i)(0) = score(i - 1)(0) + del(x(i - 1))
169       for (j <- 1 to y.length) {
170         val scoreSub = score(i - 1)(j - 1) + sub(x(i - 1), y(j - 1))
171         val scoreDel = score(i - 1)(j) + del(x(i - 1))
172         val scoreIns = score(i)(j - 1) + ins(y(j - 1))
173         score(i)(j) = scoreSub max scoreDel max scoreIns
174       }
175     }
176     Array.tabulate(y.length + 1)(j => score(x.length)(j))
177   }

179   private def needlemanWunsch(x: Array[String], y: Array[String], builder: mutable.ArrayBuilder[Patch]): Unit = {
180     def similarity(a: String, b: String) = if (a == b) 2 else -1
181     val d = 1
182     val score = Array.tabulate(x.length + 1, y.length + 1) { (i, j) =>
183       if (i == 0) d * j
184       else if (j == 0) d * i
185       else 0
186     }
187     for (i <- 1 to x.length) {
188       for (j <- 1 to y.length) {
189         val mtch = score(i - 1)(j - 1) + similarity(x(i - 1), y(j - 1))
190         val delete = score(i - 1)(j) + d
191         val insert = score(i)(j - 1) + d
192         score(i)(j) = mtch max insert max delete
193       }
194     }
195 
196     var alignment = List.empty[Patch]
197     var i = x.length
198     var j = y.length
199     while (i > 0 || j > 0) {
200       if (i > 0 && j > 0 && score(i)(j) == score(i - 1)(j - 1) + similarity(x(i - 1), y(j - 1))) {
201         val newHead =
202           if (x(i - 1) == y(j - 1)) Unmodified(x(i - 1))
203           else Modified(x(i - 1), y(j - 1))
204         alignment = newHead :: alignment
205         i = i - 1
206         j = j - 1
207       } else if (i > 0 && score(i)(j) == score(i - 1)(j) + d) {
208         alignment = Deleted(x(i - 1)) :: alignment
209         i = i - 1
210       } else {
211         alignment = Inserted(y(j - 1)) :: alignment
212         j = j - 1
213       }
214     }
215     builder ++= alignment
216   }
