18   @sharable private val headerPattern = Pattern.compile("""^(::)?!#.*(\r|\n|\r\n)""", Pattern.MULTILINE)
19   private val headerStarts  = List("#!", "::#!")
20 
21   def apply(file: AbstractFile, content: Array[Char]) = {
22     /** Length of the script header from the given content, if there is one.
23      *  The header begins with "#!" or "::#!" and ends with a line starting
24      *  with "!#" or "::!#".
25      */
26     val headerLength =
27       if (headerStarts exists (content startsWith _)) {
28         val matcher = headerPattern matcher content.mkString
29         if (matcher.find) matcher.end
30         else throw new IOException("script file does not close its header with !# or ::!#")
31       } else 0
32     new SourceFile(file, content drop headerLength) {
33       override val underlying = new SourceFile(file, content)
34     }
35   }
36 }

82   private def isLineBreak(idx: Int) =
83     if (idx >= length) false else {
84       val ch = content(idx)
85       // don't identify the CR in CR LF as a line break, since LF will do.
86       if (ch == CR) (idx + 1 == length) || (content(idx + 1) != LF)
87       else isLineBreakChar(ch)
88     }

90   private def calculateLineIndices(cs: Array[Char]) = {
91     val buf = new ArrayBuffer[Int]
92     buf += 0
93     for (i <- 0 until cs.length) if (isLineBreak(i)) buf += i + 1
94     buf += cs.length // sentinel, so that findLine below works smoother
95     buf.toArray
96   }
97   private lazy val lineIndices: Array[Int] = calculateLineIndices(content)

114   /** The index of the first character of the line containing position `offset` */

128   /** The column corresponding to `offset`, starting at 0 */
129   def column(offset: Int): Int = {
130     var idx = startOfLine(offset)
131     var col = 0
132     while (idx != offset) {
133       col += (if (idx < length && content(idx) == '\t') (tabInc - col) % tabInc else 1)
134       idx += 1
135     }

137   }

139   /** The padding of the column corresponding to `offset`, includes tabs */
140   def startColumnPadding(offset: Int): String = {

145       idx += 1
146     }

148   }
