11 /**
12  * A HashSet where the elements are stored weakly. Elements in this set are eligible for GC if no other
13  * hard references are associated with them. Its primary use case is as a canonical reference
14  * identity holder (aka "hash-consing") via findEntryOrUpdate
15  *
16  * This Set implementation cannot hold null. Any attempt to put a null in it will result in a NullPointerException
17  *
18  * This set implementation is not in general thread safe without external concurrency control. However it behaves
19  * properly when GC concurrently collects elements in this set.
20  */
21 final class WeakHashSet[A >: Null <: AnyRef](val initialCapacity: Int, val loadFactor: Double) extends Set[A] with Function1[A, Boolean] with MSet[A] {

41   /**
42    * from a specified initial capacity compute the capacity we'll use as being the next
43    * power of two equal to or greater than the specified initial capacity
44    */
45   private def computeCapacity = {
46     if (initialCapacity < 0) throw new IllegalArgumentException("initial capacity cannot be less than 0")
47     var candidate = 1
48     while (candidate < initialCapacity) {
49       candidate  *= 2
50     }
51     candidate
52   }

66   /**
67    * find the bucket associated with an element's hash code
68    */
69   private[this] def bucketFor(hash: Int): Int = {
70     // spread the bits around to try to avoid accidental collisions using the
71     // same algorithm as java.util.HashMap
72     var h = hash
73     h ^= h >>> 20 ^ h >>> 12
74     h ^= h >>> 7 ^ h >>> 4
75 
76     // this is finding h % table.length, but takes advantage of the
77     // fact that table length is a power of 2,
78     // if you don't do bit flipping in your head, if table.length
79     // is binary 100000.. (with n 0s) then table.length - 1
80     // is 1111.. with n 1's.
81     // In other words this masks on the last n bits in the hash
82     h & (table.length - 1)
83   }

88   private[this] def remove(bucket: Int, prevEntry: Entry[A], entry: Entry[A]): Unit = {
89     prevEntry match {
90       case null => table(bucket) = entry.tail
91       case _ => prevEntry.tail = entry.tail
92     }
93     count -= 1
94   }

96   /**
97    * remove entries associated with elements that have been gc'ed
98    */
99   private[this] def removeStaleEntries(): Unit = {
100     def poll(): Entry[A] = queue.poll().asInstanceOf[Entry[A]]
101 
102     @tailrec
103     def queueLoop(): Unit = {
104       val stale = poll()
105       if (stale != null) {
106         val bucket = bucketFor(stale.hash)
107 
108         @tailrec
109         def linkedListLoop(prevEntry: Entry[A], entry: Entry[A]): Unit = if (stale eq entry) remove(bucket, prevEntry, entry)
110         else if (entry != null) linkedListLoop(entry, entry.tail)
111 
112         linkedListLoop(null, table(bucket))
113 
114         queueLoop()
115       }
116     }
117 
118     queueLoop()
119   }

124   private[this] def resize(): Unit = {
125     val oldTable = table
126     table = new Array[Entry[A]](oldTable.size * 2)
127     threshold = computeThreshold
128 
129     @tailrec
130     def tableLoop(oldBucket: Int): Unit = if (oldBucket < oldTable.size) {
131       @tailrec
132       def linkedListLoop(entry: Entry[A]): Unit = entry match {
133         case null => ()
134         case _ => {
135           val bucket = bucketFor(entry.hash)
136           val oldNext = entry.tail
137           entry.tail = table(bucket)
138           table(bucket) = entry
139           linkedListLoop(oldNext)
140         }
141       }
142       linkedListLoop(oldTable(oldBucket))
143 
144       tableLoop(oldBucket + 1)
145     }
146     tableLoop(0)
147   }

288       /**
289        * the bucket currently being examined. Initially it's set past the last bucket and will be decremented
290        */
291       private[this] var currentBucket: Int = table.size
292 
293       /**
294        * the entry that was last examined
295        */
296       private[this] var entry: Entry[A] = null
297 
298       /**
299        * the element that will be the result of the next call to next()
300        */
301       private[this] var lookaheadelement: A = null.asInstanceOf[A]
302 
303       @tailrec
304       def hasNext: Boolean = {
305         while (entry == null && currentBucket > 0) {
306           currentBucket -= 1
307           entry = table(currentBucket)
308         }
309 
310         if (entry == null) false
311         else {
312           lookaheadelement = entry.get
313           if (lookaheadelement == null) {
314             // element null means the weakref has been cleared since we last did a removeStaleEntries(), move to the next entry
315             entry = entry.tail
316             hasNext
317           } else {
318             true
319           }
320         }
321       }
322 
323       def next(): A = if (lookaheadelement == null)
324         throw new IndexOutOfBoundsException("next on an empty iterator")
325       else {
326         val result = lookaheadelement
327         lookaheadelement = null.asInstanceOf[A]
328         entry = entry.tail
329         result
330       }
331     }

338   private[util] class Diagnostics {
339     /**
340      * Verify that the internal structure of this hash set is fully consistent.
341      * Throws an assertion error on any problem. In order for it to be reliable
342      * the entries must be stable. If any are garbage collected during validation
343      * then an assertion may inappropriately fire.
344      */
345     def fullyValidate: Unit = {
346       var computedCount = 0
347       var bucket = 0
348       while (bucket < table.size) {
349         var entry = table(bucket)
350         while (entry != null) {
351           assert(entry.get != null, s"$entry had a null value indicated that gc activity was happening during diagnostic validation or that a null value was inserted")
352           computedCount += 1
353           val cachedHash = entry.hash
354           val realHash = entry.get.hashCode
355           assert(cachedHash == realHash, s"for $entry cached hash was $cachedHash but should have been $realHash")
356           val computedBucket = bucketFor(realHash)
357           assert(computedBucket == bucket, s"for $entry the computed bucket was $computedBucket but should have been $bucket")
358 
359           entry = entry.tail
360         }
361 
362         bucket += 1
363       }
364 
365       assert(computedCount == count, s"The computed count was $computedCount but should have been $count")
366     }
