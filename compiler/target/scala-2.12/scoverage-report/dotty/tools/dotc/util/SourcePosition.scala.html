7 /** A source position is comprised of a position in a source file */

10   /** Is `that` a source position contained in this source position ?
11    *  `outer` is not taken into account. */
12   def contains(that: SourcePosition): Boolean =
13     this.source == that.source && this.pos.contains(that.pos)

15   def exists = pos.exists

17   def lineContent: String = source.lineContent(point)

19   def point: Int = pos.point

21   /** The line of the position, starting at 0 */
22   def line: Int = source.offsetToLine(point)

24   /** Extracts the lines from the underlying source file as `Array[Char]`*/
25   def linesSlice: Array[Char] =

28   /** The lines of the position */
29   def lines: List[Int] = {
30     val startOffset = source.offsetToLine(start)
31     val endOffset = source.offsetToLine(end + 1)
32     if (startOffset >= endOffset) line :: Nil
33     else (startOffset until endOffset).toList
34   }

36   def lineOffsets: List[Int] =
37     lines.map(source.lineToOffset(_))

39   def beforeAndAfterPoint: (List[Int], List[Int]) =
40     lineOffsets.partition(_ <= point)

42   /** The column of the position, starting at 0 */
43   def column: Int = source.column(point)
44 
45   def start: Int = pos.start
46   def startLine: Int = source.offsetToLine(start)
47   def startColumn: Int = source.column(start)
48   def startColumnPadding: String = source.startColumnPadding(start)
49 
50   def end: Int = pos.end
51   def endLine: Int = source.offsetToLine(end)
52   def endColumn: Int = source.column(end)
53 
54   def withOuter(outer: SourcePosition) = new SourcePosition(source, pos, outer)
55 
56   override def toString =
57     if (source.exists) s"${source.file}:${line + 1}"
58     else s"(no source file, offset = ${pos.point})"
59 }
