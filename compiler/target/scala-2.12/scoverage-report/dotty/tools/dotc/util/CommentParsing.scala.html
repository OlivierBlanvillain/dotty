8 /** The comment parsing in `dotc` is used by both the comment cooking and the
9   * dottydoc tool.
10   *
11   * The comment cooking is used to expand comments with `@inheritdoc` and
12   * `@define` annotations. The rest of the comment is untouched and later
13   * handled by dottydoc.
14   */
15 object CommentParsing {
16   import Chars._

18   /** Returns index of string `str` following `start` skipping longest
19    *  sequence of whitespace characters characters (but no newlines)
20    */
21   def skipWhitespace(str: String, start: Int): Int =
22     if (start < str.length && isWhitespace(str charAt start)) skipWhitespace(str, start + 1)
23     else start
24 
25   /** Returns index of string `str` following `start` skipping
26    *  sequence of identifier characters.
27    */
28   def skipIdent(str: String, start: Int): Int =
29     if (start < str.length && isIdentifierPart(str charAt start)) skipIdent(str, start + 1)
30     else start
31 
32   /** Returns index of string `str` following `start` skipping
33    *  sequence of identifier characters.
34    */
35   def skipTag(str: String, start: Int): Int =
36     if (start < str.length && (str charAt start) == '@') skipIdent(str, start + 1)
37     else start
38 
39 
40   /** Returns index of string `str` after `start` skipping longest
41    *  sequence of space and tab characters, possibly also containing
42    *  a single `*` character or the `/``**` sequence.
43    *  @pre  start == str.length || str(start) == `\n`
44    */
45   def skipLineLead(str: String, start: Int): Int =
46     if (start == str.length) start
47     else {
48       val idx = skipWhitespace(str, start + 1)
49       if (idx < str.length && (str charAt idx) == '*') skipWhitespace(str, idx + 1)
50       else if (idx + 2 < str.length && (str charAt idx) == '/' && (str charAt (idx + 1)) == '*' && (str charAt (idx + 2)) == '*')
51         skipWhitespace(str, idx + 3)
52       else idx
53     }
54 
55   /** Skips to next occurrence of `\n` or to the position after the `/``**` sequence following index `start`.
56    */
57   def skipToEol(str: String, start: Int): Int =
58     if (start + 2 < str.length && (str charAt start) == '/' && (str charAt (start + 1)) == '*' && (str charAt (start + 2)) == '*') start + 3
59     else if (start < str.length && (str charAt start) != '\n') skipToEol(str, start + 1)
60     else start
61 
62   /** Returns first index following `start` and starting a line (i.e. after skipLineLead) or starting the comment
63    *  which satisfies predicate `p`.
64    */
65   def findNext(str: String, start: Int)(p: Int => Boolean): Int = {
66     val idx = skipLineLead(str, skipToEol(str, start))
67     if (idx < str.length && !p(idx)) findNext(str, idx)(p)
68     else idx
69   }
70 
71   /** Return first index following `start` and starting a line (i.e. after skipLineLead)
72    *  which satisfies predicate `p`.
73    */
74   def findAll(str: String, start: Int)(p: Int => Boolean): List[Int] = {
75     val idx = findNext(str, start)(p)
76     if (idx == str.length) List()
77     else idx :: findAll(str, idx)(p)
78   }

80   /** Produces a string index, which is a list of `sections`, i.e
81    *  pairs of start/end positions of all tagged sections in the string.
82    *  Every section starts with an at sign and extends to the next at sign,
83    *  or to the end of the comment string, but excluding the final two
84    *  characters which terminate the comment.
85    *
86    *  Also take usecases into account - they need to expand until the next
87    *  usecase or the end of the string, as they might include other sections
88    *  of their own
89    */
90   def tagIndex(str: String, p: Int => Boolean = (idx => true)): List[(Int, Int)] = {
91     var indices = findAll(str, 0) (idx => str(idx) == '@' && p(idx))
92     indices = mergeUsecaseSections(str, indices)
93     indices = mergeInheritdocSections(str, indices)
94 
95     indices match {
96       case List() => List()
97       case idxs   => idxs zip (idxs.tail ::: List(str.length - 2))
98     }
99   }
100 
101   /**
102    * Merge sections following an usecase into the usecase comment, so they
103    * can override the parent symbol's sections
104    */
105   def mergeUsecaseSections(str: String, idxs: List[Int]): List[Int] = {
106     idxs.indexWhere(str.startsWith("@usecase", _)) match {
107       case firstUCIndex if firstUCIndex != -1 =>
108         val commentSections = idxs.take(firstUCIndex)
109         val usecaseSections = idxs.drop(firstUCIndex).filter(str.startsWith("@usecase", _))
110         commentSections ::: usecaseSections
111       case _ =>
112         idxs
113     }
114   }
115 
116   /**
117    * Merge the inheritdoc sections, as they never make sense on their own
118    */
119   def mergeInheritdocSections(str: String, idxs: List[Int]): List[Int] =
120     idxs.filterNot(str.startsWith("@inheritdoc", _))

122   /** Does interval `iv` start with given `tag`?
123    */
124   def startsWithTag(str: String, section: (Int, Int), tag: String): Boolean =
125     startsWithTag(str, section._1, tag)
126 
127   def startsWithTag(str: String, start: Int, tag: String): Boolean =
128     str.startsWith(tag, start) && !isIdentifierPart(str charAt (start + tag.length))
129 
130   /** The first start tag of a list of tag intervals,
131    *  or the end of the whole comment string - 2 if list is empty
132    */
133   def startTag(str: String, sections: List[(Int, Int)]) = sections match {
134     case Nil             => str.length - 2
135     case (start, _) :: _ => start
136   }

138   /** A map from parameter names to start/end indices describing all parameter
139    *  sections in `str` tagged with `tag`, where `sections` is the index of `str`.
140    */
141   def paramDocs(str: String, tag: String, sections: List[(Int, Int)]): Map[String, (Int, Int)] =
142     Map() ++ {
143       for (section <- sections if startsWithTag(str, section, tag)) yield {
144         val start = skipWhitespace(str, section._1 + tag.length)
145         str.substring(start, skipIdent(str, start)) -> section
146       }
147     }

149   /** Optionally start and end index of return section in `str`, or `None`
150    *  if `str` does not have a @group. */
151   def groupDoc(str: String, sections: List[(Int, Int)]): Option[(Int, Int)] =
152     sections find (startsWithTag(str, _, "@group"))

155   /** Optionally start and end index of return section in `str`, or `None`
156    *  if `str` does not have a @return.
157    */
158   def returnDoc(str: String, sections: List[(Int, Int)]): Option[(Int, Int)] =
159     sections find (startsWithTag(str, _, "@return"))

161   /** Extracts variable name from a string, stripping any pair of surrounding braces */
162   def variableName(str: String): String =
163     if (str.length >= 2 && (str charAt 0) == '{' && (str charAt (str.length - 1)) == '}')
164       str.substring(1, str.length - 1)
165     else
166       str
167 
168   /** Returns index following variable, or start index if no variable was recognized
169    */
170   def skipVariable(str: String, start: Int): Int = {
171     var idx = start
172     if (idx < str.length && (str charAt idx) == '{') {
173       do idx += 1
174       while (idx < str.length && (str charAt idx) != '}')
175       if (idx < str.length) idx + 1 else start
176     } else {
177       while (idx < str.length && isVarPart(str charAt idx))
178         idx += 1
179       idx
180     }
181   }

183   /** A map from the section tag to section parameters */
184   def sectionTagMap(str: String, sections: List[(Int, Int)]): Map[String, (Int, Int)] =
185     Map() ++ {
186       for (section <- sections) yield
187         extractSectionTag(str, section) -> section
188     }
189 
190   /** Extract the section tag, treating the section tag as an identifier */
191   def extractSectionTag(str: String, section: (Int, Int)): String =
192     str.substring(section._1, skipTag(str, section._1))
193 
194   /** Extract the section parameter */
195   def extractSectionParam(str: String, section: (Int, Int)): String = {
196     val (beg, _) = section
197     assert(str.startsWith("@param", beg) ||
198            str.startsWith("@tparam", beg) ||
199            str.startsWith("@throws", beg))
200 
201     val start = skipWhitespace(str, skipTag(str, beg))
202     val finish = skipIdent(str, start)
203 
204     str.substring(start, finish)
205   }
206 
207   /** Extract the section text, except for the tag and comment newlines */
208   def extractSectionText(str: String, section: (Int, Int)): (Int, Int) = {
209     val (beg, end) = section
210     if (str.startsWith("@param", beg) ||
211         str.startsWith("@tparam", beg) ||
212         str.startsWith("@throws", beg))
213       (skipWhitespace(str, skipIdent(str, skipWhitespace(str, skipTag(str, beg)))), end)
214     else
215       (skipWhitespace(str, skipTag(str, beg)), end)
216   }
217 
218   /** Cleanup section text */
219   def cleanupSectionText(str: String) = {
220     var result = str.trim.replaceAll("\n\\s+\\*\\s+", " \n")
221     while (result.endsWith("\n"))
222       result = result.substring(0, str.length - 1)
223     result
224   }
225 
226 
227   def removeSections(raw: String, xs: String*): String = {
228     val sections = tagIndex(raw)
229 
230     val toBeRemoved = for {
231       section <- xs
232       lines = sections filter { startsWithTag(raw, _, section) }
233     } yield lines
234 
235     val end = startTag(raw, toBeRemoved.flatten.sortBy(_._1).toList)
236 
237     if (end == raw.length - 2) raw else raw.substring(0, end) + "*/"
238   }
