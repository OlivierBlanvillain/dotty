14   case class PickleBufferEntry(num: Int, startIndex: Int, tag: Int, bytes: Array[Byte]) {
15     def isName = tag == TERMname || tag == TYPEname
16     def hasName = tag match {
17       case TYPEsym | ALIASsym | CLASSsym | MODULEsym | VALsym | EXTref | EXTMODCLASSref => true
18       case _                                                                            => false
19     }
20     def readName =
21       if (isName) new String(bytes, "UTF-8")
22       else sys.error("%s is no name" format tagName)
23     def nameIndex =
24       if (hasName) readNat(bytes, 0)
25       else sys.error("%s has no name" format tagName)
26 
27     def tagName = tag2string(tag)
28     override def toString = "%d,%d: %s".format(num, startIndex, tagName)
29   }

100   def readNat(data: Array[Byte], index: Int): Int = {
101     var idx = index
102     var result = 0L
103     var b = 0L
104     do {
105       b = data(idx)
106       idx += 1
107       result = (result << 7) + (b & 0x7f)
108     } while((b & 0x80) != 0L)
109 
110     result.toInt
111   }
