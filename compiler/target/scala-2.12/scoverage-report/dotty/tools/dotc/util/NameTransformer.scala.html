57   /** Replace operator symbols by corresponding expansion strings.
58    *
59    *  @param name the string to encode
60    *  @return     the string with all recognized opchars replaced with their encoding
61    *
62    *  Operator symbols are only recognized if they make up the whole name, or
63    *  if they make up the last part of the name which follows a `_`.
64    */
65   def encode(name: SimpleName): SimpleName = {
66     def loop(len: Int, ops: List[String]): SimpleName = {
67       def convert =
68         if (ops.isEmpty) name
69         else {
70           val buf = new java.lang.StringBuilder
71           buf.append(chrs, name.start, len)
72           for (op <- ops) buf.append(op)
73           termName(buf.toString)
74         }
75       if (len == 0 || name(len - 1) == '_') convert
76       else {
77         val ch = name(len - 1)
78         if (ch <= nops && op2code(ch) != null)
79           loop(len - 1, op2code(ch) :: ops)
80         else if (Chars.isSpecial(ch))
81           loop(len - 1, ch.toString :: ops)
82         else name
83       }
84     }
85     loop(name.length, Nil)
86   }

88   /** Replace operator expansions by the operators themselves.
89    *  Operator expansions are only recognized if they make up the whole name, or
90    *  if they make up the last part of the name which follows a `_`.
91    */
92   def decode(name: SimpleName): SimpleName = {
93     def loop(len: Int, ops: List[Char]): SimpleName = {
94       def convert =
95         if (ops.isEmpty) name
96         else {
97           val buf = new java.lang.StringBuilder
98           buf.append(chrs, name.start, len)
99           for (op <- ops) buf.append(op)
100           termName(buf.toString)
101         }
102       if (len == 0 || name(len - 1) == '_') convert
103       else if (Chars.isSpecial(name(len - 1))) loop(len - 1, name(len - 1) :: ops)
104       else {
105         val idx = name.lastIndexOf('$', len - 1)
106         if (idx >= 0 && idx + 2 < len)
107           str2op.get(name.sliceToString(idx, len)) match {
108             case Some(ch) => loop(idx, ch :: ops)
109             case None => name
110           }
111         else name
112       }
113     }
114     loop(name.length, Nil)
115   }
