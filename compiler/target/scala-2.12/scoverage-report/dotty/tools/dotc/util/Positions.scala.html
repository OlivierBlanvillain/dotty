5 /** Position format in little endian:
6  *  Start: unsigned 26 Bits (works for source files up to 64M)
7  *  End: unsigned 26 Bits
8  *  Point: unsigned 12 Bits relative to start
9  *  NoPosition encoded as -1L (this is a normally invalid position
10  *  because point would lie beyond end.
11  */
12 object Positions {
13 
14   private final val StartEndBits = 26
15   private final val StartEndMask = (1L << StartEndBits) - 1
16   private final val SyntheticPointDelta = (1 << (64 - StartEndBits * 2)) - 1
17 
18   /** The maximal representable offset in a position */
19   private final val MaxOffset = StartEndMask
20 
21   /** Convert offset `x` to an integer by sign extending the original
22    *  field of `StartEndBits` width.
23    */
24   def offsetToInt(x: Int) =
25     x << (32 - StartEndBits) >> (32 - StartEndBits)
26 
27   /** A position indicates a range between a start offset and an end offset.
28    *  Positions can be synthetic or source-derived. A source-derived position
29    *  has in addition a point lies somewhere between start and end. The point
30    *  is roughly where the ^ would go if an error was diagnosed at that position.
31    *  All quantities are encoded opaquely in a Long.
32    */
33   class Position(val coords: Long) extends AnyVal {
34 
35     /** Is this position different from NoPosition? */
36     def exists = this != NoPosition
37 
38     /** The start of this position. */
39     def start: Int = {
40       assert(exists)
41       (coords & StartEndMask).toInt
42     }
43 
44     /** The end of this position */
45     def end: Int = {
46       assert(exists)
47       ((coords >>> StartEndBits) & StartEndMask).toInt
48     }
49 
50     /** The point of this position, returns start for synthetic positions */
51     def point: Int = {
52       assert(exists)
53       val poff = pointDelta
54       if (poff == SyntheticPointDelta) start else start + poff
55     }
56 
57     /** The difference between point and start in this position */
58     def pointDelta =
59       (coords >>> (StartEndBits * 2)).toInt
60 
61     def orElse(that: Position) =
62       if (this.exists) this else that
63 
64     /** The union of two positions. This is the least range that encloses
65      *  both positions. It is always a synthetic position.
66      */
67     def union(that: Position) =
68       if (!this.exists) that
69       else if (!that.exists) this
70       else Position(this.start min that.start, this.end max that.end, this.point)
71 
72     /** Does the range of this position contain the one of that position? */
73     def contains(that: Position): Boolean =
74       !that.exists || exists && (start <= that.start && end >= that.end)
75 
76     /** Is this position synthetic? */
77     def isSynthetic = pointDelta == SyntheticPointDelta
78 
79     /** Is this position source-derived? */
80     def isSourceDerived = !isSynthetic
81 
82     /** Is this a zero-extent position? */
83     def isZeroExtent = start == end
84 
85      /** A position where all components are shifted by a given `offset`
86      *  relative to this position.
87      */
88     def shift(offset: Int) =
89       if (exists) fromOffsets(start + offset, end + offset, pointDelta)
90       else this
91 
92     /** The zero-extent position with start and end at the point of this position */
93     def focus = if (exists) Position(point) else NoPosition
94 
95     /** The zero-extent position with start and end at the start of this position */
96     def startPos = if (exists) Position(start) else NoPosition
97 
98     /** The zero-extent position with start and end at the end of this position */
99     def endPos = if (exists) Position(end) else NoPosition
100 
101     /** A copy of this position with a different start */
102     def withStart(start: Int) =
103       if (exists) fromOffsets(start, this.end, if (isSynthetic) SyntheticPointDelta else this.point - start)
104       else this
105 
106     /** A copy of this position with a different end */
107     def withEnd(end: Int) =
108       if (exists) fromOffsets(this.start, end, pointDelta)
109       else this
110 
111     /** A copy of this position with a different point */
112     def withPoint(point: Int) =
113       if (exists) fromOffsets(this.start, this.end, point - this.start)
114       else this
115 
116     /** A synthetic copy of this position */
117     def toSynthetic = if (isSynthetic) this else Position(start, end)
118 
119     override def toString = {
120       val (left, right) = if (isSynthetic) ("<", ">") else ("[", "]")
121       if (exists)
122         s"$left$start..${if (point == start) "" else s"$point.."}$end$right"
123       else
124         s"${left}no position${right}"
125     }
126   }
127 
128   private def fromOffsets(start: Int, end: Int, pointDelta: Int) = {
129     //assert(start <= end || start == 1 && end == 0, s"$start..$end")
130     new Position(
131       (start & StartEndMask).toLong |
132       ((end & StartEndMask).toLong << StartEndBits) |
133       (pointDelta.toLong << (StartEndBits * 2)))
134   }
135 
136   /** A synthetic position with given start and end */
137   def Position(start: Int, end: Int): Position =
138     fromOffsets(start, end, SyntheticPointDelta)
139 
140   /** A source-derived position with given start, end, and point delta */
141   def Position(start: Int, end: Int, point: Int): Position = {
142     val pointDelta = (point - start) max 0
143     fromOffsets(start, end, if (pointDelta >= SyntheticPointDelta) 0 else pointDelta)
144   }
145 
146   /** A synthetic zero-extent position that starts and ends at given `start`. */
147   def Position(start: Int): Position = Position(start, start)
148 
149   /** A sentinel for a non-existing position */
150   val NoPosition = Position(1, 0)
151 
152   /** The coordinate of a symbol. This is either an index or
153    *  a zero-range position.
154    */
155   class Coord(val encoding: Int) extends AnyVal {
156     def isIndex = encoding > 0
157     def isPosition = encoding <= 0
158     def toIndex: Int = {
159       assert(isIndex)
160       encoding - 1
161     }
162     def toPosition = {
163       assert(isPosition)
164       if (this == NoCoord) NoPosition else Position(-1 - encoding)
165     }
166   }
167 
168   /** An index coordinate */
169   implicit def indexCoord(n: Int): Coord = new Coord(n + 1)
170   implicit def positionCoord(pos: Position): Coord =
171     if (pos.exists) new Coord(-(pos.point + 1))
172     else NoCoord
173 
174   /** A sentinel for a missing coordinate */
175   val NoCoord = new Coord(0)
176 }
