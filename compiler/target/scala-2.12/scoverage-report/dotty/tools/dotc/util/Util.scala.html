7    *  `candidates(i)` is closest to `x`, determined by binary search, or -1
8    *  if `x < candidates(0)`.
9    *  @param  hint   If between 0 and `candidates.length` use this
10    *                 as the first search point, otherwise use
11    *                 `candidates.length/2`.
12    *  @pre   candidates is sorted
13    */
14   def bestFit(candidates: Array[Int], length: Int, x: Int, hint: Int = -1): Int = {
15     def recur(lo: Int, hi: Int, mid: Int): Int =
16       if (x < candidates(mid))
17         recur(lo, mid - 1, (lo + mid - 1) / 2)
18       else if (mid + 1 < length && x >= candidates(mid + 1))
19         recur(mid + 1, hi, (mid + 1 + hi) / 2)
20       else mid
21     val initMid = if (0 <= hint && hint < length) hint else length / 2
22     if (length == 0 || x < candidates(0)) -1
23     else recur(0, length, initMid)
24   }
