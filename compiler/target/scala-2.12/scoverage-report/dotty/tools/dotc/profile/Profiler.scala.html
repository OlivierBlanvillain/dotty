93 private [profile] class RealProfiler(reporter : ProfileReporter)(implicit ctx: Context) extends Profiler with NotificationListener {
94   def completeBackground(threadRange: ProfileRange): Unit = {

103     case gc => println(s"Cant connect gcListener to ${gc.getClass}")

106   private val mainThread = Thread.currentThread()

108   private[profile] def snapThread(idleTimeNanos: Long): ProfileSnap = {
109     import RealProfiler._
110     val current = Thread.currentThread()
111 
112     ProfileSnap(
113       threadId = current.getId,
114       threadName = current.getName,
115       snapTimeNanos = System.nanoTime(),
116       idleTimeNanos = idleTimeNanos,
117       cpuTimeNanos = threadMx.getCurrentThreadCpuTime,
118       userTimeNanos = threadMx.getCurrentThreadUserTime,
119       allocatedBytes = threadMx.getThreadAllocatedBytes(Thread.currentThread().getId),
120       heapBytes = readHeapUsage()
121     )
122   }
123   private def readHeapUsage() = RealProfiler.memoryMx.getHeapMemoryUsage.getUsed

125   private def doGC: Unit = {
126     System.gc()
127     System.runFinalization()
128   }

132   override def finished(): Unit = {
133     //we may miss a GC event if gc is occurring as we call this
134     RealProfiler.gcMx foreach {
135       case emitter: NotificationEmitter => emitter.removeNotificationListener(this)
136       case gc =>
137     }
138     reporter.close(this)
139   }

142   override def handleNotification(notification: Notification, handback: scala.Any): Unit = {
143     import java.lang.{Long => jLong}
144     import java.lang.{Integer => jInt}
145     val reportNs = System.nanoTime()
146     val data = notification.getUserData
147     val seq = notification.getSequenceNumber
148     val message = notification.getMessage
149     val tpe = notification.getType
150     val time= notification.getTimeStamp
151     data match {
152       case cd: CompositeData if tpe == "com.sun.management.gc.notification" =>
153         val name = cd.get("gcName").toString
154         val action = cd.get("gcAction").toString
155         val cause = cd.get("gcCause").toString
156         val info = cd.get("gcInfo").asInstanceOf[CompositeData]
157         val duration = info.get("duration").asInstanceOf[jLong].longValue()
158         val startTime = info.get("startTime").asInstanceOf[jLong].longValue()
159         val endTime = info.get("endTime").asInstanceOf[jLong].longValue()
160         val threads = info.get("GcThreadCount").asInstanceOf[jInt].longValue()
161         reporter.reportGc(GcEventData("", reportNs, startTime, endTime, name, action, cause, threads))
162     }
163   }

165   override def afterPhase(phase: Phase, snapBefore: ProfileSnap): Unit = {
166     assert(mainThread eq Thread.currentThread())
167     val initialSnap = snapThread(0)
168     if (ctx.settings.YprofileExternalTool.value.contains(phase.toString)) {
169       println("Profile hook stop")
170       ExternalToolHook.after()
171     }
172     val finalSnap = if (ctx.settings.YprofileRunGcBetweenPhases.value.contains(phase.toString)) {
173       doGC
174       initialSnap.updateHeap(readHeapUsage())
175     } else initialSnap
176 
177     reporter.reportForeground(this, ProfileRange(snapBefore, finalSnap, phase, "", 0, Thread.currentThread))
178   }
179 
180   override def beforePhase(phase: Phase): ProfileSnap = {
181     assert(mainThread eq Thread.currentThread())
182     if (ctx.settings.YprofileRunGcBetweenPhases.value.contains(phase.toString))
183       doGC
184     if (ctx.settings.YprofileExternalTool.value.contains(phase.toString)) {
185       println("Profile hook start")
186       ExternalToolHook.before()
187     }
188     snapThread(0)
189   }
