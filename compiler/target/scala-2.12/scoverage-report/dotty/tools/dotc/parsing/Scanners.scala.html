122     def intVal(negated: Boolean): Long = {
123       if (token == CHARLIT && !negated) {
124         charVal
125       } else {
126         var value: Long = 0
127         val divider = if (base == 10) 1 else 2
128         val limit: Long =
129           if (token == LONGLIT) Long.MaxValue else Int.MaxValue
130         var i = 0
131         val len = strVal.length
132         while (i < len) {
133           val d = digit2int(strVal charAt i, base)
134           if (d < 0) {
135             error("malformed integer number")
136             return 0
137           }
138           if (value < 0 ||
139             limit / (base / divider) < value ||
140             limit - (d / divider) < value * (base / divider) &&
141               !(negated && limit == value * base - 1 + d)) {
142             error("integer number too large")
143             return 0
144           }
145           value = value * base + d
146           i += 1
147         }
148         if (negated) -value else value
149       }
150     }

588       def appendToComment(ch: Char) =
589         if (keepComments) commentBuf.append(ch)
590       def nextChar() = {
591         appendToComment(ch)
592         Scanner.this.nextChar()
593       }
594       def skipLine(): Unit = {
595         nextChar()
596         if ((ch != CR) && (ch != LF) && (ch != SU)) skipLine()
597       }
598       @tailrec
599       def skipComment(): Unit = {
600         if (ch == '/') {
601           nextChar()
602           if (ch == '*') nestedComment()
603           skipComment()
604         }
605         else if (ch == '*') {
606           do nextChar() while (ch == '*')
607           if (ch == '/') nextChar()
608           else skipComment()
609         }
610         else if (ch == SU) incompleteInputError("unclosed comment")
611         else { nextChar(); skipComment() }

626       }

628       if (ch == '/') { skipLine(); finishComment() }
629       else if (ch == '*') { nextChar(); skipComment(); finishComment() }
630       else false
