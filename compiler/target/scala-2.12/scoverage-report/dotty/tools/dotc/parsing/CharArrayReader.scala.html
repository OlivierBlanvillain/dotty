72   /** Interpret \\uxxxx escapes */
73   private def potentialUnicode(): Unit = {
74     def evenSlashPrefix: Boolean = {
75       var p = charOffset - 2
76       while (p >= 0 && buf(p) == '\\') p -= 1
77       (charOffset - p) % 2 == 0
78     }
79     def udigit: Int = {
80       if (charOffset >= buf.length) {
81         // Since the positioning code is very insistent about throwing exceptions,
82         // we have to decrement the position so our error message can be seen, since
83         // we are one past EOF.  This happens with e.g. val x = \ u 1 <EOF>
84         error("incomplete unicode escape", charOffset - 1)
85         SU
86       }
87       else {

89         if (d >= 0) charOffset += 1

91         d
92       }
93     }
94     if (charOffset < buf.length && buf(charOffset) == 'u' && decodeUni && evenSlashPrefix) {
95       do charOffset += 1
