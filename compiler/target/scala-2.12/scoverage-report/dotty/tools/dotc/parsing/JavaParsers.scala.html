56       var nparens = 0
57       var nbraces = 0
58       while (true) {
59         in.token match {
60           case EOF =>
61             return
62           case SEMI =>
63             if (nparens == 0 && nbraces == 0) return
64           case RPAREN =>
65             nparens -= 1
66           case RBRACE =>
67             if (nbraces == 0) return
68             nbraces -= 1
69           case LPAREN =>
70             nparens += 1
71           case LBRACE =>
72             nbraces += 1
73           case _ =>
74         }
75         in.nextToken()
76       }

113     def makeTemplate(parents: List[Tree], stats: List[Tree], tparams: List[TypeDef], needsDummyConstr: Boolean) = {
114       def pullOutFirstConstr(stats: List[Tree]): (Tree, List[Tree]) = stats match {
115         case (meth: DefDef) :: rest if meth.name == nme.CONSTRUCTOR => (meth, rest)
116         case first :: rest =>
117           val (constr, tail) = pullOutFirstConstr(rest)
118           (constr, first :: tail)
119         case nil => (EmptyTree, nil)
120       }
121       var (constr1, stats1) = pullOutFirstConstr(stats)
122       if (constr1 == EmptyTree) constr1 = makeConstructor(List(), tparams)
123       // A dummy first constructor is needed for Java classes so that the real constructors see the
124       // import of the companion object. The constructor has parameter of type Unit so no Java code
125       // can call it.
126       if (needsDummyConstr) {
127         stats1 = constr1 :: stats1
128         constr1 = makeConstructor(List(scalaDot(tpnme.Unit)), tparams, Flags.JavaDefined | Flags.PrivateLocal)
129       }
130       Template(constr1.asInstanceOf[DefDef], parents, EmptyValDef, stats1)
131     }

475       val inInterface = definesInterface(parentToken)
476       val tparams = if (in.token == LT) typeParams(Flags.JavaDefined | Flags.Param) else List()

478       var rtpt =
479         if (isVoid)

481             in.nextToken()
482             TypeTree(UnitType)
483           }
484         else typ()

487         case Ident(name) => name
488         case _ => nme.EMPTY
489       }

491         // constructor declaration
492         val vparams = formalParams()
493         optThrows()

496             DefDef(nme.CONSTRUCTOR, parentTParams,
497                 List(vparams), TypeTree(), methodBody()).withMods(mods)
498           }
499         }
500       } else {

502         if (mods is Flags.Abstract) mods1 = mods &~ Flags.Abstract
503         nameOffset = in.offset

506           // method declaration
507           val vparams = formalParams()

545     /** Parse a sequence of field declarations, separated by commas.
546       *  This one is tricky because a comma might also appear in an
547       *  initializer. Since we don't parse initializers we don't know
548       *  what the comma signifies.
549       *  We solve this with a second list buffer `maybe` which contains
550       *  potential variable definitions.
551       *  Once we have reached the end of the statement, we know whether
552       *  these potential definitions are real or not.
553       */
554     def fieldDecls(start: Offset, firstNameOffset: Offset, mods: Modifiers, tpt: Tree, name: Name): List[Tree] = {
555       val buf = ListBuffer[Tree](
556           atPos(start, firstNameOffset) { varDecl(mods, tpt, name.toTermName) })
557       val maybe = new ListBuffer[Tree] // potential variable definitions.
558       while (in.token == COMMA) {
559         in.nextToken()
560         if (in.token == IDENTIFIER) { // if there's an ident after the comma ...
561         val nextNameOffset = in.offset
562         val name = ident()
563           if (in.token == EQUALS || in.token == SEMI) { // ... followed by a `=` or `;`, we know it's a real variable definition
564             buf ++= maybe
565             buf += atPos(start, nextNameOffset) { varDecl(mods, tpt, name.toTermName) }
566             maybe.clear()
567           } else if (in.token == COMMA) { // ... if there's a comma after the ident, it could be a real vardef or not.
568             maybe += atPos(start, nextNameOffset) { varDecl(mods, tpt, name.toTermName) }
569           } else { // ... if there's something else we were still in the initializer of the
570             // previous var def; skip to next comma or semicolon.
571             skipTo(COMMA, SEMI)
572             maybe.clear()
573           }
574         } else { // ... if there's no ident following the comma we were still in the initializer of the
575           // previous var def; skip to next comma or semicolon.
576           skipTo(COMMA, SEMI)
577           maybe.clear()
578         }
579       }
580       if (in.token == SEMI) {
581         buf ++= maybe // every potential vardef that survived until here is real.
582       }
583       buf.toList
584     }

613     // when each performs the import in the enclosing class's scope.
