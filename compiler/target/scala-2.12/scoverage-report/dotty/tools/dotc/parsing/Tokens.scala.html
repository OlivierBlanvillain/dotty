131   def buildKeywordArray(keywords: TokenSet) = {
132     def start(tok: Token) = tokenString(tok).toTermName.asSimpleName.start
133     def sourceKeywords = keywords.toList.filter { (kw: Token) =>
134       val ts = tokenString(kw)
135       (ts != null) && !ts.contains(' ')
136     }

138     val lastKeywordStart = sourceKeywords.map(start).max
139 
140     val arr = Array.fill(lastKeywordStart + 1)(IDENTIFIER)
141     for (kw <- sourceKeywords) arr(start(kw)) = kw
