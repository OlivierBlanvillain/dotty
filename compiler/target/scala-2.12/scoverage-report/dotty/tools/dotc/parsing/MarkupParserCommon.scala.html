96   /** actually, Name ::= (Letter | '_' | ':') (NameChar)*  but starting with ':' cannot happen
97    *  Name ::= (Letter | '_') (NameChar)*
98    *
99    *  see  [5] of XML 1.0 specification
100    *
101    *  pre-condition:  ch != ':' // assured by definition of XMLSTART token
102    *  post-condition: name does neither start, nor end in ':'
103    */
104   def xName: String = {
105     if (ch == SU)
106       truncatedError("")
107     else if (!isNameStart(ch))
108       return errorAndResult("name expected, but char '%s' cannot start a name" format ch, "")
109 
110     val buf = new StringBuilder
111 
112     do buf append ch_returning_nextch
113     while (isNameChar(ch))
114 
115     if (buf.last == ':') {
116       reportSyntaxError( "name cannot end in ':'" )
117       buf.toString dropRight 1
118     }
119     else buf.toString
120   }

122   private def attr_unescape(s: String) = s match {
123     case "lt"     => "<"

221   /** Take characters from input stream until given String "until"
222    *  is seen.  Once seen, the accumulated characters are passed
223    *  along with the current Position to the supplied handler function.
224    */
225   protected def xTakeUntil[T](
226     handler: (PositionType, String) => T,
227     positioner: () => PositionType,
228     until: String): T =
229   {
230     val sb = new StringBuilder
231     val head = until.head
232     val rest = until.tail
233 
234     while (true) {
235       if (ch == head && peek(rest))
236         return handler(positioner(), sb.toString)
237       else if (ch == SU)
238         truncatedError("")  // throws TruncatedXMLControl in compiler
239 
240       sb append ch
241       nextch()
242     }
243     unreachable
244   }

246   /** Create a non-destructive lookahead reader and see if the head
247    *  of the input would match the given String.  If yes, return true
248    *  and drop the entire String from input; if no, return false
249    *  and leave input unchanged.
250    */
251   private def peek(lookingFor: String): Boolean =
252     (lookahead() take lookingFor.length sameElements lookingFor.iterator) && {
253       // drop the chars from the real reader (all lookahead + orig)
254       (0 to lookingFor.length) foreach (_ => nextch())
255       true
256     }
