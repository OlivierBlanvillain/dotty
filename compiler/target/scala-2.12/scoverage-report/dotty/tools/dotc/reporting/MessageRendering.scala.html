18   /** Remove ANSI coloring from `str`, useful for getting real length of
19     * strings
20     *
21     * @return string stripped of ANSI escape codes
22     */
23   def stripColor(str: String): String =
24     str.replaceAll("\u001b\\[.*?m", "")

26   /** When inlining a method call, if there's an error we'd like to get the
27     * outer context and the `pos` at which the call was inlined.
28     *
29     * @return a list of strings with inline locations
30     */
31   def outer(pos: SourcePosition, prefix: String)(implicit ctx: Context): List[String] =
32     if (pos.outer.exists) {
33        s"$prefix| This location is in code that was inlined at ${pos.outer}" ::
34        outer(pos.outer, prefix)
35     } else Nil

37   /** Get the sourcelines before and after the position, as well as the offset
38     * for rendering line numbers
39     *
40     * @return (lines before error, lines after error, line numbers offset)
41     */
42   def sourceLines(pos: SourcePosition)(implicit ctx: Context): (List[String], List[String], Int) = {
43     var maxLen = Int.MinValue
44     def render(offsetAndLine: (Int, String)): String = {
45       val (offset, line) = offsetAndLine
46       val lineNbr = pos.source.offsetToLine(offset)
47       val prefix = s"${lineNbr + 1} |"
48       maxLen = math.max(maxLen, prefix.length)
49       val lnum = Red(" " * math.max(0, maxLen - prefix.length) + prefix).show
50       lnum + line.stripLineEnd
51     }
52 
53     def linesFrom(arr: Array[Char]): List[String] = {
54       def pred(c: Char) = (c: @switch) match {
55         case LF | CR | FF | SU => true
56         case _ => false
57       }
58       val (line, rest0) = arr.span(!pred(_))
59       val (_, rest) = rest0.span(pred)
60       new String(line) :: { if (rest.isEmpty) Nil else linesFrom(rest) }
61     }
62 
63     val syntax =
64       if (ctx.settings.color.value != "never")
65         SyntaxHighlighting(pos.linesSlice).toArray
66       else pos.linesSlice
67     val lines = linesFrom(syntax)
68     val (before, after) = pos.beforeAndAfterPoint
69 
70     (
71       before.zip(lines).map(render),
72       after.zip(lines.drop(before.length)).map(render),
73       maxLen
74     )
75   }

77   /** The column markers aligned under the error */
78   def columnMarker(pos: SourcePosition, offset: Int)(implicit ctx: Context): String = {
79     val prefix = " " * (offset - 1)
80     val padding = pos.startColumnPadding
81     val carets = Red {
82       if (pos.startLine == pos.endLine)
83         "^" * math.max(1, pos.endColumn - pos.startColumn)
84       else "^"
85     }
86     s"$prefix|$padding${carets.show}"
87   }

89   /** The error message (`msg`) aligned under `pos`
90     *
91     * @return aligned error message
92     */
93   def errorMsg(pos: SourcePosition, msg: String, offset: Int)(implicit ctx: Context): String = {
94     val padding = msg.lines.foldLeft(pos.startColumnPadding) { (pad, line) =>
95       val lineLength = stripColor(line).length
96       val maxPad = math.max(0, ctx.settings.pageWidth.value - offset - lineLength) - offset
97 
98       if (maxPad < pad.length) " " * maxPad
99       else pad
100     }
101 
102     msg.lines
103       .map { line => " " * (offset - 1) + "|" + padding + line}
104       .mkString(sys.props("line.separator"))
105   }

107   /** The separator between errors containing the source file and error type
108     *
109     * @return separator containing error location and kind
110     */
111   def posStr(pos: SourcePosition, diagnosticLevel: String, message: Message)(implicit ctx: Context): String =
112     if (pos.exists) Blue({
113       val file = s"${pos.source.file.toString}:${pos.line + 1}:${pos.column}"
114       val errId =
115         if (message.errorId ne ErrorMessageID.NoExplanationID) {
116           val errorNumber = message.errorId.errorNumber()
117           s"[E${"0" * (3 - errorNumber.toString.length) + errorNumber}] "
118         } else ""
119       val kind =
120         if (message.kind == "") diagnosticLevel
121         else s"${message.kind} $diagnosticLevel"
122       val prefix = s"-- ${errId}${kind}: $file "
123 
124       prefix +
125         ("-" * math.max(ctx.settings.pageWidth.value - stripColor(prefix).length, 0))
126     }).show else ""

128   /** Explanation rendered under "Explanation" header */
129   def explanation(m: Message)(implicit ctx: Context): String = {
130     val sb = new StringBuilder(
131       hl"""|
132            |${Blue("Explanation")}
133            |${Blue("===========")}"""
134     )
135     sb.append('\n').append(m.explanation)
136     if (m.explanation.lastOption != Some('\n')) sb.append('\n')
137     sb.toString
138   }

140   /** The whole message rendered from `msg` */
141   def messageAndPos(msg: Message, pos: SourcePosition, diagnosticLevel: String)(implicit ctx: Context): String = {
142     val sb = mutable.StringBuilder.newBuilder
143     val posString = posStr(pos, diagnosticLevel, msg)
144     if (posString.nonEmpty) sb.append(posString).append('\n')
145     if (pos.exists) {
146       val (srcBefore, srcAfter, offset) = sourceLines(pos)
147       val marker = columnMarker(pos, offset)
148       val err = errorMsg(pos, msg.msg, offset)
149       sb.append((srcBefore ::: marker :: err :: outer(pos, " " * (offset - 1)) ::: srcAfter).mkString("\n"))
150     } else sb.append(msg.msg)
151     sb.toString
152   }
