129 /**
130  * This interface provides methods to issue information, warning and
131  * error messages.
132  */
133 abstract class Reporter extends interfaces.ReporterResult {
134 
135   /** Report a diagnostic */
136   def doReport(m: MessageContainer)(implicit ctx: Context): Unit
137 
138   /** Whether very long lines can be truncated.  This exists so important
139    *  debugging information (like printing the classpath) is not rendered
140    *  invisible due to the max message length.
141    */
142   private[this] var _truncationOK: Boolean = true
143   def truncationOK = _truncationOK
144   def withoutTruncating[T](body: => T): T = {
145     val saved = _truncationOK
146     _truncationOK = false
147     try body
148     finally _truncationOK = saved
149   }
150 
151   type ErrorHandler = MessageContainer => Context => Unit
152   private[this] var incompleteHandler: ErrorHandler = d => c => report(d)(c)
153   def withIncompleteHandler[T](handler: ErrorHandler)(op: => T): T = {
154     val saved = incompleteHandler
155     incompleteHandler = handler
156     try op
157     finally incompleteHandler = saved
158   }
159 
160   var errorCount = 0
161   var warningCount = 0
162   def hasErrors = errorCount > 0
163   def hasWarnings = warningCount > 0
164   private[this] var errors: List[Error] = Nil
165   def allErrors = errors
166 
167   /** Have errors been reported by this reporter, or in the
168    *  case where this is a StoreReporter, by an outer reporter?
169    */
170   def errorsReported = hasErrors
171 
172   private[this] var reportedFeaturesUseSites = Set[Symbol]()
173   def isReportedFeatureUseSite(featureTrait: Symbol): Boolean = reportedFeaturesUseSites.contains(featureTrait)
174   def reportNewFeatureUseSite(featureTrait: Symbol): Unit = reportedFeaturesUseSites += featureTrait
175 
176   val unreportedWarnings = new mutable.HashMap[String, Int] {
177     override def default(key: String) = 0
178   }
179 
180   def report(m: MessageContainer)(implicit ctx: Context): Unit =
181     if (!isHidden(m)) {
182       doReport(m)(ctx.addMode(Mode.Printing))
183       m match {
184         case m: ConditionalWarning if !m.enablingOption.value => unreportedWarnings(m.enablingOption.name) += 1
185         case m: Warning => warningCount += 1
186         case m: Error =>
187           errors = m :: errors
188           errorCount += 1
189         case m: Info => // nothing to do here
190         // match error if d is something else
191       }
192     }
193 
194   def incomplete(m: MessageContainer)(implicit ctx: Context): Unit =
195     incompleteHandler(m)(ctx)
196 
197   /** Summary of warnings and errors */
198   def summary: String = {
199     val b = new mutable.ListBuffer[String]
200     if (warningCount > 0)
201       b += countString(warningCount, "warning") + " found"
202     if (errorCount > 0)
203       b += countString(errorCount, "error") + " found"
204     for ((settingName, count) <- unreportedWarnings)
205       b += s"there were $count ${settingName.tail} warning(s); re-run with $settingName for details"
206     b.mkString("\n")
207   }
208 
209   /** Print the summary of warnings and errors */
210   def printSummary(implicit ctx: Context): Unit = {
211     val s = summary
212     if (s != "") ctx.echo(s)
213   }
214 
215   /** Returns a string meaning "n elements". */
216   protected def countString(n: Int, elements: String): String = n match {
217     case 0 => "no " + elements + "s"
218     case 1 => "one " + elements
219     case 2 => "two " + elements + "s"
220     case 3 => "three " + elements + "s"
221     case 4 => "four " + elements + "s"
222     case _ => n + " " + elements + "s"
223   }
224 
225   /** Should this diagnostic not be reported at all? */
226   def isHidden(m: MessageContainer)(implicit ctx: Context): Boolean =
227     ctx.mode.is(Mode.Printing)
228 
229   /** Does this reporter contain not yet reported errors or warnings? */
230   def hasPending: Boolean = false
231 
232   /** If this reporter buffers messages, remove and return all buffered messages. */
233   def removeBufferedMessages(implicit ctx: Context): List[MessageContainer] = Nil
234 
235   /** Issue all error messages in this reporter to next outer one, or make sure they are written. */
236   def flush()(implicit ctx: Context): Unit =
237     removeBufferedMessages.foreach(ctx.reporter.report)
