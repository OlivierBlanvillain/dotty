10 /** This trait implements `isHidden` so that multiple messages per position
11   * are suppressed, unless they are of increasing severity. */
12 trait UniqueMessagePositions extends Reporter {
13 
14   private val positions = new mutable.HashMap[(SourceFile, Int), Int]
15 
16   /** Logs a position and returns true if it was already logged.
17    *  @note  Two positions are considered identical for logging if they have the same point.
18    */
19   override def isHidden(m: MessageContainer)(implicit ctx: Context): Boolean =
20     super.isHidden(m) || {
21       m.pos.exists && {
22         var shouldHide = false
23         for (pos <- m.pos.start to m.pos.end) {
24           positions get (ctx.source, pos) match {
25             case Some(level) if level >= m.level => shouldHide = true
26             case _ => positions((ctx.source, pos)) = m.level
27           }
28         }
29         shouldHide
30       }
31     }
32 }
