125   def toText(tp: Type): Text = controlled {
126     homogenize(tp) match {
127       case tp: TypeType =>
128         toTextRHS(tp)
129       case tp: TermRef
130       if !tp.denotationIsCurrent && !homogenizedView || // always print underlying when testing picklers
131          tp.symbol.is(Module) || tp.symbol.name == nme.IMPORT =>
132         toTextRef(tp) ~ ".type"
133       case tp: TermRef if tp.denot.isOverloaded =>
134         "<overloaded " ~ toTextRef(tp) ~ ">"
135       case tp: TypeRef =>
136         if (printWithoutPrefix.contains(tp.symbol))
137           toText(tp.name)
138         else
139           toTextPrefix(tp.prefix) ~ selectionString(tp)
140       case tp: TermParamRef =>
141         ParamRefNameString(tp) ~ ".type"
142       case tp: TypeParamRef =>
143         ParamRefNameString(tp) ~ lambdaHash(tp.binder)
144       case tp: SingletonType =>
145         toTextLocal(tp.underlying) ~ "(" ~ toTextRef(tp) ~ ")"
146       case AppliedType(tycon, args) =>
147         (toTextLocal(tycon) ~ "[" ~ Text(args map argText, ", ") ~ "]").close
148       case tp: RefinedType =>
149         val parent :: (refined: List[RefinedType @unchecked]) =
150           refinementChain(tp).reverse
151         toTextLocal(parent) ~ "{" ~ Text(refined map toTextRefinement, "; ").close ~ "}"
152       case tp: RecType =>
153         try {
154           openRecs = tp :: openRecs
155           "{" ~ selfRecName(openRecs.length) ~ " => " ~ toTextGlobal(tp.parent) ~ "}"
156         }
157         finally openRecs = openRecs.tail
158       case AndType(tp1, tp2) =>
159         changePrec(AndPrec) { toText(tp1) ~ " & " ~ toText(tp2) }
160       case OrType(tp1, tp2) =>
161         changePrec(OrPrec) { toText(tp1) ~ " | " ~ toText(tp2) }
162       case tp: ErrorType =>
163         s"<error ${tp.msg.msg}>"
164       case tp: WildcardType =>
165         if (tp.optBounds.exists) "(?" ~ toTextRHS(tp.bounds) ~ ")" else "?"
166       case NoType =>
167         "<notype>"
168       case NoPrefix =>
169         "<noprefix>"
170       case tp: MethodType =>
171         changePrec(GlobalPrec) {
172           (if (tp.isImplicitMethod) "(implicit " else "(") ~ paramsText(tp) ~
173           (if (tp.resultType.isInstanceOf[MethodType]) ")" else "): ") ~
174           toText(tp.resultType)
175         }
176       case tp: ExprType =>
177         changePrec(GlobalPrec) { "=> " ~ toText(tp.resultType) }
178       case tp: TypeLambda =>
179         changePrec(GlobalPrec) {
180           "[" ~ paramsText(tp) ~ "]" ~ lambdaHash(tp) ~
181           (Str(" => ") provided !tp.resultType.isInstanceOf[MethodType]) ~
182           toTextGlobal(tp.resultType)
183         }
184       case AnnotatedType(tpe, annot) =>
185         toTextLocal(tpe) ~ " " ~ toText(annot)
186       case tp: TypeVar =>
187         if (tp.isInstantiated)
188           toTextLocal(tp.instanceOpt) ~ (Str("^") provided ctx.settings.YprintDebug.value)
189         else {
190           val constr = ctx.typerState.constraint
191           val bounds =
192             if (constr.contains(tp)) constr.fullBounds(tp.origin)(ctx.addMode(Mode.Printing))
193             else TypeBounds.empty
194           if (bounds.isAlias) toText(bounds.lo) ~ (Str("^") provided ctx.settings.YprintDebug.value)
195           else if (ctx.settings.YshowVarBounds.value) "(" ~ toText(tp.origin) ~ "?" ~ toText(bounds) ~ ")"
196           else toText(tp.origin)
197         }
198       case tp: LazyRef =>
199         def refTxt =
200           try toTextGlobal(tp.ref)
201           catch {
202             case ex: Throwable => Str("...")
203           }
204         "LazyRef(" ~ refTxt ~ ")"
205       case _ =>
206         tp.fallbackToText(this)
207     }
208   }.close
