18   /** General purpose string formatter, with the following features:
19    *
20    *  1) On all Showables, `show` is called instead of `toString`
21    *  2) Exceptions raised by a `show` are handled by falling back to `toString`.
22    *  3) Sequences can be formatted using the desired separator between two `%` signs,
23    *     eg `i"myList = (${myList}%, %)"`
24    *  4) Safe handling of multi-line margins. Left margins are skipped om the parts
25    *     of the string context *before* inserting the arguments. That way, we guard
26    *     against accidentally treating an interpolated value as a margin.
27    */
28   class StringFormatter(protected val sc: StringContext) {
29 
30     protected def showArg(arg: Any)(implicit ctx: Context): String = arg match {
31       case arg: Showable =>
32         try arg.show
33         catch {
34           case NonFatal(ex)
35           if !ctx.mode.is(Mode.PrintShowExceptions) &&
36              !ctx.settings.YshowPrintErrors.value =>
37             s"[cannot display due to $ex, raw string = ${arg.toString}]"
38         }
39       case _ => arg.toString
40     }
41 
42     private def treatArg(arg: Any, suffix: String)(implicit ctx: Context): (Any, String) = arg match {
43       case arg: Seq[_] if suffix.nonEmpty && suffix.head == '%' =>
44         val (rawsep, rest) = suffix.tail.span(_ != '%')
45         val sep = StringContext.treatEscapes(rawsep)
46         if (rest.nonEmpty) (arg.map(showArg).mkString(sep), rest.tail)
47         else (arg, suffix)
48       case _ =>
49         (showArg(arg), suffix)
50     }
51 
52     def assemble(args: Seq[Any])(implicit ctx: Context): String = {
53       def isLineBreak(c: Char) = c == '\n' || c == '\f' // compatible with StringLike#isLineBreak
54       def stripTrailingPart(s: String) = {
55         val (pre, post) = s.span(c => !isLineBreak(c))
56         pre ++ post.stripMargin
57       }
58       val (prefix, suffixes) = sc.parts.toList match {
59         case head :: tail => (head.stripMargin, tail map stripTrailingPart)
60         case Nil => ("", Nil)
61       }
62       val (args1, suffixes1) = (args, suffixes).zipped.map(treatArg(_, _)).unzip
63       new StringContext(prefix :: suffixes1.toList: _*).s(args1: _*)
64     }
65   }

92   private def wrapNonSensical(arg: Any /* Type | Symbol */, str: String)(implicit ctx: Context): String = {
93     import MessageContainer._
94     def isSensical(arg: Any): Boolean = arg match {
95       case tpe: Type =>

97       case sym: Symbol if sym.isCompleted =>
98         sym.info match {

100           case _ => true

115     def record(str: String, entry: Recorded)(implicit ctx: Context): String = {
116       def followAlias(e1: Recorded): Recorded = e1 match {
117         case e1: Symbol if e1.isAliasType =>
118           val underlying = e1.typeRef.underlyingClassRef(refinementOK = false).typeSymbol
119           if (underlying.name == e1.name) underlying else e1
120         case _ => e1
121       }
122       lazy val dealiased = followAlias(entry)
123       var alts = apply(str).dropWhile(alt => dealiased ne followAlias(alt))
124       if (alts.isEmpty) {
125         alts = entry :: apply(str)
126         update(str, alts)
127       }
128       str + "'" * (alts.length - 1)
129     }

146   /** Create explanation for single `Recorded` type or symbol */
147   def explanation(entry: AnyRef)(implicit ctx: Context): String = {
148     def boundStr(bound: Type, default: ClassSymbol, cmp: String) =
149       if (bound.isRef(default)) "" else i"$cmp $bound"
150 
151     def boundsStr(bounds: TypeBounds): String = {
152       val lo = boundStr(bounds.lo, defn.NothingClass, ">:")
153       val hi = boundStr(bounds.hi, defn.AnyClass, "<:")
154       if (lo.isEmpty) hi
155       else if (hi.isEmpty) lo
156       else s"$lo and $hi"
157     }
158 
159     def addendum(cat: String, info: Type): String = info match {
160       case bounds @ TypeBounds(lo, hi) if bounds ne TypeBounds.empty =>
161         if (lo eq hi) i" which is an alias of $lo"
162         else i" with $cat ${boundsStr(bounds)}"
163       case _ =>
164         ""
165     }
166 
167     entry match {
168       case param: TypeParamRef =>
169         s"is a type variable${addendum("constraint", ctx.typeComparer.bounds(param))}"
170       case sym: Symbol =>
171         s"is a ${ctx.printer.kindString(sym)}${sym.showExtendedLocation}${addendum("bounds", sym.info)}"
172       case tp: SkolemType =>
173         s"is an unknown value of type ${tp.widen.show}"
174     }
175   }

177   /** Turns a `Seen` into a `String` to produce an explanation for types on the
178     * form `where: T is...`
179     *
180     * @return string disambiguating types
181     */
182   private def explanations(seen: Seen)(implicit ctx: Context): String = {
183     def needsExplanation(entry: Recorded) = entry match {
184       case param: TypeParamRef => ctx.typerState.constraint.contains(param)
185       case skolem: SkolemType => true
186       case _ => false
187     }
188 
189     val toExplain: List[(String, Recorded)] = seen.toList.flatMap {
190       case (str, entry :: Nil) =>
191         if (needsExplanation(entry)) (str, entry) :: Nil else Nil
192       case (str, entries) =>
193         entries.map(alt => (seen.record(str, alt), alt))
194     }.sortBy(_._1)
195 
196     def columnar(parts: List[(String, String)]): List[String] = {
197       lazy val maxLen = parts.map(_._1.length).max
198       parts.map {
199         case (leader, trailer) =>
200           val variable = hl"$leader"
201           s"""$variable${" " * (maxLen - leader.length)} $trailer"""
202       }
203     }
204 
205     val explainParts = toExplain.map { case (str, entry) => (str, explanation(entry)) }
206     val explainLines = columnar(explainParts)
207     if (explainLines.isEmpty) "" else i"where:    $explainLines%\n          %\n"
208   }

230   /** When getting a type mismatch it is useful to disambiguate placeholders like:
231     *
232     * ```
233     * found:    List[Int]
234     * required: List[T]
235     * where:    T is a type in the initalizer of value s which is an alias of
236     *           String
237     * ```
238     *
239     * @return the `where` section as well as the printing context for the
240     *         placeholders - `("T is a...", printCtx)`
241     */
242   def disambiguateTypes(args: Type*)(implicit ctx: Context): (String, Context) = {
243     val seen = new Seen
244     val printCtx = explainCtx(seen)
245     args.foreach(_.show(printCtx)) // showing each member will put it into `seen`
246     (explanations(seen), printCtx)
247   }

249   /** This method will produce a colored type diff from the given arguments.
250     * The idea is to do this for known cases that are useful and then fall back
251     * on regular syntax highlighting for the cases which are unhandled.
252     *
253     * Please not that if used in combination with `disambiguateTypes` the
254     * correct `Context` for printing should also be passed when calling the
255     * method.
256     *
257     * @return the (found, expected, changePercentage) with coloring to
258     *         highlight the difference
259     */
260   def typeDiff(found: Type, expected: Type)(implicit ctx: Context): (String, String) = {
261     val fnd = wrapNonSensical(found, found.show)
262     val exp = wrapNonSensical(expected, expected.show)
263 
264     DiffUtil.mkColoredTypeDiff(fnd, exp) match {
265       case _ if ctx.settings.color.value == "never" => (fnd, exp)
266       case (fnd, exp, change) if change < 0.5 => (fnd, exp)
267       case _ => (fnd, exp)
268     }
269   }
