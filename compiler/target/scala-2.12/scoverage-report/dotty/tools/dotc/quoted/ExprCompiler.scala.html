25 /** Compiler that takes the contents of a quoted expression `expr` and produces
26  *  a class file with `class ' { def apply: Object = expr }`.
27  */
28 class ExprCompiler(directory: AbstractFile) extends Compiler {
29   import tpd._
30 
31   /** A GenBCode phase that outputs to a virtual directory */
32   private class ExprGenBCode extends GenBCode {
33     override def phaseName = "genBCode"
34     override def outputDir(implicit ctx: Context) = directory
35   }
36 
37   override protected def frontendPhases: List[List[Phase]] =
38     List(List(new ExprFrontend(putInClass = true)))
39 
40   override protected def picklerPhases: List[List[Phase]] =
41     List(List(new ReifyQuotes))
42 
43   override protected def backendPhases: List[List[Phase]] =
44     List(List(new ExprGenBCode))
45 
46   override def newRun(implicit ctx: Context): ExprRun = {
47     reset()
48     new ExprRun(this, ctx.addMode(Mode.ReadPositions))
49   }
50 
51   def outputClassName: TypeName = "Quoted".toTypeName
52 
53   /** Frontend that receives scala.quoted.Expr as input */
54   class ExprFrontend(putInClass: Boolean) extends FrontEnd {
55     import tpd._
56 
57     override def isTyper = false
58 
59     override def runOn(units: List[CompilationUnit])(implicit ctx: Context): List[CompilationUnit] = {
60       units.map {
61         case exprUnit: ExprCompilationUnit =>
62           val tree =
63             if (putInClass) inClass(exprUnit.expr)
64             else PickledQuotes.quotedToTree(exprUnit.expr)
65           val source = new SourceFile("", Seq())
66           CompilationUnit.mkCompilationUnit(source, tree, forceTrees = true)
67       }
68     }
69 
70     /** Places the contents of expr in a compilable tree for a class
71       *  with the following format.
72       *  `package __root__ { class ' { def apply: Any = <expr> } }`
73       */
74     private def inClass(expr: Expr[_])(implicit ctx: Context): Tree = {
75       val pos = Position(0)
76       val assocFile = new PlainFile(Path("<quote>"))
77 
78       val cls = ctx.newCompleteClassSymbol(defn.RootClass, outputClassName, EmptyFlags,
79         defn.ObjectType :: Nil, newScope, coord = pos, assocFile = assocFile).entered.asClass
80       cls.enter(ctx.newDefaultConstructor(cls), EmptyScope)
81       val meth = ctx.newSymbol(cls, nme.apply, Method, ExprType(defn.AnyType), coord = pos).entered
82 
83       val quoted = PickledQuotes.quotedToTree(expr)(ctx.withOwner(meth))
84 
85       val run = DefDef(meth, quoted)
86       val classTree = ClassDef(cls, DefDef(cls.primaryConstructor.asTerm), run :: Nil)
87       PackageDef(ref(defn.RootPackage).asInstanceOf[Ident], classTree :: Nil).withPos(pos)
88     }
89   }
90 
91   class ExprRun(comp: Compiler, ictx: Context) extends Run(comp, ictx) {
92     def compileExpr(expr: Expr[_]): Unit = {
93       val units = new ExprCompilationUnit(expr) :: Nil
94       compileUnits(units)
95     }
96   }
