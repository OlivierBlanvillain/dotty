12 /** Methods for adding constraints and solving them.
13  *
14  * What goes into a Constraint as opposed to a ConstrainHandler?
15  *
16  * Constraint code is purely functional: Operations get constraints and produce new ones.
17  * Constraint code does not have access to a type-comparer. Anything regarding lubs and glbs has to be done
18  * elsewhere.

184   /** Solve constraint set for given type parameter `param`.
185    *  If `fromBelow` is true the parameter is approximated by its lower bound,
186    *  otherwise it is approximated by its upper bound. However, any occurrences
187    *  of the parameter in a refinement somewhere in the bound are removed. Also
188    *  wildcard types in bounds are approximated by their upper or lower bounds.
189    *  (Such occurrences can arise for F-bounded types).
190    *  The constraint is left unchanged.
191    *  @return the instantiating type
192    *  @pre `param` is in the constraint's domain.
193    */
194   final def approximation(param: TypeParamRef, fromBelow: Boolean): Type = {
195     val avoidParam = new TypeMap {
196       override def stopAtStatic = true
197       def avoidInArg(arg: Type): Type =
198         if (param.occursIn(arg)) TypeBounds.empty else arg
199       def apply(tp: Type) = mapOver {
200         tp match {
201           case tp @ AppliedType(tycon, args) =>
202             tp.derivedAppliedType(tycon, args.mapConserve(avoidInArg))
203           case tp: RefinedType if param occursIn tp.refinedInfo =>
204             tp.parent
205           case tp: WildcardType =>
206             val bounds = tp.optBounds.orElse(TypeBounds.empty).bounds
207             // Try to instantiate the wildcard to a type that is known to conform to it.
208             // This means:
209             //  If fromBelow is true, we minimize the type overall
210             //  Hence, if variance < 0, pick the maximal safe type: bounds.lo
211             //           (i.e. the whole bounds range is over the type)
212             //         if variance > 0, pick the minimal safe type: bounds.hi
213             //           (i.e. the whole bounds range is under the type)
214             //         if variance == 0, pick bounds.lo anyway (this is arbitrary but in line with
215             //           the principle that we pick the smaller type when in doubt).
216             //  If fromBelow is false, we maximize the type overall and reverse the bounds
217             //  if variance != 0. For variance == 0, we still minimize.
218             //  In summary we pick the bound given by this table:
219             //
220             //  variance    | -1  0   1
221             //  ------------------------
222             //  from below  | lo  lo  hi
223             //  from above  | hi  lo  lo
224             //
225             if (variance == 0 || fromBelow == (variance < 0)) bounds.lo else bounds.hi
226           case _ => tp
227         }
228       }
229     }
230     constraint.entry(param) match {
231       case _: TypeBounds =>
232         val bound = if (fromBelow) constraint.fullLowerBound(param) else constraint.fullUpperBound(param)
233         val inst = avoidParam(bound)
234         typr.println(s"approx ${param.show}, from below = $fromBelow, bound = ${bound.show}, inst = ${inst.show}")
235         inst
236       case inst =>
237         assert(inst.exists, i"param = $param\nconstraint = $constraint")
238         inst
239     }
240   }

242   /** The instance type of `param` in the current constraint (which contains `param`).
243    *  If `fromBelow` is true, the instance type is the lub of the parameter's
244    *  lower bounds; otherwise it is the glb of its upper bounds. However,
245    *  a lower bound instantiation can be a singleton type only if the upper bound
246    *  is also a singleton type.
247    */
248   def instanceType(param: TypeParamRef, fromBelow: Boolean): Type = {
249     def upperBound = constraint.fullUpperBound(param)
250     def isMultiSingleton(tp: Type): Boolean = tp.stripAnnots match {
251       case tp: SingletonType => true
252       case AndType(tp1, tp2) => isMultiSingleton(tp1) | isMultiSingleton(tp2)
253       case OrType(tp1, tp2) => isMultiSingleton(tp1) & isMultiSingleton(tp2)
254       case _ => false
255     }
256     def isFullyDefined(tp: Type): Boolean = tp match {
257       case tp: TypeVar => tp.isInstantiated && isFullyDefined(tp.instanceOpt)
258       case tp: TypeProxy => isFullyDefined(tp.underlying)
259       case tp: AndOrType => isFullyDefined(tp.tp1) && isFullyDefined(tp.tp2)
260       case _ => true
261     }
262     def isOrType(tp: Type): Boolean = tp.stripTypeVar.dealias match {
263       case tp: OrType => true
264       case tp: RefinedOrRecType => isOrType(tp.parent)
265       case AndType(tp1, tp2) => isOrType(tp1) | isOrType(tp2)
266       case WildcardType(bounds: TypeBounds) => isOrType(bounds.hi)
267       case _ => false
268     }
269 
270     // First, solve the constraint.
271     var inst = approximation(param, fromBelow).simplified
272 
273     // Then, approximate by (1.) - (3.) and simplify as follows.
274     // 1. If instance is from below and is a singleton type, yet upper bound is
275     // not a singleton type or a subtype of `scala.Singleton`, widen the
276     // instance.
277     if (fromBelow && isMultiSingleton(inst) && !isMultiSingleton(upperBound)
278         && !isSubTypeWhenFrozen(upperBound, defn.SingletonType))
279       inst = inst.widen
280 
281     // 2. If instance is from below and is a fully-defined union type, yet upper bound
282     // is not a union type, approximate the union type from above by an intersection
283     // of all common base types.
284     if (fromBelow && isOrType(inst) && !isOrType(upperBound))
285       inst = inst.widenUnion
286 
287     inst
288   }

349   /** Add constraint `param <: bound` if `fromBelow` is false, `param >: bound` otherwise.
350    *  `bound` is assumed to be in normalized form, as specified in `firstTry` and
351    *  `secondTry` of `TypeComparer`. In particular, it should not be an alias type,
352    *  lazy ref, typevar, wildcard type, error type. In addition, upper bounds may
353    *  not be AndTypes and lower bounds may not be OrTypes. This is assured by the
354    *  way isSubType is organized.
355    */
356   protected def addConstraint(param: TypeParamRef, bound: Type, fromBelow: Boolean): Boolean = {
357     def description = i"constr $param ${if (fromBelow) ">:" else "<:"} $bound:\n$constraint"
358     //checkPropagated(s"adding $description")(true) // DEBUG in case following fails
359     checkPropagated(s"added $description") {
360       addConstraintInvocations += 1
361 
362       /** When comparing lambdas we might get constraints such as
363        *  `A <: X0` or `A = List[X0]` where `A` is a constrained parameter
364        *  and `X0` is a lambda parameter. The constraint for `A` is not allowed
365        *  to refer to such a lambda parameter because the lambda parameter is
366        *  not visible where `A` is defined. Consequently, we need to
367        *  approximate the bound so that the lambda parameter does not appear in it.
368        *  If `tp` is an upper bound, we need to approximate with something smaller,
369        *  otherwise something larger.
370        *  Test case in pos/i94-nada.scala. This test crashes with an illegal instance
371        *  error in Test2 when the rest of the SI-2712 fix is applied but `pruneLambdaParams` is
372        *  missing.
373        */
374       def pruneLambdaParams(tp: Type) =
375         if (comparedTypeLambdas.nonEmpty) {
376           val approx = new ApproximatingTypeMap {
377             if (fromBelow) variance = -1
378             def apply(t: Type): Type = t match {
379               case t @ TypeParamRef(tl: TypeLambda, n) if comparedTypeLambdas contains tl =>
380                 val bounds = tl.paramInfos(n)
381                 range(bounds.lo, bounds.hi)
382               case _ =>
383                 mapOver(t)
384             }
385           }
386           approx(tp)
387         }
388         else tp
389 
390       def addParamBound(bound: TypeParamRef) =
391         constraint.entry(param) match {
392           case _: TypeBounds =>
393             if (fromBelow) addLess(bound, param) else addLess(param, bound)
394           case tp =>
395             if (fromBelow) isSubType(bound, tp) else isSubType(tp, bound)
396         }
397 
398       /** Drop all constrained parameters that occur at the toplevel in `bound` and
399        *  handle them by `addLess` calls.
400        *  The preconditions make sure that such parameters occur only
401        *  in one of two ways:
402        *
403        *  1.
404        *
405        *    P <: Ts1 | ... | Tsm   (m > 0)
406        *    Tsi = T1 & ... Tn      (n >= 0)
407        *    Some of the Ti are constrained parameters
408        *
409        *  2.
410        *
411        *    Ts1 & ... & Tsm <: P   (m > 0)
412        *    Tsi = T1 | ... | Tn    (n >= 0)
413        *    Some of the Ti are constrained parameters
414        *
415        *  In each case we cannot leave the parameter in place,
416        *  because that would risk making a parameter later a subtype or supertype
417        *  of a bound where the parameter occurs again at toplevel, which leads to cycles
418        *  in the subtyping test. So we intentionally narrow the constraint by
419        *  recording an isLess relationship instead (even though this is not implied
420        *  by the bound).
421        *
422        *  Narrowing a constraint is better than widening it, because narrowing leads
423        *  to incompleteness (which we face anyway, see for instance eitherIsSubType)
424        *  but widening leads to unsoundness.
425        *
426        *  A test case that demonstrates the problem is i864.scala.
427        *  Turn Config.checkConstraintsSeparated on to get an accurate diagnostic
428        *  of the cycle when it is created.
429        *
430        *  @return The pruned type if all `addLess` calls succeed, `NoType` otherwise.
431        */
432       def prune(bound: Type): Type = bound match {
433         case bound: AndOrType =>
434           val p1 = prune(bound.tp1)
435           val p2 = prune(bound.tp2)
436           if (p1.exists && p2.exists) bound.derivedAndOrType(p1, p2)
437           else NoType
438         case bound: TypeVar if constraint contains bound.origin =>
439           prune(bound.underlying)
440         case bound: TypeParamRef =>
441           constraint.entry(bound) match {
442             case NoType => pruneLambdaParams(bound)
443             case _: TypeBounds =>
444               if (!addParamBound(bound)) NoType
445               else if (fromBelow) defn.NothingType
446               else defn.AnyType
447             case inst =>
448               prune(inst)
449           }
450         case _ =>
451           pruneLambdaParams(bound)
452       }
453 
454       try bound match {
455         case bound: TypeParamRef if constraint contains bound =>
456           addParamBound(bound)
457         case _ =>
458           val pbound = prune(bound)
459           pbound.exists && (
460             if (fromBelow) addLowerBound(param, pbound) else addUpperBound(param, pbound))
461       }
462       finally addConstraintInvocations -= 1
463     }
464   }

477   def checkPropagated(msg: => String)(result: Boolean): Boolean = {
478     if (Config.checkConstraintsPropagated && result && addConstraintInvocations == 0) {
479       val saved = frozenConstraint
480       frozenConstraint = true
481       for (p <- constraint.domainParams) {
482         def check(cond: => Boolean, q: TypeParamRef, ordering: String, explanation: String): Unit =
483           assert(cond, i"propagation failure for $p $ordering $q: $explanation\n$msg")
484         for (u <- constraint.upper(p))
485           check(bounds(p).hi <:< bounds(u).hi, u, "<:", "upper bound not propagated")
486         for (l <- constraint.lower(p)) {
487           check(bounds(l).lo <:< bounds(p).hi, l, ">:", "lower bound not propagated")
488           check(constraint.isLess(l, p), l, ">:", "reverse ordering (<:) missing")
489         }
490       }
491       frozenConstraint = saved
492     }
