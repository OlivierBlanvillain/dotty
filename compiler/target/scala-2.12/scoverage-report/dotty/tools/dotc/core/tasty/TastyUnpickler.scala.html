38   private def check(cond: Boolean, msg: => String) =
39     if (!cond) throw new UnpickleException(msg)

41   private def readName(): TermName = nameAtRef(readNameRef())
42   private def readString(): String = readName().toString

44   private def readNameContents(): TermName = {
45     val tag = readByte()
46     val length = readNat()
47     val start = currentAddr
48     val end = start + length
49     val result = tag match {
50       case UTF8 =>
51         goto(end)
52         termName(bytes, start.index, length)
53       case QUALIFIED | EXPANDED | EXPANDPREFIX =>
54         qualifiedNameKindOfTag(tag)(readName(), readName().asSimpleName)
55       case UNIQUE =>
56         val separator = readName().toString
57         val num = readNat()
58         val originals = until(end)(readName())
59         val original = if (originals.isEmpty) EmptyTermName else originals.head
60         uniqueNameKindOfSeparator(separator)(original, num)
61       case DEFAULTGETTER | VARIANT =>
62         numberedNameKindOfTag(tag)(readName(), readNat())
63       case SIGNED =>
64         val original = readName()
65         val result = readName().toTypeName
66         val params = until(end)(readName().toTypeName)
67         var sig = Signature(params, result)
68         SignedName(original, sig)
69       case _ =>
70         simpleNameKindOfTag(tag)(readName())
71     }
72     assert(currentAddr == end, s"bad name $result $start $currentAddr $end")
73     result
74   }
75 
76   private def readHeader(): UUID = {
77     for (i <- 0 until header.length)
78       check(readByte() == header(i), "not a TASTy file")
79     val major = readNat()
80     val minor = readNat()

82       s"""TASTy signature has wrong version.
83          | expected: $MajorVersion.$MinorVersion
84          | found   : $major.$minor""".stripMargin)
85     new UUID(readUncompressedLong(), readUncompressedLong())
86   }
87 
88   private val uuid = readHeader()
89 
90   locally {

92     while (!isAtEnd) {
93       val secName = readString()
94       val secEnd = readEnd()
95       sectionReader(secName) = new TastyReader(bytes, currentAddr.index, secEnd.index, currentAddr.index)
96       goto(secEnd)
97     }
98   }
99 
100   def unpickle[R](sec: SectionUnpickler[R]): Option[R] =
101     for (reader <- sectionReader.get(sec.name)) yield
102       sec.unpickle(reader, nameAtRef)
