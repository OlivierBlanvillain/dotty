36   /** A predicate that tests whether a type is a legal erased type. Only asInstanceOf and
37    *  isInstanceOf may have types that do not satisfy the predicate.
38    *  ErasedValueType is considered an erased type because it is valid after Erasure (it is
39    *  eliminated by ElimErasedValueType).
40    */
41   def isErasedType(tp: Type)(implicit ctx: Context): Boolean = tp match {
42     case _: ErasedValueType =>
43       true
44     case tp: TypeRef =>
45       val sym = tp.symbol
46       sym.isClass &&
47       sym != defn.AnyClass && sym != defn.ArrayClass &&
48       !defn.isSyntheticFunctionClass(sym)
49     case _: TermRef =>
50       true
51     case JavaArrayType(elem) =>
52       isErasedType(elem)
53     case AnnotatedType(tp, _) =>
54       isErasedType(tp)
55     case ThisType(tref) =>
56       isErasedType(tref)
57     case tp: MethodType =>
58       tp.paramInfos.forall(isErasedType) && isErasedType(tp.resultType)
59     case tp @ ClassInfo(pre, _, parents, decls, _) =>
60       isErasedType(pre) && parents.forall(isErasedType) //&& decls.forall(sym => isErasedType(sym.info)) && isErasedType(tp.selfType)
61     case NoType | NoPrefix | WildcardType | _: ErrorType | SuperType(_, _) =>
62       true
63     case _ =>
64       false
65   }

219   /** The erased least upper bound is computed as follows
220    *  - if both argument are arrays of objects, an array of the erased lub of the element types
221    *  - if both arguments are arrays of same primitives, an array of this primitive
222    *  - if one argument is array of primitives and the other is array of objects, Object
223    *  - if one argument is an array, Object
224    *  - otherwise a common superclass or trait S of the argument classes, with the
225    *    following two properties:
226    *      S is minimal: no other common superclass or trait derives from S
227    *      S is last   : in the linearization of the first argument type `tp1`
228    *                    there are no minimal common superclasses or traits that
229    *                    come after S.
230    *  The reason to pick last is that we prefer classes over traits that way,
231    *  which leads to more predictable bytecode and (?) faster dynamic dispatch.
232    */
233   def erasedLub(tp1: Type, tp2: Type)(implicit ctx: Context): Type = tp1 match {
234     case JavaArrayType(elem1) =>
235       import dotty.tools.dotc.transform.TypeUtils._

237         case JavaArrayType(elem2) =>
238           if (elem1.isPrimitiveValueType || elem2.isPrimitiveValueType) {
239             if (elem1.classSymbol eq elem2.classSymbol) // same primitive
240               JavaArrayType(elem1)
241             else defn.ObjectType
242           } else JavaArrayType(erasedLub(elem1, elem2))
243         case _ => defn.ObjectType
244       }
245     case _ =>
246       tp2 match {
247         case JavaArrayType(_) => defn.ObjectType
248         case _ =>
249           val cls2 = tp2.classSymbol

251           /** takeWhile+1 */
252           def takeUntil[T](l: List[T])(f: T => Boolean): List[T] = {
253             @tailrec def loop(tail: List[T], acc: List[T]): List[T] =
254               tail match {
255                 case h :: t => loop(if (f(h)) t else Nil, h :: acc)
256                 case Nil    => acc.reverse
257               }
258             loop(l, Nil)
259           }
260 
261           // We are not interested in anything that is not a supertype of tp2
262           val tp2superclasses = tp1.baseClasses.filter(cls2.derivesFrom)
263 
264           // From the spec, "Linearization also satisfies the property that a
265           // linearization of a class always contains the linearization of its
266           // direct superclass as a suffix"; it's enought to consider every
267           // candidate up to the first class.
268           val candidates = takeUntil(tp2superclasses)(!_.is(Trait))
269 
270           // Candidates st "no other common superclass or trait derives from S"
271           val minimums = candidates.filter { cand =>
272             candidates.forall(x => !x.derivesFrom(cand) || x.eq(cand))
273           }
274 
275           // Pick the last minimum to prioritise classes over traits
276           minimums.lastOption match {
277             case Some(lub) if lub != defn.AnyClass && lub != defn.AnyValClass =>
278               lub.typeRef
279             case _ => // Any/AnyVal only exist before erasure
280               defn.ObjectType
281           }
282       }
283   }

285   /** The erased greatest lower bound picks one of the two argument types. It prefers, in this order:
286    *  - arrays over non-arrays
287    *  - subtypes over supertypes, unless isJava is set
288    *  - real classes over traits

291     case JavaArrayType(elem1) =>
292       tp2 match {
293         case JavaArrayType(elem2) => JavaArrayType(erasedGlb(elem1, elem2, isJava))
294         case _ => tp1
295       }
296     case _ =>
297       tp2 match {
298         case JavaArrayType(_) => tp2
299         case _ =>
300           val tsym1 = tp1.typeSymbol
301           val tsym2 = tp2.typeSymbol
302           if (!tsym2.exists) tp1
303           else if (!tsym1.exists) tp2
304           else if (!isJava && tsym1.derivesFrom(tsym2)) tp1
305           else if (!isJava && tsym2.derivesFrom(tsym1)) tp2
306           else if (tp1.typeSymbol.isRealClass) tp1
307           else if (tp2.typeSymbol.isRealClass) tp2
308           else tp1
309       }

342   /**  The erasure |T| of a type T. This is:
343    *
344    *   - For a refined type scala.Array+[T]:
345    *      - if T is Nothing or Null, []Object
346    *      - otherwise, if T <: Object, []|T|
347    *      - otherwise, if T is a type paramter coming from Java, []Object
348    *      - otherwise, Object
349    *   - For a term ref p.x, the type <noprefix> # x.
350    *   - For a typeref scala.Any, scala.AnyVal or scala.Singleton: |java.lang.Object|
351    *   - For a typeref scala.Unit, |scala.runtime.BoxedUnit|.
352    *   - For a typeref scala.FunctionN, where N > MaxImplementedFunctionArity, scala.FunctionXXL
353    *   - For a typeref scala.ImplicitFunctionN, | scala.FunctionN |
354    *   - For a typeref P.C where C refers to a class, <noprefix> # C.
355    *   - For a typeref P.C where C refers to an alias type, the erasure of C's alias.
356    *   - For a typeref P.C where C refers to an abstract type, the erasure of C's upper bound.
357    *   - For a this-type C.this, the type itself.
358    *   - For all other type proxies: The erasure of the underlying type.
359    *   - For T1 & T2, the erased glb of |T1| and |T2| (see erasedGlb)
360    *   - For T1 | T2, the first base class in the linearization of T which is also a base class of T2
361    *   - For => T, ()T
362    *   - For a method type (Fs)scala.Unit, (|Fs|)scala.Unit.
363    *   - For any other uncurried method type (Fs)T, (|Fs|)|T|.
364    *   - For a curried method type (Fs1)(Fs2)T, (|Fs1|,Es2)ET where (Es2)ET = |(Fs2)T|.
365    *   - For a polymorphic type [Ts](Ps)T, |(Ps)T|
366    *   _ For a polymorphic type [Ts]T where T is not a method type, ()|T|
367    *   - For the class info type of java.lang.Object, the same type without any parents.
368    *   - For a class info type of a value class, the same type without any parents.
369    *   - For any other class info type with parents Ps, the same type with
370    *     parents |Ps|, but with duplicate references of Object removed.
371    *   - For NoType or NoPrefix, the type itself.
372    *   - For any other type, exception.
373    */
374   private def apply(tp: Type)(implicit ctx: Context): Type = tp match {
375     case _: ErasedValueType =>
376       tp
377     case tp: TypeRef =>
378       val sym = tp.symbol
379       if (!sym.isClass) this(tp.info)
380       else if (semiEraseVCs && isDerivedValueClass(sym)) eraseDerivedValueClassRef(tp)
381       else if (sym == defn.ArrayClass) apply(tp.appliedTo(TypeBounds.empty)) // i966 shows that we can hit a raw Array type.
382       else if (defn.isSyntheticFunctionClass(sym)) defn.erasedFunctionType(sym)
383       else if (defn.isPhantomTerminalClass(sym)) PhantomErasure.erasedPhantomType
384       else if (sym eq defn.PhantomClass) defn.ObjectType // To erase the definitions of Phantom.{assume, Any, Nothing}
385       else eraseNormalClassRef(tp)
386     case tp: AppliedType =>
387       if (tp.tycon.isRef(defn.ArrayClass)) eraseArray(tp)
388       else apply(tp.superType)
389     case _: TermRef | _: ThisType =>
390       this(tp.widen)
391     case SuperType(thistpe, supertpe) =>
392       SuperType(this(thistpe), this(supertpe))
393     case ExprType(rt) =>
394       defn.FunctionType(0)
395     case tp: TypeProxy =>
396       this(tp.underlying)
397     case AndType(tp1, tp2) =>
398       erasedGlb(this(tp1), this(tp2), isJava)
399     case OrType(tp1, tp2) =>
400       ctx.typeComparer.orType(this(tp1), this(tp2), erased = true)
401     case tp: MethodType =>
402       def paramErasure(tpToErase: Type) =
403         erasureFn(tp.isJavaMethod, semiEraseVCs, isConstructor, wildcardOK)(tpToErase)
404       val (names, formals0) =
405         if (tp.paramInfos.exists(_.isPhantom)) tp.paramNames.zip(tp.paramInfos).filterNot(_._2.isPhantom).unzip
406         else (tp.paramNames, tp.paramInfos)
407       val formals = formals0.mapConserve(paramErasure)
408       eraseResult(tp.resultType) match {
409         case rt: MethodType =>
410           tp.derivedLambdaType(names ++ rt.paramNames, formals ++ rt.paramInfos, rt.resultType)
411         case rt =>
412           tp.derivedLambdaType(names, formals, rt)
413       }
414     case tp: PolyType =>
415       this(tp.resultType)
416     case tp @ ClassInfo(pre, cls, parents, decls, _) =>
417       if (cls is Package) tp
418       else {
419         val erasedParents: List[Type] =
420           if ((cls eq defn.ObjectClass) || cls.isPrimitiveValueClass) Nil
421           else parents.mapConserve(apply) match {
422             case tr :: trs1 =>
423               assert(!tr.classSymbol.is(Trait), cls)
424               val tr1 = if (cls is Trait) defn.ObjectType else tr
425               tr1 :: trs1.filterNot(_ isRef defn.ObjectClass)
426             case nil => nil
427           }
428         val erasedDecls = decls.filteredScope(sym => !sym.isType || sym.isClass)
429         tp.derivedClassInfo(NoPrefix, erasedParents, erasedDecls, erasedRef(tp.selfType))
430           // can't replace selftype by NoType because this would lose the sourceModule link
431       }
432     case NoType | NoPrefix | _: ErrorType | JavaArrayType(_) =>
433       tp
434     case tp: WildcardType if wildcardOK =>
435       tp
436   }
