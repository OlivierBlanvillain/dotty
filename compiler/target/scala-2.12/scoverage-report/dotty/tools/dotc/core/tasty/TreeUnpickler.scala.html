112      *  as `OwnerTree`s in `buf`.
113      *  A complication concerns member definitions. These are lexically nested in a
114      *  Template node, but need to be listed separately in the OwnerTree of the enclosing class
115      *  in order not to confuse owner chains.

117     def scanTree(buf: ListBuffer[OwnerTree], mode: MemberDefMode = AllDefs): Unit = {
118       val start = currentAddr
119       val tag = readByte()
120       tag match {
121         case VALDEF | DEFDEF | TYPEDEF | TYPEPARAM | PARAM | TEMPLATE =>
122           val end = readEnd()
123           for (i <- 0 until numRefs(tag)) readNat()
124           if (tag == TEMPLATE) {
125             // Read all member definitions now, whereas non-members are children of
126             // template's owner tree.
127             val nonMemberReader = fork
128             scanTrees(buf, end, MemberDefsOnly)
129             buf += new OwnerTree(start, tag, nonMemberReader, end)
130           }
131           else if (mode != NoMemberDefs)
132             buf += new OwnerTree(start, tag, fork, end)
133           goto(end)
134         case tag =>
135           if (mode == MemberDefsOnly) skipTree(tag)
136           else if (tag >= firstLengthTreeTag) {
137             val end = readEnd()
138             var nrefs = numRefs(tag)
139             if (nrefs < 0) {
140               for (i <- nrefs until 0) scanTree(buf)
141               goto(end)
142             }
143             else {
144               for (i <- 0 until nrefs) readNat()
145               if (tag == BIND) {
146                 buf += new OwnerTree(start, tag, fork, end)
147                 goto(end)
148               }
149               else scanTrees(buf, end)
150             }
151           }
152           else if (tag >= firstNatASTTreeTag) { readNat(); scanTree(buf) }
153           else if (tag >= firstASTTreeTag) scanTree(buf)
154           else if (tag >= firstNatTreeTag) readNat()
155       }

601      *  current address and `end`.

603      *           trait owning the indexed statements can have as flags.

605     def indexStats(end: Addr)(implicit ctx: Context): FlagSet = {
606       var initsFlags = NoInitsInterface
607       while (currentAddr.index < end.index) {
608         nextByte match {
609           case VALDEF | DEFDEF | TYPEDEF | TYPEPARAM | PARAM =>
610             val sym = symbolAtCurrent()
611             skipTree()
612             if (sym.isTerm && !sym.is(MethodOrLazyOrDeferred))
613               initsFlags = EmptyFlags
614             else if (sym.isClass ||
615               sym.is(Method, butNot = Deferred) && !sym.isConstructor)
616               initsFlags &= NoInits
617           case IMPORT =>
618             skipTree()
619           case PACKAGE =>
620             processPackage { (pid, end) => implicit ctx => indexStats(end) }
621           case _ =>
622             skipTree()
623             initsFlags = EmptyFlags
624         }
625       }
626       assert(currentAddr.index == end.index)
627       initsFlags

677     private def readNewDef()(implicit ctx: Context): Tree = {
678       val start = currentAddr
679       val sym = symAtAddr(start)
680       val tag = readByte()
681       val end = readEnd()
682 
683       def readParamss(implicit ctx: Context): List[List[ValDef]] = {
684         collectWhile(nextByte == PARAMS) {
685           readByte()
686           readEnd()
687           readParams[ValDef](PARAM)
688         }
689       }
690 
691       val localCtx = localContext(sym)
692 
693       def readRhs(implicit ctx: Context) =
694         if (noRhs(end)) EmptyTree
695         else readLater(end, rdr => ctx => rdr.readTerm()(ctx))
696 
697       def ValDef(tpt: Tree) =
698         ta.assignType(untpd.ValDef(sym.name.asTermName, tpt, readRhs(localCtx)), sym)
699 
700       def DefDef(tparams: List[TypeDef], vparamss: List[List[ValDef]], tpt: Tree) =
701          ta.assignType(
702             untpd.DefDef(sym.name.asTermName, tparams, vparamss, tpt, readRhs(localCtx)),
703             sym)
704 
705       def TypeDef(rhs: Tree) =
706         ta.assignType(untpd.TypeDef(sym.name.asTypeName, rhs), sym)
707 
708       def ta =  ctx.typeAssigner
709 
710       val name = readName()
711       pickling.println(s"reading def of $name at $start")
712       val tree: MemberDef = tag match {
713         case DEFDEF =>
714           val tparams = readParams[TypeDef](TYPEPARAM)(localCtx)
715           val vparamss = readParamss(localCtx)
716           val tpt = readTpt()(localCtx)
717           val typeParams = tparams.map(_.symbol)
718           val valueParamss = ctx.normalizeIfConstructor(
719               vparamss.nestedMap(_.symbol), name == nme.CONSTRUCTOR)
720           val resType = ctx.effectiveResultType(sym, typeParams, tpt.tpe)
721           sym.info = ctx.methodType(typeParams, valueParamss, resType)
722           if (sym.isSetter && sym.accessedFieldOrGetter.is(ParamAccessor)) {
723             // reconstitute ParamAccessor flag of setters for var parameters, which is not pickled
724             sym.setFlag(ParamAccessor)
725             sym.resetFlag(Deferred)
726           }
727           DefDef(tparams, vparamss, tpt)
728         case VALDEF =>
729           val tpt = readTpt()(localCtx)
730           sym.info = tpt.tpe
731           ValDef(tpt)
732         case TYPEDEF | TYPEPARAM =>
733           if (sym.isClass) {
734             val companion = sym.scalacLinkedClass
735 
736             // Is the companion defined in the same Tasty file as `sym`?
737             // The only case to check here is if `sym` is a root. In this case
738             // `companion` might have been entered by the environment but it might
739             // be missing from the Tasty file. So we check explicitly for that.
740             def isCodefined =
741               roots.contains(companion.denot) == seenRoots.contains(companion)
742             if (companion.exists && isCodefined) {
743               if (sym is Flags.ModuleClass) sym.registerCompanionMethod(nme.COMPANION_CLASS_METHOD, companion)
744               else sym.registerCompanionMethod(nme.COMPANION_MODULE_METHOD, companion)
745             }
746             TypeDef(readTemplate(localCtx))
747           } else {
748             sym.info = TypeBounds.empty // needed to avoid cyclic references when unpicklin rhs, see i3816.scala
749             sym.setFlag(Provisional)
750             val rhs = readTpt()(localCtx)
751             sym.info = new NoCompleter {
752               override def completerTypeParams(sym: Symbol)(implicit ctx: Context) =
753                 rhs.tpe.typeParams
754             }
755             sym.info = rhs.tpe match {
756               case _: TypeBounds | _: ClassInfo => checkNonCyclic(sym, rhs.tpe, reportErrors = false)
757               case _ => TypeAlias(rhs.tpe)
758             }
759             sym.resetFlag(Provisional)
760             TypeDef(rhs)
761           }
762         case PARAM =>
763           val tpt = readTpt()(localCtx)
764           if (noRhs(end)) {
765             sym.info = tpt.tpe
766             ValDef(tpt)
767           }
768           else {
769             sym.setFlag(Method)
770             sym.info = ExprType(tpt.tpe)
771             pickling.println(i"reading param alias $name -> $currentAddr")
772             DefDef(Nil, Nil, tpt)
773           }
774       }
775       val mods =
776         if (sym.annotations.isEmpty) untpd.EmptyModifiers
777         else untpd.Modifiers(annotations = sym.annotations.map(_.tree))
778       tree.withMods(mods)
779         // record annotations in tree so that tree positions can be filled in.
780         // Note: Once the inline PR with its changes to positions is in, this should be
781         // no longer necessary.
782       goto(end)
783       setPos(start, tree)
784       if (!sym.isType) { // Only terms might have leaky aliases, see the documentation of `checkNoPrivateLeaks`
785         sym.info = ta.avoidPrivateLeaks(sym, tree.pos)
786       }
787       if ((sym.isClass || sym.is(CaseVal)) && sym.isLocal)
788         // Child annotations for local classes and enum values are not pickled, so
789         // need to be re-established here.
790         sym.registerIfChild(late = true)
791       tree
792     }

794     private def readTemplate(implicit ctx: Context): Template = {
795       val start = currentAddr
796       val cls = ctx.owner.asClass
797       val assumedSelfType =
798         if (cls.is(Module) && cls.owner.isClass) TermRef(cls.owner.thisType, cls.name.sourceModuleName)
799         else NoType
800       cls.info = new TempClassInfo(cls.owner.thisType, cls, cls.unforcedDecls, assumedSelfType)
801       val localDummy = symbolAtCurrent()
802       val parentCtx = ctx.withOwner(localDummy)
803       assert(readByte() == TEMPLATE)
804       val end = readEnd()
805       val tparams = readIndexedParams[TypeDef](TYPEPARAM)
806       val vparams = readIndexedParams[ValDef](PARAM)
807       val parents = collectWhile(nextByte != SELFDEF && nextByte != DEFDEF) {
808         nextUnsharedTag match {
809           case APPLY | TYPEAPPLY => readTerm()(parentCtx)
810           case _ => readTpt()(parentCtx)
811         }
812       }
813       val parentTypes = parents.map(_.tpe.dealias)
814       val self =
815         if (nextByte == SELFDEF) {
816           readByte()
817           untpd.ValDef(readName(), readTpt(), EmptyTree).withType(NoType)
818         }
819         else EmptyValDef
820       cls.info = ClassInfo(cls.owner.thisType, cls, parentTypes, cls.unforcedDecls,
821         if (self.isEmpty) NoType else self.tpt.tpe)
822       cls.setNoInitsFlags(fork.indexStats(end))
823       val constr = readIndexedDef().asInstanceOf[DefDef]
824       val mappedParents = parents.map(_.changeOwner(localDummy, constr.symbol))
825 
826       val lazyStats = readLater(end, rdr => implicit ctx => {
827         val stats = rdr.readIndexedStats(localDummy, end)
828         tparams ++ vparams ++ stats
829       })
830       setPos(start,
831         untpd.Template(constr, mappedParents, self, lazyStats)
832           .withType(localDummy.termRef))
833     }

913     def readTerm()(implicit ctx: Context): Tree = {  // TODO: rename to readTree
914       val start = currentAddr
915       val tag = readByte()
916       pickling.println(s"reading term ${astTagToString(tag)} at $start")
917 
918       def readPathTerm(): Tree = {
919         goto(start)
920         readType() match {
921           case path: TypeRef => TypeTree(path)
922           case path: TermRef => ref(path)
923           case path: ThisType => untpd.This(untpd.EmptyTypeIdent).withType(path)
924           case path: ConstantType => Literal(path.value)
925         }
926       }

928       def completeSelect(name: Name, tpf: Type => NamedType): Select = {
929         val localCtx =
930           if (name == nme.CONSTRUCTOR) ctx.addMode(Mode.InSuperCall) else ctx
931         val qual = readTerm()(localCtx)
932         untpd.Select(qual, name).withType(tpf(qual.tpe.widenIfUnstable))
933       }
934 
935       def readQualId(): (untpd.Ident, TypeRef) = {
936         val qual = readTerm().asInstanceOf[untpd.Ident]
937          (untpd.Ident(qual.name).withPos(qual.pos), qual.tpe.asInstanceOf[TypeRef])
938       }

940       def accessibleDenot(pre: Type, name: Name, sig: Signature) = {
941         val d = pre.member(name).atSignature(sig)
942         if (!d.symbol.exists || d.symbol.isAccessibleFrom(pre)) d
943         else pre.nonPrivateMember(name).atSignature(sig)
944       }

946       def readSimpleTerm(): Tree = tag match {
947         case SHAREDterm =>
948           forkAt(readAddr()).readTerm()
949         case IDENT =>
950           untpd.Ident(readName()).withType(readType())
951         case IDENTtpt =>
952           untpd.Ident(readName().toTypeName).withType(readType())
953         case SELECT =>
954           def readRest(name: TermName, sig: Signature): Tree =
955             completeSelect(name, pre => TermRef(pre, name, accessibleDenot(pre, name, sig)))
956           readName() match {
957             case SignedName(name, sig) => readRest(name, sig)
958             case name => readRest(name, Signature.NotAMethod)
959           }
960         case SELECTtpt =>
961           val name = readName().toTypeName
962           completeSelect(name, pre => TypeRef(pre, name, accessibleDenot(pre, name, Signature.NotAMethod)))
963         case QUALTHIS =>
964           val (qual, tref) = readQualId()
965           untpd.This(qual).withType(ThisType.raw(tref))
966         case NEW =>
967           New(readTpt())
968         case SINGLETONtpt =>
969           SingletonTypeTree(readTerm())
970         case BYNAMEtpt =>
971           ByNameTypeTree(readTpt())
972         case NAMEDARG =>
973           NamedArg(readName(), readTerm())
974         case _ =>
975           readPathTerm()
976       }

978       def readLengthTerm(): Tree = {
979         val end = readEnd()
980 
981         def readBlock(mkTree: (List[Tree], Tree) => Tree): Tree = {
982           val exprReader = fork
983           skipTree()
984           val stats = readStats(ctx.owner, end)
985           val expr = exprReader.readTerm()
986           mkTree(stats, expr)
987         }
988 
989         val result =
990           (tag: @switch) match {
991             case SUPER =>
992               val qual = readTerm()
993               val (mixId, mixTpe) = ifBefore(end)(readQualId(), (untpd.EmptyTypeIdent, NoType))
994               tpd.Super(qual, mixId, ctx.mode.is(Mode.InSuperCall), mixTpe.typeSymbol)
995             case APPLY =>
996               val fn = readTerm()
997               tpd.Apply(fn, until(end)(readTerm()))
998             case TYPEAPPLY =>
999               tpd.TypeApply(readTerm(), until(end)(readTpt()))
1000             case TYPED =>
1001               val expr = readTerm()
1002               val tpt = readTpt()
1003               Typed(expr, tpt)
1004             case ASSIGN =>
1005               Assign(readTerm(), readTerm())
1006             case BLOCK =>
1007               readBlock(Block)
1008             case INLINED =>
1009               val call = readTerm()
1010               readBlock((defs, expr) => Inlined(call, defs.asInstanceOf[List[MemberDef]], expr))
1011             case IF =>
1012               If(readTerm(), readTerm(), readTerm())
1013             case LAMBDA =>
1014               val meth = readTerm()
1015               val tpt = ifBefore(end)(readTpt(), EmptyTree)
1016               Closure(Nil, meth, tpt)
1017             case MATCH =>
1018               Match(readTerm(), readCases(end))
1019             case RETURN =>
1020               val from = readSymRef()
1021               val expr = ifBefore(end)(readTerm(), EmptyTree)
1022               Return(expr, Ident(from.termRef))
1023             case TRY =>
1024               Try(readTerm(), readCases(end), ifBefore(end)(readTerm(), EmptyTree))
1025             case SELECTouter =>
1026               val levels = readNat()
1027               readTerm().outerSelect(levels, SkolemType(readType()))
1028             case REPEATED =>
1029               val elemtpt = readTpt()
1030               SeqLiteral(until(end)(readTerm()), elemtpt)
1031             case BIND =>
1032               val sym = symAtAddr.getOrElse(start, forkAt(start).createSymbol())
1033               readName()
1034               readType()
1035               Bind(sym, readTerm())
1036             case ALTERNATIVE =>
1037               Alternative(until(end)(readTerm()))
1038             case UNAPPLY =>
1039               val fn = readTerm()
1040               val implicitArgs =
1041                 collectWhile(nextByte == IMPLICITarg) {
1042                   readByte()
1043                   readTerm()
1044                 }
1045               val patType = readType()
1046               val argPats = until(end)(readTerm())
1047               UnApply(fn, implicitArgs, argPats, patType)
1048             case REFINEDtpt =>
1049               val refineCls = ctx.newRefinedClassSymbol(coordAt(start))
1050               typeAtAddr(start) = refineCls.typeRef
1051               val parent = readTpt()
1052               val refinements = readStats(refineCls, end)(localContext(refineCls))
1053               RefinedTypeTree(parent, refinements, refineCls)
1054             case APPLIEDtpt =>
1055               // If we do directly a tpd.AppliedType tree we might get a
1056               // wrong number of arguments in some scenarios reading F-bounded
1057               // types. This came up in #137 of collection strawman.
1058               val tycon = readTpt()
1059               val args = until(end)(readTpt())
1060               untpd.AppliedTypeTree(tycon, args).withType(tycon.tpe.safeAppliedTo(args.tpes))
1061             case ANDtpt =>
1062               val tpt1 = readTpt()
1063               val tpt2 = readTpt()
1064               // FIXME: We need to do this instead of "AndType(tpt1, tpt2)" to avoid self-type cyclic reference in tasty_tools
1065               untpd.AndTypeTree(tpt1, tpt2).withType(AndType(tpt1.tpe, tpt2.tpe))
1066             case ORtpt =>
1067               OrTypeTree(readTpt(), readTpt())
1068             case ANNOTATEDtpt =>
1069               Annotated(readTpt(), readTerm())
1070             case LAMBDAtpt =>
1071               val tparams = readParams[TypeDef](TYPEPARAM)
1072               val body = readTpt()
1073               LambdaTypeTree(tparams, body)
1074             case TYPEBOUNDStpt =>
1075               val lo = readTpt()
1076               val hi = if (currentAddr == end) lo else readTpt()
1077               TypeBoundsTree(lo, hi)
1078             case HOLE =>
1079               readHole(end)
1080             case _ =>
1081               readPathTerm()
1082           }
1083         assert(currentAddr == end, s"$start $currentAddr $end ${astTagToString(tag)}")
1084         result
1085       }
1086 
1087       val tree = if (tag < firstLengthTreeTag) readSimpleTerm() else readLengthTerm()
1088       if (!tree.isInstanceOf[TypTree]) // FIXME: Necessary to avoid self-type cyclic reference in tasty_tools
1089         tree.overwriteType(tree.tpe.simplified)
1090       setPos(start, tree)
1091     }

1177   /** A lazy datastructure that records how definitions are nested in TASTY data.
1178    *  The structure is lazy because it needs to be computed only for forward references
1179    *  to symbols that happen before the referenced symbol is created (see `symbolAt`).
1180    *  Such forward references are rare.
1181    *
1182    *  @param   addr    The address of tree representing an owning definition, NoAddr for root tree
1183    *  @param   tag     The tag at `addr`. Used to determine which subtrees to scan for children
1184    *                   (i.e. if `tag` is template, don't scan member defs, as these belong already
1185    *                    to enclosing class).
1186    *  @param   reader  The reader to be used for scanning for children
1187    *  @param   end     The end of the owning definition
1188    */
1189   class OwnerTree(val addr: Addr, tag: Int, reader: TreeReader, val end: Addr) {
1190 
1191     private var myChildren: List[OwnerTree] = null
1192 
1193     /** All definitions that have the definition at `addr` as closest enclosing definition */
1194     def children: List[OwnerTree] = {
1195       if (myChildren == null) myChildren = {
1196         val buf = new ListBuffer[OwnerTree]
1197         reader.scanTrees(buf, end, if (tag == TEMPLATE) NoMemberDefs else AllDefs)
1198         buf.toList
1199       }
1200       myChildren
1201     }
1202 
1203     /** Find the owner of definition at `addr` */
1204     def findOwner(addr: Addr)(implicit ctx: Context): Symbol = {
1205       def search(cs: List[OwnerTree], current: Symbol): Symbol =
1206         try cs match {
1207         case ot :: cs1 =>
1208           if (ot.addr.index == addr.index)
1209             current
1210           else if (ot.addr.index < addr.index && addr.index < ot.end.index)
1211             search(ot.children, reader.symbolAt(ot.addr))
1212           else
1213             search(cs1, current)
1214         case Nil =>
1215           throw new TreeWithoutOwner
1216       }
1217       catch {
1218         case ex: TreeWithoutOwner =>
1219           pickling.println(i"no owner for $addr among $cs%, %")
1220           throw ex
1221       }
1222       try search(children, NoSymbol)
1223       catch {
1224         case ex: TreeWithoutOwner =>
1225           pickling.println(s"ownerTree = $ownerTree")
1226           throw ex
1227       }
1228     }
1229 
1230     override def toString =
1231       s"OwnerTree(${addr.index}, ${end.index}, ${if (myChildren == null) "?" else myChildren.mkString(" ")})"
1232   }
