9 /** A byte array buffer that can be filled with bytes or natural numbers in TASTY format,
10  *  and that supports reading and patching addresses represented as natural numbers.
11  *
12  *  @param bytes    The array containing data
13  *  @param start    The position from which to read
14  *  @param end      The position one greater than the last byte to be read
15  *  @param base     The index referenced by the logical zero address Addr(0)
16  */

86   /** Read a long integer number in 2's complement big endian format, base 128. */
