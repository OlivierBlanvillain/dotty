91   /** The trait FunctionN or ImplicitFunctionN, for some N
92    *  @param  name   The name of the trait to be created
93    *
94    *  FunctionN traits follow this template:
95    *
96    *      trait FunctionN[T0,...T{N-1}, R] extends Object {
97    *        def apply($x0: T0, ..., $x{N_1}: T{N-1}): R
98    *      }
99    *
100    *  That is, they follow the template given for Function2..Function22 in the
101    *  standard library, but without `tupled` and `curried` methods and without
102    *  a `toString`.
103    *
104    *  ImplicitFunctionN traits follow this template:
105    *
106    *      trait ImplicitFunctionN[T0,...,T{N-1}, R] extends Object with FunctionN[T0,...,T{N-1}, R] {
107    *        def apply(implicit $x0: T0, ..., $x{N_1}: T{N-1}): R
108    *      }
109    */
110   def newFunctionNTrait(name: TypeName): ClassSymbol = {
111     val completer = new LazyType {
112       def complete(denot: SymDenotation)(implicit ctx: Context): Unit = {
113         val cls = denot.asClass.classSymbol
114         val decls = newScope
115         val arity = name.functionArity
116         val paramNamePrefix = tpnme.scala_ ++ str.NAME_JOIN ++ name ++ str.EXPAND_SEPARATOR
117         val argParamRefs = List.tabulate(arity) { i =>
118           enterTypeParam(cls, paramNamePrefix ++ "T" ++ (i + 1).toString, Contravariant, decls).typeRef
119         }
120         val resParam = enterTypeParam(cls, paramNamePrefix ++ "R", Covariant, decls)
121         val (methodType, parentTraits) =
122           if (name.firstPart.startsWith(str.ImplicitFunction)) {
123             val superTrait =
124               FunctionType(arity).appliedTo(argParamRefs ::: resParam.typeRef :: Nil)
125             (ImplicitMethodType, superTrait :: Nil)
126           }
127           else (MethodType, Nil)
128         val applyMeth =
129           decls.enter(
130             newMethod(cls, nme.apply,
131               methodType(argParamRefs, resParam.typeRef), Deferred))
132         denot.info =
133           ClassInfo(ScalaPackageClass.thisType, cls, ObjectType :: parentTraits, decls)
134       }
135     }
136     newClassSymbol(ScalaPackageClass, name, Trait | NoInits, completer)
137   }

227   /** Note: We cannot have same named methods defined in Object and Any (and AnyVal, for that matter)
228    *  because after erasure the Any and AnyVal references get remapped to the Object methods
229    *  which would result in a double binding assertion failure.
230    * Instead we do the following:
231    *
232    *  - Have some methods exist only in Any, and remap them with the Erasure denotation
233    *    transformer to be owned by Object.
234    *  - Have other methods exist only in Object.
235    * To achieve this, we synthesize all Any and Object methods; Object methods no longer get
236    * loaded from a classfile.
237    *
238    * There's a remaining question about `getClass`. In Scala2.x `getClass` was handled by compiler magic.
239    * This is deemed too cumersome for Dotty and therefore right now `getClass` gets no special treatment;
240    * it's just a method on `Any` which returns the raw type `java.lang.Class`. An alternative
241    * way to get better `getClass` typing would be to treat `getClass` as a method of a generic
242    * decorator which gets remapped in a later phase to Object#getClass. Then we could give it
243    * the right type without changing the typechecker:
244    *
245    *     implicit class AnyGetClass[T](val x: T) extends AnyVal {
246    *       def getClass: java.lang.Class[T] = ???
247    *     }
248    */
