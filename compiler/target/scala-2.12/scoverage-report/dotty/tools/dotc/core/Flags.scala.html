8   /** A FlagSet represents a set of flags. Flags are encoded as follows:
9    *  The first two bits indicate whether a flagset applies to terms,
10    *  to types, or to both.  Bits 2..63 are available for properties
11    *  and can be doubly used for terms and types.
12    *  Combining two FlagSets with `|` will give a FlagSet
13    *  that has the intersection of the applicability to terms/types
14    *  of the two flag sets. It is checked that the intersection is not empty.
15    */
16   case class FlagSet(val bits: Long) extends AnyVal {
17 
18     /** The union of this flag set and the given flag set
19      */
20     def | (that: FlagSet): FlagSet =
21       if (bits == 0) that
22       else if (that.bits == 0) this
23       else {
24         val tbits = bits & that.bits & KINDFLAGS
25         if (tbits == 0)
26           assert(false, s"illegal flagset combination: $this and $that")
27         FlagSet(tbits | ((this.bits | that.bits) & ~KINDFLAGS))
28       }
29 
30     /** The intersection of this flag set and the given flag set */
31     def & (that: FlagSet) = FlagSet(bits & that.bits)
32 
33     /** The intersection of this flag set with the complement of the given flag set */
34     def &~ (that: FlagSet) = {
35       val tbits = bits & KINDFLAGS
36       if ((tbits & that.bits) == 0) this
37       else FlagSet(tbits | ((this.bits & ~that.bits) & ~KINDFLAGS))
38     }
39 
40     /** Does this flag set have a non-empty intersection with the given flag set?
41      *  This means that both the kind flags and the carrier bits have non-empty intersection.
42      */
43     def is(flags: FlagSet): Boolean = {
44       val fs = bits & flags.bits
45       (fs & KINDFLAGS) != 0 && (fs & ~KINDFLAGS) != 0
46     }
47 
48    /** Does this flag set have a non-empty intersection with the given flag set,
49     *  and at the same time contain none of the flags in the `butNot` set?
50     */
51     def is(flags: FlagSet, butNot: FlagSet): Boolean = is(flags) && !is(butNot)
52 
53     /** Does this flag set have all of the flags in given flag conjunction?
54      *  Pre: The intersection of the typeflags of both sets must be non-empty.
55      */
56     def is(flags: FlagConjunction): Boolean = {
57       val fs = bits & flags.bits
58       (fs & KINDFLAGS) != 0 &&
59       (fs >>> TYPESHIFT) == (flags.bits >>> TYPESHIFT)
60     }
61 
62     /** Does this flag set have all of the flags in given flag conjunction?
63      *  and at the same time contain none of the flags in the `butNot` set?
64      *  Pre: The intersection of the typeflags of both sets must be non-empty.
65      */
66     def is(flags: FlagConjunction, butNot: FlagSet): Boolean = is(flags) && !is(butNot)
67 
68     def isEmpty = (bits & ~KINDFLAGS) == 0
69 
70     /** Is this flag set a subset of that one? */
71     def <= (that: FlagSet) = (bits & that.bits) == bits
72 
73     /** Does this flag set apply to terms? */
74     def isTermFlags = (bits & TERMS) != 0
75 
76     /** Does this flag set apply to terms? */
77     def isTypeFlags = (bits & TYPES) != 0
78 
79     /** This flag set with all flags transposed to be type flags */
80     def toTypeFlags = if (bits == 0) this else FlagSet(bits & ~KINDFLAGS | TYPES)
81 
82     /** This flag set with all flags transposed to be term flags */
83     def toTermFlags = if (bits == 0) this else FlagSet(bits & ~KINDFLAGS | TERMS)
84 
85     /** This flag set with all flags transposed to be common flags */
86     def toCommonFlags = if (bits == 0) this else FlagSet(bits | KINDFLAGS)
87 
88     /** The number of non-kind flags in this set */
89     def numFlags: Int = java.lang.Long.bitCount(bits & ~KINDFLAGS)
90 
91     /** The lowest non-kind bit set in this flagset */
92     def firstBit: Int = java.lang.Long.numberOfTrailingZeros(bits & ~KINDFLAGS)
93 
94     /** The  list of non-empty names of flags with given index idx that are set in this FlagSet */
95     private def flagString(idx: Int): List[String] =
96       if ((bits & (1L << idx)) == 0) Nil
97       else {
98         def halfString(kind: Int) =
99           if ((bits & (1L << kind)) != 0) flagName(idx)(kind) else ""
100         val termFS = halfString(TERMindex)
101         val typeFS = halfString(TYPEindex)
102         val strs = termFS :: (if (termFS == typeFS) Nil else typeFS :: Nil)
103         strs filter (_.nonEmpty)
104       }
105 
106     /** The list of non-empty names of flags that are set in this FlagSet */
107     def flagStrings: Seq[String] = {
108       val rawStrings = (2 to MaxFlag).flatMap(flagString)
109       if (this is Local)
110         rawStrings.filter(_ != "<local>").map {
111           case "private" => "private[this]"
112           case "protected" => "protected[this]"
113           case str => str
114         }
115       else rawStrings
116     }
117 
118     /** The string representation of this flag set */
119     override def toString = flagStrings.mkString(" ")
120   }

174   /** The union of all flags in given flag set */
175   def union(flagss: FlagSet*): FlagSet = {
176     var flag = EmptyFlags
177     for (f <- flagss)
178       flag |= f
179     flag
180   }

182   /** The conjunction of all flags in given flag set */
183   def allOf(flags1: FlagSet, flags2: FlagSet): FlagConjunction = {
184     assert(flags1.numFlags == 1 && flags2.numFlags == 1, "Flags.allOf doesn't support flag " + (if (flags1.numFlags != 1) flags1 else flags2))
185     FlagConjunction((flags1 | flags2).bits)
186   }
187 
188   /** The conjunction of all flags in given flag set */
189   def allOf(flags1: FlagSet, flags2: FlagSet, flags3: FlagSet, flagss: FlagSet*): FlagConjunction = {
190     val flags0 = allOf(flags1, flags2) | flags3
191     assert(flags3.numFlags == 1 && flagss.forall(_.numFlags == 1), "Flags.allOf doesn't support flag " + (if (flags3.numFlags != 1) flags3 else flagss.find(_.numFlags != 1)))
192     FlagConjunction((flags0 | union(flagss: _*)).bits)
193   }

195   def commonFlags(flagss: FlagSet*) = union(flagss.map(_.toCommonFlags): _*)
