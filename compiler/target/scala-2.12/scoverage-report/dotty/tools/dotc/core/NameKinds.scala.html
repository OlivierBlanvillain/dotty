38     /** A simple info type; some subclasses of Kind define more refined versions */

172   abstract class NumberedNameKind(tag: Int, val infoString: String) extends NameKind(tag) { self =>
173     type ThisInfo = NumberedInfo

175       override def toString = s"$infoString $num"
176     }
177     def apply(qual: TermName, num: Int) =
178       qual.derived(new NumberedInfo(num))
179     def unapply(name: DerivedName): Option[(TermName, Int)] = name match {
180       case DerivedName(underlying, info: this.NumberedInfo) => Some((underlying, info.num))
181       case _ => None

183     protected def skipSeparatorAndNum(name: SimpleName, separator: String): Int = {
184       var i = name.length
185       while (i > 0 && name(i - 1).isDigit) i -= 1
186       if (i > separator.length && i < name.length &&
187           name.slice(i - separator.length, i).toString == separator) i
188       else -1
189     }

192   }

250       var i = name.lastIndexOfSlice(str.EXPAND_SEPARATOR)
251       if (i < 0) name

253         // Hack to make super accessors from traits work. They would otherwise fail because of #765
254         // The problem is that in `x$$super$$plus` the expansion prefix needs to be `x`
255         // instead of `x$$super`.
256         if (i > FalseSuperLength && name.slice(i - FalseSuperLength, i) == FalseSuper)
257           i -= FalseSuper.length
