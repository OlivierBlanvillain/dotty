74   /** Compute all shift-deltas */
75   private def computeDeltas() = {
76     delta = new Array[Int](numOffsets)
77     var lastDelta = 0
78     var i = 0
79     while (i < numOffsets) {
80       val off = offset(i)
81       val skippedOff = skipZeroes(off)
82       val skippedCount = skippedOff.index - off.index
83       assert(skippedCount < AddrWidth, s"unset field at position $off")
84       lastDelta += skippedCount
85       delta(i) = lastDelta
86       i += 1
87     }
88   }

90   /** The absolute or relative adjusted address at index `i` of `offsets` array*/
91   private def adjustedOffset(i: Int): Addr = {
92     val at = offset(i)
93     val original = getAddr(at)
94     if (isRelative(i)) {
95       val start = skipNat(at)
96       val len1 = original + delta(i) - deltaAt(original + start.index)
97       val len2 = adjusted(original + start.index) - adjusted(start).index
98       assert(len1 == len2,
99           s"adjusting offset #$i: $at, original = $original, len1 = $len1, len2 = $len2")
100       len1
101     } else adjusted(original)
102   }

104   /** Adjust all offsets according to previously computed deltas */
105   private def adjustOffsets(): Unit = {
106     for (i <- 0 until numOffsets) {

109     }
110   }

112   /** Adjust deltas to also take account references that will shrink (and thereby
113    *  generate additional zeroes that can be skipped) due to previously
114    *  computed adjustments.
115    */
116   private def adjustDeltas(): Int = {
117     val delta1 = new Array[Int](delta.length)
118     var lastDelta = 0
119     var i = 0
120     while (i < numOffsets) {
121       val corrected = adjustedOffset(i)
122       lastDelta += AddrWidth - TastyBuffer.natSize(corrected.index)
123       delta1(i) = lastDelta
124       i += 1
125     }
126     val saved =
127       if (numOffsets == 0) 0
128       else delta1(numOffsets - 1) - delta(numOffsets - 1)
129     delta = delta1
130     saved
131   }

133   /** Compress pickle buffer, shifting bytes to close all skipped zeroes. */
134   private def compress(): Int = {
135     var lastDelta = 0
136     var start = 0
137     var i = 0
138     var wasted = 0
139     def shift(end: Int) =
140       Array.copy(bytes, start, bytes, start - lastDelta, end - start)
141     while (i < numOffsets) {
142       val next = offsets(i)
143       shift(next)
144       start = next + delta(i) - lastDelta
145       val pastZeroes = skipZeroes(Addr(next)).index
146       assert(pastZeroes >= start, s"something's wrong: eliminated non-zero")
147       wasted += (pastZeroes - start)
148       lastDelta = delta(i)
149       i += 1
150     }
151     shift(length)
152     length -= lastDelta
153     wasted
154   }

156   def adjustTreeAddrs(): Unit = {
157     val it = treeAddrs.keySet.iterator
158     while (it.hasNext) {
159       val tree = it.next
160       treeAddrs.get(tree) match {
161         case addr: Addr => treeAddrs.put(tree, adjusted(addr))
162       }
163     }
164   }
165 
166   /** Final assembly, involving the following steps:
167    *   - compute deltas
168    *   - adjust deltas until additional savings are < 1% of total
169    *   - adjust offsets according to the adjusted deltas
170    *   - shrink buffer, skipping zeroes.
171    */
172   def compactify(): Unit = {
173     val origLength = length
174     computeDeltas()
175     //println(s"offsets: ${offsets.take(numOffsets).deep}")
176     //println(s"deltas: ${delta.take(numOffsets).deep}")
177     var saved = 0
178     do {
179       saved = adjustDeltas()
180       pickling.println(s"adjusting deltas, saved = $saved")
181     } while (saved > 0 && length / saved < 100)
182     adjustOffsets()
183     adjustTreeAddrs()
184     val wasted = compress()
185     pickling.println(s"original length: $origLength, compressed to: $length, wasted: $wasted") // DEBUG, for now.
186   }
