295     private def enterAllInHash(e: ScopeEntry, n: Int = 0)(implicit ctx: Context): Unit = {
296       if (e ne null) {
297         if (n < MaxRecursions) {
298           enterAllInHash(e.prev, n + 1)
299           enterInHash(e)
300         } else {
301           var entries: List[ScopeEntry] = List()
302           var ee = e
303           while (ee ne null) {
304             entries = ee :: entries
305             ee = ee.prev
306           }
307           entries foreach enterInHash
308         }
309       }

312     /** Remove entry from this scope (which is required to be present) */
313     final def unlink(e: ScopeEntry)(implicit ctx: Context): Unit = {
314       if (lastEntry == e) {
315         lastEntry = e.prev
316       } else {
317         var e1 = lastEntry
318         while (e1.prev != e) e1 = e1.prev
319         e1.prev = e.prev
320       }
321       if (hashTable ne null) {
322         val index = e.name.hashCode & (hashTable.length - 1)
323         var e1 = hashTable(index)
324         if (e1 == e)
325           hashTable(index) = e.tail
326         else {
327           while (e1.tail != e) e1 = e1.tail
328           e1.tail = e.tail
329         }
330       }
331       elemsCache = null
332       size -= 1
333     }

363     override def lookupEntry(name: Name)(implicit ctx: Context): ScopeEntry = {
364       var e: ScopeEntry = null
365       if (hashTable ne null) {
366         e = hashTable(name.hashCode & (hashTable.length - 1))
367         while ((e ne null) && e.name != name) {
368           e = e.tail
369         }
370       } else {
371         e = lastEntry
372         while ((e ne null) && e.name != name) {
373           e = e.prev
374         }
375       }
376       if ((e eq null) && (synthesize != null)) {
377         val sym = synthesize(name)(ctx)
378         if (sym.exists) newScopeEntry(sym) else e
379       }
380       else e
381     }
