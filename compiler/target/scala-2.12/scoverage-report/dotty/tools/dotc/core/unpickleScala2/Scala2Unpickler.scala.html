626   /** Convert
627    *    tp { type name = sym } forSome { sym >: L <: H }
628    *  to
629    *    tp { name >: L <: H }
630    *  and
631    *    tp { name: sym } forSome { sym <: T with Singleton }
632    *  to
633    *    tp { name: T }
634    */
635   def elimExistentials(boundSyms: List[Symbol], tp: Type)(implicit ctx: Context): Type = {
636     // Need to be careful not to run into cyclic references here (observed when
637     // compiling t247.scala). That's why we avoid taking `symbol` of a TypeRef
638     // unless names match up.
639     val isBound = (tp: Type) => {
640       def refersTo(tp: Type, sym: Symbol): Boolean = tp match {
641         case tp: TypeRef => sym.name == tp.name && sym == tp.symbol
642         case tp: TypeVar => refersTo(tp.underlying, sym)
643         case tp : LazyRef => refersTo(tp.ref, sym)
644         case _ => false
645       }
646       boundSyms.exists(refersTo(tp, _))
647     }
648     // Cannot use standard `existsPart` method because it calls `lookupRefined`
649     // which can cause CyclicReference errors.
650     val isBoundAccumulator = new ExistsAccumulator(isBound) {
651       override def foldOver(x: Boolean, tp: Type): Boolean = tp match {
652         case tp: TypeRef => applyToPrefix(x, tp)
653         case _ => super.foldOver(x, tp)
654       }
655     }
656     def removeSingleton(tp: Type): Type =
657       if (tp isRef defn.SingletonClass) defn.AnyType else tp
658     def elim(tp: Type): Type = tp match {
659       case tp @ RefinedType(parent, name, rinfo) =>
660         val parent1 = elim(tp.parent)
661         rinfo match {
662           case TypeAlias(info: TypeRef) if isBound(info) =>
663             RefinedType(parent1, name, info.symbol.info)
664           case info: TypeRef if isBound(info) =>
665             val info1 = info.symbol.info
666             assert(info1.derivesFrom(defn.SingletonClass))
667             RefinedType(parent1, name, info1.mapReduceAnd(removeSingleton)(_ & _))
668           case info =>
669             tp.derivedRefinedType(parent1, name, info)
670         }
671       case tp @ AppliedType(tycon, args) =>
672         val tycon1 = tycon.safeDealias
673         def mapArg(arg: Type) = arg match {
674           case arg: TypeRef if isBound(arg) => arg.symbol.info
675           case _ => arg
676         }
677         if (tycon1 ne tycon) elim(tycon1.appliedTo(args))
678         else tp.derivedAppliedType(tycon, args.map(mapArg))
679       case _ =>
680         tp
681     }
682     val tp1 = elim(tp)
683     if (isBoundAccumulator(false, tp1)) {
684       val anyTypes = boundSyms map (_ => defn.AnyType)
685       val boundBounds = boundSyms map (_.info.bounds.hi)
686       val tp2 = tp1.subst(boundSyms, boundBounds).subst(boundSyms, anyTypes)
687       ctx.warning(FailureToEliminateExistential(tp, tp1, tp2, boundSyms))
688       tp2
689     } else tp1
690   }

692   /** Read a type
693    *
694    *  @param forceProperType is used to ease the transition to NullaryMethodTypes (commentmarker: NMT_TRANSITION)
695    *        the flag say that a type of kind * is expected, so that PolyType(tps, restpe) can be disambiguated to PolyType(tps, NullaryMethodType(restpe))
696    *        (if restpe is not a ClassInfoType, a MethodType or a NullaryMethodType, which leaves TypeRef/SingletonType -- the latter would make the polytype a type constructor)
697    */
698   protected def readType()(implicit ctx: Context): Type = {
699     val tag = readByte()
700     val end = readNat() + readIndex
701     (tag: @switch) match {
702       case NOtpe =>
703         NoType
704       case NOPREFIXtpe =>
705         NoPrefix
706       case THIStpe =>
707         readSymbolRef().thisType
708       case SINGLEtpe =>
709         val pre = readTypeRef()
710         val sym = readDisambiguatedSymbolRef(_.info.isParameterless)
711         pre.select(sym)
712       case SUPERtpe =>
713         val thistpe = readTypeRef()
714         val supertpe = readTypeRef()
715         SuperType(thistpe, supertpe)
716       case CONSTANTtpe =>
717         ConstantType(readConstantRef())
718       case TYPEREFtpe =>
719         var pre = readTypeRef()
720         val sym = readSymbolRef()
721         pre match {
722           case thispre: ThisType =>
723             // The problem is that class references super.C get pickled as
724             // this.C. Dereferencing the member might then get an overriding class
725             // instance. The problem arises for instance for LinkedHashMap#MapValues
726             // and also for the inner Transform class in all views. We fix it by
727             // replacing the this with the appropriate super.
728             if (sym.owner != thispre.cls) {
729               val overriding = thispre.cls.info.decls.lookup(sym.name)
730               if (overriding.exists && overriding != sym) {
731                 val base = pre.baseType(sym.owner)
732                 assert(base.exists)
733                 pre = SuperType(thispre, base)
734               }
735             }
736           case NoPrefix if sym is TypeParam =>
737             pre = sym.owner.thisType
738           case _ =>
739         }
740         val tycon = pre.select(sym)
741         val args = until(end, () => readTypeRef())
742         if (sym == defn.ByNameParamClass2x) ExprType(args.head)
743         else if (args.nonEmpty) tycon.safeAppliedTo(EtaExpandIfHK(sym.typeParams, args.map(translateTempPoly)))
744         else if (sym.typeParams.nonEmpty) tycon.EtaExpand(sym.typeParams)
745         else tycon
746       case TYPEBOUNDStpe =>
747         TypeBounds(readTypeRef(), readTypeRef())
748       case REFINEDtpe =>
749         val clazz = readSymbolRef()
750         val decls = symScope(clazz)
751         symScopes(clazz) = EmptyScope // prevent further additions
752         val parents = until(end, () => readTypeRef())
753         val parent = parents.reduceLeft(AndType(_, _))
754         if (decls.isEmpty) parent
755         else {
756           def subst(info: Type, rt: RecType) =
757             if (clazz.isClass) info.substThis(clazz.asClass, rt.recThis)
758             else info // turns out some symbols read into `clazz` are not classes, not sure why this is the case.
759           def addRefinement(tp: Type, sym: Symbol) = RefinedType(tp, sym.name, sym.info)
760           val refined = (parent /: decls.toList)(addRefinement)
761           RecType.closeOver(rt => subst(refined, rt))
762         }
763       case CLASSINFOtpe =>
764         val clazz = readSymbolRef()
765         TempClassInfoType(until(end, () => readTypeRef()), symScope(clazz), clazz)
766       case METHODtpe | IMPLICITMETHODtpe =>
767         val restpe = readTypeRef()
768         val params = until(end, () => readSymbolRef()).asInstanceOf[List[TermSymbol]]
769         def isImplicit =
770           tag == IMPLICITMETHODtpe ||
771           params.nonEmpty && (params.head is Implicit)
772         val maker = if (isImplicit) ImplicitMethodType else MethodType
773         maker.fromSymbols(params, restpe)
774       case POLYtpe =>
775         val restpe = readTypeRef()
776         val typeParams = until(end, () => readSymbolRef())
777         if (typeParams.nonEmpty) TempPolyType(typeParams.asInstanceOf[List[TypeSymbol]], restpe.widenExpr)
778         else ExprType(restpe)
779       case EXISTENTIALtpe =>
780         val restpe = readTypeRef()
781         val boundSyms = until(end, () => readSymbolRef())
782         elimExistentials(boundSyms, restpe)
783       case ANNOTATEDtpe =>
784         AnnotatedType.make(readTypeRef(), until(end, () => readAnnotationRef()))
785       case _ =>
786         noSuchTypeTag(tag, end)
787     }
788   }
