6 /** Periods are the central "clock" of the compiler.
7  *  A period consists of a run id and a phase id.
8  *  run ids represent compiler runs
9  *  phase ids represent compiler phases
10  */
11 abstract class Periods extends DotClass { self: Context =>
12   import Periods._
13 
14   /** The current phase identifier */
15   def phaseId: Int = period.phaseId
16 
17   /** The current run identifier */
18   def runId: Int = period.runId
19 
20   /** Execute `op` at given period */
21   def atPeriod[T](pd: Period)(op: Context => T): T =
22     op(ctx.fresh.setPeriod(pd))
23 
24   /** Execute `op` at given phase id */
25   def atPhase[T](pid: PhaseId)(op: Context => T): T =
26     op(ctx.withPhase(pid))

28   /** The period containing the current period where denotations do not change.
29    *  We compute this by taking as first phase the first phase less or equal to
30    *  the current phase that has the same "nextTransformerId". As last phase
31    *  we take the next transformer id following the current phase.
32    */
33   def stablePeriod = {
34     var first = phaseId
35     val nxTrans = ctx.base.nextDenotTransformerId(first)
36     while (first - 1 > NoPhaseId && (ctx.base.nextDenotTransformerId(first - 1) == nxTrans)) {
37       first -= 1
38     }
39     Period(runId, first, nxTrans)
40   }
41 
42   /** Are all base types in the current period guaranteed to be the same as in period `p`? */
43   def hasSameBaseTypesAs(p: Period) = {
44     val period = this.period
45     period == p ||
46     period.runId == p.runId &&
47       this.phases(period.phaseId).sameParentsStartId ==
48       this.phases(p.phaseId).sameParentsStartId
49   }
50 }

52 object Periods {
53 
54   /** A period is a contiguous sequence of phase ids in some run.
55    *  It is coded as follows:
56    *
57    *     sign, always 0        1 bit
58    *     runid                17 bits
59    *     last phase id:        7 bits
60    *     #phases before last:  7 bits
61    *
62    *     // Dmitry: sign == 0 isn't actually always true, in some cases phaseId == -1 is used for shifts, that easily creates code < 0
63    */
64   class Period(val code: Int) extends AnyVal {
65 
66     /** The run identifier of this period. */
67     def runId: RunId = code >>> (PhaseWidth * 2)
68 
69     /** The phase identifier of this single-phase period. */
70     def phaseId: PhaseId = (code >>> PhaseWidth) & PhaseMask
71 
72     /** The last phase of this period */
73     def lastPhaseId: PhaseId =
74       (code >>> PhaseWidth) & PhaseMask
75 
76     /** The first phase of this period */
77     def firstPhaseId = lastPhaseId - (code & PhaseMask)
78 
79     def containsPhaseId(id: PhaseId) = firstPhaseId <= id && id <= lastPhaseId
80 
81     /** Does this period contain given period? */
82     def contains(that: Period): Boolean = {
83       // Let    this = (r1, l1, d1), that = (r2, l2, d2)
84       // where  r = runid, l = last phase, d = duration - 1
85       // Then seen as intervals:
86       //
87       //  this = r1 / (l1 - d1) .. l1
88       //  that = r2 / (l2 - d2) .. l2
89       //
90       // Let's compute:
91       //
92       //  lastDiff = X * 2^5 + (l1 - l2) mod 2^5
93       //             where X >= 0, X == 0 iff r1 == r2 & l1 - l2 >= 0
94       //  result = lastDiff + d2 <= d1
95       //  We have:
96       //      lastDiff + d2 <= d1
97       //  iff X == 0 && l1 - l2 >= 0 && l1 - l2 + d2 <= d1
98       //  iff r1 == r2 & l1 >= l2 && l1 - d1 <= l2 - d2
99       //  q.e.d
100       val lastDiff = (code - that.code) >>> PhaseWidth
101       lastDiff + (that.code & PhaseMask ) <= (this.code & PhaseMask)
102     }
103 
104     /** Does this period overlap with given period? */
105     def overlaps(that: Period): Boolean =
106       this.runId == that.runId &&
107       this.firstPhaseId <= that.lastPhaseId &&
108       that.firstPhaseId <= this.lastPhaseId
109 
110     /** The intersection of two periods */
111     def & (that: Period): Period =
112       if (this overlaps that)
113         Period(
114           this.runId,
115           this.firstPhaseId max that.firstPhaseId,
116           this.lastPhaseId min that.lastPhaseId)
117       else
118         Nowhere
119 
120     /** The smallest period containing two periods */
121     def | (that: Period): Period =
122       Period(this.runId,
123           this.firstPhaseId min that.firstPhaseId,
124           this.lastPhaseId max that.lastPhaseId)
125 
126     override def toString = s"Period($firstPhaseId..$lastPhaseId, run = $runId)"
127   }
128 
129   object Period {
130 
131     /** The single-phase period consisting of given run id and phase id */
132     def apply(rid: RunId, pid: PhaseId): Period = {
133       new Period(((rid << PhaseWidth) | pid) << PhaseWidth)
134     }
135 
136     /** The period consisting of given run id, and lo/hi phase ids */
137     def apply(rid: RunId, loPid: PhaseId, hiPid: PhaseId): Period = {
138       new Period(((rid << PhaseWidth) | hiPid) << PhaseWidth | (hiPid - loPid))
139     }
140 
141     /** The interval consisting of all periods of given run id */
142     def allInRun(rid: RunId) = {
143       apply(rid, 0, PhaseMask)
144     }
145   }
146 
147   final val Nowhere = new Period(0)
148 
149   final val InitialPeriod = Period(InitialRunId, FirstPhaseId)
150 
151   final val InvalidPeriod = Period(NoRunId, NoPhaseId)
152 
153   /** An ordinal number for compiler runs. First run has number 1. */
154   type RunId = Int
155   final val NoRunId = 0
156   final val InitialRunId = 1
157   final val RunWidth = java.lang.Integer.SIZE - PhaseWidth * 2 - 1/* sign */
158   final val MaxPossibleRunId = (1 << RunWidth) - 1
159 
160   /** An ordinal number for phases. First phase has number 1. */
161   type PhaseId = Int
162   final val NoPhaseId = 0
163   final val FirstPhaseId = 1
164 
165   /** The number of bits needed to encode a phase identifier. */
166   final val PhaseWidth = 7
167   final val PhaseMask = (1 << PhaseWidth) - 1
168   final val MaxPossiblePhaseId = PhaseMask
169 }
