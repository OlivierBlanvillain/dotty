22   private val pickledIndices = new mutable.BitSet

24   def header(addrDelta: Int, hasStartDelta: Boolean, hasEndDelta: Boolean, hasPoint: Boolean) = {
25     def toInt(b: Boolean) = if (b) 1 else 0
26     (addrDelta << 3) | (toInt(hasStartDelta) << 2) | (toInt(hasEndDelta) << 1) | toInt(hasPoint)
27   }
28 
29   def picklePositions(roots: List[Tree])(implicit ctx: Context) = {
30     var lastIndex = 0
31     var lastPos = Position(0, 0)
32     def pickleDeltas(index: Int, pos: Position) = {
33       val addrDelta = index - lastIndex
34       val startDelta = pos.start - lastPos.start
35       val endDelta = pos.end - lastPos.end
36       buf.writeInt(header(addrDelta, startDelta != 0, endDelta != 0, !pos.isSynthetic))
37       if (startDelta != 0) buf.writeInt(startDelta)
38       if (endDelta != 0) buf.writeInt(endDelta)
39       if (!pos.isSynthetic) buf.writeInt(pos.pointDelta)
40       lastIndex = index
41       lastPos = pos
42 
43       pickledIndices += index
44     }
45 
46     /** True if x's position shouldn't be reconstructed automatically from its initialPos
47      */
48     def alwaysNeedsPos(x: Positioned) = x match {
49       case
50           // initialPos is inaccurate for trees with lazy field
51           _: WithLazyField[_]
52 
53           // A symbol is created before the corresponding tree is unpickled,
54           // and its position cannot be changed afterwards.
55           // so we cannot use the tree initialPos to set the symbol position.
56           // Instead, we always pickle the position of definitions.
57           | _: Trees.DefTree[_]
58 
59           // package defs might be split into several Tasty files
60           | _: Trees.PackageDef[_] => true
61       case _ => false
62     }
63 
64     def traverse(x: Any): Unit = x match {
65       case x: Tree @unchecked =>
66         val pos = if (x.isInstanceOf[MemberDef]) x.pos else x.pos.toSynthetic
67         if (pos.exists && (pos != x.initialPos.toSynthetic || alwaysNeedsPos(x))) {
68           addrOfTree(x) match {
69             case Some(addr) if !pickledIndices.contains(addr.index) =>
70               //println(i"pickling $x with $pos at $addr")
71               pickleDeltas(addr.index, pos)
72             case _ =>
73               //println(i"no address for $x")
74           }
75         }
76         //else if (x.pos.exists) println(i"skipping $x")
77         x match {
78           case x: MemberDef @unchecked =>
79             for (ann <- x.symbol.annotations) traverse(ann.tree)
80           case _ =>
81         }
82         traverse(x.productIterator)
83       case xs: TraversableOnce[_] =>
84         xs.foreach(traverse)
85       case x: Annotation =>
86         traverse(x.tree)
87       case _ =>
88     }
89     traverse(roots)
90   }
91 }
