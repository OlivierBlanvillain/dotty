135   /** Commit typer state so that its information is copied into current typer state
136    *  In addition (1) the owning state of undetermined or temporarily instantiated
137    *  type variables changes from this typer state to the current one. (2) Variables
138    *  that were temporarily instantiated in the current typer state are permanently
139    *  instantiated instead.
140    *
141    *  A note on merging: An interesting test case is isApplicableSafe.scala. It turns out that this
142    *  requires a context merge using the new `&' operator. Sequence of actions:
143    *  1) Typecheck argument in typerstate 1.
144    *  2) Cache argument.
145    *  3) Evolve same typer state (to typecheck other arguments, say)
146    *     leading to a different constraint.
147    *  4) Take typechecked argument in same state.
148    *
149    * It turns out that the merge is needed not just for
150    * isApplicableSafe but also for (e.g. erased-lubs.scala) as well as
151    * many parts of dotty itself.
152    */
153   def commit()(implicit ctx: Context) = {
154     val targetState = ctx.typerState
155     assert(isCommittable)
156     targetState.constraint =
157       if (targetState.constraint eq previousConstraint) constraint
158       else targetState.constraint & constraint
159     constraint foreachTypeVar { tvar =>
160       if (tvar.owningState.get eq this) tvar.owningState = new WeakReference(targetState)
161     }
162     targetState.ephemeral |= ephemeral
163     targetState.gc()
164     reporter.flush()
165     isCommitted = true
166   }

172   def gc()(implicit ctx: Context): Unit = {
173     val toCollect = new mutable.ListBuffer[TypeLambda]
174     constraint foreachTypeVar { tvar =>
175       if (!tvar.inst.exists) {
176         val inst = instType(tvar)
177         if (inst.exists && (tvar.owningState.get eq this)) {
178           tvar.inst = inst
179           val lam = tvar.origin.binder
180           if (constraint.isRemovable(lam)) toCollect += lam
181         }
182       }
183     }
184     for (poly <- toCollect)
185       constraint = constraint.remove(poly)
186   }
