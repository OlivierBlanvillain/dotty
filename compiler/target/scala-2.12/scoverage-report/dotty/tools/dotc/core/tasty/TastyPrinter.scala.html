39   class TreeSectionUnpickler extends SectionUnpickler[Unit]("ASTs") {
40     import TastyFormat._
41     def unpickle(reader: TastyReader, tastyName: NameTable): Unit = {
42       import reader._
43       var indent = 0
44       def newLine() = {
45         val length = treeColor("%5d".format(index(currentAddr) - index(startAddr)))
46         print(s"\n $length:" + " " * indent)
47       }
48       def printNat() = print(Yellow(" " + readNat()).show)
49       def printName() = {

53       def printTree(): Unit = {
54         newLine()

56         print(" ");print(astTagToString(tag))

58         if (tag >= firstLengthTreeTag) {

62           def printTrees() = until(end)(printTree())

67               printName(); printTrees()
68             case REFINEDtype =>

71               printNat(); printTrees()
72             case METHODtype | POLYtype | TYPELAMBDAtype =>
73               printTree()

75             case PARAMtype =>
76               printNat(); printNat()
77             case _ =>
78               printTrees()
79           }
80           if (currentAddr != end) {
81             println(s"incomplete read, current = $currentAddr, end = $end")
82             goto(end)
83           }
84         }
85         else if (tag >= firstNatASTTreeTag) {
86           tag match {
87             case IDENT | IDENTtpt | SELECT | TERMREF | TYPEREF | SELFDEF => printName()
88             case _ => printNat()
89           }

91         }
92         else if (tag >= firstASTTreeTag)
93           printTree()
94         else if (tag >= firstNatTreeTag)
95           tag match {
96             case TERMREFpkg | TYPEREFpkg | STRINGconst | IMPORTED => printName()
97             case _ => printNat()
98           }
99         indent -= 2
100       }
101       println(i"start = ${reader.startAddr}, base = $base, current = $currentAddr, end = $endAddr")
102       println(s"${endAddr.index - startAddr.index} bytes of AST, base = $currentAddr")
103       while (!isAtEnd) {

105         newLine()
106       }
107     }
