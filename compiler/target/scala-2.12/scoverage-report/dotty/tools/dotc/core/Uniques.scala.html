9 /** Defines operation `unique` for hash-consing types.
10  *  Also defines specialized hash sets for hash consing uniques of a specific type.
11  *  All sets offer a `enterIfNew` method which checks whether a type
12  *  with the given parts exists already and creates a new one if not.
13  */
14 object Uniques {
15 
16   private def recordCaching(tp: Type): Unit = recordCaching(tp.hash, tp.getClass)
17   private def recordCaching(h: Int, clazz: Class[_]): Unit =
18     if (h == NotCached) {
19       record("uncached-types")
20       record(s"uncached: $clazz")
21     } else {
22       record("cached-types")
23       record(s"cached: $clazz")
24     }
25 
26   def unique[T <: Type](tp: T)(implicit ctx: Context): T = {
27     if (monitored) recordCaching(tp)
28     if (tp.hash == NotCached) tp
29     else if (monitored) {
30       val size = ctx.uniques.size
31       val result = ctx.uniques.findEntryOrUpdate(tp).asInstanceOf[T]
32       if (ctx.uniques.size > size) record(s"fresh unique ${tp.getClass}")
33       result
34     } else ctx.uniques.findEntryOrUpdate(tp).asInstanceOf[T]
35   } /* !!! DEBUG
36   ensuring (
37     result => tp.toString == result.toString || {
38       println(s"cache mismatch; tp = $tp, cached = $result")
39       false
40     }
41   )
42  */

56     def enterIfNew(prefix: Type, designator: Designator, isTerm: Boolean)(implicit ctx: Context): NamedType = {
57       val h = doHash(designator, prefix)
58       if (monitored) recordCaching(h, classOf[NamedType])

60         if (isTerm) new CachedTermRef(prefix, designator, h)
61         else new CachedTypeRef(prefix, designator, h)
62       if (h == NotCached) newType
63       else {
64         val r = findPrevious(h, prefix, designator)
65         if (r ne null) r else addEntryAfterScan(newType)
66       }
67     }
68   }
