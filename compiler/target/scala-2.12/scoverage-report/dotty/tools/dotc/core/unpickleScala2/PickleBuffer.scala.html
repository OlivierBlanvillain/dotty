8 /** Variable length byte arrays, with methods for basic pickling and unpickling.

32   /** Write a byte of data */
33   def writeByte(b: Int): Unit = {
34     if (writeIndex == bytes.length) dble()
35     bytes(writeIndex) = b.toByte
36     writeIndex += 1

39   /** Write a natural number in big endian format, base 128.
40    *  All but the last digits have bit 0x80 set.
41    */
42   def writeNat(x: Int): Unit =
43     writeLongNat(x.toLong & 0x00000000FFFFFFFFL)
44 
45   /**
46    * Like writeNat, but for longs. This is not the same as
47    * writeLong, which writes in base 256. Note that the
48    * binary representation of LongNat is identical to Nat
49    * if the long value is in the range Int.MIN_VALUE to
50    * Int.MAX_VALUE.
51    */
52   def writeLongNat(x: Long): Unit = {
53     def writeNatPrefix(x: Long): Unit = {
54       val y = x >>> 7
55       if (y != 0L) writeNatPrefix(y)
56       writeByte(((x & 0x7f) | 0x80).toInt)
57     }
58     val y = x >>> 7
59     if (y != 0L) writeNatPrefix(y)
60     writeByte((x & 0x7f).toInt)
61   }
62 
63   /** Write a natural number <code>x</code> at position <code>pos</code>.
64    *  If number is more than one byte, shift rest of array to make space.
65    *
66    *  @param pos ...
67    *  @param x   ...
68    */
69   def patchNat(pos: Int, x: Int): Unit = {
70     def patchNatPrefix(x: Int): Unit = {
71       writeByte(0)
72       Array.copy(bytes, pos, bytes, pos + 1, writeIndex - (pos + 1))
73       bytes(pos) = ((x & 0x7f) | 0x80).toByte
74       val y = x >>> 7
75       if (y != 0) patchNatPrefix(y)
76     }
77     bytes(pos) = (x & 0x7f).toByte
78     val y = x >>> 7
79     if (y != 0) patchNatPrefix(y)
80   }

82   /** Write a long number <code>x</code> in signed big endian format, base 256.
83    *
84    *  @param x The long number to be written.
85    */
86   def writeLong(x: Long): Unit = {
87     val y = x >> 8
88     val z = x & 0xff
89     if (-y != (z >> 7)) writeLong(y)
90     writeByte(z.toInt)

107   def readLongNat(): Long = {
108     var b = 0L
109     var x = 0L
110     do {
111       b = readByte()
112       x = (x << 7) + (b & 0x7f)
113     } while ((b & 0x80) != 0L)
114     x
115   }

117   /** Read a long number in signed big endian format, base 256. */
118   def readLong(len: Int): Long = {
119     var x = 0L
120     var i = 0
121     while (i < len) {
122       x = (x << 8) + (readByte() & 0xff)
123       i += 1
124     }
125     val leading = 64 - (len << 3)
126     x << leading >> leading
127   }

129   /** Returns the buffer as a sequence of (Int, Array[Byte]) representing
130    *  (tag, data) of the individual entries.  Saves and restores buffer state.

133   def toIndexedSeq: IndexedSeq[(Int, Array[Byte])] = {
134     val saved = readIndex
135     readIndex = 0
136     readNat() ; readNat()     // discarding version
137     val result = new Array[(Int, Array[Byte])](readNat())

139     result.indices foreach { index =>

141       val len = readNat()

143       readIndex += len

146     }

152   /** Perform operation <code>op</code> until the condition
153    *  <code>readIndex == end</code> is satisfied.
154    *  Concatenate results into a list.
155    *
156    *  @param end ...
157    *  @param op  ...
158    *  @return    ...
159    */
160   def until[T](end: Int, op: () => T): List[T] =
161     if (readIndex == end) List() else op() :: until(end, op)
162 
163   /** Perform operation <code>op</code> the number of
164    *  times specified.  Concatenate the results into a list.
165    */
166   def times[T](n: Int, op: ()=>T): List[T] =
167     if (n == 0) List() else op() :: times(n-1, op)

169   /** Pickle = majorVersion_Nat minorVersion_Nat nbEntries_Nat {Entry}
170    *  Entry  = type_Nat length_Nat [actual entries]
171    *
172    *  Assumes that the ..Version_Nat are already consumed.
173    *
174    *  @return an array mapping entry numbers to locations in
175    *  the byte array where the entries start.

177   def createIndex: Array[Int] = {
178     val index = new Array[Int](readNat()) // nbEntries_Nat
179     for (i <- 0 until index.length) {
180       index(i) = readIndex
181       readByte() // skip type_Nat
182       readIndex = readNat() + readIndex // read length_Nat, jump to next entry
183     }
184     index
185   }

239     // generate initial maps from Scala flags to Dotty flags
240     val termMap, typeMap = new Array[Long](64)
241     for (idx <- 0 until ScalaFlagEnd)
242       corr get (1L << idx) match {
243         case Some((termFlag: FlagSet, typeFlag: FlagSet)) =>
244           termMap(idx) |= termFlag.bits
245           typeMap(idx) |= typeFlag.bits
246         case Some(commonFlag: FlagSet) =>
247           termMap(idx) |= commonFlag.toTermFlags.bits
248           typeMap(idx) |= commonFlag.toTypeFlags.bits
249         case _ =>
250       }
251 
252     // Convert map so that it maps chunks of ChunkBits size at once
253     // instead of single bits.
254     def chunkMap(xs: Array[Long]): FlagMap = {
255       val chunked = Array.ofDim[Long](
256         (xs.length + ChunkBits - 1) / ChunkBits, ChunkSize)
257       for (i <- 0 until chunked.length)
258         for (j <- 0 until ChunkSize)
259           for (k <- 0 until ChunkBits)
260             if ((j & (1 << k)) != 0)
261               chunked(i)(j) |= xs(i * ChunkBits + k)
262       chunked
263     }
264 
265     (chunkMap(termMap), chunkMap(typeMap))
266   }

268   def unpickleScalaFlags(sflags: Long, isType: Boolean): FlagSet = {
269     val map: FlagMap = if (isType) scalaTypeFlagMap else scalaTermFlagMap
270     val shift = ChunkBits
271     val mask = ChunkSize - 1
272     assert(6 * ChunkBits == ScalaFlagEnd)
273     FlagSet(
274       map(0)((sflags >>> (shift * 0)).toInt & mask) |
275       map(1)((sflags >>> (shift * 1)).toInt & mask) |
276       map(2)((sflags >>> (shift * 2)).toInt & mask) |
277       map(3)((sflags >>> (shift * 3)).toInt & mask) |
278       map(4)((sflags >>> (shift * 4)).toInt & mask) |
279       map(5)((sflags >>> (shift * 5)).toInt & mask)
280     )
281   }
