181   def initializeFromClassPath(owner: Symbol, classRep: ClassRepresentation)(implicit ctx: Context): Unit = {
182     ((classRep.binary, classRep.source): @unchecked) match {
183       case (Some(bin), Some(src)) if needCompile(bin, src) && !binaryOnly(owner, classRep.name) =>

241       val idx = classRep.name.indexOf('$')

247     private def enterClasses(root: SymDenotation, packageName: String, flat: Boolean)(implicit ctx: Context) = {
248       def isAbsent(classRep: ClassRepresentation) =
249         !root.unforcedDecls.lookup(classRep.name.toTypeName).exists
250 
251       if (!root.isRoot) {
252         val classReps = classPath.list(packageName).classesAndSources
253 
254         for (classRep <- classReps)
255           if (!maybeModuleClass(classRep) && isFlatName(classRep) == flat &&
256             (!flat || isAbsent(classRep))) // on 2nd enter of flat names, check that the name has not been entered before
257             initializeFromClassPath(root.symbol, classRep)
258         for (classRep <- classReps)
259           if (maybeModuleClass(classRep) && isFlatName(classRep) == flat &&
260               isAbsent(classRep))
261             initializeFromClassPath(root.symbol, classRep)
262       }
263     }
264 
265     def doComplete(root: SymDenotation)(implicit ctx: Context): Unit = {
266       assert(root is PackageClass, root)
267       val pre = root.owner.thisType
268       root.info = ClassInfo(pre, root.symbol.asClass, Nil, currentDecls, pre select sourceModule)
269       if (!sourceModule.isCompleted)
270         sourceModule.completer.complete(sourceModule)
271 
272       val packageName = if (root.isEffectiveRoot) "" else root.fullName.mangledString
273 
274       enterFlatClasses = Some { ctx =>
275         enterFlatClasses = None
276         enterClasses(root, packageName, flat = true)(ctx)
277       }
278       enterClasses(root, packageName, flat = false)
279       if (!root.isEmptyPackage)
280         for (pkg <- classPath.packages(packageName)) {
281           val fullName = pkg.name
282           val name =
283             if (packageName.isEmpty) fullName
284             else fullName.substring(packageName.length + 1)
285 
286           enterPackage(root.symbol, name.toTermName,
287             (module, modcls) => new PackageLoader(module, classPath))
288         }
289     }

308   override def complete(root: SymDenotation)(implicit ctx: Context): Unit = {
309     def signalError(ex: Exception): Unit = {
310       if (ctx.debug) ex.printStackTrace()
311       val msg = ex.getMessage()
312       ctx.error(
313         if (msg eq null) "i/o error while loading " + root.name
314         else "error while loading " + root.name + ",\n" + msg)
315     }
316     try {
317       val start = currentTime
318       if (ctx.settings.YdebugTrace.value)
319         trace(s">>>> loading ${root.debugString}", _ => s"<<<< loaded ${root.debugString}") {
320           doComplete(root)
321         }
322       else
323         doComplete(root)
324       ctx.informTime("loaded " + description, start)
325     } catch {
326       case ex: IOException =>
327         signalError(ex)
328       case NonFatal(ex) =>
329         println(s"exception caught when loading $root: $ex")
330         throw ex
331     } finally {
332       def postProcess(denot: SymDenotation) =
333         if (!denot.isCompleted &&
334             !denot.completer.isInstanceOf[SymbolLoaders.SecondCompleter])
335           denot.markAbsent()
336       postProcess(root)
337       if (!root.isRoot)
338         postProcess(root.scalacLinkedClass.denot)
339     }
340   }

365             (module, _) => new NoCompleter() withDecls newScope withSourceModule (_ => module))
366             .moduleClass.denot.asClass
367     }
368     if (rootDenot is ModuleClass) (linkedDenot, rootDenot)
369     else (rootDenot, linkedDenot)
370   }
