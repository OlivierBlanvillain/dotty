148       def fail = throw new AssertionError(i"Failure to join alternatives $tp1 and $tp2")

177         case tp: TypeRef => tp.symbol.isClass
178         case tp: AppliedType => isClassRef(tp.tycon)
179         case tp: RefinedType => isClassRef(tp.parent)

186         case tp1: TypeProxy if !isClassRef(tp1) =>

190         case _ =>

193               tp2.rebind(approximateOr(tp1, tp2.parent))

195               orDominator(tp1 | tp2.superType)

200               val doms = dominators(commonBaseClasses, Nil)

202                 tp.baseType(cls).mapReduceOr(identity)(mergeRefinedOrApplied)
203               doms.map(baseTp).reduceLeft(AndType.apply)

205       }

245   /** The list of violations where arguments are not within bounds.
246    *  @param  args          The arguments
247    *  @param  boundss       The list of type bounds
248    *  @param  instantiate   A function that maps a bound type and the list of argument types to a resulting type.
249    *                        Needed to handle bounds that refer to other bounds.
250    */
251   def boundsViolations(args: List[Tree], boundss: List[TypeBounds], instantiate: (Type, List[Type]) => Type)(implicit ctx: Context): List[BoundsViolation] = {
252     val argTypes = args.tpes
253     val violations = new mutable.ListBuffer[BoundsViolation]
254     for ((arg, bounds) <- args zip boundss) {
255       def checkOverlapsBounds(lo: Type, hi: Type): Unit = {
256         //println(i"instantiating ${bounds.hi} with $argTypes")
257         //println(i" = ${instantiate(bounds.hi, argTypes)}")
258         val hiBound = instantiate(bounds.hi, argTypes.mapConserve(_.bounds.hi))
259         val loBound = instantiate(bounds.lo, argTypes.mapConserve(_.bounds.lo))
260           // Note that argTypes can contain a TypeBounds type for arguments that are
261           // not fully determined. In that case we need to check against the hi bound of the argument.
262         if (!(lo <:< hiBound)) violations += ((arg, "upper", hiBound))
263         if (!(loBound <:< hi)) violations += ((arg, "lower", bounds.lo))
264       }
265       arg.tpe match {
266         case TypeBounds(lo, hi) => checkOverlapsBounds(lo, hi)
267         case tp => checkOverlapsBounds(tp, tp)
268       }
269     }
270     violations.toList
271   }

273   /** Is `feature` enabled in class `owner`?
274    *  This is the case if one of the following two alternatives holds:
275    *
276    *  1. The feature is imported by a named import
277    *
278    *       import owner.feature
279    *
280    *     and there is no visible nested import that excludes the feature, as in
281    *
282    *       import owner.{ feature => _ }
283    *
284    *  The feature may be bunched with others, or renamed, but wildcard imports don't count.
285    *
286    *  2. The feature is enabled by a compiler option
287    *
288    *       - language:<prefix>feature
289    *
290    *  where <prefix> is the full name of the owner followed by a "." minus
291    *  the prefix "dotty.language.".
292    */
293   def featureEnabled(owner: ClassSymbol, feature: TermName): Boolean = {
294     def toPrefix(sym: Symbol): String =
295       if (!sym.exists || (sym eq defn.LanguageModuleClass)) ""
296       else toPrefix(sym.owner) + sym.name + "."
297     def featureName = toPrefix(owner) + feature
298     def hasImport(implicit ctx: Context): Boolean = {
299       if (ctx.importInfo eq null) false
300       else {
301         val isImportOwner = ctx.importInfo.site.widen.typeSymbol eq owner
302         if (isImportOwner && ctx.importInfo.originals.contains(feature)) true
303         else if (isImportOwner && ctx.importInfo.excluded.contains(feature)) false
304         else {
305           var c = ctx.outer
306           while (c.importInfo eq ctx.importInfo) c = c.outer
307           hasImport(c)
308         }
309       }
310     }
311     def hasOption = ctx.base.settings.language.value exists (s => s == featureName || s == "_")
312     hasImport(ctx.withPhase(ctx.typerPhase)) || hasOption
313   }
