74   private def pickleSymRef(sym: Symbol)(implicit ctx: Context) = symRefs.get(sym) match {
75     case Some(label) =>
76       if (label != NoAddr) writeRef(label) else pickleForwardSymRef(sym)
77     case None =>
78       // See pos/t1957.scala for an example where this can happen.
79       // I believe it's a bug in typer: the type of an implicit argument refers
80       // to a closure parameter outside the closure itself. TODO: track this down, so that we
81       // can eliminate this case.
82       ctx.log(i"pickling reference to as yet undefined $sym in ${sym.owner}", sym.pos)
83       pickleForwardSymRef(sym)
84   }

95   def pickleConstant(c: Constant)(implicit ctx: Context): Unit = c.tag match {
96     case UnitTag =>
97       writeByte(UNITconst)
98     case BooleanTag =>
99       writeByte(if (c.booleanValue) TRUEconst else FALSEconst)
100     case ByteTag =>
101       writeByte(BYTEconst)
102       writeInt(c.byteValue)
103     case ShortTag =>
104       writeByte(SHORTconst)
105       writeInt(c.shortValue)
106     case CharTag =>
107       writeByte(CHARconst)
108       writeNat(c.charValue)
109     case IntTag =>
110       writeByte(INTconst)
111       writeInt(c.intValue)
112     case LongTag =>
113       writeByte(LONGconst)
114       writeLongInt(c.longValue)
115     case FloatTag =>
116       writeByte(FLOATconst)
117       writeInt(java.lang.Float.floatToRawIntBits(c.floatValue))
118     case DoubleTag =>
119       writeByte(DOUBLEconst)
120       writeLongInt(java.lang.Double.doubleToRawLongBits(c.doubleValue))
121     case StringTag =>
122       writeByte(STRINGconst)
123       pickleName(c.stringValue.toTermName)
124     case NullTag =>
125       writeByte(NULLconst)
126     case ClazzTag =>
127       writeByte(CLASSconst)
128       pickleType(c.typeValue)
129     case EnumTag =>
130       writeByte(ENUMconst)
131       pickleType(c.symbolValue.termRef)
132     case ScalaSymbolTag =>
133       writeByte(SYMBOLconst)
134       pickleName(c.scalaSymbolValue.name.toTermName)
135   }

137   def pickleType(tpe0: Type, richTypes: Boolean = false)(implicit ctx: Context): Unit = {
138     val tpe = tpe0.stripTypeVar
139     try {
140       val prev = pickledTypes.get(tpe)
141       if (prev == null) {
142         pickledTypes.put(tpe, currentAddr)
143         pickleNewType(tpe, richTypes)
144       }
145       else {
146         writeByte(SHAREDtype)
147         writeRef(prev.asInstanceOf[Addr])
148       }
149     } catch {
150       case ex: AssertionError =>
151         println(i"error when pickling type $tpe")
152         throw ex
153     }
154   }

156   private def pickleNewType(tpe: Type, richTypes: Boolean)(implicit ctx: Context): Unit = tpe match {
157     case AppliedType(tycon, args) =>
158       writeByte(APPLIEDtype)
159       withLength { pickleType(tycon); args.foreach(pickleType(_)) }
160     case ConstantType(value) =>
161       pickleConstant(value)
162     case tpe: NamedType =>
163       val sym = tpe.symbol
164       def pickleExternalRef(sym: Symbol) = {
165         def pickleCore() = {
166           pickleNameAndSig(sym.name, tpe.signature)
167           pickleType(tpe.prefix)
168         }
169         val isShadowedRef =
170           sym.isClass && tpe.prefix.member(sym.name).symbol != sym
171         if (sym.is(Flags.Private) || isShadowedRef) {
172           writeByte(if (tpe.isType) TYPEREFin else TERMREFin)
173           withLength {
174             pickleCore()
175             pickleType(sym.owner.typeRef)
176           }
177         }
178         else {
179           writeByte(if (tpe.isType) TYPEREF else TERMREF)
180           pickleCore()
181         }
182       }
183       if (sym.is(Flags.Package)) {
184         writeByte(if (tpe.isType) TYPEREFpkg else TERMREFpkg)
185         pickleName(sym.fullName)
186       }
187       else if (tpe.prefix == NoPrefix) {
188         writeByte(if (tpe.isType) TYPEREFdirect else TERMREFdirect)
189         pickleSymRef(sym)
190       }
191       else if (isLocallyDefined(sym)) {
192         writeByte(if (tpe.isType) TYPEREFsymbol else TERMREFsymbol)
193         pickleSymRef(sym); pickleType(tpe.prefix)
194       }
195       else tpe.designator match {
196         case name: Name =>
197           writeByte(if (tpe.isType) TYPEREF else TERMREF)
198           pickleName(name); pickleType(tpe.prefix)
199         case sym: Symbol =>
200           pickleExternalRef(sym)
201       }
202     case tpe: ThisType =>
203       if (tpe.cls.is(Flags.Package) && !tpe.cls.isEffectiveRoot) {
204         writeByte(TERMREFpkg)
205         pickleName(tpe.cls.fullName)
206       }
207       else {
208         writeByte(THIS)
209         pickleType(tpe.tref)
210       }
211     case tpe: SuperType =>
212       writeByte(SUPERtype)
213       withLength { pickleType(tpe.thistpe); pickleType(tpe.supertpe) }
214     case tpe: RecThis =>
215       writeByte(RECthis)
216       val binderAddr = pickledTypes.get(tpe.binder)
217       assert(binderAddr != null, tpe.binder)
218       writeRef(binderAddr.asInstanceOf[Addr])
219     case tpe: SkolemType =>
220       pickleType(tpe.info)
221     case tpe: RefinedType =>
222       writeByte(REFINEDtype)
223       withLength {
224         pickleName(tpe.refinedName)
225         pickleType(tpe.parent)
226         pickleType(tpe.refinedInfo, richTypes = true)
227       }
228     case tpe: RecType =>
229       writeByte(RECtype)
230       pickleType(tpe.parent)
231     case tpe: TypeAlias =>
232       writeByte(TYPEALIAS)
233       pickleType(tpe.alias, richTypes)
234     case tpe: TypeBounds =>
235       writeByte(TYPEBOUNDS)
236       withLength { pickleType(tpe.lo, richTypes); pickleType(tpe.hi, richTypes) }
237     case tpe: AnnotatedType =>
238       writeByte(ANNOTATEDtype)
239       withLength { pickleType(tpe.tpe, richTypes); pickleTree(tpe.annot.tree) }
240     case tpe: AndOrType =>
241       writeByte(if (tpe.isAnd) ANDtype else ORtype)
242       withLength { pickleType(tpe.tp1, richTypes); pickleType(tpe.tp2, richTypes) }
243     case tpe: ExprType =>
244       writeByte(BYNAMEtype)
245       pickleType(tpe.underlying)
246     case tpe: HKTypeLambda =>
247       pickleMethodic(TYPELAMBDAtype, tpe)
248     case tpe: PolyType if richTypes =>
249       pickleMethodic(POLYtype, tpe)
250     case tpe: MethodType if richTypes =>
251       pickleMethodic(METHODtype, tpe)
252     case tpe: ParamRef =>
253       assert(pickleParamRef(tpe), s"orphan parameter reference: $tpe")
254     case tpe: LazyRef =>
255       pickleType(tpe.ref)
256   }

319   def pickleTree(tree: Tree)(implicit ctx: Context): Unit = {
320     val addr = registerTreeAddr(tree)
321     if (addr != currentAddr) {
322       writeByte(SHAREDterm)
323       writeRef(addr)
324     }
325     else
326       try tree match {
327         case Ident(name) =>
328           tree.tpe match {
329             case tp: TermRef if name != nme.WILDCARD =>
330               // wildcards are pattern bound, need to be preserved as ids.
331               pickleType(tp)
332             case tp =>
333               writeByte(if (tree.isType) IDENTtpt else IDENT)
334               pickleName(name)
335               pickleType(tp)
336           }
337         case This(qual) =>
338           if (qual.isEmpty) pickleType(tree.tpe)
339           else {
340             writeByte(QUALTHIS)
341             val ThisType(tref) = tree.tpe
342             pickleTree(qual.withType(tref))
343           }
344         case Select(qual, name) =>
345           name match {
346             case OuterSelectName(_, levels) =>
347               writeByte(SELECTouter)
348               withLength {
349                 writeNat(levels)
350                 pickleTree(qual)
351                 val SkolemType(tp) = tree.tpe
352                 pickleType(tp)
353               }
354             case _ =>
355               writeByte(if (name.isTypeName) SELECTtpt else SELECT)
356               val sig = tree.tpe.signature
357               pickleNameAndSig(name, sig)
358               pickleTree(qual)
359           }
360         case Apply(fun, args) =>
361           writeByte(APPLY)
362           withLength {
363             pickleTree(fun)
364             args.foreach(pickleTree)
365           }
366         case TypeApply(fun, args) =>
367           writeByte(TYPEAPPLY)
368           withLength {
369             pickleTree(fun)
370             args.foreach(pickleTpt)
371           }
372         case Literal(const1) =>
373           pickleConstant {
374             tree.tpe match {
375               case ConstantType(const2) => const2
376               case _ => const1
377             }
378           }
379         case Super(qual, mix) =>
380           writeByte(SUPER)
381           withLength {
382             pickleTree(qual);
383             if (!mix.isEmpty) {
384               val SuperType(_, mixinType: TypeRef) = tree.tpe
385               pickleTree(mix.withType(mixinType))
386             }
387           }
388         case New(tpt) =>
389           writeByte(NEW)
390           pickleTpt(tpt)
391         case Typed(expr, tpt) =>
392           writeByte(TYPED)
393           withLength { pickleTree(expr); pickleTpt(tpt) }
394         case NamedArg(name, arg) =>
395           writeByte(NAMEDARG)
396           pickleName(name)
397           pickleTree(arg)
398         case Assign(lhs, rhs) =>
399           writeByte(ASSIGN)
400           withLength { pickleTree(lhs); pickleTree(rhs) }
401         case Block(stats, expr) =>
402           writeByte(BLOCK)
403           stats.foreach(preRegister)
404           withLength { pickleTree(expr); stats.foreach(pickleTree) }
405         case If(cond, thenp, elsep) =>
406           writeByte(IF)
407           withLength { pickleTree(cond); pickleTree(thenp); pickleTree(elsep) }
408         case Closure(env, meth, tpt) =>
409           writeByte(LAMBDA)
410           assert(env.isEmpty)
411           withLength {
412             pickleTree(meth)
413             if (tpt.tpe.exists) pickleTpt(tpt)
414           }
415         case Match(selector, cases) =>
416           writeByte(MATCH)
417           withLength { pickleTree(selector); cases.foreach(pickleTree) }
418         case CaseDef(pat, guard, rhs) =>
419           writeByte(CASEDEF)
420           withLength { pickleTree(pat); pickleTree(rhs); pickleTreeUnlessEmpty(guard) }
421         case Return(expr, from) =>
422           writeByte(RETURN)
423           withLength { pickleSymRef(from.symbol); pickleTreeUnlessEmpty(expr) }
424         case Try(block, cases, finalizer) =>
425           writeByte(TRY)
426           withLength { pickleTree(block); cases.foreach(pickleTree); pickleTreeUnlessEmpty(finalizer) }
427         case SeqLiteral(elems, elemtpt) =>
428           writeByte(REPEATED)
429           withLength { pickleTree(elemtpt); elems.foreach(pickleTree) }
430         case Inlined(call, bindings, expansion) =>
431           writeByte(INLINED)
432           bindings.foreach(preRegister)
433           withLength { pickleTree(call); pickleTree(expansion); bindings.foreach(pickleTree) }
434         case Bind(name, body) =>
435           registerDef(tree.symbol)
436           writeByte(BIND)
437           withLength {
438             pickleName(name); pickleType(tree.symbol.info); pickleTree(body)
439           }
440         case Alternative(alts) =>
441           writeByte(ALTERNATIVE)
442           withLength { alts.foreach(pickleTree) }
443         case UnApply(fun, implicits, patterns) =>
444           writeByte(UNAPPLY)
445           withLength {
446             pickleTree(fun)
447             for (implicitArg <- implicits) {
448               writeByte(IMPLICITarg)
449               pickleTree(implicitArg)
450             }
451             pickleType(tree.tpe)
452             patterns.foreach(pickleTree)
453           }
454         case tree: ValDef =>
455           pickleDef(VALDEF, tree.symbol, tree.tpt, tree.rhs)
456         case tree: DefDef =>
457           def pickleAllParams = {
458             pickleParams(tree.tparams)
459             for (vparams <- tree.vparamss) {
460               writeByte(PARAMS)
461               withLength { pickleParams(vparams) }
462             }
463           }
464           pickleDef(DEFDEF, tree.symbol, tree.tpt, tree.rhs, pickleAllParams)
465         case tree: TypeDef =>
466           pickleDef(TYPEDEF, tree.symbol, tree.rhs)
467         case tree: Template =>
468           registerDef(tree.symbol)
469           writeByte(TEMPLATE)
470           val (params, rest) = tree.body partition {
471             case stat: TypeDef => stat.symbol is Flags.Param
472             case stat: ValOrDefDef =>
473               stat.symbol.is(Flags.ParamAccessor) && !stat.symbol.isSetter
474             case _ => false
475           }
476           withLength {
477             pickleParams(params)
478             tree.parents.foreach(pickleTree)
479             val cinfo @ ClassInfo(_, _, _, _, selfInfo) = tree.symbol.owner.info
480             if ((selfInfo ne NoType) || !tree.self.isEmpty) {
481               writeByte(SELFDEF)
482               pickleName(tree.self.name)
483 
484               if (!tree.self.tpt.isEmpty) pickleTree(tree.self.tpt)
485               else {
486                 if (!tree.self.isEmpty) registerTreeAddr(tree.self)
487                 pickleType {
488                   selfInfo match {
489                     case sym: Symbol => sym.info
490                     case tp: Type => tp
491                   }
492                 }
493               }
494             }
495             pickleStats(tree.constr :: rest)
496           }
497         case Import(expr, selectors) =>
498           writeByte(IMPORT)
499           withLength {
500             pickleTree(expr)
501             selectors foreach {
502               case Thicket((from @ Ident(_)) :: (to @ Ident(_)) :: Nil) =>
503                 pickleSelector(IMPORTED, from)
504                 pickleSelector(RENAMED, to)
505               case id @ Ident(_) =>
506                 pickleSelector(IMPORTED, id)
507             }
508           }
509         case PackageDef(pid, stats) =>
510           writeByte(PACKAGE)
511           withLength { pickleType(pid.tpe); pickleStats(stats) }
512         case tree: TypeTree =>
513           pickleType(tree.tpe)
514         case SingletonTypeTree(ref) =>
515           writeByte(SINGLETONtpt)
516           pickleTree(ref)
517         case RefinedTypeTree(parent, refinements) =>
518           if (refinements.isEmpty) pickleTree(parent)
519           else {
520             val refineCls = refinements.head.symbol.owner.asClass
521             pickledTypes.put(refineCls.typeRef, currentAddr)
522             writeByte(REFINEDtpt)
523             refinements.foreach(preRegister)
524             withLength { pickleTree(parent); refinements.foreach(pickleTree) }
525           }
526         case AppliedTypeTree(tycon, args) =>
527           writeByte(APPLIEDtpt)
528           withLength { pickleTree(tycon); args.foreach(pickleTree) }
529         case AndTypeTree(tp1, tp2) =>
530           writeByte(ANDtpt)
531           withLength { pickleTree(tp1); pickleTree(tp2) }
532         case OrTypeTree(tp1, tp2) =>
533           writeByte(ORtpt)
534           withLength { pickleTree(tp1); pickleTree(tp2) }
535         case ByNameTypeTree(tp) =>
536           writeByte(BYNAMEtpt)
537           pickleTree(tp)
538         case Annotated(tree, annot) =>
539           writeByte(ANNOTATEDtpt)
540           withLength { pickleTree(tree); pickleTree(annot.tree) }
541         case LambdaTypeTree(tparams, body) =>
542           writeByte(LAMBDAtpt)
543           withLength { pickleParams(tparams); pickleTree(body) }
544         case TypeBoundsTree(lo, hi) =>
545           writeByte(TYPEBOUNDStpt)
546           withLength {
547             pickleTree(lo);
548             if (hi ne lo) pickleTree(hi)
549           }
550         case Hole(idx, args) =>
551           writeByte(HOLE)
552           withLength {
553             writeNat(idx)
554             args.foreach(pickleTree)
555           }
556       }
557       catch {
558         case ex: AssertionError =>
559           println(i"error when pickling tree $tree")
560           throw ex
561       }
562   }

570   def pickleModifiers(sym: Symbol)(implicit ctx: Context): Unit = {
571     import Flags._
572     val flags = sym.flags
573     val privateWithin = sym.privateWithin
574     if (privateWithin.exists) {
575       writeByte(if (flags is Protected) PROTECTEDqualified else PRIVATEqualified)
576       pickleType(privateWithin.typeRef)
577     }
578     if (flags is Private) writeByte(PRIVATE)
579     if (flags is Protected) if (!privateWithin.exists) writeByte(PROTECTED)
580     if ((flags is Final) && !(sym is Module)) writeByte(FINAL)
581     if (flags is Case) writeByte(CASE)
582     if (flags is Override) writeByte(OVERRIDE)
583     if (flags is Inline) writeByte(INLINE)
584     if (flags is Macro) writeByte(MACRO)
585     if (flags is JavaStatic) writeByte(STATIC)
586     if (flags is Module) writeByte(OBJECT)
587     if (flags is Local) writeByte(LOCAL)
588     if (flags is Synthetic) writeByte(SYNTHETIC)
589     if (flags is Artifact) writeByte(ARTIFACT)
590     if (flags is Scala2x) writeByte(SCALA2X)
591     if (sym.isTerm) {
592       if (flags is Implicit) writeByte(IMPLICIT)
593       if ((flags is Lazy) && !(sym is Module)) writeByte(LAZY)
594       if (flags is AbsOverride) { writeByte(ABSTRACT); writeByte(OVERRIDE) }
595       if (flags is Mutable) writeByte(MUTABLE)
596       if (flags is Accessor) writeByte(FIELDaccessor)
597       if (flags is CaseAccessor) writeByte(CASEaccessor)
598       if (flags is DefaultParameterized) writeByte(DEFAULTparameterized)
599       if (flags is Stable) writeByte(STABLE)
600     } else {
601       if (flags is Sealed) writeByte(SEALED)
602       if (flags is Abstract) writeByte(ABSTRACT)
603       if (flags is Trait) writeByte(TRAIT)
604       if (flags is Covariant) writeByte(COVARIANT)
605       if (flags is Contravariant) writeByte(CONTRAVARIANT)
606     }
607     sym.annotations.foreach(pickleAnnotation(sym, _))
608   }

610   private def isUnpicklable(owner: Symbol, ann: Annotation)(implicit ctx: Context) = ann match {
611     case Annotation.Child(sym) => sym.isInaccessibleChildOf(owner)
612       // If child annotation refers to a local class or enum value under
613       // a different toplevel class, it is impossible to pickle a reference to it.
614       // Such annotations will be reconstituted when unpickling the child class.
615       // See tests/pickling/i3149.scala
616     case _ => ann.symbol == defn.BodyAnnot
617       // inline bodies are reconstituted automatically when unpickling
618   }
