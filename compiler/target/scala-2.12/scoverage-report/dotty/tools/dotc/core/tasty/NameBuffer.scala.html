20   def nameIndex(name: Name): NameRef = {

34           case DerivedName(original, _) =>

36           case _ =>
37         }
38         val ref = NameRef(nameRefs.size)

41       }
42   }

44   private def withLength(op: => Unit, lengthWidth: Int = 1): Unit = {
45     val lengthAddr = currentAddr
46     for (i <- 0 until lengthWidth) writeByte(0)

48     val length = currentAddr.index - lengthAddr.index - lengthWidth

50   }

52   def writeNameRef(ref: NameRef): Unit = writeNat(ref.index)
53   def writeNameRef(name: Name): Unit = writeNameRef(nameRefs(name.toTermName))

55   def pickleNameContents(name: Name): Unit = {
56     val tag = name.toTermName.info.kind.tag
57     writeByte(tag)
58     name.toTermName match {
59       case name: SimpleName =>
60         val bytes =
61           if (name.length == 0) new Array[Byte](0)
62           else Codec.toUTF8(chrs, name.start, name.length)
63         writeNat(bytes.length)
64         writeBytes(bytes, bytes.length)
65       case AnyQualifiedName(prefix, name) =>
66         withLength { writeNameRef(prefix); writeNameRef(name) }
67       case AnyUniqueName(original, separator, num) =>
68         withLength {
69           writeNameRef(separator.toTermName)
70           writeNat(num)
71           if (!original.isEmpty) writeNameRef(original)
72         }
73       case AnyNumberedName(original, num) =>
74         withLength { writeNameRef(original); writeNat(num) }
75       case SignedName(original, Signature(params, result)) =>
76         withLength(
77           { writeNameRef(original); writeNameRef(result); params.foreach(writeNameRef) },
78           if ((params.length + 2) * maxIndexWidth <= maxNumInByte) 1 else 2)
79       case DerivedName(original, _) =>
80         withLength { writeNameRef(original) }
81     }
82   }
83 
84   override def assemble(): Unit = {
85     var i = 0
86     for ((name, ref) <- nameRefs) {
87       assert(ref.index == i)
88       i += 1
89       pickleNameContents(name)
90     }
91   }
92 }
