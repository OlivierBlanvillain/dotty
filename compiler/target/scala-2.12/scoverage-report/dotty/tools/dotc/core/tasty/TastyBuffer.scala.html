38 /** A byte array buffer that can be filled with bytes or natural numbers in TASTY format,
39  *  and that supports reading and patching addresses represented as natural numbers.
40  */
41 class TastyBuffer(initialSize: Int) {
42 
43   /** The current byte array, will be expanded as needed */
44   var bytes = new Array[Byte](initialSize)
45 
46   /** The number of bytes written */
47   var length = 0
48 
49   // -- Output routines --------------------------------------------
50 
51   /** Write a byte of data. */
52   def writeByte(b: Int): Unit = {
53     if (length >= bytes.length)
54       bytes = dble(bytes)
55     bytes(length) = b.toByte
56     length += 1
57   }
58 
59   /** Write the first `n` bytes of `data`. */
60   def writeBytes(data: Array[Byte], n: Int): Unit = {
61     while (bytes.length < length + n) bytes = dble(bytes)
62     Array.copy(data, 0, bytes, length, n)
63     length += n
64   }
65 
66   /** Write a natural number in big endian format, base 128.
67    *  All but the last digits have bit 0x80 set.
68    */
69   def writeNat(x: Int): Unit =
70     writeLongNat(x.toLong & 0x00000000FFFFFFFFL)
71 
72   /** Write a natural number in 2's complement big endian format, base 128.
73    *  All but the last digits have bit 0x80 set.
74    */
75   def writeInt(x: Int): Unit =
76     writeLongInt(x)
77 
78   /**
79    * Like writeNat, but for longs. Note that the
80    * binary representation of LongNat is identical to Nat
81    * if the long value is in the range Int.MIN_VALUE to
82    * Int.MAX_VALUE.
83    */
84   def writeLongNat(x: Long): Unit = {
85     def writePrefix(x: Long): Unit = {
86       val y = x >>> 7
87       if (y != 0L) writePrefix(y)
88       writeByte((x & 0x7f).toInt)
89     }
90     val y = x >>> 7
91     if (y != 0L) writePrefix(y)
92     writeByte(((x & 0x7f) | 0x80).toInt)
93   }
94 
95   /** Like writeInt, but for longs */
96   def writeLongInt(x: Long): Unit = {
97     def writePrefix(x: Long): Unit = {
98       val y = x >> 7
99       if (y != 0L - ((x >> 6) & 1)) writePrefix(y)
100       writeByte((x & 0x7f).toInt)
101     }
102     val y = x >> 7
103     if (y != 0L - ((x >> 6) & 1)) writePrefix(y)
104     writeByte(((x & 0x7f) | 0x80).toInt)
105   }
106 
107   /** Write an uncompressed Long stored in 8 bytes in big endian format */
108   def writeUncompressedLong(x: Long): Unit = {
109     var y = x
110     val bytes = new Array[Byte](8)
111     for (i <- 7 to 0 by -1) {
112       bytes(i) = (y & 0xff).toByte
113       y = y >>> 8
114     }
115     writeBytes(bytes, 8)
116   }
117 
118   // -- Address handling --------------------------------------------
119 
120   /** Write natural number `x` right-adjusted in a field of `width` bytes
121    *  starting with address `at`.
122    */
123   def putNat(at: Addr, x: Int, width: Int): Unit = {
124     var y = x
125     var w = width
126     if(at.index + w >= bytes.length)
127       bytes = dble(bytes)
128     var digit = y & 0x7f | 0x80
129     while (w > 0) {
130       w -= 1
131       bytes(at.index + w) = digit.toByte
132       y >>>= 7
133       digit = y & 0x7f
134     }
135     assert(y == 0, s"number $x too large to fit in $width bytes")
136   }
137 
138   /** The byte at given address */
139   def getByte(at: Addr): Int = bytes(at.index)
140 
141   /** The natural number at address `at` */
142   def getNat(at: Addr): Int = getLongNat(at).toInt
143 
144   /** The long natural number at address `at` */
145   def getLongNat(at: Addr): Long = {
146     var b = 0L
147     var x = 0L
148     var idx = at.index
149     do {
150       b = bytes(idx)
151       x = (x << 7) | (b & 0x7f)
152       idx += 1
153     } while ((b & 0x80) == 0)
154     x
155   }
156 
157   /** The address (represented as a natural number) at address `at` */
158   def getAddr(at: Addr) = Addr(getNat(at))
159 
160   /** The smallest address equal to or following `at` which points to a non-zero byte */
161   final def skipZeroes(at: Addr): Addr =
162     if (getByte(at) != 0) at else skipZeroes(at + 1)
163 
164   /** The address after the natural number found at address `at`. */
165   final def skipNat(at: Addr): Addr = {
166     val next = at + 1
167     if ((getByte(at) & 0x80) != 0) next else skipNat(next)
168   }
169 
170   /** The address referring to the end of data written so far */
171   def currentAddr: Addr = Addr(length)
172 
173   /** Reserve `AddrWidth` bytes to write an address into */
174   def reserveAddr(): Addr = {
175     val result = currentAddr
176     length += AddrWidth
177     result
178   }
179 
180   /** Fill reserved space at address `at` with address `target` */
181   def fillAddr(at: Addr, target: Addr) =
182     putNat(at, target.index, AddrWidth)
183 
184   /** Write address without leading zeroes */
185   def writeAddr(addr: Addr): Unit = writeNat(addr.index)
186 
187   // -- Finalization --------------------------------------------
188 
189   /** Hook to be overridden in subclasses.
190    *  Perform all actions necessary to assemble the final byte array.
191    *  After `assemble` no more output actions to this buffer are permitted.
192    */
193   def assemble(): Unit = ()
