346     protected def expandVariables(initialStr: String, sym: Symbol, site: Symbol)(implicit ctx: Context): String = {
347       val expandLimit = 10
348 
349       def expandInternal(str: String, depth: Int): String = {
350         if (depth >= expandLimit)
351           throw new ExpansionLimitExceeded(str)
352 
353         val out         = new StringBuilder
354         var copied, idx = 0
355         // excluding variables written as \$foo so we can use them when
356         // necessary to document things like Symbol#decode
357         def isEscaped = idx > 0 && str.charAt(idx - 1) == '\\'
358         while (idx < str.length) {
359           if ((str charAt idx) != '$' || isEscaped)
360             idx += 1
361           else {
362             val vstart = idx
363             idx = skipVariable(str, idx + 1)
364             def replaceWith(repl: String) = {
365               out append str.substring(copied, vstart)
366               out append repl
367               copied = idx
368             }
369             variableName(str.substring(vstart + 1, idx)) match {
370               case "super"    =>
371                 superComment(sym) foreach { sc =>
372                   val superSections = tagIndex(sc)
373                   replaceWith(sc.substring(3, startTag(sc, superSections)))
374                   for (sec @ (start, end) <- superSections)
375                     if (!isMovable(sc, sec)) out append sc.substring(start, end)
376                 }
377               case "" => idx += 1
378               case vname  =>
379                 lookupVariable(vname, site) match {
380                   case Some(replacement) => replaceWith(replacement)
381                   case None              =>
382                     dottydoc.println(s"Variable $vname undefined in comment for $sym in $site")
383                 }
384               }
385           }
386         }
387         if (out.length == 0) str
388         else {
389           out append str.substring(copied)
390           expandInternal(out.toString, depth + 1)
391         }
392       }
393 
394       // We suppressed expanding \$ throughout the recursion, and now we
395       // need to replace \$ with $ so it looks as intended.
396       expandInternal(initialStr, 0).replaceAllLiterally("""\$""", "$")
397     }
