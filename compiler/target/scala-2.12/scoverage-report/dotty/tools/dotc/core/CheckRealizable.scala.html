57 /** Compute realizability status */
58 class CheckRealizable(implicit ctx: Context) {
59   import CheckRealizable._
60 
61   /** A set of all fields that have already been checked. Used
62    *  to avoid infinite recursions when analyzing recursive types.
63    */
64   private val checkedFields: mutable.Set[Symbol] = mutable.LinkedHashSet[Symbol]()
65 
66   /** Is symbol's definitition a lazy val?
67    *  (note we exclude modules here, because their realizability is ensured separately)
68    */
69   private def isLateInitialized(sym: Symbol) = sym.is(Lazy, butNot = Module)
70 
71   /** The realizability status of given type `tp`*/
72   def realizability(tp: Type): Realizability = tp.dealias match {
73     case tp: TermRef =>
74       val sym = tp.symbol
75       if (sym.is(Stable)) realizability(tp.prefix)
76       else {
77         val r =
78           if (!sym.isStable) NotStable
79           else if (!isLateInitialized(sym)) realizability(tp.prefix)
80           else if (!sym.isEffectivelyFinal) new NotFinal(sym)
81           else realizability(tp.info).mapError(r => new ProblemInUnderlying(tp.info, r))
82         if (r == Realizable) sym.setFlag(Stable)
83         r
84       }
85     case _: SingletonType | NoPrefix =>
86       Realizable
87     case tp =>
88       def isConcrete(tp: Type): Boolean = tp.dealias match {
89         case tp: TypeRef => tp.symbol.isClass
90         case tp: TypeProxy => isConcrete(tp.underlying)
91         case tp: AndOrType => isConcrete(tp.tp1) && isConcrete(tp.tp2)
92         case _ => false
93       }
94       if (!isConcrete(tp)) NotConcrete
95       else boundsRealizability(tp).andAlso(memberRealizability(tp))
96   }
97 
98   /** `Realizable` if `tp` has good bounds, a `HasProblem...` instance
99    *  pointing to a bad bounds member otherwise. "Has good bounds" means:
100    *
101    *    - all type members have good bounds
102    *    - all base types are class types, and if their arguments are wildcards
103    *      they have good bounds.
104    *    - base types do not appear in multiple instances with different arguments.
105    *      (depending on the simplification scheme for AndTypes employed, this could
106    *       also lead to base types with bad bounds).
107    */
108   private def boundsRealizability(tp: Type) = {
109     val mbrProblems =
110       for {
111         mbr <- tp.nonClassTypeMembers
112         if !(mbr.info.loBound <:< mbr.info.hiBound)
113       }
114       yield new HasProblemBounds(mbr)
115 
116     def baseTypeProblems(base: Type) = base match {
117       case AndType(base1, base2) =>
118         new HasProblemBase(base1, base2) :: Nil
119       case base =>
120         base.argInfos.collect {
121           case bounds @ TypeBounds(lo, hi) if !(lo <:< hi) =>
122             new HasProblemBaseArg(base, bounds)
123         }
124     }
125     val baseProblems =
126       tp.baseClasses.map(_.baseTypeOf(tp)).flatMap(baseTypeProblems)
127 
128     (((Realizable: Realizability)
129       /: mbrProblems)(_ andAlso _)
130       /: baseProblems)(_ andAlso _)
131   }
132 
133   /** `Realizable` if all of `tp`'s non-struct fields have realizable types,
134    *  a `HasProblemField` instance pointing to a bad field otherwise.
135    */
136   private def memberRealizability(tp: Type) = {
137     def checkField(sofar: Realizability, fld: SingleDenotation): Realizability =
138       sofar andAlso {
139         if (checkedFields.contains(fld.symbol) || fld.symbol.is(Private | Mutable | Lazy))
140           // if field is private it cannot be part of a visible path
141           // if field is mutable it cannot be part of a path
142           // if field is lazy it does not need to be initialized when the owning object is
143           // so in all cases the field does not influence realizability of the enclosing object.
144           Realizable
145         else {
146           checkedFields += fld.symbol
147           realizability(fld.info).mapError(r => new HasProblemField(fld, r))
148         }
149       }
150     if (ctx.settings.strict.value)
151       // check fields only under strict mode for now.
152       // Reason: An embedded field could well be nullable, which means it
153       // should not be part of a path and need not be checked; but we cannot recognize
154       // this situation until we have a typesystem that tracks nullability.
155       ((Realizable: Realizability) /: tp.fields)(checkField)
156     else
157       Realizable
158   }
159 }
