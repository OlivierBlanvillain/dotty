19     /** COMPACTIFY
20      *
21      *  The hashed name has the form (prefix + marker + md5 + marker + suffix), where
22      *   - prefix/suffix.length = MaxNameLength / 4
23      *   - md5.length = 32
24      *
25      *  We obtain the formula:
26      *
27      *   FileNameLength = 2*(MaxNameLength / 4) + 2.marker.length + 32 + 6
28      *
29      *  (+6 for ".class"). MaxNameLength can therefore be computed as follows:
30      */
31     def apply(s: String)(implicit ctx: Context): String = {
32       val marker = "$$$$"
33       val limit: Int = ctx.settings.XmaxClassfileName.value
34       val MaxNameLength = (limit - 6) min 2 * (limit - 6 - 2 * marker.length - 32)
35 
36       def toMD5(s: String, edge: Int): String = {
37         val prefix = s take edge
38         val suffix = s takeRight edge
39 
40         val cs = s.toArray
41         val bytes = Codec toUTF8 cs
42         md5 update bytes
43         val md5chars = (md5.digest() map (b => (b & 0xFF).toHexString)).mkString
44 
45         prefix + marker + md5chars + marker + suffix
46       }
47 
48       if (s.length <= MaxNameLength) s else toMD5(s, MaxNameLength / 4)
49     }
