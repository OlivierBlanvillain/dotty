12   def avoidZero(src: Array[Byte]): Array[Byte] = {
13     var i = 0
14     val srclen = src.length
15     var count = 0
16     while (i < srclen) {
17       if (src(i) == 0x7f) count += 1
18       i += 1
19     }
20     val dst = new Array[Byte](srclen + count)
21     i = 0
22     var j = 0
23     while (i < srclen) {
24       val in = src(i)
25       if (in == 0x7f) {
26         dst(j) = (0xc0).toByte
27         dst(j + 1) = (0x80).toByte
28         j += 2
29       } else {
30         dst(j) = (in + 1).toByte
31         j += 1
32       }
33       i += 1
34     }
35     dst
36   }

38   def regenerateZero(src: Array[Byte]): Int = {
39     var i = 0
40     val srclen = src.length
41     var j = 0
42     while (i < srclen) {
43       val in: Int = src(i) & 0xff
44       if (in == 0xc0 && (src(i + 1) & 0xff) == 0x80) {
45         src(j) = 0x7f
46         i += 2
47       } else if (in == 0) {

49         i += 1
50       } else {
51         src(j) = (in - 1).toByte
52         i += 1
53       }
54       j += 1
55     }

193   /**

206    *
207    * However, this does not always happen.

209   def decode(xs: Array[Byte]): Int = {

211     decode7to8(xs, len)
