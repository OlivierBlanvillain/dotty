10   final def subst(tp: Type, from: BindingType, to: BindingType, theMap: SubstBindingMap): Type =
11     tp match {
12       case tp: BoundType =>
13         if (tp.binder eq from) tp.copyBoundType(to.asInstanceOf[tp.BT]) else tp
14       case tp: NamedType =>
15         if (tp.currentSymbol.isStatic) tp
16         else tp.derivedSelect(subst(tp.prefix, from, to, theMap))
17       case _: ThisType | NoPrefix =>
18         tp
19       case _ =>
20         (if (theMap != null) theMap else new SubstBindingMap(from, to))
21           .mapOver(tp)
22     }

24   final def subst1(tp: Type, from: Symbol, to: Type, theMap: Subst1Map): Type = {
25     tp match {
26       case tp: NamedType =>
27         val sym = tp.symbol
28         if (sym eq from) return to
29         if (sym.isStatic && !from.isStatic) tp
30         else tp.derivedSelect(subst1(tp.prefix, from, to, theMap))
31       case _: ThisType | _: BoundType | NoPrefix =>
32         tp
33       case _ =>
34         (if (theMap != null) theMap else new Subst1Map(from, to))
35           .mapOver(tp)
36     }

39   final def subst2(tp: Type, from1: Symbol, to1: Type, from2: Symbol, to2: Type, theMap: Subst2Map): Type = {
40     tp match {
41       case tp: NamedType =>
42         val sym = tp.symbol
43         if (sym eq from1) return to1
44         if (sym eq from2) return to2
45         if (sym.isStatic && !from1.isStatic && !from2.isStatic) tp
46         else tp.derivedSelect(subst2(tp.prefix, from1, to1, from2, to2, theMap))
47       case _: ThisType | _: BoundType | NoPrefix =>
48         tp
49       case _ =>
50         (if (theMap != null) theMap else new Subst2Map(from1, to1, from2, to2))
51           .mapOver(tp)

55   final def subst(tp: Type, from: List[Symbol], to: List[Type], theMap: SubstMap): Type = {
56     tp match {
57       case tp: NamedType =>
58         val sym = tp.symbol
59         var fs = from
60         var ts = to
61         while (fs.nonEmpty) {
62           if (fs.head eq sym) return ts.head
63           fs = fs.tail
64           ts = ts.tail
65         }
66         if (sym.isStatic && !existsStatic(from)) tp
67         else tp.derivedSelect(subst(tp.prefix, from, to, theMap))
68       case _: ThisType | _: BoundType | NoPrefix =>
69         tp
70       case _ =>
71         (if (theMap != null) theMap else new SubstMap(from, to))
72           .mapOver(tp)
73     }
74   }

106     tp match {
