3 /** A collection of mode bits that are part of a context */
4 case class Mode(val bits: Int) extends AnyVal {
5   import Mode._
6   def | (that: Mode) = Mode(bits | that.bits)
7   def & (that: Mode) = Mode(bits & that.bits)
8   def &~ (that: Mode) = Mode(bits & ~that.bits)
9   def is (that: Mode) = (bits & that.bits) == that.bits

11   def isExpr = (this & PatternOrTypeBits) == None
