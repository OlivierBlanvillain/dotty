28   /** Replace raw types with wildcard applications */
29   def cook(implicit ctx: Context) = new TypeMap {
30     def apply(tp: Type): Type = tp match {
31       case tp: TypeRef if tp.symbol.typeParams.nonEmpty =>
32         AppliedType(tp, tp.symbol.typeParams.map(Function.const(TypeBounds.empty)))
33       case tp @ AppliedType(tycon, args) =>
34         // disregard tycon itself, but map over it to visit the prefix
35         tp.derivedAppliedType(mapOver(tycon), args.mapConserve(this))
36       case tp @ TempPolyType(_, tpe) =>
37         val tpe1 = this(tpe)
38         if (tpe1 eq tpe) tp else tp.copy(tpe = tpe1)
39       case tp @ TempClassInfoType(parents, _, _) =>
40         val parents1 = parents.mapConserve(this)
41         if (parents eq parents1) tp else tp.copy(parentTypes = parents1)
42       case _ =>
43         mapOver(tp)
44     }
45   }

307     // So local methods need their own ctx parameters.
308     def sig2type(tparams: immutable.Map[Name, Symbol], skiptvs: Boolean)(implicit ctx: Context): Type = {
309       val tag = sig(index); index += 1
310       (tag: @switch) match {
311         case BYTE_TAG   => defn.ByteType
312         case CHAR_TAG   => defn.CharType
313         case DOUBLE_TAG => defn.DoubleType
314         case FLOAT_TAG  => defn.FloatType
315         case INT_TAG    => defn.IntType
316         case LONG_TAG   => defn.LongType
317         case SHORT_TAG  => defn.ShortType
318         case VOID_TAG   => defn.UnitType
319         case BOOL_TAG   => defn.BooleanType
320         case 'L' =>
321           def processInner(tp: Type): Type = tp match {
322             case tp: TypeRef if !(tp.symbol.owner is Flags.ModuleClass) =>
323               TypeRef(processInner(tp.prefix.widen), tp.symbol.asType)
324             case _ =>
325               tp
326           }
327           def processClassType(tp: Type): Type = tp match {
328             case tp: TypeRef =>
329               if (sig(index) == '<') {
330                 accept('<')
331                 val argsBuf = if (skiptvs) null else new ListBuffer[Type]
332                 while (sig(index) != '>') {
333                   val arg = sig(index) match {
334                     case variance @ ('+' | '-' | '*') =>
335                       index += 1
336                       variance match {
337                         case '+' => objToAny(TypeBounds.upper(sig2type(tparams, skiptvs)))
338                         case '-' =>
339                           val tp = sig2type(tparams, skiptvs)
340                           // sig2type seems to return AnyClass regardless of the situation:
341                           // we don't want Any as a LOWER bound.
342                           if (tp.isDirectRef(defn.AnyClass)) TypeBounds.empty
343                           else TypeBounds.lower(tp)
344                         case '*' => TypeBounds.empty
345                       }
346                     case _ => sig2type(tparams, skiptvs)
347                   }
348                   if (argsBuf != null) argsBuf += arg
349                 }
350                 accept('>')
351                 if (skiptvs) tp else tp.appliedTo(argsBuf.toList)
352               } else tp
353             case tp =>
354               assert(sig(index) != '<', tp)
355               tp
356           }
357 
358           val classSym = classNameToSymbol(subName(c => c == ';' || c == '<'))
359           var tpe = processClassType(processInner(classSym.typeRef))
360           while (sig(index) == '.') {
361             accept('.')
362             val name = subName(c => c == ';' || c == '<' || c == '.').toTypeName
363             val clazz = tpe.member(name).symbol
364             tpe = processClassType(processInner(clazz.typeRef))
365           }
366           accept(';')
367           tpe
368         case ARRAY_TAG =>
369           while ('0' <= sig(index) && sig(index) <= '9') index += 1
370           var elemtp = sig2type(tparams, skiptvs)
371           // make unbounded Array[T] where T is a type variable into Array[T with Object]
372           // (this is necessary because such arrays have a representation which is incompatible
373           // with arrays of primitive types.
374           // NOTE that the comparison to Object only works for abstract types bounded by classes that are strict subclasses of Object
375           // if the bound is exactly Object, it will have been converted to Any, and the comparison will fail
376           // see also RestrictJavaArraysMap (when compiling java sources directly)
377           if (elemtp.typeSymbol.isAbstractOrParamType && !(elemtp.derivesFrom(defn.ObjectClass))) {
378             elemtp = AndType(elemtp, defn.ObjectType)
379           }
380           defn.ArrayOf(elemtp)
381         case '(' =>
382           // we need a method symbol. given in line 486 by calling getType(methodSym, ..)
383           val paramtypes = new ListBuffer[Type]()
384           var paramnames = new ListBuffer[TermName]()
385           while (sig(index) != ')') {
386             paramnames += nme.syntheticParamName(paramtypes.length)
387             paramtypes += objToAny(sig2type(tparams, skiptvs))
388           }
389           index += 1
390           val restype = sig2type(tparams, skiptvs)
391           JavaMethodType(paramnames.toList, paramtypes.toList, restype)
392         case 'T' =>
393           val n = subName(';'.==).toTypeName
394           index += 1
395           //assert(tparams contains n, s"classTparams = $classTParams, tparams = $tparams, key = $n")
396           if (skiptvs) defn.AnyType else tparams(n).typeRef
397       }
398     } // sig2type(tparams, skiptvs)

460   def parseAnnotArg(skip: Boolean = false)(implicit ctx: Context): Option[Tree] = {
461     val tag = in.nextByte.toChar
462     val index = in.nextChar
463     tag match {
464       case STRING_TAG =>
465         if (skip) None else Some(Literal(Constant(pool.getName(index).toString)))
466       case BOOL_TAG | BYTE_TAG | CHAR_TAG | SHORT_TAG =>
467         if (skip) None else Some(Literal(pool.getConstant(index, tag)))
468       case INT_TAG | LONG_TAG | FLOAT_TAG | DOUBLE_TAG =>
469         if (skip) None else Some(Literal(pool.getConstant(index)))
470       case CLASS_TAG =>
471         if (skip) None else Some(Literal(Constant(pool.getType(index))))
472       case ENUM_TAG =>
473         val t = pool.getType(index)
474         val n = pool.getName(in.nextChar)
475         val module = t.typeSymbol.companionModule
476         val s = module.info.decls.lookup(n)
477         if (skip) {
478           None
479         } else if (s != NoSymbol) {
480           Some(Literal(Constant(s)))
481         } else {
482           ctx.warning(s"""While parsing annotations in ${in.file}, could not find $n in enum $module.\nThis is likely due to an implementation restriction: an annotation argument cannot refer to a member of the annotated class (SI-7014).""")
483           None
484         }
485       case ARRAY_TAG =>
486         val arr = new ArrayBuffer[Tree]()
487         var hasError = false
488         for (i <- 0 until index)
489           parseAnnotArg(skip) match {
490             case Some(c) => arr += c
491             case None => hasError = true
492           }
493         if (hasError) None
494         else if (skip) None
495         else {
496           val elems = arr.toList
497           val elemType =
498             if (elems.isEmpty) defn.ObjectType
499             else ctx.typeComparer.lub(elems.tpes).widen
500           Some(JavaSeqLiteral(elems, TypeTree(elemType)))
501         }
502       case ANNOTATION_TAG =>
503         parseAnnotation(index, skip) map (_.tree)
504     }
505   }

510   def parseAnnotation(attrNameIndex: Char, skip: Boolean = false)(implicit ctx: Context): Option[Annotation] = try {
511     val attrType = pool.getType(attrNameIndex)
512     val nargs = in.nextChar
513     val argbuf = new ListBuffer[Tree]
514     var hasError = false
515     for (i <- 0 until nargs) {
516       val name = pool.getName(in.nextChar)
517       parseAnnotArg(skip) match {
518         case Some(arg) => argbuf += NamedArg(name, arg)
519         case None => hasError = !skip
520       }
521     }
522     if (hasError || skip) None
523     else Some(Annotation.deferredResolve(attrType, argbuf.toList))
524   } catch {
525     case f: FatalError => throw f // don't eat fatal errors, they mean a class was not found
526     case NonFatal(ex) =>
527       // We want to be robust when annotations are unavailable, so the very least
528       // we can do is warn the user about the exception
529       // There was a reference to ticket 1135, but that is outdated: a reference to a class not on
530       // the classpath would *not* end up here. A class not found is signaled
531       // with a `FatalError` exception, handled above. Here you'd end up after a NPE (for example),
532       // and that should never be swallowed silently.
533       ctx.warning("Caught: " + ex + " while parsing annotations in " + in.file)
534       if (ctx.debug) ex.printStackTrace()
535 
536       None // ignore malformed annotations
537   }
