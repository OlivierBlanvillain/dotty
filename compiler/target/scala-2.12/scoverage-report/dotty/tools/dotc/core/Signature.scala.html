9 /** The signature of a denotation.
10  *  Overloaded denotations with the same name are distinguished by
11  *  their signatures. A signature of a method (of type PolyType,MethodType, or ExprType) is
12  *  composed of a list of signature names, one for each parameter type, plus a signature for
13  *  the result type. Methods are uncurried before taking their signatures.
14  *  The signature name of a type is the fully qualified name of the type symbol of the type's erasure.
15  *
16  *  For instance a definition
17  *
18  *      def f(x: Int)(y: List[String]): String
19  *
20  *  would have signature
21  *
22  *      Signature(
23  *        List("scala.Int".toTypeName, "scala.collection.immutable.List".toTypeName),
24  *        "scala.String".toTypeName)
25  *
26  *  The signatures of non-method types are always `NotAMethod`.
27  *
28  *  There are three kinds of "missing" parts of signatures:
29  *
30  *   - tpnme.EMPTY          Result type marker for NotAMethod and OverloadedSignature
31  *   - tpnme.WILDCARD       Arises from a Wildcard or error type
32  *   - tpnme.Uninstantiated Arises from an uninstantiated type variable
33  */
34 case class Signature(paramsSig: List[TypeName], resSig: TypeName) {
35   import Signature._
36 
37   /** Two names are consistent if they are the same or one of them is tpnme.Uninstantiated */
38   private def consistent(name1: TypeName, name2: TypeName) =
39     name1 == name2 || name1 == tpnme.Uninstantiated || name2 == tpnme.Uninstantiated
40 
41   /** Does this signature coincide with that signature on their parameter parts?
42    *  This is the case if all parameter names are _consistent_, i.e. they are either
43    *  equal or on of them is tpnme.Uninstantiated.
44    */
45   final def consistentParams(that: Signature): Boolean = {
46     @tailrec def loop(names1: List[TypeName], names2: List[TypeName]): Boolean =
47       if (names1.isEmpty) names2.isEmpty
48       else names2.nonEmpty && consistent(names1.head, names2.head) && loop(names1.tail, names2.tail)
49     loop(this.paramsSig, that.paramsSig)
50   }
51 
52   /** `that` signature, but keeping all corresponding parts of `this` signature. */
53   final def updateWith(that: Signature): Signature = {
54     def update(name1: TypeName, name2: TypeName): TypeName =
55       if (consistent(name1, name2)) name1 else name2
56     if (this == that) this
57     else if (!this.paramsSig.hasSameLengthAs(that.paramsSig)) that
58     else {
59       val mapped = Signature(
60           this.paramsSig.zipWithConserve(that.paramsSig)(update),
61           update(this.resSig, that.resSig))
62       if (mapped == this) this else mapped
63     }
64   }
65 
66   /** The degree to which this signature matches `that`.
67    *  If parameter names are consistent and result types names match (i.e. they are the same
68    *  or one is a wildcard), the result is `FullMatch`.
69    *  If only the parameter names are consistent, the result is `ParamMatch` before erasure and
70    *  `NoMatch` otherwise.
71    *  If the parameters are inconsistent, the result is always `NoMatch`.
72    */
73   final def matchDegree(that: Signature)(implicit ctx: Context): MatchDegree =
74     if (consistentParams(that))
75       if (resSig == that.resSig || isWildcard(resSig) || isWildcard(that.resSig)) FullMatch
76       else if (!ctx.erasedTypes) ParamMatch
77       else NoMatch
78     else NoMatch
79 
80   /** name.toString == "" or name.toString == "_" */
81   private def isWildcard(name: TypeName) = name.isEmpty || name == tpnme.WILDCARD
82 
83   /** Construct a signature by prepending the signature names of the given `params`
84    *  to the parameter part of this signature.
85    */
86   def prepend(params: List[Type], isJava: Boolean)(implicit ctx: Context) =
87     Signature(params.collect { case p if !p.isPhantom => sigName(p, isJava) } ++ paramsSig, resSig)
88 
89   /** A signature is under-defined if its paramsSig part contains at least one
90    *  `tpnme.Uninstantiated`. Under-defined signatures arise when taking a signature
91    *  of a type that still contains uninstantiated type variables. They are eliminated
92    *  by `fixSignature` in `PostTyper`.
93    */
94   def isUnderDefined(implicit ctx: Context) =
95     paramsSig.contains(tpnme.Uninstantiated) || resSig == tpnme.Uninstantiated
96 }
97 
98 object Signature {
99 
100   type MatchDegree = Int
101   val NoMatch = 0
102   val ParamMatch = 1
103   val FullMatch = 2
104 
105   /** The signature of everything that's not a method, i.e. that has
106    *  a type different from PolyType, MethodType, or ExprType.
107    */
108   val NotAMethod = Signature(List(), EmptyTypeName)
109 
110   /** The signature of an overloaded denotation.
111    */
112   val OverloadedSignature = Signature(List(tpnme.OVERLOADED), EmptyTypeName)
113 
114   /** The signature of a method with no parameters and result type `resultType`. */
115   def apply(resultType: Type, isJava: Boolean)(implicit ctx: Context): Signature = {
116     assert(!resultType.isInstanceOf[ExprType])
117     apply(Nil, sigName(resultType, isJava))
118   }
