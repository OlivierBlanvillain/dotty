574     /** Make sure the capacity of the character array is at least `n` */
575     def ensureCapacity(n: Int) =
576       if (n > chrs.length) {
577         val newchrs = new Array[Char](chrs.length * 2)
578         chrs.copyToArray(newchrs)
579         chrs = newchrs
580       }

582     /** Enter characters into chrs array. */
583     def enterChars(): Unit = {
584       ensureCapacity(nc + len)
585       var i = 0
586       while (i < len) {
587         chrs(nc + i) = cs(offset + i)
588         i += 1
589       }
590       nc += len
591     }

593     /** Rehash chain of names */
594     def rehash(name: SimpleName): Unit =
595       if (name != null) {
596         val oldNext = name.next
597         val h = hashValue(chrs, name.start, name.length) & (table.size - 1)
598         name.next = table(h)
599         table(h) = name
600         rehash(oldNext)
601       }

603     /** Make sure the hash table is large enough for the given load factor */
604     def incTableSize() = {
605       size += 1

607         val oldTable = table
608         table = new Array[SimpleName](table.size * 2)
609         for (i <- 0 until oldTable.size) rehash(oldTable(i))
610       }
611     }

613     val next = table(h)
614     var name = next
615     while (name ne null) {
616       if (name.length == len && equals(name.start, cs, offset, len))
617         return name
618       name = name.next
619     }
620     name = new SimpleName(nc, len, next)
621     enterChars()
622     table(h) = name
623     incTableSize()
624     name
