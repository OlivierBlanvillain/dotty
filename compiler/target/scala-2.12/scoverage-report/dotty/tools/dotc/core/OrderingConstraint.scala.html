40       val es = entries(c, poly)

44     /** The `current` constraint but with the entry for `param` updated to `entry`.
45      *  `current` is used linearly. If it is different from `prev` it is
46      *  known to be dead after the call. Hence it is OK to update destructively
47      *  parts of `current` which are not shared by `prev`.
48      */
49     def update(prev: OrderingConstraint, current: OrderingConstraint,
50         poly: TypeLambda, idx: Int, entry: T)(implicit ctx: Context): OrderingConstraint = {
51       var es = entries(current, poly)
52       if (es != null && (es(idx) eq entry)) current
53       else {
54         val result =
55           if (es == null) {
56             es = Array.fill(poly.paramNames.length)(initial)
57             updateEntries(current, poly, es)
58           }
59           else if (es ne entries(prev, poly))
60             current // can re-use existing entries array.
61           else {
62             es = es.clone
63             updateEntries(current, poly, es)
64           }
65         es(idx) = entry
66         result
67       }
68     }

71         param: TypeParamRef, entry: T)(implicit ctx: Context): OrderingConstraint =
72       update(prev, current, param.binder, param.paramNum, entry)

110 /** Constraint over undetermined type parameters that keeps separate maps to
111  *  reflect parameter orderings.
112  *  @param boundsMap a map from TypeLambda to arrays.
113  *               Each array contains twice the number of entries as there a type parameters
114  *               in the TypeLambda. The first half of the array contains the type bounds that constrain the
115  *               lambda's type parameters. The second half might contain type variables that
116  *               track the corresponding parameters, or is left empty (filled with nulls).
117  *               An instantiated type parameter is represented by having its instance type in
118  *               the corresponding array entry. The dual use of arrays for poly params
119  *               and typevars is to save space and hopefully gain some speed.
120  *
121  *  @param lowerMap a map from TypeLambdas to arrays. Each array entry corresponds
122  *               to a parameter P of the type lambda; it contains all constrained parameters
123  *               Q that are known to be smaller than P, i.e. Q <: P.
124  *  @param upperMap a map from TypeLambdas to arrays. Each array entry corresponds
125  *               to a parameter P of the type lambda; it contains all constrained parameters
126  *               Q that are known to be greater than P, i.e. P <: Q.
127  */

217   /** The list of parameters P such that, for a fresh type parameter Q:
218    *
219    *    Q <: tp  implies  Q <: P      and isUpper = true, or
220    *    tp <: Q  implies  P <: Q      and isUpper = false

222   def dependentParams(tp: Type, isUpper: Boolean): List[TypeParamRef] = tp match {
223     case param: TypeParamRef if contains(param) =>
224       param :: (if (isUpper) upper(param) else lower(param))
225     case tp: AndOrType =>
226       val ps1 = dependentParams(tp.tp1, isUpper)
227       val ps2 = dependentParams(tp.tp2, isUpper)
228       if (isUpper == tp.isAnd) ps1.union(ps2) else ps1.intersect(ps2)
229     case _ =>
230       Nil

233   /** The bound type `tp` without constrained parameters which are clearly
234    *  dependent. A parameter in an upper bound is clearly dependent if it appears
235    *  in a hole of a context H given by:
236    *
237    *      H = []
238    *          H & T
239    *          T & H
240    *
241    *  (the idea is that a parameter P in a H context is guaranteed to be a supertype of the
242    *   bounded parameter.)
243    *  Analogously, a parameter in a lower bound is clearly dependent if it appears
244    *  in a hole of a context H given by:
245    *
246    *      L = []
247    *          L | T
248    *          T | L
249    *
250    *  "Clearly dependent" is not synonymous with "dependent" in the sense
251    *  it is defined in `dependentParams`. Dependent parameters are handled
252    *  in `updateEntry`. The idea of stripping off clearly dependent parameters
253    *  and to handle them separately is for efficiency, so that type expressions
254    *  used as bounds become smaller.
255    *
256    *  @param isUpper   If true, `bound` is an upper bound, else a lower bound.
257    */
258   private def stripParams(tp: Type, paramBuf: mutable.ListBuffer[TypeParamRef],
259       isUpper: Boolean)(implicit ctx: Context): Type = tp match {
260     case param: TypeParamRef if contains(param) =>
261       if (!paramBuf.contains(param)) paramBuf += param
262       NoType
263     case tp: AndOrType if isUpper == tp.isAnd =>
264       val tp1 = stripParams(tp.tp1, paramBuf, isUpper)
265       val tp2 = stripParams(tp.tp2, paramBuf, isUpper)
266       if (tp1.exists)
267         if (tp2.exists) tp.derivedAndOrType(tp1, tp2)
268         else tp1
269       else tp2
270     case _ =>
271       tp
272   }

322     if (param1 == param2 || current.isLess(param1, param2)) this
323     else {

359   /** A new constraint which is derived from this constraint by removing
360    *  the type parameter `param` from the domain and replacing all top-level occurrences
361    *  of the parameter elsewhere in the constraint by type `tp`, or a conservative
362    *  approximation of it if that is needed to avoid cycles.
363    *  Occurrences nested inside a refinement or prefix are not affected.
364    *
365    *  The reason we need to substitute top-level occurrences of the parameter
366    *  is to deal with situations like the following. Say we have in the constraint
367    *
368    *      P <: Q & String
369    *      Q
370    *
371    *  and we replace Q with P. Then substitution gives
372    *
373    *      P <: P & String
374    *
375    *  this would be a cyclic constraint is therefore changed by `normalize` and
376    *  `recombine` below to
377    *
378    *      P <: String
379    *
380    *  approximating the RHS occurrence of P with Any. Without the substitution we
381    *  would not find out where we need to approximate. Occurrences of parameters
382    *  that are not top-level are not affected.

395       def replaceParam(tp: Type, atPoly: TypeLambda, atIdx: Int): Type = tp match {
396         case bounds @ TypeBounds(lo, hi) =>

398           def recombine(andor: AndOrType, op: (Type, Boolean) => Type, isUpper: Boolean): Type = {
399             val tp1 = op(andor.tp1, isUpper)
400             val tp2 = op(andor.tp2, isUpper)
401             if ((tp1 eq andor.tp1) && (tp2 eq andor.tp2)) andor
402             else if (andor.isAnd) tp1 & tp2
403             else tp1 | tp2
404           }

407             case p: TypeParamRef if p.binder == atPoly && p.paramNum == atIdx =>
