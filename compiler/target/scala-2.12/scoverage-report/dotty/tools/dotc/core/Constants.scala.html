172     def convertTo(pt: Type)(implicit ctx: Context): Constant = {
173       def classBound(pt: Type): Type = pt.dealias.stripTypeVar match {
174         case tref: TypeRef if !tref.symbol.isClass => classBound(tref.info.bounds.lo)
175         case param: TypeParamRef =>
176           ctx.typerState.constraint.entry(param) match {
177             case TypeBounds(lo, hi) =>
178               if (hi.classSymbol.isPrimitiveValueClass) hi //constrain further with high bound
179               else classBound(lo)
180             case NoType => classBound(param.binder.paramInfos(param.paramNum).lo)
181             case inst => classBound(inst)
182           }
183         case pt => pt
184       }
185       val target = classBound(pt).typeSymbol
186       if (target == tpe.typeSymbol)
187         this
188       else if ((target == defn.ByteClass) && isByteRange)
189         Constant(byteValue)
190       else if (target == defn.ShortClass && isShortRange)
191         Constant(shortValue)
192       else if (target == defn.CharClass && isCharRange)
193         Constant(charValue)
194       else if (target == defn.IntClass && isIntRange)
195         Constant(intValue)
196       else if (target == defn.LongClass && isLongRange)
197         Constant(longValue)
198       else if (target == defn.FloatClass && isFloatRange)
199         Constant(floatValue)
200       else if (target == defn.DoubleClass && isNumeric)

203         null
204     }

214     /**

216      * Consider -0d to be distinct from 0d, despite equality
