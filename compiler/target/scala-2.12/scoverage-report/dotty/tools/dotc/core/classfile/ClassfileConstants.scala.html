334   import Flags._
335   abstract class FlagTranslation {
336 
337     protected def baseFlags(jflags: Int) = EmptyFlags
338     protected def isClass: Boolean = false
339 
340     private def translateFlag(jflag: Int): FlagSet = (jflag: @switch) match {
341       case JAVA_ACC_PRIVATE    => Private
342       case JAVA_ACC_PROTECTED  => Protected
343       case JAVA_ACC_FINAL      => Final
344       case JAVA_ACC_SYNTHETIC  => Synthetic
345       case JAVA_ACC_STATIC     => JavaStatic
346       case JAVA_ACC_ABSTRACT   => if (isClass) Abstract else Deferred
347       case JAVA_ACC_INTERFACE  => PureInterfaceCreationFlags | JavaDefined
348       case _                   => EmptyFlags
349     }
350 
351     private def addFlag(base: FlagSet, jflag: Int): FlagSet =
352       if (jflag == 0) base else base | translateFlag(jflag)
353 
354     private def translateFlags(jflags: Int, baseFlags: FlagSet): FlagSet = {
355       val nflags =
356         if ((jflags & JAVA_ACC_ANNOTATION) == 0) jflags
357         else jflags & ~(JAVA_ACC_ABSTRACT | JAVA_ACC_INTERFACE) // annotations are neither abstract nor interfaces
358       var res: FlagSet = baseFlags | JavaDefined
359       res = addFlag(res, nflags & JAVA_ACC_PRIVATE)
360       res = addFlag(res, nflags & JAVA_ACC_PROTECTED)
361       res = addFlag(res, nflags & JAVA_ACC_FINAL)
362       res = addFlag(res, nflags & JAVA_ACC_SYNTHETIC)
363       res = addFlag(res, nflags & JAVA_ACC_STATIC)
364       res = addFlag(res, nflags & JAVA_ACC_ABSTRACT)
365       res = addFlag(res, nflags & JAVA_ACC_INTERFACE)
366       res
367     }
368 
369     def flags(jflags: Int): FlagSet = translateFlags(jflags, baseFlags(jflags))
370   }
371   val classTranslation = new FlagTranslation {
372     override def isClass = true
373   }
374   val fieldTranslation = new FlagTranslation {
375     override def baseFlags(jflags: Int) = if ((jflags & JAVA_ACC_FINAL) == 0) Mutable else EmptyFlags
376   }
