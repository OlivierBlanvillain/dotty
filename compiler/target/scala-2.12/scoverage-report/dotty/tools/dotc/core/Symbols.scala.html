320   /** Map given symbols, subjecting their attributes to the mappings
321    *  defined in the given TreeTypeMap `ttmap`.
322    *  Cross symbol references are brought over from originals to copies.
323    *  Do not copy any symbols if all attributes of all symbols stay the same.
324    */
325   def mapSymbols(originals: List[Symbol], ttmap: TreeTypeMap, mapAlways: Boolean = false): List[Symbol] =
326     if (originals.forall(sym =>
327         (ttmap.mapType(sym.info) eq sym.info) &&
328         !(ttmap.oldOwners contains sym.owner)) && !mapAlways)
329       originals
330     else {
331       val copies: List[Symbol] = for (original <- originals) yield
332         original match {
333           case original: ClassSymbol =>
334             newNakedClassSymbol(original.coord, original.assocFile)
335           case _ =>
336             newNakedSymbol[original.ThisName](original.coord)
337         }
338       val ttmap1 = ttmap.withSubstitution(originals, copies)
339       (originals, copies).zipped foreach { (original, copy) =>
340         val odenot = original.denot
341         val oinfo = original.info match {
342           case ClassInfo(pre, _, parents, decls, selfInfo) =>
343             assert(original.isClass)
344             ClassInfo(pre, copy.asClass, parents, decls.cloneScope, selfInfo)
345           case oinfo => oinfo
346         }
347 
348         val completer = new LazyType {
349           def complete(denot: SymDenotation)(implicit ctx: Context): Unit = {
350             denot.info = oinfo // needed as otherwise we won't be able to go from Sym -> parents & etc
351                                // Note that this is a hack, but hack commonly used in Dotty
352                                // The same thing is done by other completers all the time
353             denot.info = ttmap1.mapType(oinfo)
354             denot.annotations = odenot.annotations.mapConserve(ttmap1.apply)
355           }
356         }
357 
358         copy.denot = odenot.copySymDenotation(
359           symbol = copy,
360           owner = ttmap1.mapOwner(odenot.owner),
361           initFlags = odenot.flags &~ Touched,
362           info = completer,
363           privateWithin = ttmap1.mapOwner(odenot.privateWithin), // since this refers to outer symbols, need not include copies (from->to) in ownermap here.
364           annotations = odenot.annotations)
365       }
366 
367       copies.foreach(_.ensureCompleted()) // avoid memory leak
368       copies
369     }
