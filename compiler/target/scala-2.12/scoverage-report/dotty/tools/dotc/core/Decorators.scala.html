49   /** Implements filterConserve, zipWithConserve methods
50    *  on lists that avoid duplication of list nodes where feasible.
51    */
52   implicit class ListDecorator[T](val xs: List[T]) extends AnyVal {
53 
54     final def mapconserve[U](f: T => U): List[U] = {
55       @tailrec
56       def loop(mapped: ListBuffer[U], unchanged: List[U], pending: List[T]): List[U] =
57         if (pending.isEmpty) {
58           if (mapped eq null) unchanged
59           else mapped.prependToList(unchanged)
60         } else {
61           val head0 = pending.head
62           val head1 = f(head0)
63 
64           if (head1.asInstanceOf[AnyRef] eq head0.asInstanceOf[AnyRef])
65             loop(mapped, unchanged, pending.tail)
66           else {
67             val b = if (mapped eq null) new ListBuffer[U] else mapped
68             var xc = unchanged
69             while (xc ne pending) {
70               b += xc.head
71               xc = xc.tail
72             }
73             b += head1
74             val tail0 = pending.tail
75             loop(b, tail0.asInstanceOf[List[U]], tail0)
76           }
77         }
78       loop(null, xs.asInstanceOf[List[U]], xs)
79     }
80 
81     /** Like `xs filter p` but returns list `xs` itself  - instead of a copy -
82      *  if `p` is true for all elements and `xs` is not longer
83      *  than `MaxFilterRecursions`.
84      */
85     def filterConserve(p: T => Boolean): List[T] = {
86       def loop(xs: List[T], nrec: Int): List[T] = xs match {
87         case Nil => xs
88         case x :: xs1 =>
89           if (nrec < MaxFilterRecursions) {
90             val ys1 = loop(xs1, nrec + 1)
91             if (p(x))
92               if (ys1 eq xs1) xs else x :: ys1
93             else
94               ys1
95           } else xs filter p
96       }
97       loop(xs, 0)
98     }
99 
100     /** Like `(xs, ys).zipped.map(f)`, but returns list `xs` itself
101      *  - instead of a copy - if function `f` maps all elements of
102      *  `xs` to themselves. Also, it is required that `ys` is at least
103      *  as long as `xs`.
104      */
105     def zipWithConserve[U](ys: List[U])(f: (T, U) => T): List[T] =
106       if (xs.isEmpty || ys.isEmpty) Nil
107       else {
108         val x1 = f(xs.head, ys.head)
109         val xs1 = xs.tail.zipWithConserve(ys.tail)(f)
110         if ((x1.asInstanceOf[AnyRef] eq xs.head.asInstanceOf[AnyRef]) &&
111             (xs1 eq xs.tail)) xs
112         else x1 :: xs1
113       }
114 
115     final def hasSameLengthAs[U](ys: List[U]): Boolean = {
116       @tailrec def loop(xs: List[T], ys: List[U]): Boolean =
117         if (xs.isEmpty) ys.isEmpty
118         else ys.nonEmpty && loop(xs.tail, ys.tail)
119       loop(xs, ys)
120     }
121 
122     @tailrec final def eqElements(ys: List[AnyRef]): Boolean = xs match {
123       case x :: _ =>
124         ys match {
125           case y :: _ =>
126             x.asInstanceOf[AnyRef].eq(y) &&
127             xs.tail.eqElements(ys.tail)
128           case _ => false
129         }
130       case nil => ys.isEmpty
131     }
132 
133     /** Union on lists seen as sets */
134     def | (ys: List[T]): List[T] = xs ++ (ys filterNot (xs contains _))
135 
136     /** Intersection on lists seen as sets */
137     def & (ys: List[T]): List[T] = xs filter (ys contains _)
138   }

153   implicit class PhaseListDecorator(val names: List[String]) extends AnyVal {
154     def containsPhase(phase: Phase): Boolean = phase match {
155       case phase: MegaPhase => phase.miniPhases.exists(containsPhase)
156       case _ =>
157         names exists { name =>
158           name == "all" || {
159             val strippedName = name.stripSuffix("+")
160             val logNextPhase = name ne strippedName
161             phase.phaseName.startsWith(strippedName) ||
162               (logNextPhase && phase.prev.phaseName.startsWith(strippedName))
163           }
164         }
165     }
