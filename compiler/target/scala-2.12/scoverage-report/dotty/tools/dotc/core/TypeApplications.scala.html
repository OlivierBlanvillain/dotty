78   /** A type map that tries to reduce (part of) the result type of the type lambda `tycon`
79    *  with the given `args`(some of which are wildcard arguments represented by type bounds).
80    *  Non-wildcard arguments are substituted everywhere as usual. A wildcard argument
81    *  `>: L <: H` is substituted for a type lambda parameter `X` only under certain conditions.
82    *
83    *  1. If Mode.AllowLambdaWildcardApply is set:
84    *  The wildcard argument is substituted only if `X` appears in a toplevel application of the form
85    *
86    *        C[..., X, ...]
87    *
88    *  and there are no other occurrences of `X` in the reduced type. In that case
89    *  the refinement above is replaced by
90    *
91    *        C[..., _ >: L <: H, ...]
92    *
93    *  The `allReplaced` field indicates whether all occurrences of type lambda parameters
94    *  in the reduced type have been replaced with arguments.
95    *
96    *  2. If Mode.AllowLambdaWildcardApply is not set:
97    *  All `X` arguments are replaced by:
98    *
99    *        _ >: L <: H
100    *
101    *  Any other occurrence of `X` in `tycon` is replaced by `U`, if the
102    *  occurrence of `X` in `tycon` is covariant, or nonvariant, or by `L`,
103    *  if the occurrence is contravariant.
104    *
105    *  The idea is that the `AllowLambdaWildcardApply` mode is used to check whether
106    *  a type can be soundly reduced, and to give an error or warning if that
107    *  is not the case. By contrast, the default mode, with `AllowLambdaWildcardApply`
108    *  not set, reduces all applications even if this yields a different type, so
109    *  its postcondition is that no type parameters of `tycon` appear in the
110    *  result type. Using this mode, we can guarantee that `appliedTo` will never
111    *  produce a higher-kinded application with a type lambda as type constructor.
112    */
113   class Reducer(tycon: TypeLambda, args: List[Type])(implicit ctx: Context) extends TypeMap {
114     private[this] var available = (0 until args.length).toSet
115     var allReplaced = true
116     def hasWildcardArg(p: TypeParamRef) =
117       p.binder == tycon && args(p.paramNum).isInstanceOf[TypeBounds]
118     def canReduceWildcard(p: TypeParamRef) =
119       !ctx.mode.is(Mode.AllowLambdaWildcardApply) || available.contains(p.paramNum)
120     def atNestedLevel(op: => Type): Type = {
121       val saved = available
122       available = Set()
123       try op
124       finally available = saved
125     }
126 
127     // If this is a reference to a reducable type parameter corresponding to a
128     // wildcard argument, return the wildcard argument, otherwise apply recursively.
129     def applyArg(arg: Type): Type = arg match {
130       case p: TypeParamRef if hasWildcardArg(p) && canReduceWildcard(p) =>
131         available -= p.paramNum
132         args(p.paramNum)
133       case _ =>
134         atNestedLevel(apply(arg))
135     }
136 
137     def apply(t: Type) = t match {
138       case t @ AppliedType(tycon, args1) if tycon.typeSymbol.isClass =>
139         t.derivedAppliedType(apply(tycon), args1.mapConserve(applyArg))
140       case p: TypeParamRef if p.binder == tycon =>
141         args(p.paramNum) match {
142           case TypeBounds(lo, hi) =>
143             if (ctx.mode.is(Mode.AllowLambdaWildcardApply)) { allReplaced = false; p }
144             else if (variance < 0) lo
145             else hi
146           case arg =>
147             arg
148         }
149       case _: TypeBounds | _: AppliedType =>
150         atNestedLevel(mapOver(t))
151       case _ =>
152         mapOver(t)
153     }
154   }

274   /** If argument A and type parameter P are higher-kinded, adapt the variances
275    *  of A to those of P, ensuring that the variances of the type lambda A
276    *  agree with the variances of corresponding higher-kinded type parameters of P. Example:
277    *
278    *     class GenericCompanion[+CC[X]]
279    *     GenericCompanion[List]
280    *
281    *  with adaptHkVariances, the argument `List` will expand to
282    *
283    *     [X] => List[X]
284    *
285    *  instead of
286    *
287    *     [+X] => List[X]
288    *
289    *  even though `List` is covariant. This adaptation is necessary to ignore conflicting
290    *  variances in overriding members that have types of hk-type parameters such as
291    *  `GenericCompanion[GenTraversable]` or `GenericCompanion[ListBuffer]`.
292    *  When checking overriding, we need to validate the subtype relationship
293    *
294    *      GenericCompanion[[X] -> ListBuffer[X]] <: GenericCompanion[[+X] -> GenTraversable[X]]
295    *
296    *   Without adaptation, this would be false, and hence an overriding error would
297    *   result. But with adaptation, the rhs argument will be adapted to
298    *
299    *     [X] -> GenTraversable[X]
300    *
301    *   which makes the subtype test succeed. The crucial point here is that, since
302    *   GenericCompanion only expects a non-variant CC, the fact that GenTraversable
303    *   is covariant is irrelevant, so can be ignored.
304    */

306     val hkParams = bound.hkTypeParams
307     if (hkParams.isEmpty) self
308     else {

310         case arg @ HKTypeLambda(tparams, body) if
311              !tparams.corresponds(hkParams)(_.paramVariance == _.paramVariance) &&
312              tparams.corresponds(hkParams)(varianceConforms) =>

316             tl => arg.paramInfos.map(_.subst(arg, tl).bounds),
317             tl => arg.resultType.subst(arg, tl)
