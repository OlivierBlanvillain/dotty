51   /** Entry point to the compiler that can be conveniently used with Java reflection.
52    *
53    *  This entry point can easily be used without depending on the `dotty` package,
54    *  you only need to depend on `dotty-interfaces` and call this method using
55    *  reflection. This allows you to write code that will work against multiple
56    *  versions of dotty without recompilation.
57    *
58    *  The trade-off is that you can only pass a SimpleReporter to this method
59    *  and not a normal Reporter which is more powerful.
60    *
61    *  Usage example: [[https://github.com/lampepfl/dotty/tree/master/test/test/InterfaceEntryPointTest.scala]]
62    *
63    *  @param args       Arguments to pass to the compiler.
64    *  @param simple     Used to log errors, warnings, and info messages.
65    *                    The default reporter is used if this is `null`.
66    *  @param callback   Used to execute custom code during the compilation
67    *                    process. No callbacks will be executed if this is `null`.
68    *  @return
69    */
70   final def process(args: Array[String], simple: interfaces.SimpleReporter,
71     callback: interfaces.CompilerCallback): interfaces.ReporterResult = {
72     val reporter = if (simple == null) null else Reporter.fromSimpleReporter(simple)
73     process(args, reporter, callback)
74   }

77    *

79    *  in method `runCompiler`
80    *
81    *  @param args       Arguments to pass to the compiler.
82    *  @param reporter   Used to log errors, warnings, and info messages.
83    *                    The default reporter is used if this is `null`.
84    *  @param callback   Used to execute custom code during the compilation
85    *                    process. No callbacks will be executed if this is `null`.
86    *  @return           The `Reporter` used. Use `Reporter#hasErrors` to check
87    *                    if compilation succeeded.
88    */
89   final def process(args: Array[String], reporter: Reporter = null,
90     callback: interfaces.CompilerCallback = null): Reporter = {
91     val ctx = initCtx.fresh
92     if (reporter != null)
93       ctx.setReporter(reporter)
94     if (callback != null)
95       ctx.setCompilerCallback(callback)
96     process(args, ctx)
97   }

100    *
101    *  This overload is provided for compatibility reasons: the
102    *  `RawCompiler` of sbt expects this method to exist and calls
103    *  it using reflection. Keeping it means that we can change
104    *  the other overloads without worrying about breaking compatibility
105    *  with sbt.
106    */

108     process(args, null: Reporter, null: interfaces.CompilerCallback)

110   /** Entry point to the compiler using a custom `Context`.
111    *
112    *  In most cases, you do not need a custom `Context` and should
113    *  instead use one of the other overloads of `process`. However,
114    *  the other overloads cannot be overriden, instead you
115    *  should override this one which they call internally.
116    *
117    *  Usage example: [[https://github.com/lampepfl/dotty/tree/master/test/test/OtherEntryPointsTest.scala]]
118    *  in method `runCompilerWithContext`
119    *
120    *  @param args       Arguments to pass to the compiler.
121    *  @param rootCtx    The root Context to use.
122    *  @return           The `Reporter` used. Use `Reporter#hasErrors` to check
123    *                    if compilation succeeded.
124    */
125   def process(args: Array[String], rootCtx: Context): Reporter = {
126     val (fileNames, ctx) = setup(args, rootCtx)
127     doCompile(newCompiler(ctx), fileNames)(ctx)
128   }
129 
130   def main(args: Array[String]): Unit = {
131     // Preload scala.util.control.NonFatal. Otherwise, when trying to catch a StackOverflowError,
132     // we may try to load it but fail with another StackOverflowError and lose the original exception,
133     // see <https://groups.google.com/forum/#!topic/scala-user/kte6nak-zPM>.
134     val _ = NonFatal
135     sys.exit(if (process(args).hasErrors) 1 else 0)
136   }
