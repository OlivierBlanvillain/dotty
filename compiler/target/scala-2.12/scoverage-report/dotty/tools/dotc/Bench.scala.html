15   private def ntimes(n: Int)(op: => Reporter): Reporter =
16     (emptyReporter /: (0 until n)) ((_, _) => op)

18   override def doCompile(compiler: Compiler, fileNames: List[String])(implicit ctx: Context): Reporter =
19     ntimes(numRuns) {
20       val start = System.nanoTime()

31   def extractNumArg(args: Array[String], name: String, default: Int = 1): (Int, Array[String]) = {
32     val pos = args indexOf name
33     if (pos < 0) (default, args)
34     else (args(pos + 1).toInt, (args take pos) ++ (args drop (pos + 2)))
35   }

37   override def process(args: Array[String], rootCtx: Context): Reporter = {
38     val (numCompilers, args1) = extractNumArg(args, "#compilers")
39     val (numRuns, args2) = extractNumArg(args1, "#runs")
40     this.numRuns = numRuns
41     ntimes(numCompilers)(super.process(args2, rootCtx))
42   }
