36   /** Produces the following contexts, from outermost to innermost
37    *
38    *    bootStrap:   A context with next available runId and a scope consisting of
39    *                 the RootPackage _root_
40    *    start        A context with RootClass as owner and the necessary initializations
41    *                 for type checking.
42    *    imports      For each element of RootImports, an import context
43    */
44   protected[this] def rootContext(implicit ctx: Context): Context = {
45     ctx.initialize()(ctx)
46     ctx.setPhasePlan(comp.phases)
47     val rootScope = new MutableScope
48     val bootstrap = ctx.fresh
49       .setPeriod(Period(comp.nextRunId, FirstPhaseId))
50       .setScope(rootScope)
51     rootScope.enter(ctx.definitions.RootPackage)(bootstrap)
52     val start = bootstrap.fresh
53       .setOwner(defn.RootClass)
54       .setTyper(new Typer)
55       .addMode(Mode.ImplicitsEnabled)
56       .setTyperState(new TyperState(ctx.typerState))
57       .setFreshNames(new FreshNameCreator.Default)
58     ctx.initialize()(start) // re-initialize the base context with start
59     def addImport(ctx: Context, refFn: () => TermRef) =
60       ctx.fresh.setImportInfo(ImportInfo.rootImport(refFn)(ctx))
61     (start.setRun(this) /: defn.RootImportFns)(addImport)
62   }

104   def getSource(fileName: String): SourceFile = {
105     val f = new PlainFile(io.Path(fileName))
106     if (f.isDirectory) {
107       ctx.error(s"expected file, received directory '$fileName'")
108       NoSource
109     } else if (f.exists) {
110       val encoding = ctx.settings.encoding.value
111       new SourceFile(f, Codec(encoding))
112     } else {
113       ctx.error(s"not found: $fileName")
114       NoSource
115     }
116   }

149   protected def compileUnits()(implicit ctx: Context) = Stats.maybeMonitored {
150     ctx.checkSingleThreaded()
151 
152     // If testing pickler, make sure to stop after pickling phase:
153     val stopAfter =
154       if (ctx.settings.YtestPickler.value) List("pickler")
155       else ctx.settings.YstopAfter.value
156 
157     val phases = ctx.squashPhases(ctx.phasePlan,
158       ctx.settings.Yskip.value, ctx.settings.YstopBefore.value, stopAfter, ctx.settings.Ycheck.value)
159     ctx.usePhases(phases)
160 
161     def runPhases(implicit ctx: Context) = {
162       var lastPrintedTree: PrintedTree = NoPrintedTree
163       val profiler = ctx.profiler
164 
165       for (phase <- ctx.allPhases)
166         if (phase.isRunnable)
167           Stats.trackTime(s"$phase ms ") {
168             val start = System.currentTimeMillis
169             val profileBefore = profiler.beforePhase(phase)
170             units = phase.runOn(units)
171             profiler.afterPhase(phase, profileBefore)
172             if (ctx.settings.Xprint.value.containsPhase(phase)) {
173               for (unit <- units) {
174                 lastPrintedTree =
175                   printTree(lastPrintedTree)(ctx.fresh.setPhase(phase.next).setCompilationUnit(unit))
176               }
177             }
178             ctx.informTime(s"$phase ", start)
179             Stats.record(s"total trees at end of $phase", ast.Trees.ntrees)
180             for (unit <- units)
181               Stats.record(s"retained typed trees at end of $phase", unit.tpdTree.treeSize)
182           }
183 
184       profiler.finished()
185     }
186 
187     val runCtx = ctx.fresh
188     runCtx.setProfiler(Profiler())
189     ctx.phases.foreach(_.initContext(runCtx))
190     runPhases(runCtx)
191     if (!ctx.reporter.hasErrors) Rewrites.writeBack()
192   }

198     if (!files.contains(file) && !lateFiles.contains(file)) {

214   private def printTree(last: PrintedTree)(implicit ctx: Context): PrintedTree = {
215     val unit = ctx.compilationUnit
216     val prevPhase = ctx.phase.prev // can be a mini-phase
217     val squashedPhase = ctx.squashed(prevPhase)
218     val treeString = unit.tpdTree.show(ctx.withProperty(XprintMode, Some(())))
219 
220     ctx.echo(s"result of $unit after $squashedPhase:")
221 
222     last match {
223       case SomePrintedTree(phase, lastTreeSting) if lastTreeSting != treeString =>
224         val msg =
225           if (!ctx.settings.XprintDiff.value && !ctx.settings.XprintDiffDel.value) treeString
226           else DiffUtil.mkColoredCodeDiff(treeString, lastTreeSting, ctx.settings.XprintDiffDel.value)
227         ctx.echo(msg)
228         SomePrintedTree(squashedPhase.toString, treeString)
229 
230       case SomePrintedTree(phase, lastTreeSting) =>
231         ctx.echo("  Unchanged since " + phase)
232         last
233 
234       case NoPrintedTree =>
235         ctx.echo(treeString)
236         SomePrintedTree(squashedPhase.toString, treeString)
237     }
238   }

240   def compile(sourceCode: String): Unit = {
241     val virtualFile = new VirtualFile(sourceCode) // use source code as name as it's used for equals
242     val writer = new BufferedWriter(new OutputStreamWriter(virtualFile.output, "UTF-8")) // buffering is still advised by javadoc
243     writer.write(sourceCode)
244     writer.close()
245     compileSources(List(new SourceFile(virtualFile, Codec.UTF8)))
246   }
