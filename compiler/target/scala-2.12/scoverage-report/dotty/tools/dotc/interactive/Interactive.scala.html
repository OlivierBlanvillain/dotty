14 /** High-level API to get information out of typed trees, designed to be used by IDEs.
15  *
16  *  @see `InteractiveDriver` to get typed trees from code.
17  */

21   /** Does this tree define a symbol ? */
22   def isDefinition(tree: Tree) =
23     tree.isInstanceOf[DefTree with NameTree]

25   /** The type of the closest enclosing tree with a type containing position `pos`. */
26   def enclosingType(trees: List[SourceTree], pos: SourcePosition)(implicit ctx: Context): Type = {
27     val path = pathTo(trees, pos)

32   /** The source symbol of the closest enclosing tree with a symbol containing position `pos`.
33    *
34    *  @see sourceSymbol
35    */
36   def enclosingSourceSymbol(trees: List[SourceTree], pos: SourcePosition)(implicit ctx: Context): Symbol = {
37     pathTo(trees, pos).dropWhile(!_.symbol.exists).headOption match {
38       case Some(tree) =>

40       case None =>
41         NoSymbol
42     }
43   }

45   /** A symbol related to `sym` that is defined in source code.
46    *
47    *  @see enclosingSourceSymbol
48    */
49   @tailrec def sourceSymbol(sym: Symbol)(implicit ctx: Context): Symbol =
50     if (!sym.exists)
51       sym
52     else if (sym.is(ModuleVal))
53       sourceSymbol(sym.moduleClass) // The module val always has a zero-extent position
54     else if (sym.is(Synthetic)) {
55       val linked = sym.linkedClass
56       if (linked.exists && !linked.is(Synthetic))
57         linked
58       else
59         sourceSymbol(sym.owner)
60     }
61     else if (sym.isPrimaryConstructor)
62       sourceSymbol(sym.owner)
63     else sym

65   private def safely[T](op: => List[T]): List[T] =
66     try op catch { case ex: TypeError => Nil }

68   /** Get possible completions from tree at `pos`
69    *
70    *  @return offset and list of symbols for possible completions
71    */
72   def completions(trees: List[SourceTree], pos: SourcePosition)(implicit ctx: Context): (Int, List[Symbol]) = {
73     val path = pathTo(trees, pos)
74     val boundary = enclosingDefinitionInPath(path).symbol
75 
76     // FIXME: Get all declarations available in the current scope, not just
77     // those from the enclosing class
78     def scopeCompletions: List[Symbol] =
79       boundary.enclosingClass match {
80         case csym: ClassSymbol =>
81           val classRef = csym.classInfo.appliedRef
82           completions(classRef, boundary)
83         case _ =>
84           Nil
85       }
86 
87     path.headOption.map {
88       case sel @ Select(qual, name) =>
89         // When completing "`a.foo`, return the members of `a`
90         (sel.pos.point, completions(qual.tpe, boundary))
91       case id @ Ident(name) =>
92         (id.pos.point, scopeCompletions)
93       case _ =>
94         (0, scopeCompletions)
95     }
96     .getOrElse((0, Nil))
97   }
98 
99   /** Possible completions of members of `prefix` which are accessible when called inside `boundary` */
100   def completions(prefix: Type, boundary: Symbol)(implicit ctx: Context): List[Symbol] =
101     safely {
102       if (boundary != NoSymbol) {
103         val boundaryCtx = ctx.withOwner(boundary)
104         prefix.memberDenots(completionsFilter, (name, buf) =>
105           buf ++= prefix.member(name).altsWith{ d =>
106             !d.isAbsent &&
107             !d.is(Synthetic) && !d.is(Artifact) &&
108             d.symbol.isAccessibleFrom(prefix)(boundaryCtx)
109           }
110         ).map(_.symbol).toList
111       }
112       else Nil
113     }
114 
115   /** Filter for names that should appear when looking for completions. */
116   private[this] object completionsFilter extends NameFilter {
117     def apply(pre: Type, name: Name)(implicit ctx: Context): Boolean =
118       !name.isConstructorName && name.toTermName.info.kind == SimpleNameKind
119   }

121   /** Find named trees with a non-empty position whose symbol match `sym` in `trees`.
122    *
123    *  Note that nothing will be found for symbols not defined in source code,
124    *  use `sourceSymbol` to get a symbol related to `sym` that is defined in
125    *  source code.
126    *
127    *  @param includeReferences  If true, include references and not just definitions
128    *  @param includeOverriden   If true, include trees whose symbol is overriden by `sym`
129    */
130   def namedTrees(trees: List[SourceTree], includeReferences: Boolean, includeOverriden: Boolean, sym: Symbol)
131    (implicit ctx: Context): List[SourceTree] =
132     if (!sym.exists)
133       Nil
134     else
135       namedTrees(trees, includeReferences, matchSymbol(_, sym, includeOverriden))

137   /** Find named trees with a non-empty position whose name contains `nameSubstring` in `trees`.
138    *
139    *  @param includeReferences  If true, include references and not just definitions
140    */
141   def namedTrees(trees: List[SourceTree], includeReferences: Boolean, nameSubstring: String)
142    (implicit ctx: Context): List[SourceTree] =
143     namedTrees(trees, includeReferences, _.show.toString.contains(nameSubstring))

145   /** Find named trees with a non-empty position satisfying `treePredicate` in `trees`.
146    *
147    *  @param includeReferences  If true, include references and not just definitions
148    */
149   def namedTrees(trees: List[SourceTree], includeReferences: Boolean, treePredicate: NameTree => Boolean)
150     (implicit ctx: Context): List[SourceTree] = safely {
151     val buf = new mutable.ListBuffer[SourceTree]
152 
153     trees foreach { case SourceTree(topTree, source) =>
154       (new untpd.TreeTraverser {
155         override def traverse(tree: untpd.Tree)(implicit ctx: Context) = {
156           tree match {
157             case _: untpd.Inlined =>
158               // Skip inlined trees
159             case utree: untpd.NameTree if tree.hasType =>
160               val tree = utree.asInstanceOf[tpd.NameTree]
161               if (tree.symbol.exists
162                    && !tree.symbol.is(Synthetic)
163                    && tree.pos.exists
164                    && !tree.pos.isZeroExtent
165                    && (includeReferences || isDefinition(tree))
166                    && treePredicate(tree))
167                 buf += SourceTree(tree, source)
168             case _ =>
169           }
170           traverseChildren(tree)
171         }
172       }).traverse(topTree)
173     }
174 
175     buf.toList
176   }

178   /** Check if `tree` matches `sym`.
179    *  This is the case if `sym` is the symbol of `tree` or, if `includeOverriden`
180    *  is true, if `sym` is overriden by `tree`.
181    */
182   def matchSymbol(tree: Tree, sym: Symbol, includeOverriden: Boolean)(implicit ctx: Context): Boolean =
183     (sym == tree.symbol) || (includeOverriden && tree.symbol.allOverriddenSymbols.contains(sym))

186   /** The reverse path to the node that closest encloses position `pos`,
187    *  or `Nil` if no such path exists. If a non-empty path is returned it starts with
188    *  the tree closest enclosing `pos` and ends with an element of `trees`.
189    */
190   def pathTo(trees: List[SourceTree], pos: SourcePosition)(implicit ctx: Context): List[Tree] =
191     trees.find(_.pos.contains(pos)) match {
192       case Some(tree) =>
193         // FIXME: We shouldn't need a cast. Change NavigateAST.pathTo to return a List of Tree?
194         val path = NavigateAST.pathTo(pos.pos, tree.tree, skipZeroExtent = true).asInstanceOf[List[untpd.Tree]]
195 
196         path.dropWhile(!_.hasType).asInstanceOf[List[tpd.Tree]]
197       case None =>
198         Nil
199     }
200 
201   /** The first tree in the path that is a definition. */
202   def enclosingDefinitionInPath(path: List[Tree])(implicit ctx: Context): Tree =
203     path.find(_.isInstanceOf[DefTree]).getOrElse(EmptyTree)
