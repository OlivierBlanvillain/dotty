13 case class SourceTree(tree: tpd.NameTree, source: SourceFile) {
14   /** The position of `tree` */
15   def pos(implicit ctx: Context): SourcePosition = source.atPos(tree.pos)
16 
17   /** The position of the name in `tree` */
18   def namePos(implicit ctx: Context): SourcePosition = {
19     // FIXME: Merge with NameTree#namePos ?
20     val treePos = tree.pos
21     if (treePos.isZeroExtent)
22       NoSourcePosition
23     else {
24       val nameLength = tree.name.stripModuleClassSuffix.show.toString.length
25       val position = {
26         // FIXME: This is incorrect in some cases, like with backquoted identifiers,
27         //        see https://github.com/lampepfl/dotty/pull/1634#issuecomment-257079436
28         val (start, end) =
29           if (!treePos.isSynthetic)
30             (treePos.point, treePos.point + nameLength)
31           else
32             // If we don't have a point, we need to find it
33             (treePos.end - nameLength, treePos.end)
34         Position(start, end, start)
35       }
36       source.atPos(position)
37     }
38   }
39 }
40 object SourceTree {
41   def fromSymbol(sym: ClassSymbol)(implicit ctx: Context): Option[SourceTree] = {
42     if (sym == defn.SourceFileAnnot || // FIXME: No SourceFile annotation on SourceFile itself
43         sym.sourceFile == null) // FIXME: We cannot deal with external projects yet
44       None
45     else {
46       import ast.Trees._
47       def sourceTreeOfClass(tree: tpd.Tree): Option[SourceTree] = tree match {
48         case PackageDef(_, stats) => stats.flatMap(sourceTreeOfClass).headOption
49         case tree: tpd.TypeDef if tree.symbol == sym =>
50           val sourceFile = new SourceFile(sym.sourceFile, Codec.UTF8)
51           Some(SourceTree(tree, sourceFile))
52         case _ => None
53       }
54       sourceTreeOfClass(sym.tree)
55     }
56   }
