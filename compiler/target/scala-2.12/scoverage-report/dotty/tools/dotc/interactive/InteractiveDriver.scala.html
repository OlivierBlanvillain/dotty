79   private def classNames(cp: ClassPath, packageName: String): List[String] = {
80     def className(classSegments: List[String]) =
81       classSegments.mkString(".").stripSuffix(".class")
82 
83     val ClassPathEntries(pkgs, classReps) = cp.list(packageName)
84 
85     classReps
86       .filter((classRep: ClassRepresentation) => classRep.binary match {
87         case None =>
88           true
89         case Some(binFile) =>
90           val prefix =
91             if (binFile.name.endsWith(".class"))
92               binFile.name.stripSuffix(".class")
93             else
94               null
95           prefix != null && {
96             binFile match {
97               case pf: PlainFile =>
98                 tastySuffixes.map(suffix => pf.givenPath.parent / (prefix + suffix)).exists(_.exists)
99               case _ =>
100                 sys.error(s"Unhandled file type: $binFile [getClass = ${binFile.getClass}]")
101             }
102           }
103       })
104       .map(classRep => (packageName ++ (if (packageName != "") "." else "") ++ classRep.name)).toList ++
105     pkgs.flatMap(pkg => classNames(cp, pkg.name))
106   }

113   private val (zipClassPaths, dirClassPaths) = currentCtx.platform.classPath(currentCtx) match {
114     case AggregateClassPath(cps) =>

123   // Like in `ZipArchiveFileLookup` we assume that zips are immutable
124   private val zipClassPathClasses: Seq[String] = zipClassPaths.flatMap { zipCp =>
125     // Working with Java 8 stream without SAMs and scala-java8-compat is awful.
126     val entries = new ZipFile(zipCp.zipFile)
127       .stream
128       .toArray(new IntFunction[Array[ZipEntry]] { def apply(size: Int) = new Array(size) })
129       .toSeq
130     for {
131       entry <- entries
132       name = entry.getName
133       tastySuffix <- tastySuffixes
134       if name.endsWith(tastySuffix)
135     } yield name.replace("/", ".").stripSuffix(tastySuffix)
136   }

138   // FIXME: classfiles in directories may change at any point, so we retraverse
139   // the directories each time, if we knew when classfiles changed (sbt
140   // server-mode might help here), we could do cache invalidation instead.
141   private def dirClassPathClasses: Seq[String] = {
142     val names = new mutable.ListBuffer[String]
143     dirClassPaths.foreach { dirCp =>
144       val root = dirCp.dir.toPath
145       Files.walkFileTree(root, new SimpleFileVisitor[Path] {
146         override def visitFile(path: Path, attrs: BasicFileAttributes) = {
147           if (!attrs.isDirectory) {
148             val name = path.getFileName.toString
149             for {
150               tastySuffix <- tastySuffixes
151               if name.endsWith(tastySuffix)
152             } {
153               names += root.relativize(path).toString.replace("/", ".").stripSuffix(tastySuffix)
154             }
155           }
156           FileVisitResult.CONTINUE
157         }
158       })
159     }
160     names.toList
161   }

180   /** Remove attachments and error out completers. The goal is to avoid
181    *  having a completer hanging in a typed tree which can capture the context
182    *  of a previous run. Note that typed trees can have untyped or partially
183    *  typed children if the source contains errors.
184    */
185   private def cleanup(tree: tpd.Tree)(implicit ctx: Context): Unit = {
186     val seen = mutable.Set.empty[tpd.Tree]
187     def cleanupTree(tree: tpd.Tree): Unit = {
188       seen += tree
189       tree.foreachSubTree { t =>
190         if (t.symbol.exists && t.hasType) {
191           if (!t.symbol.isCompleted) t.symbol.info = UnspecifiedErrorType
192           t.symbol.annotations.foreach { annot =>
193             /* In some cases annotations are are used on themself (possibly larger cycles).
194             *  This is the case with the java.lang.annotation.Target annotation, would end
195             *  in an infinite loop while cleaning. The `seen` is added to ensure that those
196             *  trees are not cleand twice.
197             *  TODO: Find a less expensive way to check for those cycles.
198             */
199             if (!seen(annot.tree))
200               cleanupTree(annot.tree)
201           }
202         }
203         t.removeAllAttachments()
204       }
205     }
206     cleanupTree(tree)
207   }

209   def run(uri: URI, sourceCode: String): List[MessageContainer] = {
210     val previousCtx = myCtx
211     try {
212       val reporter =
213         new StoreReporter(null) with UniqueMessagePositions with HideNonSensicalMessages
214 
215       val run = compiler.newRun(myInitCtx.fresh.setReporter(reporter))
216       myCtx = run.runContext
217 
218       implicit val ctx = myCtx
219 
220       val virtualFile = new VirtualFile(uri.toString, Paths.get(uri).toString)
221       val writer = new BufferedWriter(new OutputStreamWriter(virtualFile.output, "UTF-8"))
222       writer.write(sourceCode)
223       writer.close()
224       val source = new SourceFile(virtualFile, Codec.UTF8)
225       myOpenedFiles(uri) = source
226 
227       run.compileSources(List(source))
228       run.printSummary()
229       val t = ctx.run.units.head.tpdTree
230       cleanup(t)
231       myOpenedTrees(uri) = topLevelClassTrees(t, source)
232 
233       reporter.removeBufferedMessages
234     }
235     catch {
236       case ex: FatalError  =>
237         myCtx = previousCtx
238         close(uri)
239         Nil
240     }
241   }
242 
243   def close(uri: URI): Unit = {
244     myOpenedFiles.remove(uri)
245     myOpenedTrees.remove(uri)
246   }
